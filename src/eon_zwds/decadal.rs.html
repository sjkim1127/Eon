1//! 자미두수 대한(大限, 10년 대운) 연산 모듈
2//!
3//! 성별과 생년 천간의 음양에 따라 순행/역행 방향을 정하고, 12개 궁에 대한의 나이 범위와 대간(大干)을 설정합니다.
4
5use eon_core::birth::Gender;
6use eon_saju::core::stem::HeavenlyStem;
7use eon_saju::core::branch::EarthlyBranch;
8use crate::types::{PalaceIndex, FiveElementsClass, DaXian};
9use crate::palace::{fix_index, get_palace_stem, zwds_idx_to_std_idx};
10
11/// 대한 목록을 생성하여 반환합니다.
12///
13/// * `gender` - 성별 (남성/여성)
14/// * `year_stem` - 출생년도 천간
15/// * `soul_idx` - 명궁의 ZWDS 지지 인덱스
16/// * `five_elements` - 오행국
17pub fn calculate_daxian(
18    gender: Gender,
19    year_stem: HeavenlyStem,
20    soul_idx: PalaceIndex,
21    five_elements: FiveElementsClass,
22) -> Vec<DaXian> {
23    let mut daxian_list = Vec::new();
24
25    // 1. 순행/역행 방향 결정
26    // 양남음녀(陽男陰女) -> 순행, 음남양녀(陰男陽女) -> 역행
27    // 양천간: 甲, 丙, 戊, 庚, 壬 (index % 2 == 0)
28    // 음천간: 乙, 丁, 己, 辛, 癸 (index % 2 == 1)
29    let is_yang_stem = year_stem.index() % 2 == 0;
30    let is_male = matches!(gender, Gender::Male);
31
32    let is_forward = if is_male {
33        is_yang_stem // 양남 -> 순행, 음남 -> 역행
34    } else {
35        !is_yang_stem // 음녀 -> 순행, 양녀 -> 역행
36    };
37
38    let step = if is_forward { 1 } else { -1 };
39    let start_age = five_elements.starting_age() as u32;
40
41    for i in 0..12 {
42        let age_start = start_age + i * 10;
43        let age_end = age_start + 9;
44
45        // 명궁(soul_idx)에서 시작하여 순행/역행
46        let p_idx = fix_index(soul_idx as i32 + step * i as i32);
47
48        // 해당 궁의 천간과 지지 구하기
49        let stem = get_palace_stem(year_stem, p_idx);
50        let branch_std = zwds_idx_to_std_idx(p_idx);
51        let branch = EarthlyBranch::from_index(branch_std as i32);
52
53        daxian_list.push(DaXian {
54            index: i,
55            age_start,
56            age_end,
57            palace_idx: p_idx,
58            stem_hanja: stem.hanja().to_string(),
59            branch_hanja: branch.hanja().to_string(),
60        });
61    }
62
63    daxian_list
64}
