1//! 자미두수 궁(Palace) 연산 모듈
2//!
3//! 명궁(命宮)과 신궁(身宮) 계산, 오행국(五行局) 결정, 12궁명 매핑을 처리합니다.
4
5use crate::types::{FiveElementsClass, PalaceIndex, PalaceName};
6use eon_saju::core::branch::EarthlyBranch;
7use eon_saju::core::stem::HeavenlyStem;

9/// 인덱스 보정 (0~11 범위)
10pub fn fix_index(i: i32) -> usize {
11    let r = i % 12;
12    if r < 0 {
13        (r + 12) as usize
14    } else {
15        r as usize
16    }
17}
18
19/// ZWDS 지지 인덱스 (0=寅) -> 표준 지지 인덱스 (0=子)
20pub fn zwds_idx_to_std_idx(zwds_idx: usize) -> usize {
21    (zwds_idx + 2) % 12
22}
23
24/// 표준 지지 인덱스 (0=子) -> ZWDS 지지 인덱스 (0=寅)
25pub fn std_idx_to_zwds_idx(std_idx: usize) -> usize {
26    (std_idx + 10) % 12
27}
28
29/// 명궁 및 신궁 인덱스 계산 (ZWDS 인덱스 반환)
30///
31/// * `lunar_month` - 음력 월 (1~12)
32/// * `time_branch_std_idx` - 태어난 시의 표준 지지 인덱스 (0=子)
33pub fn get_soul_and_body_index(
34    lunar_month: u32,
35    time_branch_std_idx: usize,
36) -> (PalaceIndex, PalaceIndex) {
37    let month_idx = (lunar_month - 1) as i32;
38    let time_idx = time_branch_std_idx as i32;
39
40    let soul = fix_index(month_idx - time_idx);
41    let body = fix_index(month_idx + time_idx);
42
43    (soul, body)
44}

46/// 각 궁의 천간 계산 (五虎遁年起月법 적용)
47///
48/// * `year_stem` - 출생년도의 천간
49/// * `palace_zwds_idx` - ZWDS 지지 인덱스 (0=寅, 1=卯...)
50pub fn get_palace_stem(year_stem: HeavenlyStem, palace_zwds_idx: PalaceIndex) -> HeavenlyStem {
51    let start_stem_idx = match year_stem {
52        HeavenlyStem::Jia | HeavenlyStem::Ji => 2,   // 丙
53        HeavenlyStem::Yi | HeavenlyStem::Geng => 4,  // 戊
54        HeavenlyStem::Bing | HeavenlyStem::Xin => 6, // 庚
55        HeavenlyStem::Ding | HeavenlyStem::Ren => 8, // 壬
56        HeavenlyStem::Wu | HeavenlyStem::Gui => 0,   // 甲
57    };
58    let stem_idx = (start_stem_idx + palace_zwds_idx) % 10;
59    HeavenlyStem::from_index(stem_idx as i32)
60}

62/// 오행국(Five Elements Class) 결정
63///
64/// * `soul_stem` - 명궁의 천간
65/// * `soul_zwds_idx` - 명궁의 ZWDS 지지 인덱스
66pub fn get_five_elements(soul_stem: HeavenlyStem, soul_zwds_idx: PalaceIndex) -> FiveElementsClass {
67    let stem_num = match soul_stem {
68        HeavenlyStem::Jia | HeavenlyStem::Yi => 1,
69        HeavenlyStem::Bing | HeavenlyStem::Ding => 2,
70        HeavenlyStem::Wu | HeavenlyStem::Ji => 3,
71        HeavenlyStem::Geng | HeavenlyStem::Xin => 4,
72        HeavenlyStem::Ren | HeavenlyStem::Gui => 5,
73    };
74
75    let soul_branch_std_idx = zwds_idx_to_std_idx(soul_zwds_idx);
76    let branch = EarthlyBranch::from_index(soul_branch_std_idx as i32);
77
78    let branch_num = match branch {
79        EarthlyBranch::Zi | EarthlyBranch::Wu | EarthlyBranch::Chou | EarthlyBranch::Wei => 1,
80        EarthlyBranch::Yin | EarthlyBranch::Shen | EarthlyBranch::Mao | EarthlyBranch::You => 2,
81        EarthlyBranch::Chen | EarthlyBranch::Xu | EarthlyBranch::Si | EarthlyBranch::Hai => 3,
82    };
83
84    let sum = stem_num + branch_num;
85    let rem = sum % 5;
86
87    match rem {
88        1 => FiveElementsClass::Wood3,
89        2 => FiveElementsClass::Metal4,
90        3 => FiveElementsClass::Water2,
91        4 => FiveElementsClass::Fire6,
92        0 => FiveElementsClass::Earth5,
93        _ => unreachable!(),
94    }
95}

97/// 각 ZWDS 인덱스의 궁명(PalaceName) 결정
98///
99/// * `soul_zwds_idx` - 명궁의 ZWDS 지지 인덱스
100/// * `palace_zwds_idx` - 현재 궁의 ZWDS 지지 인덱스
101pub fn get_palace_name(soul_zwds_idx: PalaceIndex, palace_zwds_idx: PalaceIndex) -> PalaceName {
102    let offset = fix_index(palace_zwds_idx as i32 - soul_zwds_idx as i32);
103    PalaceName::ALL[offset]
104}

106#[cfg(test)]
107mod tests {
108    use super::*;
109
110    #[test]
111    fn test_soul_body_calc() {
112        // 음력 10월, 亥시생 (lunar_month = 10, time_branch_std_idx = 11)
113        let (soul, body) = get_soul_and_body_index(10, 11);
114        // soul = 9 - 11 = -2 -> 10 (子궁)
115        // body = 9 + 11 = 20 -> 8 (戌궁)
116        assert_eq!(soul, 10);
117        assert_eq!(body, 8);
118    }
119
120    #[test]
121    fn test_palace_stem() {
122        // 甲년, ZWDS 0(寅)의 천간 -> 丙
123        assert_eq!(get_palace_stem(HeavenlyStem::Jia, 0), HeavenlyStem::Bing);
124        // 甲년, ZWDS 10(子)의 천간 -> 甲 (start 2 + 10 = 12 % 10 = 2 -> 丙인데? 아! 10 % 10 = 0 -> 丙 + 10 = 丙(2) + 10 = 12 -> 丙?)
125        // 잠시, 공식이 맞는지 검증해보자.
126        // start_stem_idx:
127        // 甲/己年 -> 寅궁(0) 천간 = 丙 (2)
128        // 卯궁(1) 천간 = 丁 (3)
129        // 辰궁(2) 천간 = 戊 (4)
130        // 巳궁(3) 천간 = 己 (5)
131        // 午궁(4) 천간 = 庚 (6)
132        // 未궁(5) 천간 = 辛 (7)
133        // 申궁(6) 천간 = 壬 (8)
134        // 酉궁(7) 천간 = 癸 (9)
135        // 戌궁(8) 천간 = 甲 (0)
136        // 亥궁(9) 천간 = 乙 (1)
137        // 子궁(10) 천간 = 丙 (2)
138        // 丑궁(11) 천간 = 丁 (3)
139        // 공식: (start_stem_idx + palace_zwds_idx) % 10
140        // (2 + 0) % 10 = 2 -> 丙. 맞음.
141        // (2 + 10) % 10 = 2 -> 丙. 맞음! (子궁의 천간은 丙이다. 五虎遁법에 의하면 甲己년의 자궁 천간은 丙이 맞음).
142        // (2 + 8) % 10 = 0 -> 甲. 戌궁의 천간은 甲. 맞음!
143        assert_eq!(get_palace_stem(HeavenlyStem::Jia, 10), HeavenlyStem::Bing);
144    }

146    #[test]
147    fn test_five_elements() {
148        // 명궁 천간: 丙, 명궁 ZWDS: 10(子)
149        // stem_num = 丙(2)
150        // ZWDS 10 -> 표준 0(子) -> branch_num = 1
151        // sum = 3 -> rem = 3 -> Water2 (수이국)
152        let class = get_five_elements(HeavenlyStem::Bing, 10);
153        assert_eq!(class, FiveElementsClass::Water2);
154    }
