1//! 자미두수 격국(Destiny Patterns) 감지 모듈
2//!
3//! 명반의 삼방사정(본궁, 대궁, 재백궁, 관록궁)을 연산하고, 
4//! 주요 6대 전통 길흉 격국을 감지하여 반환합니다.
5
6use std::collections::HashMap;
7use crate::types::{DestinyPattern, PalaceData, ZwdsStar, SiHuaType};
8
9/// 명궁의 삼방사정을 분석하여 길격/흉격 격국 리스트를 반환합니다.
10pub fn analyze_destiny_patterns(
11    soul_idx: usize,
12    star_positions: &HashMap<ZwdsStar, usize>,
13    palaces: &[PalaceData; 12],
14) -> Vec<DestinyPattern> {
15    let mut patterns = Vec::new();
16
17    // 삼방사정 (본궁, 대궁/천이궁, 재백궁, 관록궁) 구하기
18    // ZWDS 좌표계: 0=寅, 1=卯, ..., 11=丑 (반시계 방향)
19    // Ming = soul_idx
20    // Caibo = (soul_idx + 4) % 12 (재백)
21    // Qianyi = (soul_idx + 6) % 12 (천이/대궁)
22    // Guanlu = (soul_idx + 8) % 12 (관록)
23    let ming_idx = soul_idx;
24    let qianyi_idx = (soul_idx + 6) % 12;
25    let caibo_idx = (soul_idx + 4) % 12;
26    let guanlu_idx = (soul_idx + 8) % 12;
27
28    let three_four_directions = [ming_idx, qianyi_idx, caibo_idx, guanlu_idx];
29
30    // Helper to check if a star is in the three-party/four-direction
31    let has_star_in_three_four = |star: ZwdsStar| -> bool {
32        if let Some(&pos) = star_positions.get(&star) {
33            three_four_directions.contains(&pos)
34        } else {
35            false
36        }
37    };
38
39    // Helper to check if a star is in a specific palace index
40    let has_star_in_palace = |star: ZwdsStar, palace_idx: usize| -> bool {
41        star_positions.get(&star) == Some(&palace_idx)
42    };
43
44    // 1. 자부조원격 (紫府朝垣格)
45    // 조건: 자미(ZiWei)와 천부(TianFu)가 삼방사정(명궁, 재백궁, 관록궁, 천이궁)에 모두 있으면서, 둘 다 명궁(Ming)에 있지 않은 경우.
46    if has_star_in_three_four(ZwdsStar::ZiWei) && has_star_in_three_four(ZwdsStar::TianFu) {
47        let both_in_ming = has_star_in_palace(ZwdsStar::ZiWei, ming_idx) && has_star_in_palace(ZwdsStar::TianFu, ming_idx);
48        if !both_in_ming {
49            patterns.push(DestinyPattern {
50                name_hanja: "紫府朝垣".to_string(),
51                name_korean: "자부조원격".to_string(),
52                is_auspicious: true,
53                description_korean: "명궁의 삼방사정에서 자미성과 천부성을 모두 비추어, 평생 물질적 풍요와 높은 명예를 누리는 대표적인 길격입니다.".to_string(),
54                description_english: "The Emperor (Ziwei) and Treasury (Tianfu) both shine on the Life Palace from the three-party and four-direction, indicating lifelong wealth and high social status.".to_string(),
55            });
56        }
57    }
58
59    // 2. 삼기조명격 (三奇加會格 / 三奇朝命格)
60    // 조건: 명궁의 삼방사정에서 화록(HuaLu), 화권(HuaQuan), 화과(HuaKe)를 모두 보는 경우.
61    let mut has_lu = false;
62    let mut has_quan = false;
63    let mut has_ke = false;
64    for &p_idx in &three_four_directions {
65        for star_in_p in &palaces[p_idx].stars {
66            if let Some(sihua) = star_in_p.si_hua {
67                match sihua {
68                    SiHuaType::HuaLu => has_lu = true,
69                    SiHuaType::HuaQuan => has_quan = true,
70                    SiHuaType::HuaKe => has_ke = true,
71                    _ => {}
72                }
73            }
74        }
75    }
76    if has_lu && has_quan && has_ke {
77        patterns.push(DestinyPattern {
78            name_hanja: "三奇加會".to_string(),
79            name_korean: "삼기조명격".to_string(),
80            is_auspicious: true,
81            description_korean: "명궁의 삼방사정에서 화록, 화권, 화과를 모두 만나 학문적 성취, 권력, 재물을 크게 이루며 사회적으로 대성하는 길격입니다.".to_string(),
82            description_english: "The three transformations Hua Lu, Hua Quan, and Hua Ke meet in the three-party and four-direction, leading to outstanding achievements, authority, and prosperity.".to_string(),
83        });
84    }
85
86    // 3. 양량창록격 (陽梁昌祿格)
87    // 조건: 태양(TaiYang), 천량(TianLiang), 문창(WenChang), 록존(LuCun) 또는 화록(HuaLu)이 삼방사정에서 모두 만나는 경우.
88    if has_star_in_three_four(ZwdsStar::TaiYang)
89        && has_star_in_three_four(ZwdsStar::TianLiang)
90        && has_star_in_three_four(ZwdsStar::WenChang)
91    {
92        let has_lu_or_lucun = has_star_in_three_four(ZwdsStar::LuCun) || has_lu;
93        if has_lu_or_lucun {
94            patterns.push(DestinyPattern {
95                name_hanja: "陽梁昌祿".to_string(),
96                name_korean: "양량창록격".to_string(),
97                is_auspicious: true,
98                description_korean: "태양, 천량, 문창, 록존(혹은 화록)이 삼방사정에서 결합하여 시험에서 우수한 성적을 거두거나 학술적, 공직 분야에서 크게 이름을 떨치는 길격입니다.".to_string(),
99                description_english: "The Sun (Taiyang), Blessing (Tianliang), Intellect (Wenchang), and wealth (Lucun/Hua Lu) combine in the three-party and four-direction, indicating exceptional academic success and official honors.".to_string(),
100            });
101        }
102    }
103
104    // 4. 극향리명격 (極向離明格)
105    // 조건: 자미(ZiWei)가 午宮(index 4)에 단독으로 배치되고, 이 午宮이 명궁(Ming Palace)인 경우.
106    if ming_idx == 4 && has_star_in_palace(ZwdsStar::ZiWei, 4) {
107        let mut main_stars_in_wu = 0;
108        for star_in_p in &palaces[4].stars {
109            if star_in_p.star.is_main_star() {
110                main_stars_in_wu += 1;
111            }
112        }
113        if main_stars_in_wu == 1 {
114            patterns.push(DestinyPattern {
115                name_hanja: "極向離明".to_string(),
116                name_korean: "극향리명격".to_string(),
117                is_auspicious: true,
118                description_korean: "자미성이 남쪽인 오궁(午宮)에서 제왕의 자리에 앉아 명궁이 되는 격국으로, 뛰어난 통솔력과 고귀한 지위를 얻게 되는 길격입니다.".to_string(),
119                description_english: "The Emperor (Ziwei) sits alone in the South (Wu Palace) as the Life Palace, symbolizing a king in his high court, representing strong leadership and noble status.".to_string(),
120            });
121        }
122    }
123
124    // 5. 석중은옥격 (石中隱玉格)
125    // 조건: 거문(JuMen)이 子宮(index 10) 또는 午宮(index 4)에 있으며 명궁인 경우.
126    if (ming_idx == 10 || ming_idx == 4) && has_star_in_palace(ZwdsStar::JuMen, ming_idx) {
127        patterns.push(DestinyPattern {
128            name_hanja: "石中隱玉".to_string(),
129            name_korean: "석중은옥격".to_string(),
130            is_auspicious: true,
131            description_korean: "거문성이 자궁이나 오궁의 명궁에 임하여 돌 속에 옥이 감추어져 있는 것처럼, 처음에는 고생을 겪으나 결국 재능을 발현하여 크게 성공하는 길격입니다.".to_string(),
132            description_english: "The Gate (Jumen) resides in the Child (Zi) or Sun (Wu) Palace as the Life Palace. Like a jade hidden inside a stone, talents are revealed over time, leading to great success.".to_string(),
133        });
134    }
135
136    // 6. 살공겁조격 (殺拱劫照格)
137    // 조건: 명궁 삼방사정에 칠살(QiSha)이 있고, 지공(DiKong)이나 지겁(DiJie) 중 하나 이상이 명궁 혹은 삼방사정에 있는 경우.
138    if has_star_in_three_four(ZwdsStar::QiSha)
139        && (has_star_in_three_four(ZwdsStar::DiKong) || has_star_in_three_four(ZwdsStar::DiJie))
140    {
141        patterns.push(DestinyPattern {
142            name_hanja: "殺拱劫照".to_string(),
143            name_korean: "살공겁조격".to_string(),
144            is_auspicious: false,
145            description_korean: "칠살의 파괴력과 지공·지겁의 공허함이 결합하여, 재물의 기복이 크고 삶에 풍파와 급격한 하락을 겪기 쉬운 대표적인 흉격입니다.".to_string(),
146            description_english: "The unstable General (Qisha) meets Void (Dikong) or Robbery (Dijie) in the three-party and four-direction, indicating volatile fortune, sudden drops, and life challenges.".to_string(),
147        });
148    }
149
150    // 7. 기월동량격 (機月同梁格)
151    // 조건: 명궁에 천기/태음/천동/천량 중 하나가 있고, 삼방사정에 이 네 별이 모두 모이는 격국.
152    let is_ming_gi_wol_dong_ryang = [ZwdsStar::TianJi, ZwdsStar::TaiYin, ZwdsStar::TianTong, ZwdsStar::TianLiang]
153        .iter()
154        .any(|&s| has_star_in_palace(s, ming_idx));
155    let has_all_four_gi_wol_dong_ryang = [ZwdsStar::TianJi, ZwdsStar::TaiYin, ZwdsStar::TianTong, ZwdsStar::TianLiang]
156        .iter()
157        .all(|&s| has_star_in_three_four(s));
158    if is_ming_gi_wol_dong_ryang && has_all_four_gi_wol_dong_ryang {
159        patterns.push(DestinyPattern {
160            name_hanja: "機月同梁".to_string(),
161            name_korean: "기월동량격".to_string(),
162            is_auspicious: true,
163            description_korean: "천기, 태음, 천동, 천량이 삼방사정에서 회조하여, 공직이나 학술, 대기업 등 안정적인 조직에서 기획력과 행정력으로 성공을 거두는 전형적인 샐러리맨/참모형 길격입니다.".to_string(),
164            description_english: "Plan (Tianji), Moon (Taiyin), Harmony (Tiantong), and Blessing (Tianliang) gather in the three-party, representing a highly successful administrative or advisory career with stable growth.".to_string(),
165        });
166    }
167
168    // 8. 자부염무상격 (紫府廉武相格)
169    // 조건: 명궁에 자미/천부/염정/무곡/천상 중 하나가 있고, 삼방사정에 이 다섯 별이 모두 모이는 격국.
170    let is_ming_zi_fu_lian_wu_xiang = [ZwdsStar::ZiWei, ZwdsStar::TianFu, ZwdsStar::LianZhen, ZwdsStar::WuQu, ZwdsStar::TianXiang]
171        .iter()
172        .any(|&s| has_star_in_palace(s, ming_idx));
173    let has_all_five_zi_fu_lian_wu_xiang = [ZwdsStar::ZiWei, ZwdsStar::TianFu, ZwdsStar::LianZhen, ZwdsStar::WuQu, ZwdsStar::TianXiang]
174        .iter()
175        .all(|&s| has_star_in_three_four(s));
176    if is_ming_zi_fu_lian_wu_xiang && has_all_five_zi_fu_lian_wu_xiang {
177        patterns.push(DestinyPattern {
178            name_hanja: "紫府廉武相".to_string(),
179            name_korean: "자부염무상격".to_string(),
180            is_auspicious: true,
181            description_korean: "자미, 천부, 염정, 무곡, 천상이 삼방사정에서 강력한 세력을 이루어, 뛰어난 사회적 지위, 강력한 리더십, 풍부한 재정적 결실을 거두는 최상위 길격 중 하나입니다.".to_string(),
182            description_english: "The Emperor, Treasury, Judge, Soldier, and Minister align in the three-party, generating high-class status, leadership potential, and outstanding material wealth.".to_string(),
183        });
184    }
185
186    // 9. 거일동궁격 (巨日同宮格)
187    // 조건: 태양과 거문이 寅궁(0)이나 申궁(6)에 동궁하며 명궁이 되는 격국.
188    if (ming_idx == 0 || ming_idx == 6)
189        && has_star_in_palace(ZwdsStar::TaiYang, ming_idx)
190        && has_star_in_palace(ZwdsStar::JuMen, ming_idx)
191    {
192        patterns.push(DestinyPattern {
193            name_hanja: "巨日同宮".to_string(),
194            name_korean: "거일동궁격".to_string(),
195            is_auspicious: true,
196            description_korean: "태양의 양광이 거문의 어두움을 걷어내는 격국으로, 주로 언론, 학술, 교육, 외교 등 말과 재능을 펼치는 분야나 외국과의 교류에서 대성하는 길격입니다.".to_string(),
197            description_english: "The Sun and Gate occupy the same Palace (Yin/Shen) as the Life Palace, scattering dark clouds with bright light, symbolizing great success in communication, diplomacy, or academic fields.".to_string(),
198        });
199    }
200
201    // 10. 일월동궁격 (日月同宮格)
202    // 조건: 태양과 태음이 丑궁(11)이나 未궁(5)에 동궁하며 명궁이 되는 격국.
203    if (ming_idx == 5 || ming_idx == 11)
204        && has_star_in_palace(ZwdsStar::TaiYang, ming_idx)
205        && has_star_in_palace(ZwdsStar::TaiYin, ming_idx)
206    {
207        patterns.push(DestinyPattern {
208            name_hanja: "日月同宮".to_string(),
209            name_korean: "일월동궁격".to_string(),
210            is_auspicious: true,
211            description_korean: "태양(해)과 태음(달)이 동궁하여 명궁을 비추는 격국으로, 성격이 다재다능하고 지혜로우며 공직이나 연구 분야 등에서 점진적인 명예와 성취를 거둡니다.".to_string(),
212            description_english: "The Sun and Moon occupy the same Palace (Chou/Wei) as the Life Palace, granting deep wisdom, versatile talents, and steady success in official or scholarly careers.".to_string(),
213        });
214    }
215
216    // 11. 일월협명격 (日月夾命格)
217    // 조건: 명궁의 양 옆 궁에 태양과 태음이 나란히 배치되어 명궁을 협하는 격국.
218    let adj1 = (ming_idx + 1) % 12;
219    let adj2 = (ming_idx + 11) % 12;
220    let has_sun_yin_adj = (has_star_in_palace(ZwdsStar::TaiYang, adj1) && has_star_in_palace(ZwdsStar::TaiYin, adj2))
221        || (has_star_in_palace(ZwdsStar::TaiYang, adj2) && has_star_in_palace(ZwdsStar::TaiYin, adj1));
222    if has_sun_yin_adj {
223        patterns.push(DestinyPattern {
224            name_hanja: "日月夾命".to_string(),
225            name_korean: "일월협명격".to_string(),
226            is_auspicious: true,
227            description_korean: "명궁이 태양과 태음 사이에 자리하여 양옆의 은혜로운 빛을 입는 격국으로, 귀인의 전폭적인 도움이나 상사의 후원을 받아 평생 귀한 신분으로 순조롭게 출세하는 길격입니다.".to_string(),
228            description_english: "The Sun and Moon sandwich the Life Palace, bestowing light and support, resulting in strong helpers, high prestige, and an exceptionally smooth career path.".to_string(),
229        });
230    }
231
232    // 12. 쌍록조원격 (雙祿朝垣格)
233    // 조건: 록존과 화록이 명궁 삼방사정에 배치되나, 둘 다 명궁(Ming) 자체에는 없는 경우.
234    let mut has_hua_lu_in_three_directions = false;
235    for &p_idx in &[qianyi_idx, caibo_idx, guanlu_idx] {
236        for star_in_p in &palaces[p_idx].stars {
237            if star_in_p.si_hua == Some(SiHuaType::HuaLu) {
238                has_hua_lu_in_three_directions = true;
239            }
240        }
241    }
242    let has_lucun_in_three_directions = [qianyi_idx, caibo_idx, guanlu_idx]
243        .iter()
244        .any(|&idx| has_star_in_palace(ZwdsStar::LuCun, idx));
245    if has_hua_lu_in_three_directions && has_lucun_in_three_directions {
246        patterns.push(DestinyPattern {
247            name_hanja: "雙祿朝垣".to_string(),
248            name_korean: "쌍록조원격".to_string(),
249            is_auspicious: true,
250            description_korean: "록존과 화록이 삼방사정에서 명궁을 든든하게 비추어 도와주는 격국으로, 일생 재무적 기회가 끊이지 않으며 자수성가하여 큰 부를 축적하는 대표적인 부유 격국입니다.".to_string(),
251            description_english: "Lucun and Hua Lu both shine on the Life Palace from the three-party (excluding the Life Palace itself), bringing abundant financial fortunes and successful self-made wealth.".to_string(),
252        });
253    }
254
255    // 13. 백관조공격 (百官朝拱格)
256    // 조건: 명궁에 자미가 있고, 좌보/우필/천괴/천월/문창/문곡 중 4개 이상의 보좌성이 삼방사정에 있는 경우.
257    if has_star_in_palace(ZwdsStar::ZiWei, ming_idx) {
258        let assistants = [
259            ZwdsStar::ZuoFu,
260            ZwdsStar::YouBi,
261            ZwdsStar::TianKui,
262            ZwdsStar::TianYue,
263            ZwdsStar::WenChang,
264            ZwdsStar::WenQu,
265        ];
266        let assistant_count = assistants.iter().filter(|&&s| has_star_in_three_four(s)).count();
267        if assistant_count >= 4 {
268            patterns.push(DestinyPattern {
269                name_hanja: "百官朝拱".to_string(),
270                name_korean: "백관조공격".to_string(),
271                is_auspicious: true,
272                description_korean: "제왕인 자미성이 명궁에 자리하고 여러 보좌길성(신하)들이 주위를 둘러싸고 옹위하는 형국으로, 막강한 권력, 강한 조력자, 높은 지위에 오르게 되는 제왕의 명조입니다.".to_string(),
273                description_english: "The Emperor (Ziwei) sits in the Life Palace surrounded by at least 4 major assistant stars (ministers), signifying great authority, social status, and supreme support.".to_string(),
274            });
275        }
276    }
277
278    // 14. 령창타무격 (鈴昌陀武格)
279    // 조건: 영성(LingXing), 문창(WenChang), 타라(TuoLuo), 무곡(WuQu)이 삼방사정에서 모두 모이는 흉격.
280    let has_ling_chang_tuo_wu = [ZwdsStar::LingXing, ZwdsStar::WenChang, ZwdsStar::TuoLuo, ZwdsStar::WuQu]
281        .iter()
282        .all(|&s| has_star_in_three_four(s));
283    if has_ling_chang_tuo_wu {
284        patterns.push(DestinyPattern {
285            name_hanja: "鈴昌陀武".to_string(),
286            name_korean: "령창타무격".to_string(),
287            is_auspicious: false,
288            description_korean: "무곡의 금기운에 영성, 타라, 문창의 수살이 결합하여 갑작스러운 재난, 금융상의 대패, 혹은 중대한 실패를 암시하므로 일생 투자와 건강에 극도로 신중해야 하는 흉격입니다.".to_string(),
289            description_english: "Wuqu, Wenchang, Tuoluo, and Lingxing gather in the three-party, representing a risky astrological configuration that warns against sudden downfalls, legal issues, or financial collapses.".to_string(),
290        });
291    }
292
293    // 15. 형기협인격 (刑忌夾印格)
294    // 조건: 천상(TianXiang, 印)이 명궁이고, 인접한 두 궁에 각각 화기(또는 거문화기 등)와 천형(또는 경양/타라)이 배치되어 협하는 흉격.
295    if has_star_in_palace(ZwdsStar::TianXiang, ming_idx) {
296        let adj_palaces = [(ming_idx + 1) % 12, (ming_idx + 11) % 12];
297        let mut has_ji = false;
298        let mut has_xing_or_yang = false;
299        for &p_idx in &adj_palaces {
300            for star_in_p in &palaces[p_idx].stars {
301                if star_in_p.si_hua == Some(SiHuaType::HuaJi) {
302                    has_ji = true;
303                }
304                if star_in_p.star == ZwdsStar::QingYang || star_in_p.star == ZwdsStar::TuoLuo || star_in_p.star == ZwdsStar::TianXing {
305                    has_xing_or_yang = true;
306                }
307            }
308        }
309        if has_ji && has_xing_or_yang {
310            patterns.push(DestinyPattern {
311                name_hanja: "刑忌夾印".to_string(),
312                name_korean: "형기협인격".to_string(),
313                is_auspicious: false,
314                description_korean: "관인(도장)인 천상성이 형성(경양/천형)과 기성(화기) 사이에 협여 억압받는 격국으로, 송사, 배신, 감금, 또는 예상치 못한 재난에 휘말리기 쉬우며 문서나 계약 시 각별한 주의가 필요합니다.".to_string(),
315                description_english: "The Minister (Tianxiang) in the Life Palace is sandwiched by punishment (Xing/Qingyang) and jealousy (Ji), warning of legal disputes, document traps, or unexpected betrayals.".to_string(),
316            });
317        }
318    }
319
320    // 16. 마두대검격 (馬頭帶劍格)
321    // 조건: 경양이 午宮(4)에 있고, 오궁이 명궁이면서 동시에 천동이나 태음과 동궁하는 격국.
322    if ming_idx == 4
323        && has_star_in_palace(ZwdsStar::QingYang, 4)
324        && (has_star_in_palace(ZwdsStar::TianTong, 4) || has_star_in_palace(ZwdsStar::TaiYin, 4))
325    {
326        patterns.push(DestinyPattern {
327            name_hanja: "馬頭帶劍".to_string(),
328            name_korean: "마두대검격".to_string(),
329            is_auspicious: false, // 흉격 성향이 강하나, 난세의 영웅격
330            description_korean: "칼에 해당하는 경양이 남쪽 오궁에서 칼날을 치켜세우는 형국으로, 삶의 기복과 고난이 대단히 격렬하지만 군인, 경찰, 외과 의사 등 개척적인 특수 직업군에서 큰 권력을 장악하기도 합니다.".to_string(),
331            description_english: "Qingyang resides in the Wu Palace (South) as the Life Palace with Tiantong or Taiyin, representing a high-risk but high-reward destiny of military authority or major breakthroughs through hardships.".to_string(),
332        });
333    }
334
335    // 17. 범수도화격 (泛水桃花格)
336    // 조건: 탐랑(TanLang)이 子宮(10)의 명궁에 있는 격국.
337    if ming_idx == 10 && has_star_in_palace(ZwdsStar::TanLang, 10) {
338        patterns.push(DestinyPattern {
339            name_hanja: "泛水桃花".to_string(),
340            name_korean: "범수도화격".to_string(),
341            is_auspicious: false,
342            description_korean: "도화의 별인 탐랑이 물(子궁)을 만나 도화 기운이 넘쳐흐르는 격국으로, 매력이 뛰어나 예체능이나 인기 업종에서 주목을 받을 수 있으나, 이성 관계나 주색으로 인한 낭패와 풍파를 경계해야 합니다.".to_string(),
343            description_english: "The Wolf (Tanlang) resides in the Child (Zi Palace - Water) as the Life Palace, amplifying natural charm and emotional entanglements, which requires self-discipline to prevent personal scanadals.".to_string(),
344        });
345    }
346
347    patterns
348}
