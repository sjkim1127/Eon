3//! 자미성과 천부성의 위치를 계산하고, 이를 기준으로 14주성을 각 궁에 배치합니다.

5use crate::palace::fix_index;
6use crate::types::{PalaceIndex, ZwdsStar};
7use std::collections::HashMap;

9/// 자미성(紫微星)의 ZWDS 지지 인덱스 (0=寅) 계산
10///
11/// * `lunar_day` - 음력 일 (1~30)
12/// * `five_elements_val` - 오행국 수치 (2, 3, 4, 5, 6)
13pub fn get_ziwei_index(lunar_day: u32, five_elements_val: u32) -> PalaceIndex {
14    let mut offset: i32 = -1;
15    loop {
16        offset += 1;
17        let divisor = lunar_day as i32 + offset;
18        let quotient = divisor / five_elements_val as i32;
19        let remainder = divisor % five_elements_val as i32;
20        if remainder == 0 {
21            let mut ziwei_idx = quotient % 12 - 1;
22            if offset % 2 == 0 {
23                ziwei_idx += offset;
24            } else {
25                ziwei_idx -= offset;
26            }
27            return fix_index(ziwei_idx);
28        }
29    }
30}
31
32/// 천부성(天府星)의 ZWDS 지지 인덱스 계산
33///
34/// 자미성과 천부성은 寅-申 축 대칭입니다.
35/// (ziwei_index + tianfu_index) % 12 == 0
36pub fn get_tianfu_index(ziwei_idx: PalaceIndex) -> PalaceIndex {
37    fix_index(12 - ziwei_idx as i32)
38}

40/// 자미성과 천부성을 기준으로 14주성의 위치를 맵으로 리턴
41pub fn place_14_main_stars(
42    ziwei_idx: PalaceIndex,
43    tianfu_idx: PalaceIndex,
44) -> HashMap<ZwdsStar, PalaceIndex> {
45    let mut stars = HashMap::new();
46
47    // 1. 자미성계 (역행: 시계 반대 방향, 인덱스 감소)
48    let z = ziwei_idx as i32;
49    stars.insert(ZwdsStar::ZiWei, fix_index(z));
50    stars.insert(ZwdsStar::TianJi, fix_index(z - 1));
51    stars.insert(ZwdsStar::TaiYang, fix_index(z - 3));
52    stars.insert(ZwdsStar::WuQu, fix_index(z - 4));
53    stars.insert(ZwdsStar::TianTong, fix_index(z - 5));
54    stars.insert(ZwdsStar::LianZhen, fix_index(z - 8));
55
56    // 2. 천부성계 (순행: 시계 방향, 인덱스 증가)
57    let f = tianfu_idx as i32;
58    stars.insert(ZwdsStar::TianFu, fix_index(f));
59    stars.insert(ZwdsStar::TaiYin, fix_index(f + 1));
60    stars.insert(ZwdsStar::TanLang, fix_index(f + 2));
61    stars.insert(ZwdsStar::JuMen, fix_index(f + 3));
62    stars.insert(ZwdsStar::TianXiang, fix_index(f + 4));
63    stars.insert(ZwdsStar::TianLiang, fix_index(f + 5));
64    stars.insert(ZwdsStar::QiSha, fix_index(f + 6));
65    stars.insert(ZwdsStar::PoJun, fix_index(f + 10));
66
67    stars
68}

70#[cfg(test)]
71mod tests {
72    use super::*;
73
74    #[test]
75    fn test_ziwei_pos() {
76        // 음력 16일, 수이국(2) -> quotient = 16/2 = 8, rem = 0, offset = 0.
77        // ziwei = 8 % 12 - 1 = 7 (酉). offset=0이므로 ziwei = 7 (酉).
78        assert_eq!(get_ziwei_index(16, 2), 7); // 酉
79
80        // 음력 16일, 토오국(5) -> divisor = 16 + 4 = 20 (offset = 4).
81        // quotient = 20 / 5 = 4. ziwei = 4 % 12 - 1 = 3. offset=4이므로 ziwei = 3 + 4 = 7 (酉).
82        assert_eq!(get_ziwei_index(16, 5), 7); // 酉
83    }
84
85    #[test]
86    fn test_tianfu_pos() {
87        // 자미가 寅(0) -> 천부는 寅(0)
88        assert_eq!(get_tianfu_index(0), 0);
89        // 자미가 子(10) -> 천부는 辰(2)
90        assert_eq!(get_tianfu_index(10), 2);
91    }
92}
