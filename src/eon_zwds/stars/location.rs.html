10/// 록존(禄存), 경양(擎羊), 타라(陀羅)의 ZWDS 지지 인덱스 계산
11pub fn place_lucun_qingyang_tuoluo(
12    year_stem: HeavenlyStem,
13) -> (PalaceIndex, PalaceIndex, PalaceIndex) {
14    let lucun = match year_stem {
15        HeavenlyStem::Jia => 0,  // 寅
16        HeavenlyStem::Yi => 1,   // 卯
17        HeavenlyStem::Bing => 3, // 巳
18        HeavenlyStem::Ding => 4, // 午
19        HeavenlyStem::Wu => 3,   // 巳
20        HeavenlyStem::Ji => 4,   // 午
21        HeavenlyStem::Geng => 6, // 申
22        HeavenlyStem::Xin => 7,  // 酉
23        HeavenlyStem::Ren => 9,  // 亥
24        HeavenlyStem::Gui => 10, // 子
25    };
26    let qingyang = fix_index(lucun as i32 + 1);
27    let tuoluo = fix_index(lucun as i32 - 1);
28    (lucun, qingyang, tuoluo)
29}

31/// 천마(天馬)의 ZWDS 지지 인덱스 계산
32pub fn place_tianma(year_branch_std_idx: usize) -> PalaceIndex {
33    let branch = EarthlyBranch::from_index(year_branch_std_idx as i32);

37        EarthlyBranch::Si | EarthlyBranch::You | EarthlyBranch::Chou => 9, // 亥
38        EarthlyBranch::Hai | EarthlyBranch::Mao | EarthlyBranch::Wei => 3, // 巳
39    }

42/// 화성(火星)과 영성(鈴星)의 ZWDS 지지 인덱스 계산
43pub fn place_huoxing_lingxing(
44    year_branch_std_idx: usize,
45    time_branch_std_idx: usize,
46) -> (PalaceIndex, PalaceIndex) {
47    let branch = EarthlyBranch::from_index(year_branch_std_idx as i32);
48    let t = time_branch_std_idx as i32;
49
50    let (huo_start, ling_start) = match branch {
51        EarthlyBranch::Yin | EarthlyBranch::Wu | EarthlyBranch::Xu => (11, 1), // 화=丑(11), 영=卯(1)
52        EarthlyBranch::Shen | EarthlyBranch::Zi | EarthlyBranch::Chen => (0, 8), // 화=寅(0), 영=戌(8)
53        EarthlyBranch::Si | EarthlyBranch::You | EarthlyBranch::Chou => (1, 8), // 화=卯(1), 영=戌(8)
54        EarthlyBranch::Hai | EarthlyBranch::Mao | EarthlyBranch::Wei => (7, 8), // 화=酉(7), 영=戌(8)
55    };
56
57    let huoxing = fix_index(huo_start + t);
58    let lingxing = fix_index(ling_start + t);
59
60    (huoxing, lingxing)
61}

63/// 지겁(地劫)과 지공(地空)의 ZWDS 지지 인덱스 계산
64pub fn place_dijie_dikong(time_branch_std_idx: usize) -> (PalaceIndex, PalaceIndex) {

71/// 음력 월 기준 잡성 배치
72pub fn place_month_stars(lunar_month: u32) -> Vec<(ZwdsStar, PalaceIndex)> {
73    let m = lunar_month as i32;
74    let mut stars = Vec::new();
75
76    // 천형(天刑): 酉궁(7)에서 순행
77    stars.push((ZwdsStar::TianXing, fix_index(7 + m - 1)));
78    // 천요(天姚): 丑궁(11)에서 순행
79    stars.push((ZwdsStar::TianYao, fix_index(11 + m - 1)));
80
81    // 해신(解神): 1,2월->申(6), 3,4월->酉(7)...
82    let jieshen = fix_index(((lunar_month - 1) / 2) as i32 + 6);
83    stars.push((ZwdsStar::JieShen, jieshen));
84
85    // 음살(陰煞): 1월->寅(0), 2월->子(10)...
86    let yinsha = fix_index(-2 * (((lunar_month - 1) % 6) as i32));
87    stars.push((ZwdsStar::YinSha, yinsha));
88
89    // 천월(天月): 룩업 테이블
90    let tianyue_table = [8, 3, 2, 0, 5, 7, 6, 5, 9, 4, 0, 7];
91    let tianyue = tianyue_table[(lunar_month as usize - 1) % 12];
92    stars.push((ZwdsStar::TianYueStar, tianyue));
93
94    // 천무(天巫): 1,5,9월->巳(3)...
95    let tianwu = match (lunar_month - 1) % 4 {
96        0 => 3, // 巳
97        1 => 6, // 申
98        2 => 0, // 寅
99        3 => 9, // 亥
100        _ => unreachable!(),
101    };
102    stars.push((ZwdsStar::TianWu, tianwu));
103
104    stars
105}

107/// 출생시 기준 잡성 배치
108pub fn place_hour_stars(time_branch_std_idx: usize) -> Vec<(ZwdsStar, PalaceIndex)> {
109    let t = time_branch_std_idx as i32;
110    let mut stars = Vec::new();
111
112    // 대보(台輔): 午궁(4)에서 순행
113    stars.push((ZwdsStar::TaiFu, fix_index(4 + t)));
114    // 봉고(封誥): 寅궁(0)에서 순행
115    stars.push((ZwdsStar::FengGao, fix_index(t)));
116
117    stars
118}

120/// 음력 일 기준 잡성 배치
121pub fn place_day_stars(
122    lunar_day: u32,
123    zuofu_idx: PalaceIndex,
124    youbi_idx: PalaceIndex,
125    wenchang_idx: PalaceIndex,
126    wenqu_idx: PalaceIndex,
127) -> Vec<(ZwdsStar, PalaceIndex)> {
128    let d = lunar_day as i32;
129    let mut stars = Vec::new();
130
131    // 삼태(三台): 좌보에서 순행
132    stars.push((ZwdsStar::SanTai, fix_index(zuofu_idx as i32 + d - 1)));
133    // 팔좌(八座): 우필에서 역행
134    stars.push((ZwdsStar::BaZuo, fix_index(youbi_idx as i32 - d + 1)));
135    // 은광(恩光): 문창에서 순행
136    stars.push((ZwdsStar::EnGuang, fix_index(wenchang_idx as i32 + d - 2)));
137    // 천귀(天貴): 문곡에서 순행
138    stars.push((ZwdsStar::TianGui, fix_index(wenqu_idx as i32 + d - 2)));
139
140    stars
141}

143/// 출생년 지지 기준 잡성 배치
144pub fn place_year_branch_stars(year_branch_std_idx: usize) -> Vec<(ZwdsStar, PalaceIndex)> {
145    // ZWDS 년지 지지 인덱스 (0=寅)
146    let y_zwds = ((year_branch_std_idx + 10) % 12) as i32;
147    let mut stars = Vec::new();
148
149    // 홍란(紅鸞): 卯궁(1)에서 역행
150    let hongluan = fix_index(1 - y_zwds);
151    stars.push((ZwdsStar::HongLuan, hongluan));
152    // 천희(天喜): 홍란 대궁
153    stars.push((ZwdsStar::TianXi, fix_index(hongluan as i32 + 6)));
154
155    // 화개(華蓋) & 함지(咸池) & 겁살(劫殺)
156    let branch = EarthlyBranch::from_index(year_branch_std_idx as i32);
157    let (huagai, xianchi, jiesha) = match branch {
158        EarthlyBranch::Yin | EarthlyBranch::Wu | EarthlyBranch::Xu => (8, 1, 9), // 戌, 卯, 亥
159        EarthlyBranch::Shen | EarthlyBranch::Zi | EarthlyBranch::Chen => (2, 7, 3), // 辰, 酉, 巳
160        EarthlyBranch::Si | EarthlyBranch::You | EarthlyBranch::Chou => (11, 4, 0), // 丑, 午, 寅
161        EarthlyBranch::Hai | EarthlyBranch::Mao | EarthlyBranch::Wei => (5, 10, 6), // 未, 子, 申
162    };
163    stars.push((ZwdsStar::HuaGai, huagai));
164    stars.push((ZwdsStar::XianChi, xianchi));
165    stars.push((ZwdsStar::JieSha, jiesha));
166
167    // 고신(孤辰) & 과숙(寡宿)
168    let (guchen, guasu) = match branch {
169        EarthlyBranch::Yin | EarthlyBranch::Mao | EarthlyBranch::Chen => (3, 11), // 巳, 丑
170        EarthlyBranch::Si | EarthlyBranch::Wu | EarthlyBranch::Wei => (6, 2),     // 申, 辰
171        EarthlyBranch::Shen | EarthlyBranch::You | EarthlyBranch::Xu => (9, 5),   // 亥, 未
172        EarthlyBranch::Hai | EarthlyBranch::Zi | EarthlyBranch::Chou => (0, 8),   // 寅, 戌
173    };
174    stars.push((ZwdsStar::GuChen, guchen));
175    stars.push((ZwdsStar::GuaSu, guasu));
176
177    // 용지(龍池): 辰궁(2)에서 순행
178    stars.push((ZwdsStar::LongChi, fix_index(2 + y_zwds)));
179    // 봉각(鳳閣): 戌궁(8)에서 역행
180    stars.push((ZwdsStar::FengGe, fix_index(8 - y_zwds)));
181
182    // 천곡(天哭): 午궁(4)에서 역행
183    stars.push((ZwdsStar::TianKu, fix_index(4 - y_zwds)));
184    // 천허(天虛): 午궁(4)에서 순행
185    stars.push((ZwdsStar::TianXu, fix_index(4 + y_zwds)));
186
187    // 천공(天空): 卯궁(1)에서 순행 (년지 ZWDS 인덱스 + 1)
188    stars.push((ZwdsStar::TianKong, fix_index(y_zwds + 1)));
189
190    stars
191}

193/// 출생년 천간 기준 잡성 배치
194pub fn place_year_stem_stars(
195    year_stem: HeavenlyStem,
196    year_branch_std_idx: usize,
197    soul_idx: PalaceIndex,
198    body_idx: PalaceIndex,
199) -> Vec<(ZwdsStar, PalaceIndex)> {
200    let y_zwds = ((year_branch_std_idx + 10) % 12) as i32;
201    let mut stars = Vec::new();
202
203    // 천재(天才): 명궁에서 년지 인덱스만큼 순행
204    stars.push((ZwdsStar::TianCai, fix_index(soul_idx as i32 + y_zwds)));
205    // 천수(天壽): 신궁에서 년지 인덱스만큼 순행
206    stars.push((ZwdsStar::TianShou, fix_index(body_idx as i32 + y_zwds)));
207
208    let s_idx = year_stem.index() as usize;
209
210    // 천관(天官)
211    let tianguan_table = [5, 2, 3, 0, 1, 7, 9, 7, 8, 4];
212    stars.push((ZwdsStar::TianGuan, tianguan_table[s_idx]));
213
214    // 천복(天福)
215    let tianfu_table = [7, 6, 10, 9, 1, 0, 4, 3, 4, 3];
216    stars.push((ZwdsStar::TianFu2, tianfu_table[s_idx]));
217
218    // 천주(天廚)
219    let tianchu_table = [3, 4, 10, 3, 4, 6, 0, 4, 7, 9];
220    stars.push((ZwdsStar::TianChu, tianchu_table[s_idx]));
221
222    stars
