4pub enum ZodiacQuarter {

11impl ZodiacQuarter {
12    /// Determines the Quarter based on the Personality Sun Gate.
13    pub fn from_gate(gate: u8) -> Self {
14        // Quarter of Initiation: Gates 13 to 24 (13, 49, 30, 55, 37, 63, 22, 36, 25, 17, 21, 51, 42, 3, 27, 24)
15        const INITIATION: [u8; 16] = [
16            13, 49, 30, 55, 37, 63, 22, 36, 25, 17, 21, 51, 42, 3, 27, 24,
17        ];
18
19        // Quarter of Civilization: Gates 2 to 33 (2, 23, 8, 20, 16, 35, 45, 12, 15, 52, 39, 53, 62, 56, 31, 33)
20        const CIVILIZATION: [u8; 16] =
21            [2, 23, 8, 20, 16, 35, 45, 12, 15, 52, 39, 53, 62, 56, 31, 33];
22
23        // Quarter of Duality: Gates 7 to 44 (7, 4, 29, 59, 40, 64, 47, 6, 46, 18, 48, 57, 32, 50, 28, 44)
24        const DUALITY: [u8; 16] = [7, 4, 29, 59, 40, 64, 47, 6, 46, 18, 48, 57, 32, 50, 28, 44];
25
26        // Quarter of Mutation: Gates 1 to 19 (1, 43, 14, 34, 9, 5, 26, 11, 10, 58, 38, 54, 61, 60, 41, 19)
27        const MUTATION: [u8; 16] = [1, 43, 14, 34, 9, 5, 26, 11, 10, 58, 38, 54, 61, 60, 41, 19];
28
29        if INITIATION.contains(&gate) {
30            Self::Initiation
31        } else if CIVILIZATION.contains(&gate) {
32            Self::Civilization
33        } else if DUALITY.contains(&gate) {
34            Self::Duality
35        } else if MUTATION.contains(&gate) {
36            Self::Mutation
37        } else {
38            // Fallback (should never happen if 1..64 is passed)
39            Self::Initiation
40        }
41    }
42
