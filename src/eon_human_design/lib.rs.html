132pub fn degree_to_gate_line(degree: f64) -> (u8, u8) {
133    let gate_size = 360.0 / 64.0;
134    let line_size = gate_size / 6.0;
135    let adjusted = (degree - HD_START_DEGREE + 360.0) % 360.0;
136    let index = (adjusted / gate_size).floor() as usize;
137    let line = ((adjusted % gate_size) / line_size).floor() as u8 + 1;
138    let gate = GATE_SEQUENCE[index % 64];
139    (gate, line)
140}

142pub fn get_planet_positions(
143    engine: &AstroEngine,
144    datetime: DateTime<Utc>,
145) -> Result<HashMap<String, HdPlanetData>, AstroError> {
146    let mut results = HashMap::new();
147
148    // 1. Sun & Earth
149    // Flag: SEFLG_SWIEPH = 2, but get_sun_longitude uses FFI internally or we can use swe_calc_ut.
150    // Let's use get_sun_longitude or swe_calc_ut.
151    let sun_deg = engine.get_sun_longitude(datetime)?;
152    let (gate, line) = degree_to_gate_line(sun_deg);
153    results.insert("Sun".to_string(), HdPlanetData {
154        name: "Sun".to_string(),
155        degree: sun_deg,
156        gate,
157        line,
158    });
159
160    let earth_deg = (sun_deg + 180.0) % 360.0;
161    let (gate, line) = degree_to_gate_line(earth_deg);
162    results.insert("Earth".to_string(), HdPlanetData {
163        name: "Earth".to_string(),
164        degree: earth_deg,
165        gate,
166        line,
167    });
168
169    // 2. Nodes (True Node)
170    // Planet id: 11 is SE_TRUE_NODE
171    let nn_deg = engine.get_planet_position(datetime, 11, 2)?;// SEFLG_SWIEPH = 2
172    let (gate, line) = degree_to_gate_line(nn_deg);
173    results.insert("N.Node".to_string(), HdPlanetData {
174        name: "N.Node".to_string(),
175        degree: nn_deg,
176        gate,
177        line,
178    });
179
180    let sn_deg = (nn_deg + 180.0) % 360.0;
181    let (gate, line) = degree_to_gate_line(sn_deg);
182    results.insert("S.Node".to_string(), HdPlanetData {
183        name: "S.Node".to_string(),
184        degree: sn_deg,
185        gate,
186        line,
187    });
188
189    // 3. Remaining Planets
190    let planets = vec![
191        ("Moon", 1),
192        ("Mercury", 2),
193        ("Venus", 3),
194        ("Mars", 4),
195        ("Jupiter", 5),
196        ("Saturn", 6),
197        ("Uranus", 7),
198        ("Neptune", 8),
199        ("Pluto", 9),
200    ];
201
202    for (name, planet_id) in planets {
203        let pos = engine.get_planet_position(datetime, planet_id, 2)?;
204        let (gate, line) = degree_to_gate_line(pos);
205        results.insert(name.to_string(), HdPlanetData {
206            name: name.to_string(),
207            degree: pos,
208            gate,
209            line,
210        });
211    }
212
213    Ok(results)
214}

216pub fn get_defined_centers_and_channels(all_gates: &HashSet<u8>) -> (HashSet<HdCenter>, Vec<(u8, u8)>) {
217    let mut defined_centers = HashSet::new();
218    let mut active_channels = Vec::new();
219
220    for &(g1, g2) in &CHANNELS {
221        if all_gates.contains(&g1) && all_gates.contains(&g2) {
222            if let Some((c1, c2)) = get_channel_centers(g1, g2) {
223                defined_centers.insert(c1);
224                defined_centers.insert(c2);
225                active_channels.push((g1, g2));
226            }
227        }
228    }
229
230    (defined_centers, active_channels)
231}

233pub fn determine_type(defined_centers: &HashSet<HdCenter>, active_channels: &[(u8, u8)]) -> String {
234    let has_sacral = defined_centers.contains(&HdCenter::Sacral);
235    let has_throat = defined_centers.contains(&HdCenter::Throat);
236
237    let motor_centers: HashSet<HdCenter> = [
238        HdCenter::Sacral,
239        HdCenter::Heart,
240        HdCenter::SolarPlexus,
241        HdCenter::Root,
242    ]
243    .iter()
244    .cloned()
245    .collect();
246
247    let mut motor_to_throat = false;
248
249    if has_throat {
250        // Build graph of defined centers
251        let mut adj = HashMap::new();
252        for &center in &[
253            HdCenter::Head, HdCenter::Ajna, HdCenter::Throat, HdCenter::SelfG,
254            HdCenter::Heart, HdCenter::Sacral, HdCenter::Root, HdCenter::Spleen, HdCenter::SolarPlexus
255        ] {
256            adj.insert(center, HashSet::new());
257        }
258
259        for &(g1, g2) in active_channels {
260            if let Some((c1, c2)) = get_channel_centers(g1, g2) {
261                adj.entry(c1).or_insert_with(HashSet::new).insert(c2);
262                adj.entry(c2).or_insert_with(HashSet::new).insert(c1);
263            }
264        }
265
266        // BFS to find if a motor center connects to Throat
267        let mut visited = HashSet::new();
268        let mut queue = VecDeque::new();
269        queue.push_back(HdCenter::Throat);
270
271        while let Some(current) = queue.pop_front() {
272            if !visited.contains(&current) {
273                visited.insert(current);
274                if motor_centers.contains(&current) {
275                    motor_to_throat = true;
276                    break;
277                }
278                if let Some(neighbors) = adj.get(&current) {
279                    for &neigh in neighbors {
280                        if !visited.contains(&neigh) {
281                            queue.push_back(neigh);
282                        }
283                    }
284                }
285            }
286        }
287    }
288
289    if defined_centers.is_empty() {
290        "Reflector".to_string()
291    } else if has_sacral && motor_to_throat {
292        "Manifesting Generator".to_string()
293    } else if has_sacral {
294        "Generator".to_string()
295    } else if motor_to_throat {
296        "Manifestor".to_string()
297    } else {
298        "Projector".to_string()
299    }
300}

302pub fn determine_authority(defined_centers: &HashSet<HdCenter>) -> String {
303    let priority = vec![
304        (HdCenter::SolarPlexus, "Emotional"),
305        (HdCenter::Sacral, "Sacral"),
306        (HdCenter::Spleen, "Splenic"),
307        (HdCenter::Heart, "Ego"),
308        (HdCenter::SelfG, "Self-Projected"),
309    ];
310
311    for (center, authority) in priority {
312        if defined_centers.contains(&center) {
313            return authority.to_string();
314        }
315    }
316
317    if defined_centers.contains(&HdCenter::Throat)
318        || defined_centers.contains(&HdCenter::Ajna)
319        || defined_centers.contains(&HdCenter::Head)
320    {
321        "Mental/Outer".to_string()
322    } else {
323        "None/Outer".to_string()
324    }
325}

327pub fn calculate_human_design(
328    engine: &AstroEngine,
329    birth_time: DateTime<Utc>,
330) -> Result<HumanDesignResult, HdError> {
331    // 1. Personality Positions
332    let personality = get_planet_positions(engine, birth_time)?;
333
334    // 2. Design Date Calculation
335    // Design time is when the Sun was exactly 88 degrees prior to its birth position.
336    let p_sun = personality.get("Sun").unwrap();
337    let target_design_deg = (p_sun.degree - 88.0 + 360.0) % 360.0;
338
339    // Use AstroEngine's find_time_for_longitude starting approx 88 days before birth
340    use chrono::Duration;
341    let approx_design_time = birth_time - Duration::days(88);
342    let design_time = engine.find_time_for_longitude(approx_design_time, target_design_deg)?;
343
344    // Calculate Design Positions
345    let design = get_planet_positions(engine, design_time)?;
346
347    // 3. Collect active gates
348    let mut all_gates = HashSet::new();
349    for p in personality.values() {
350        all_gates.insert(p.gate);
351    }
352    for p in design.values() {
353        all_gates.insert(p.gate);
354    }
355
356    // 4. Calculate centers and channels
357    let (defined_set, active_channels) = get_defined_centers_and_channels(&all_gates);
358
359    let mut defined_centers: Vec<HdCenter> = defined_set.iter().cloned().collect();
360    defined_centers.sort_by_key(|c| *c as u8);
361
362    let mut undefined_centers: Vec<HdCenter> = [
363        HdCenter::Head, HdCenter::Ajna, HdCenter::Throat, HdCenter::SelfG,
364        HdCenter::Heart, HdCenter::Sacral, HdCenter::Root, HdCenter::Spleen, HdCenter::SolarPlexus
365    ]
366    .iter()
367    .filter(|c| !defined_set.contains(c))
368    .cloned()
369    .collect();
370    undefined_centers.sort_by_key(|c| *c as u8);
371
372    // 5. Profile
373    let p_sun_line = personality.get("Sun").unwrap().line;
374    let d_sun_line = design.get("Sun").unwrap().line;
375    let profile = format!("{}/{}", p_sun_line, d_sun_line);
376
377    // 6. Type and Authority
378    let chart_type = determine_type(&defined_set, &active_channels);
379    let authority = determine_authority(&defined_set);
380
381    let active_gates: Vec<u8> = all_gates.into_iter().collect();
382    let mut active_gates = active_gates;
383    active_gates.sort();
384
385    Ok(HumanDesignResult {
386        chart_type,
387        profile,
388        authority,
389        defined_centers,
390        undefined_centers,
391        personality,
392        design,
393        active_gates,
394        active_channels,
395    })
396}
