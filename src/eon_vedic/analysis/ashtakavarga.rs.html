25    pub fn calculate_sav(chart: &VedicChart) -> Sarvashtakavarga {
26        let planets = [
27            VedicPlanet::Sun,
28            VedicPlanet::Moon,
29            VedicPlanet::Mars,
30            VedicPlanet::Mercury,
31            VedicPlanet::Jupiter,
32            VedicPlanet::Venus,
33            VedicPlanet::Saturn,
34        ];
35
36        let mut total_points = [0u8; 12];
37
38        for &p in &planets {
39            let bav = Self::calculate_bav(p, chart);
40            for i in 0..12 {
41                total_points[i] += bav.points[i];
42            }
43        }
44
45        Sarvashtakavarga {
46            points: total_points,
47        }
48    }

50    /// Bhinnashtakavarga (BAV) for a specific planet using BPHS tables.
51    pub fn calculate_bav(target_planet: VedicPlanet, chart: &VedicChart) -> AshtakavargaPoints {
52        let mut points = [0u8; 12];
53
54        let sun_pos = chart
55            .planets
56            .iter()
57            .find(|p| p.planet == VedicPlanet::Sun)
58            .unwrap()
59            .rasi;
60        let moon_pos = chart
61            .planets
62            .iter()
63            .find(|p| p.planet == VedicPlanet::Moon)
64            .unwrap()
65            .rasi;
66        let mars_pos = chart
67            .planets
68            .iter()
69            .find(|p| p.planet == VedicPlanet::Mars)
70            .unwrap()
71            .rasi;
72        let merc_pos = chart
73            .planets
74            .iter()
75            .find(|p| p.planet == VedicPlanet::Mercury)
76            .unwrap()
77            .rasi;
78        let jup_pos = chart
79            .planets
80            .iter()
81            .find(|p| p.planet == VedicPlanet::Jupiter)
82            .unwrap()
83            .rasi;
84        let ven_pos = chart
85            .planets
86            .iter()
87            .find(|p| p.planet == VedicPlanet::Venus)
88            .unwrap()
89            .rasi;
90        let sat_pos = chart
91            .planets
92            .iter()
93            .find(|p| p.planet == VedicPlanet::Saturn)
94            .unwrap()
95            .rasi;
96        let lagna_pos = chart.ascendant.rasi;
97
98        let refs = [
99            (VedicPlanet::Sun, sun_pos),
100            (VedicPlanet::Moon, moon_pos),
101            (VedicPlanet::Mars, mars_pos),
102            (VedicPlanet::Mercury, merc_pos),
103            (VedicPlanet::Jupiter, jup_pos),
104            (VedicPlanet::Venus, ven_pos),
105            (VedicPlanet::Saturn, sat_pos),
106            (VedicPlanet::Ascendant, lagna_pos),
107        ];
108
109        for (ref_planet, ref_rasi) in refs {
110            let offsets = Self::get_offsets(target_planet, ref_planet);
111            for &offset in offsets {
112                // target_rasi = (ref_rasi + offset - 1) % 12 + 1
113                let target_rasi = (ref_rasi + offset - 2) % 12 + 1;
114                points[target_rasi as usize - 1] += 1;
115            }
116        }
117
118        let trikona_points = Self::apply_triangular_reduction(points);
119        let shodhana_points = Self::apply_ekadhipatya_reduction(trikona_points, chart);
120        let sodya_pinda = Self::calculate_pinda(target_planet, shodhana_points, chart);
121
122        AshtakavargaPoints {
123            planet: target_planet,
124            points,
125            trikona_points,
126            shodhana_points,
127            sodya_pinda,
128        }
129    }

131    /// Trikona Shodhana (Triangular Reduction)
132    /// Reduces points based on triplicities (Elements)
133    pub fn apply_triangular_reduction(points: [u8; 12]) -> [u8; 12] {
134        let mut reduced = points;
135        let triplicities = [
136            [0, 4, 8],  // Fire (1, 5, 9)
137            [1, 5, 9],  // Earth (2, 6, 10)
138            [2, 6, 10], // Air (3, 7, 11)
139            [3, 7, 11], // Water (4, 8, 12)
140        ];
141
142        for trip in &triplicities {
143            let p1 = reduced[trip[0]];
144            let p2 = reduced[trip[1]];
145            let p3 = reduced[trip[2]];
146
147            let zeros = (if p1 == 0 { 1 } else { 0 })
148                + (if p2 == 0 { 1 } else { 0 })
149                + (if p3 == 0 { 1 } else { 0 });
150
151            if zeros == 0 {
152                // All have points: subtract minimum
153                let min_val = p1.min(p2).min(p3);
154                reduced[trip[0]] -= min_val;
155                reduced[trip[1]] -= min_val;
156                reduced[trip[2]] -= min_val;
157            } else if zeros == 1 {
158                // One is zero: no reduction
159            } else if zeros == 2 {
160                // Two are zero: third becomes zero too
161                reduced[trip[0]] = 0;
162                reduced[trip[1]] = 0;
163                reduced[trip[2]] = 0;
164            } else {
165                // All are zero: already 0
166            }
167        }
168        reduced
169    }

171    /// Ekadhipatya Shodhana (Reduction for planets owning two signs)
172    pub fn apply_ekadhipatya_reduction(points: [u8; 12], chart: &VedicChart) -> [u8; 12] {
173        let mut reduced = points;
174        let pairs = [
175            (0, 7),  // Mars: Aries (1) & Scorpio (8)
176            (1, 6),  // Venus: Taurus (2) & Libra (7)
177            (2, 5),  // Mercury: Gemini (3) & Virgo (6)
178            (8, 11), // Jupiter: Sag (9) & Pisces (12)
179            (9, 10), // Saturn: Cap (10) & Aqua (11)
180        ];
181
182        let is_occupied = |rasi_idx: usize| -> bool {
183            chart.planets.iter().any(|p| p.rasi == (rasi_idx as u8 + 1))
184        };
185
186        for (r1, r2) in pairs {
187            let p1 = reduced[r1];
188            let p2 = reduced[r2];
189            let occ1 = is_occupied(r1);
190            let occ2 = is_occupied(r2);
191
192            if p1 == 0 || p2 == 0 {
193                if p1 == 0 && p2 == 0 {
194                    continue;
195                }
196                // One has points, the other doesn't
197                let has_pts_idx = if p1 == 0 { r2 } else { r1 };
198                if !is_occupied(has_pts_idx) {
199                    reduced[has_pts_idx] = 0;
200                }
201            } else {
202                // Both have points
203                if occ1 && occ2 {
204                    // Both occupied, no reduction
205                } else if !occ1 && !occ2 {
206                    // Neither occupied
207                    if p1 == p2 {
208                        // BPHS: If both are equal and unoccupied, both become 0
209                        reduced[r1] = 0;
210                        reduced[r2] = 0;
211                    } else {
212                        // Both to lower value
213                        let min_val = p1.min(p2);
214                        reduced[r1] = min_val;
215                        reduced[r2] = min_val;
216                    }
217                } else {
218                    // One occupied
219                    let (occ_idx, unocc_idx) = if occ1 { (r1, r2) } else { (r2, r1) };
220                    if reduced[unocc_idx] > reduced[occ_idx] {
221                        reduced[unocc_idx] = reduced[occ_idx];
222                    }
223                }
224            }
225        }
226        reduced
227    }

229    /// Calculate Sodya Pinda (Final sum of reduced points multiplied by rasi/planet factors)
230    pub fn calculate_pinda(_target: VedicPlanet, points: [u8; 12], chart: &VedicChart) -> u32 {
231        let rasi_multipliers = [7, 10, 8, 4, 10, 5, 7, 8, 9, 5, 11, 12];
232        let graha_multipliers = [
233            (VedicPlanet::Sun, 5),
234            (VedicPlanet::Moon, 5),
235            (VedicPlanet::Mars, 8),
236            (VedicPlanet::Mercury, 5),
237            (VedicPlanet::Jupiter, 10),
238            (VedicPlanet::Venus, 7),
239            (VedicPlanet::Saturn, 5),
240        ];
241
242        // 1. Rasi Pinda: Points in each rasi * Rasi multiplier
243        let mut rasi_pinda = 0u32;
244        for i in 0..12 {
245            rasi_pinda += (points[i] as u32) * rasi_multipliers[i];
246        }
247
248        // 2. Graha Pinda: Reduced points in rasis where planets are * Graha multiplier
249        let mut graha_pinda = 0u32;
250        for (p, mult) in graha_multipliers {
251            if let Some(pos) = chart.planets.iter().find(|pos| pos.planet == p) {
252                let rasi_idx = (pos.rasi as usize).wrapping_sub(1);
253                if rasi_idx < 12 {
254                    graha_pinda += (points[rasi_idx] as u32) * mult;
255                }
256            }
257        }
258
259        rasi_pinda + graha_pinda
260    }
