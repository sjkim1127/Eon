55    pub fn calculate_karakas(chart: &VedicChart, use_8_karakas: bool) -> Vec<KarakaAssignment> {
56        let mut planets_data: Vec<(VedicPlanet, f64)> = chart
57            .planets
58            .iter()
59            .filter(|p| {
60                let is_base = matches!(
61                    p.planet,
62                    VedicPlanet::Sun
63                        | VedicPlanet::Moon
64                        | VedicPlanet::Mars
65                        | VedicPlanet::Mercury
66                        | VedicPlanet::Jupiter
67                        | VedicPlanet::Venus
68                        | VedicPlanet::Saturn
69                );
70                if use_8_karakas {
71                    is_base || p.planet == VedicPlanet::Rahu
72                } else {
73                    is_base
74                }
75            })
76            .map(|p| {
77                let mut deg = p.sidereal_deg % 30.0;
78                // Special Rule for Rahu in Jaimini: Reverse degree because Node is retrograde
79                if p.planet == VedicPlanet::Rahu {
80                    deg = 30.0 - deg;
81                }
82                (p.planet, deg)
83            })
84            .collect();
85
86        // Sort by degree within the Rasi (descending: AK first)
87        planets_data.sort_by(|a, b| b.1.partial_cmp(&a.1).unwrap_or(std::cmp::Ordering::Equal));
88
89        let mut roles = vec![
90            JaiminiKarakaRole::Atmakaraka,
91            JaiminiKarakaRole::Amatyakaraka,
92            JaiminiKarakaRole::Bhratrukaraka,
93            JaiminiKarakaRole::Matrukaraka,
94        ];
95
96        if use_8_karakas {
97            roles.push(JaiminiKarakaRole::Pitrikaraka);
98        }
99
100        roles.extend([
101            JaiminiKarakaRole::Putrakaraka,
102            JaiminiKarakaRole::Gnatikaraka,
103            JaiminiKarakaRole::Darakaraka,
104        ]);
105
106        let mut assignments = Vec::new();
107        for (idx, (planet, deg)) in planets_data.iter().enumerate() {
108            if idx < roles.len() {
109                assignments.push(KarakaAssignment {
110                    planet: *planet,
111                    role: roles[idx].clone(),
112                    degree_in_rasi: *deg,
113                });
114            }
115        }
116
117        assignments
118    }

120    /// Calculate Arudha Padas for all 12 houses
121    pub fn calculate_arudha_padas(chart: &VedicChart) -> Vec<ArudhaPada> {
122        let mut results = Vec::new();
123        let lagna_rasi = chart.ascendant.rasi;
124
125        let names = [
126            "Arudha Lagna (AL)", "Dhanapada (A2)", "Vikramapada (A3)", "Matrupada (A4)",
127            "Putrapada (A5)", "Shatrupada (A6)", "Darapada (A7)", "Mrityupada (A8)",
128            "Bhagyapada (A9)", "Rajyapada (A10)", "Labhapada (A11)", "Upapada Lagna (UL/A12)"
129        ];
130
131        for house in 1..=12 {
132            let house_rasi = ((lagna_rasi as i16 + house as i16 - 2) % 12 + 1) as u8;
133            let lord = VedicPlanet::get_ruler_of(house_rasi);
134            
135            // Find lord's position in D1
136            if let Some(lord_pos) = chart.planets.iter().find(|p| p.planet == lord) {
137                let lord_rasi = lord_pos.rasi;
138                
139                // Distance from house to lord
140                let dist = (lord_rasi as i16 - house_rasi as i16 + 12) % 12;
141                
142                // Arudha = Lord + Distance
143                let mut arudha_rasi = ((lord_rasi as i16 + dist - 1) % 12 + 1) as u8;
144                
145                if arudha_rasi == house_rasi {
146                    // If Arudha is in the house itself, final Arudha is 10th from house
147                    arudha_rasi = ((house_rasi as i16 + 10 - 2) % 12 + 1) as u8;
148                } else if arudha_rasi == ((house_rasi as i16 + 7 - 2) % 12 + 1) as u8 {
149                    // If Arudha is in the 7th from house, final Arudha is 4th from house
150                    arudha_rasi = ((house_rasi as i16 + 4 - 2) % 12 + 1) as u8;
151                }
152
153                results.push(ArudhaPada {
154                    house,
155                    rasi: arudha_rasi,
156                    name: names[house as usize - 1].to_string(),
157                });
158            }
159        }
160
161        results
162    }

164    /// Calculate Special Lagnas from BPHS
165    pub fn calculate_special_lagnas(chart: &VedicChart) -> Vec<SpecialLagna> {
166        let mut results = Vec::new();
167        
168        let sun_pos = chart.planets.iter().find(|p| p.planet == VedicPlanet::Sun);
169        if let (Some(sun), sunrise) = (sun_pos, chart.panchanga.sunrise) {
170            let birth_time = chart.panchanga.current_time;
171            let diff_mins = birth_time.signed_duration_since(sunrise).num_minutes() as f64;
172            
173            // Bhava Lagna (BL): 1 Rasi (30 deg) per 24 mins (1 Ghati)
174            let bl_long = (sun.sidereal_deg + diff_mins * (30.0 / 24.0)) % 360.0;
175            results.push(SpecialLagna {
176                name: "Bhava Lagna (BL)".to_string(),
177                longitude: bl_long,
178                rasi: (bl_long / 30.0).floor() as u8 + 1,
179            });
180
181            // Hora Lagna (HL): 1 Rasi (30 deg) per 60 mins (2.5 Ghati)
182            let hl_long = (sun.sidereal_deg + diff_mins * (30.0 / 60.0)) % 360.0;
183            results.push(SpecialLagna {
184                name: "Hora Lagna (HL)".to_string(),
185                longitude: hl_long,
186                rasi: (hl_long / 30.0).floor() as u8 + 1,
187            });
188
189            // Ghati Lagna (GL): Rate of 1.25 signs per Ghati (24 mins)
190            // This is equivalent to 1 sign per 19.2 mins.
191            // BPHS standard: 1.25 * (30 deg) / 24 mins = 1.5625 deg / min.
192            let gl_long = (sun.sidereal_deg + diff_mins * 1.5625) % 360.0;
193            results.push(SpecialLagna {
194                name: "Ghati Lagna (GL)".to_string(),
195                longitude: gl_long,
196                rasi: (gl_long / 30.0).floor() as u8 + 1,
197            });
198        }
199        
200        results
201    }

203    /// Calculate Chara Dasha (True KN Rao method)
204    /// 
205    /// Sequence (KN Rao):
206    /// - Forward: Ar, Le, Vi, Li, Aq, Pi (1, 5, 6, 7, 11, 12)
207    /// - Backward: Ta, Ge, Cn, Sc, Sg, Cp (2, 3, 4, 8, 9, 10)
208    pub fn calculate_chara_dasha(chart: &VedicChart) -> Vec<SignDashaPeriod> {
209        let lagna_rasi = chart.ascendant.rasi;
210        let birth_time = chart.panchanga.current_time;
211        
212        let forward_signs = [1, 5, 6, 7, 11, 12];
213        let is_forward_seq = forward_signs.contains(&lagna_rasi);
214        
215        let mut sequence = Vec::new();
216        for i in 0..12 {
217            let rasi = if is_forward_seq {
218                ((lagna_rasi as i16 + i - 1) % 12 + 1) as u8
219            } else {
220                ((lagna_rasi as i16 - i + 11) % 12 + 1) as u8
221            };
222            sequence.push(rasi);
223        }
224
225        let mut timeline = Vec::new();
226        let mut current_start = birth_time;
227
228        for rasi in sequence {
229            let years = Self::calculate_chara_dasha_years(chart, rasi);
230
231            let end_time = current_start + Duration::seconds((years as f64 * 365.2425 * 24.0 * 60.0 * 60.0) as i64);
232            
233            timeline.push(SignDashaPeriod {
234                rasi,
235                start_time: current_start,
236                end_time,
237            });
238            
239            current_start = end_time;
240        }
241
242        timeline
243    }

245    /// Calculate years for a sign in Chara Dasha
246    fn calculate_chara_dasha_years(chart: &VedicChart, rasi: u8) -> u32 {
247        let forward_signs = [1, 5, 6, 7, 11, 12];
248        let is_counting_forward = forward_signs.contains(&rasi);
249
250        let lord_rasi = match rasi {
251            8 => Self::evaluate_co_ruler_strength(chart, 8, VedicPlanet::Mars, VedicPlanet::Ketu),
252            11 => Self::evaluate_co_ruler_strength(chart, 11, VedicPlanet::Saturn, VedicPlanet::Rahu),
253            _ => {
254                let lord = VedicPlanet::get_ruler_of(rasi);
255                chart.planets.iter().find(|p| p.planet == lord).map(|p| p.rasi).unwrap_or(rasi)
256            }
257        };
258
259        let dist = if is_counting_forward {
260            (lord_rasi as i16 - rasi as i16 + 12) % 12
261        } else {
262            (rasi as i16 - lord_rasi as i16 + 12) % 12
263        };
264
265        if dist == 0 { 12 } else { dist as u32 }
266    }

268    /// Selection rules for Scorpio/Aquarius co-rulers (Jaimini / KN Rao)
269    /// 1. Sign with more planets (excluding the sign itself) wins.
270    /// 2. If equal, the node/planet that is stronger wins (Exaltation > Own Sign > Others).
271    /// 3. If still equal, the one with more degrees in the sign wins (Jaimini logic).
272    fn evaluate_co_ruler_strength(chart: &VedicChart, rasi: u8, p1: VedicPlanet, p2: VedicPlanet) -> u8 {
273        let pos1 = match chart.planets.iter().find(|p| p.planet == p1) {
274            Some(p) => p,
275            None => return 1,
276        };
277        let pos2 = match chart.planets.iter().find(|p| p.planet == p2) {
278            Some(p) => p,
279            None => return 2,
280        };
281
282        // Count conjunctions in the rasi where the lord is placed (excluding the lord itself)
283        let count_conj = |p_rasi: u8, planet: VedicPlanet| {
284            chart.planets.iter()
285                .filter(|p| p.rasi == p_rasi && p.planet != planet)
286                .count()
287        };
288
289        let c1 = count_conj(pos1.rasi, p1);
290        let c2 = count_conj(pos2.rasi, p2);
291
292        if c1 > c2 { return pos1.rasi; }
293        if c2 > c1 { return pos2.rasi; }
294
295        // If conjunctions are equal, check if one is in their own sign/exalted
296        let is_stronger = |p: VedicPlanet, prasi: u8| {
297            if prasi == rasi { 2 } // Own sign
298            else if prasi == p.exaltation_rasi() { 3 } // Exalted
299            else { 1 }
300        };
301
302        let s1 = is_stronger(p1, pos1.rasi);
303        let s2 = is_stronger(p2, pos2.rasi);
304
305        if s1 > s2 { return pos1.rasi; }
306        if s2 > s1 { return pos2.rasi; }
307
308        // Last resort: Higher degree in the rasi
309        if pos1.sidereal_deg % 30.0 > pos2.sidereal_deg % 30.0 { pos1.rasi } else { pos2.rasi }
310    }

312    /// Jaimini Rashi Drishti (Sign Aspects)
313    /// Rule: 
314    /// - Movable (Ar, Cn, Li, Cp) aspects all Fixed (Ta, Le, Sc, Aq) except the adjacent one.
315    /// - Fixed aspects all Movable except the adjacent one.
316    /// - Dual (Ge, Vi, Sg, Pi) aspects all other Dual signs.
317    pub fn get_rashi_drishti(rasi: u8) -> Vec<u8> {
318        match rasi {
319            1 => vec![5, 8, 11], // Ar (M) -> Le, Sc, Aq (not Ta)
320            2 => vec![4, 7, 10], // Ta (F) -> Cn, Li, Cp (not Ar)
321            3 => vec![6, 9, 12], // Ge (D) -> Vi, Sg, Pi (not self)
322            4 => vec![2, 8, 11], // Cn (M) -> Ta, Sc, Aq (not Le)
323            5 => vec![1, 7, 10], // Le (F) -> Ar, Li, Cp (not Cn)
324            6 => vec![3, 9, 12], // Vi (D) -> Ge, Sg, Pi
325            7 => vec![2, 5, 11], // Li (M) -> Ta, Le, Aq (not Sc)
326            8 => vec![1, 4, 10], // Sc (F) -> Ar, Cn, Cp (not Li)
327            9 => vec![3, 6, 12], // Sg (D) -> Ge, Vi, Pi
328            10 => vec![2, 5, 8], // Cp (M) -> Ta, Le, Sc (not Aq)
329            11 => vec![1, 4, 7], // Aq (F) -> Ar, Cn, Li (not Cp)
330            12 => vec![3, 6, 9], // Pi (D) -> Ge, Vi, Sg
331            _ => vec![],
332        }
333    }

335    /// Calculate Argala (Intervention) for a sign/planet position
336    pub fn get_argala(rasi: u8) -> Vec<(u8, String)> {
337        let mut results = Vec::new();
338        
339        // Primary Argala: 2, 4, 11
340        // Obstruction (Virodhargala): 12, 10, 3
341        let primary = [(2, "Wealth/Speech"), (4, "Happiness/Home"), (11, "Gains")];
342        for (dist, desc) in primary {
343            let target = ((rasi as i16 + dist - 2) % 12 + 1) as u8;
344            results.push((target, format!("Primary Argala ({})", desc)));
345        }
346
347        // Secondary Argala: 5 (Obstruction: 9)
348        let secondary = (5, "Knowledge/Children");
349        let target = ((rasi as i16 + secondary.0 - 2) % 12 + 1) as u8;
350        results.push((target, format!("Secondary Argala ({})", secondary.1)));
351
352        results
353    }
