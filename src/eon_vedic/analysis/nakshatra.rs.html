6pub enum TarabalaCategory {
7    Janma = 1,    // Danger to self
8    Sampat = 2,   // Wealth and Prosperity
9    Vipat = 3,    // Obstacles and Losses
10    Kshema = 4,   // Well-being and Security
11    Pratyak = 5,  // Opposition and Enemies
12    Sadhana = 6,  // Achievement and Success
13    Naidhana = 7, // Danger and Endings
14    Mitra = 8,    // Friendship and Cooperation
15    AtiMitra = 9, // Great Friendship
16}
17
18impl TarabalaCategory {
19    pub fn description(&self) -> &'static str {
20        match self {
21            Self::Janma => "Janma (Birth) - Danger to self/health",
22            Self::Sampat => "Sampat (Wealth) - Prosperity and gain",
23            Self::Vipat => "Vipat (Danger) - Obstacles and accidents",
24            Self::Kshema => "Kshema (Safety) - Security and comfort",
25            Self::Pratyak => "Pratyak (Obstacle) - Resistance and conflicts",
26            Self::Sadhana => "Sadhana (Effort) - Success through hard work",
27            Self::Naidhana => "Naidhana (Ending) - Critical danger or finality",
28            Self::Mitra => "Mitra (Friend) - Support and help",
29            Self::AtiMitra => "Ati-Mitra (Great Friend) - Exceptional gains",
30        }
31    }
32
33    pub fn is_auspicious(&self) -> bool {
34        matches!(
35            self,
36            Self::Sampat | Self::Kshema | Self::Sadhana | Self::Mitra | Self::AtiMitra
37        )
38    }
39}

175    /// Calculate Tarabala from natal Moon Nakshatra to target Nakshatra
176    pub fn calculate_tarabala(natal_moon_nak: u8, target_nak: u8) -> TarabalaCategory {
177        let count = if target_nak >= natal_moon_nak {
178            target_nak - natal_moon_nak + 1
179        } else {
180            (27 - natal_moon_nak) + target_nak + 1
181        };
182
183        let cat = (count - 1) % 9;
184        match cat {
185            0 => TarabalaCategory::Janma,
186            1 => TarabalaCategory::Sampat,
187            2 => TarabalaCategory::Vipat,
188            3 => TarabalaCategory::Kshema,
189            4 => TarabalaCategory::Pratyak,
190            5 => TarabalaCategory::Sadhana,
191            6 => TarabalaCategory::Naidhana,
192            7 => TarabalaCategory::Mitra,
193            8 | _ => TarabalaCategory::AtiMitra, // (count % 9 == 0)
194        }
195    }

197    /// Calculate nakshatra (1..27) and pada (1..4) from sidereal longitude (0..360)
198    pub fn nakshatra_and_pada(longitude: f64) -> (u8, u8) {
199        let deg = (longitude % 360.0 + 360.0) % 360.0;
200        let nak_pos = deg / (360.0 / 27.0);
201        let nakshatra = (nak_pos.floor() as u8).saturating_add(1).min(27);
202        let pada_pos = (deg % (360.0 / 27.0)) / (360.0 / 108.0);
203        let pada = (pada_pos.floor() as u8).saturating_add(1).min(4);
204        (nakshatra, pada)
205    }
