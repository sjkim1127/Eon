32        match house_diff_1_indexed {
33            3 | 11 => TajikaAspectType::Mitra(false),
34            5 | 9 => TajikaAspectType::Mitra(true),
35            1 | 7 => TajikaAspectType::Shatru(true),
36            4 | 10 => TajikaAspectType::Shatru(false),
37            _ => TajikaAspectType::Sama,

41    /// Ithasala Yoga (Applying Aspect)
42    pub fn check_ithasala(
43        p1_long: f64,
44        p1_speed: f64,
45        p2_long: f64,
46        p2_speed: f64,
47        orb: f64,
48    ) -> bool {
49        let (faster_long, _faster_speed, slower_long, _slower_speed) = if p1_speed > p2_speed {
50            (p1_long, p1_speed, p2_long, p2_speed)
51        } else {
52            (p2_long, p2_speed, p1_long, p1_speed)
53        };
54
55        let diff = (slower_long - faster_long + 360.0) % 360.0;
56        diff > 0.0 && diff <= orb
57    }

59    /// Calculate Sahams (Sensitive Points in Annual Chart)
60    pub fn calculate_sahams(chart: &VedicChart) -> Vec<Saham> {
61        let mut results = Vec::new();
62        let lagna = chart.ascendant.sidereal_deg;
63
64        let get_p = |p: VedicPlanet| {
65            chart
66                .planets
67                .iter()
68                .find(|pos| pos.planet == p)
69                .map(|pos| pos.sidereal_deg)
70                .unwrap_or(0.0)
71        };
72
73        let sun = get_p(VedicPlanet::Sun);
74        let moon = get_p(VedicPlanet::Moon);
75        let jupiter = get_p(VedicPlanet::Jupiter);
76
77        let is_day = chart.panchanga.is_day_birth;
78
79        // Punya Saham (Fortune)
80        let punya = if is_day {
81            (moon - sun + lagna + 360.0) % 360.0
82        } else {
83            (sun - moon + lagna + 360.0) % 360.0
84        };
85        results.push(Saham {
86            name: "Punya (Fortune)".to_string(),
87            longitude: punya,
88            rasi: (punya / 30.0).floor() as u8 + 1,
89        });
90
91        // Vidya Saham (Knowledge)
92        let vidya = if is_day {
93            (sun - moon + lagna + 360.0) % 360.0
94        } else {
95            (moon - sun + lagna + 360.0) % 360.0
96        };
97        results.push(Saham {
98            name: "Vidya (Knowledge)".to_string(),
99            longitude: vidya,
100            rasi: (vidya / 30.0).floor() as u8 + 1,
101        });
102
103        // Yash Saham (Fame)
104        let yash = (jupiter - sun + lagna + 360.0) % 360.0;
105        results.push(Saham {
106            name: "Yash (Fame)".to_string(),
107            longitude: yash,
108            rasi: (yash / 30.0).floor() as u8 + 1,
109        });
110
111        results
112    }

146    /// Selection of Year Lord (Varsheshwara) - Full Orthodox Tajika Implementation
147    pub fn select_year_lord(
148        chart: &VedicChart,
149        birth_lagna_rasi: u8,
150        age_years: u32,
151    ) -> VedicPlanet {
152        let annual_lagna_rasi = chart.ascendant.rasi;
153        let muntha_rasi = Self::calculate_muntha(birth_lagna_rasi, age_years);
154        let is_day = chart.panchanga.is_day_birth;
155
156        // 1. Pancha Adhikaris (5 Candidates)
157        let mut candidates = Vec::new();
158
159        // 1.1 Muntha Lord
160        candidates.push(VedicPlanet::get_ruler_of(muntha_rasi));
161
162        // 1.2 Birth Lagna Lord
163        candidates.push(VedicPlanet::get_ruler_of(birth_lagna_rasi));
164
165        // 1.3 Varsha (Annual) Lagna Lord
166        candidates.push(VedicPlanet::get_ruler_of(annual_lagna_rasi));
167
168        // 1.4 Dina/Ratri Pati (Day/Night Lord)
169        if is_day {
170            if let Some(sun) = chart.planets.iter().find(|p| p.planet == VedicPlanet::Sun) {
171                candidates.push(VedicPlanet::get_ruler_of(sun.rasi));
172            }
173        } else {
174            if let Some(moon) = chart.planets.iter().find(|p| p.planet == VedicPlanet::Moon) {
175                candidates.push(VedicPlanet::get_ruler_of(moon.rasi));
176            }
177        }
178
179        // 1.5 Tri-Rashi Pati
180        candidates.push(Self::get_tri_rashi_pati(annual_lagna_rasi, is_day));
181
182        // 2. Filter candidates who aspect the Annual Lagna
183        // (In Tajika, any aspect makes it eligible)
184        let eligible: Vec<VedicPlanet> = candidates
185            .into_iter()
186            .filter(|&p| {
187                if let Some(pos) = chart.planets.iter().find(|pos| pos.planet == p) {
188                    let dist = (pos.rasi as i16 - annual_lagna_rasi as i16 + 12) % 12;
189                    let aspect = Self::get_aspect_type((dist + 1) as u8);
190                    !matches!(aspect, TajikaAspectType::Sama)
191                        || p == VedicPlanet::get_ruler_of(annual_lagna_rasi)
192                } else {
193                    false
194                }
195            })
196            .collect();
197
198        // 3. Selection: Strongest among eligible by Harsha Bala
199        if eligible.is_empty() {
200            return VedicPlanet::get_ruler_of(muntha_rasi);
201        }
202
203        let mut strongest = eligible[0];
204        let mut max_bala = 0;
205
206        for p in eligible {
207            let bala = TajikaBala::calculate_harsha_bala(chart, p);
208            if bala >= max_bala {
209                max_bala = bala;
210                strongest = p;
211            }
212        }
213
214        strongest
215    }

221    /// Harsha Bala (4 factors of Joy)
222    pub fn calculate_harsha_bala(chart: &VedicChart, planet: VedicPlanet) -> u32 {
223        let mut score = 0;
224        let p_pos = chart.planets.iter().find(|p| p.planet == planet);
225
226        if let Some(p) = p_pos {
227            // 1. Sthana (House)
228            let h = p.house_index;
229            match planet {
230                VedicPlanet::Sun
231                    if h == 9 => {
232                        score += 5;
233                    }
234                VedicPlanet::Moon
235                    if h == 4 => {
236                        score += 5;
237                    }
238                VedicPlanet::Mars
239                    if h == 6 => {
240                        score += 5;
241                    }
242                VedicPlanet::Mercury
243                    if h == 1 => {
244                        score += 5;
245                    }
246                VedicPlanet::Jupiter
247                    if h == 11 => {
248                        score += 5;
249                    }
250                VedicPlanet::Venus
251                    if h == 5 => {
252                        score += 5;
253                    }
254                VedicPlanet::Saturn
255                    if h == 12 => {
256                        score += 5;
257                    }
258                _ => {}
259            }
260
261            // 2. Swavarga (Own/Exaltation Sign in Annual Chart)
262            let lord = VedicPlanet::get_ruler_of(p.rasi);
263            if lord == planet || p.rasi == planet.exaltation_rasi() {
264                score += 5;
265            }
266
267            // 3. Stri-Purusha (Gender/Sect)
268            let is_day = chart.panchanga.is_day_birth;
269            match planet {
270                VedicPlanet::Sun | VedicPlanet::Mars | VedicPlanet::Jupiter
271                    if is_day => {
272                        score += 5;
273                    }
274                VedicPlanet::Moon | VedicPlanet::Venus | VedicPlanet::Saturn
275                    if !is_day => {
276                        score += 5;
277                    }
278                _ => {}
279            }
280
281            // 4. Appearance (In Kendra)
282            if [1, 4, 7, 10].contains(&h) {
283                score += 5;
284            }
285        }
286        score
287    }
