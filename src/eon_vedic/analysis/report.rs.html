90    pub fn generate(chart: &VedicChart, birth_time: chrono::DateTime<Utc>, _birth_lagna_rasi: u8) -> Self {
91        let get_karaka = |role: crate::analysis::jaimini::JaiminiKarakaRole| {
92            chart.karakas.iter()
93                .find(|k| std::mem::discriminant(&k.role) == std::mem::discriminant(&role))
94                .map(|k| k.planet)
95        };
96
97        let ak = get_karaka(crate::analysis::jaimini::JaiminiKarakaRole::Atmakaraka).unwrap_or(VedicPlanet::Sun);
98        let amk = get_karaka(crate::analysis::jaimini::JaiminiKarakaRole::Amatyakaraka).unwrap_or(VedicPlanet::Sun);
99        let bk = get_karaka(crate::analysis::jaimini::JaiminiKarakaRole::Bhratrukaraka);
100        let mk = get_karaka(crate::analysis::jaimini::JaiminiKarakaRole::Matrukaraka);
101        let pik = get_karaka(crate::analysis::jaimini::JaiminiKarakaRole::Pitrikaraka);
102        let pk = get_karaka(crate::analysis::jaimini::JaiminiKarakaRole::Putrakaraka);
103        let gk = get_karaka(crate::analysis::jaimini::JaiminiKarakaRole::Gnatikaraka);
104        let dk = get_karaka(crate::analysis::jaimini::JaiminiKarakaRole::Darakaraka).unwrap_or(VedicPlanet::Sun);
105
106        let mut house_summary = Vec::new();
107        for h in &chart.bhava_strengths {
108            let rating = if h.total_score > 400.0 {
109                "Excellent"
110            } else if h.total_score > 300.0 {
111                "Strong"
112            } else if h.total_score > 200.0 {
113                "Average"
114            } else {
115                "Weak"
116            };
117
118            let summary = match h.house {
119                1 => "Vitality & Personality",
120                2 => "Wealth & Speech",
121                3 => "Effort & Siblings",
122                4 => "Mother & Domestic Joy",
123                5 => "Creativity & Progeny",
124                6 => "Enemies, Debt & Health",
125                7 => "Partnership & Public",
126                8 => "Longevity & Transformation",
127                9 => "Wisdom & Fortune",
128                10 => "Career & Visibility",
129                11 => "Gains & Fulfillment",
130                12 => "Expenses & Spiritual Depth",
131                _ => "Life Area",
132            }.to_string();
133
134            house_summary.push(HouseRating {
135                house: h.house,
136                rating: rating.to_string(),
137                total_score: h.total_score,
138                summary,
139                description: format!("Capacity for specific life themes related to House {}.", h.house),
140                reasons: h.reasons.clone(),
141            });
142        }
143
144        // Calculate Dasha Focus (Vimshottari)
145        let moon_pos = chart.planets.iter().find(|p| p.planet == VedicPlanet::Moon);
146        let (dasha_focus, dasha_timeline) = if let Some(m) = moon_pos {
147            let timeline = crate::analysis::dasha::VimshottariDasha::calculate_timeline(
148                birth_time,
149                m.sidereal_deg,
150                3, // Maha + Antar + Pratyantar
151            );
152            let focus = if let Some(d) = timeline.first() {
153                format!("Current Major Period: {:?}", d.lord)
154            } else {
155                "Dasha info unavailable".to_string()
156            };
157            (focus, timeline)
158        } else {
159            ("Moon position required for Dasha".to_string(), Vec::new())
160        };
161
162        // Yogini Dasha
163        let yogini_timeline = if let Some(m) = moon_pos {
164            YoginiDasha::calculate_timeline(birth_time, m.sidereal_deg)
165        } else {
166            Vec::new()
167        };
168
169        // Nakshatra Info
170        let nakshatra_info = if let Some(m) = moon_pos {
171            let name = crate::analysis::nakshatra::NakshatraEngine::get_name(m.nakshatra);
172            let pada_desc =
173                crate::analysis::nakshatra::NakshatraEngine::get_pada_description(m.pada);
174            format!("Birth Star: {} (Pada {}) - {}", name, m.pada, pada_desc)
175        } else {
176            "Nakshatra info unavailable".to_string()
177        };
178
179        // Yoga 계산
180        let yogas = YogaEngine::check_yogas(chart);
181
182        // Jaimini Integration
183        let al = chart.arudha_padas.iter().find(|a| a.house == 1).map(|a| a.rasi).unwrap_or(0);
184        let ul = chart.arudha_padas.iter().find(|a| a.house == 12).map(|a| a.rasi).unwrap_or(0);
185        let special_lagnas_summary: Vec<_> = chart.special_lagnas.iter().map(|s| (s.name.clone(), s.rasi)).collect();
186
187        // Varga Interpretation Integration
188        let planets_to_check = [
189            VedicPlanet::Sun, VedicPlanet::Moon, VedicPlanet::Mars,
190            VedicPlanet::Mercury, VedicPlanet::Jupiter, VedicPlanet::Venus, VedicPlanet::Saturn
191        ];
192        let mut varga_interpretations = Vec::new();
193        for p in planets_to_check {
194            varga_interpretations.push(VargaInterpretationEngine::interpret_planet(chart, p));
195        }
196        let d9_marriage_analysis = VargaInterpretationEngine::analyze_marriage(chart);
197        let d10_career_analysis = VargaInterpretationEngine::analyze_career(chart);
198
199        // Chara Dasha (Jaimini)
200        let chara_dasha_timeline = crate::analysis::jaimini::JaiminiEngine::calculate_chara_dasha(chart);
201
202        // Kala Chakra Dasha
203        let kalachakra_timeline = if let Some(m) = moon_pos {
204            crate::prediction::kalachakra::KalaChakraDasha::calculate(m.sidereal_deg, birth_time)
205        } else {
206            Vec::new()
207        };
208
209        Self {
210            primary_karakas: KarakaSummary {
211                atmakaraka: ak,
212                amatyakaraka: amk,
213                bhratrukaraka: bk,
214                matrukaraka: mk,
215                pitrikaraka: pik,
216                putrakaraka: pk,
217                gnatikaraka: gk,
218                darakaraka: dk,
219            },
220            house_summary,
221            dasha_focus,
222            dasha_timeline,
223            yogini_timeline,
224            chara_dasha_timeline,
225            all_karakas: chart.karakas.clone(),
226            nakshatra_info,
227            overall_strength_score: chart
228                .bhava_strengths
229                .iter()
230                .map(|h| h.total_score)
231                .sum::<f64>()
232                / 12.0,
233            sade_sati: crate::analysis::gochara::SadeSatiPhase::None, // Expected to be populated by service/caller
234            yogas,
235            arudha_lagna: al,
236            upapada_lagna: ul,
237            special_lagnas_summary,
238            varga_interpretations,
239            d9_marriage_analysis,
240            d10_career_analysis,
241            kalachakra_timeline,
242        }
243    }
244}

246impl TajikaReport {
247    pub fn generate(chart: &VedicChart, birth_lagna_rasi: u8, age_years: u32) -> Self {
248        let sahams = TajikaEngine::calculate_sahams(chart);
249        let mut harsha_bala_summary = Vec::new();
250        let planets_to_check = [
251            VedicPlanet::Sun, VedicPlanet::Moon, VedicPlanet::Mars,
252            VedicPlanet::Mercury, VedicPlanet::Jupiter, VedicPlanet::Venus, VedicPlanet::Saturn
253        ];
254        for p in planets_to_check {
255            harsha_bala_summary.push((p, TajikaBala::calculate_harsha_bala(chart, p)));
256        }
257
258        let muntha_rasi = TajikaEngine::calculate_muntha(birth_lagna_rasi, age_years);
259        let year_lord = TajikaEngine::select_year_lord(chart, birth_lagna_rasi, age_years);
260
261        Self {
262            year_lord: Some(year_lord),
263            muntha_rasi,
264            sahams,
265            harsha_bala_summary,
266            summary: format!("Annual Chart Summary: Year Lord is {:?}, Muntha in Sign {}.", year_lord, muntha_rasi),
267        }
268    }
269
270    pub fn to_text_summary(&self) -> String {
271        self.summary.clone()
272    }
273}

275impl VedicAnalysisReport {
276    pub fn to_text_summary(&self) -> String {
277        let mut s = String::new();
278        s.push_str("# Vedic Chart Analysis Summary\n\n");
279
280        s.push_str("## 💠 Key Indicators (Jaimini Karakas)\n");
281        s.push_str(&format!("- **Atmakaraka (Self)**: {:?}\n", self.primary_karakas.atmakaraka));
282        s.push_str(&format!("- **Amatyakaraka (Career)**: {:?}\n", self.primary_karakas.amatyakaraka));
283        if let Some(bk) = self.primary_karakas.bhratrukaraka { s.push_str(&format!("- **Bhratrukaraka (Siblings/Effort)**: {:?}\n", bk)); }
284        if let Some(mk) = self.primary_karakas.matrukaraka { s.push_str(&format!("- **Matrukaraka (Mother/Home)**: {:?}\n", mk)); }
285        if let Some(pk) = self.primary_karakas.putrakaraka { s.push_str(&format!("- **Putrakaraka (Children/Intelligence)**: {:?}\n", pk)); }
286        s.push_str(&format!("- **Darakaraka (Partner)**: {:?}\n", self.primary_karakas.darakaraka));
287        s.push('\n');
288
289        s.push_str("## 🧭 Jaimini Indicators\n");
290        if self.arudha_lagna > 0 {
291            s.push_str(&format!("- **Arudha Lagna (AL)**: Sign {}\n", self.arudha_lagna));
292        }
293        if self.upapada_lagna > 0 {
294            s.push_str(&format!("- **Upapada Lagna (UL)**: Sign {}\n", self.upapada_lagna));
295        }
296        for (name, rasi) in &self.special_lagnas_summary {
297            s.push_str(&format!("- **{}**: Sign {}\n", name, rasi));
298        }
299        s.push('\n');
300
301        s.push_str(&format!("- **Nakshatra**: {}\n", self.nakshatra_info));
302        s.push_str(&format!("- **Vimshottari Dasha**: {}\n", self.dasha_focus));
303        if let Some(y) = self.yogini_timeline.first() {
304            s.push_str(&format!("- **Current Yogini**: {} (ruled by {:?})\n", y.name.as_ref().unwrap_or(&"Unknown".to_string()), y.lord));
305        }
306        if let Some(c) = self.chara_dasha_timeline.first() {
307            s.push_str(&format!("- **Chara Dasha**: Sign {} until {}\n", c.rasi, c.end_time.format("%Y-%m-%d")));
308        }
309        s.push('\n');
310
311        s.push_str("## 🏠 House Capacities\n");
312        let mut strong_houses: Vec<_> = self
313            .house_summary
314            .iter()
315            .filter(|h| h.total_score > 300.0)
316            .collect();
317        strong_houses.sort_by(|a, b| b.total_score.partial_cmp(&a.total_score).unwrap_or(std::cmp::Ordering::Equal));
318
319        if strong_houses.is_empty() {
320            s.push_str("- No exceptionally strong houses detected.\n");
321        } else {
322            for h in strong_houses.iter().take(3) {
323                s.push_str(&format!(
324                    "- **House {}**: {} ({:.1})\n",
325                    h.house, h.rating, h.total_score
326                ));
327            }
328        }
329        s.push('\n');
330
331        s.push_str("## 💍 Marriage & Career Insights (Divisional Charts)\n");
332        s.push_str(&format!("- **D9 Navamsa**: {}\n", self.d9_marriage_analysis));
333        s.push_str(&format!("- **D10 Dasamsa**: {}\n\n", self.d10_career_analysis));
334
335        s.push_str("## 🚀 Special Status (Vargottama / Pushkar)\n");
336        let special_planets: Vec<String> = self.varga_interpretations.iter()
337            .filter(|vi| vi.is_vargottama || vi.is_pushkar_navamsa)
338            .map(|vi| {
339                let mut status = Vec::new();
340                if vi.is_vargottama { status.push("Vargottama"); }
341                if vi.is_pushkar_navamsa { status.push("Pushkar Navamsa"); }
342                format!("{:?}({})", vi.planet, status.join("+"))
343            }).collect();
344        if special_planets.is_empty() {
345            s.push_str("- No planets in special varga status.\n");
346        } else {
347            s.push_str(&format!("- **Planets**: {}\n", special_planets.join(", ")));
348        }
349        s.push('\n');
350
351        s.push_str("## 🪐 Transit Alerts & Gochara\n");
352        // Sade Sati
353        match self.sade_sati {
354            crate::analysis::gochara::SadeSatiPhase::Rising => s.push_str("- **Sade Sati (Rising)**: Saturn has entered the 12th from Moon. A period of internal shifts begins.\n"),
355            crate::analysis::gochara::SadeSatiPhase::Peak => s.push_str("- **Sade Sati (Peak)**: Saturn is over the Moon. Focus on emotional resilience and discipline.\n"),
356            crate::analysis::gochara::SadeSatiPhase::Setting => s.push_str("- **Sade Sati (Setting)**: The intensity is fading as Saturn reaches the 2nd from Moon.\n"),
357            crate::analysis::gochara::SadeSatiPhase::None => s.push_str("- No Sade Sati active.\n"),
358        }
359        s.push('\n');
360        s.push_str("Detailed planetary transits including Murti Nirnaya and Vedha obstructions are available in the interactive dashboard.\n\n");
361
362        s.push_str(&format!(
363            "## 📊 Overall Chart Strength: {:.1}/600\n",
364            self.overall_strength_score
365        ));
366
367        s
368    }
