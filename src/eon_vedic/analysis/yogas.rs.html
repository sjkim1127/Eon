138    pub fn check_yogas(chart: &VedicChart) -> Vec<YogaResult> {
139        let rules = Self::get_rules();
140        let mut results = Vec::new();
141
142        for rule in rules {
143            // Parivartana is handled separately through classification logic
144            if rule.condition == YogaCondition::ParivartanaCheck {
145                continue;
146            }
147
148            if let Some(mut planets) = Self::evaluate_condition(&rule.condition, chart) {
149                // Determine Quality
150                let mut quality = Self::assess_quality(chart, &planets);
151
152                // [BPHS 고도화] Vipareeta Raja Yoga: Shadbala 기반 최종 품질 조정
153                // evaluate_condition 에서 이미 score < 80 행성은 필터됐으므로
154                // 여기서는 80~180 구간(약한 Yoga)을 Weak 으로 다운그레이드
155                if rule.yoga_type == YogaType::VipareetaRajaYoga {
156                    let avg_shadbala = planets
157                        .iter()
158                        .filter_map(|&pl| {
159                            chart
160                                .planets
161                                .iter()
162                                .find(|p| p.planet == pl)
163                                .map(|pos| StrengthEngine::calculate(pos, chart).total_score)
164                        })
165                        .sum::<f64>()
166                        / planets.len().max(1) as f64;
167
168                    if avg_shadbala < 180.0 {
169                        quality = YogaQuality::Weak(format!(
170                            "Shadbala 강도 부족 ({:.0}/180 미만) — Yoga 발현 약함",
171                            avg_shadbala
172                        ));
173                    }
174                }
175
176                if rule.yoga_type == YogaType::PanchaMahapurusha {
177                    let avg_shadbala = planets
178                        .iter()
179                        .filter_map(|&pl| {
180                            chart
181                                .planets
182                                .iter()
183                                .find(|p| p.planet == pl)
184                                .map(|pos| StrengthEngine::calculate(pos, chart).total_score)
185                        })
186                        .sum::<f64>()
187                        / planets.len().max(1) as f64;
188
189                    if avg_shadbala < 100.0 {
190                        quality = YogaQuality::Weak(format!(
191                            "Shadbala 강도 부족 ({:.0}/100 미만) — Mahapurusha 발현 불가",
192                            avg_shadbala
193                        ));
194                    }
195                }
196
197                if rule.yoga_type == YogaType::NeechaBhanga {
198                    let mut has_kendra_exaltation = false;
199                    for pl in &planets {
200                        if let Some(pos) = chart.planets.iter().find(|p| p.planet == *pl) {
201                            if [1, 4, 7, 10].contains(&pos.house_index) {
202                                has_kendra_exaltation = true;
203                            }
204                        }
205                    }
206                    if !has_kendra_exaltation {
207                        quality = YogaQuality::Weak(
208                            "고양성/지배성이 켄드라에 위치하지 않아 약한 니차방가".to_string(),
209                        );
210                    } else {
211                        quality = YogaQuality::VeryHigh; // 확실한 역전
212                    }
213                }
214
215                // Dedup planets
216                planets.sort();
217                planets.dedup();
218
219                results.push(YogaResult {
220                    name: rule.name,
221                    yoga_type: rule.yoga_type,
222                    description: rule.description,
223                    planets_involved: planets,
224                    quality,
225                });
226            }
227        }
228
229        // Include Parivartana Yogas with proper classification (Maha, Khala, Dainya)
230        results.extend(Self::find_parivartana_exchanges(chart));
231
232        // Include Nabhasa Yogas (BPHS planet distribution patterns)
233        results.extend(Self::evaluate_nabhasa_yogas(chart));
234
235        results
236    }

451    /// Find and classify all Parivartana Yoga exchanges
452    fn find_parivartana_exchanges(chart: &VedicChart) -> Vec<YogaResult> {
453        let mut results = Vec::new();
454        let mut processed = std::collections::HashSet::new();
455
456        // Pre-build a map from planet to its position for O(1) lookups instead of
457        // repeated O(n) linear scans inside the nested loop.
458        let planet_map: std::collections::HashMap<_, _> =
459            chart.planets.iter().map(|p| (p.planet, p)).collect();
460
461        for h1 in 1..=12 {
462            let lord1 = Self::get_lord_of_house(chart.ascendant.rasi, h1);
463            if let Some(pos1) = planet_map.get(&lord1) {
464                let h2 = pos1.house_index;
465                if h1 != h2 {
466                    let lord2 = Self::get_lord_of_house(chart.ascendant.rasi, h2);
467                    if let Some(pos2) = planet_map.get(&lord2) {
468                        if pos2.house_index == h1 && !processed.contains(&(h1.min(h2), h1.max(h2)))
469                        {
470                            // This is a valid exchange
471                            processed.insert((h1.min(h2), h1.max(h2)));
472
473                            let (yoga_type, quality_desc) = Self::classify_parivartana(h1, h2);
474                            let planets = vec![lord1, lord2];
475                            let quality = Self::assess_quality(chart, &planets);
476
477                            results.push(YogaResult {
478                                name: format!("Parivartana Yoga: House {} ↔ House {}", h1, h2),
479                                yoga_type,
480                                description: format!(
481                                    "{} ({} exchange)",
482                                    quality_desc,
483                                    if h1 == h2 { "same house" } else { "cross" }
484                                ),
485                                planets_involved: planets,
486                                quality,
487                            });
488                        }
489                    }
490                }
491            }
492        }
493
494        results
495    }

497    /// Classify Parivartana Yoga type based on house types (BPHS Standard)
498    fn classify_parivartana(h1: u8, h2: u8) -> (YogaType, String) {
499        let is_kendra = |h: u8| matches!(h, 1 | 4 | 7 | 10);
500        let is_trikona = |h: u8| matches!(h, 1 | 5 | 9);
501        let is_dusthana = |h: u8| matches!(h, 6 | 8 | 12);
502
503        let d1 = is_dusthana(h1);
504        let d2 = is_dusthana(h2);
505
506        if d1 && d2 {
507            // Both dusthana - Dainya (Difficult)
508            (
509                YogaType::ParivartanaDainya,
510                "Dainya Parivartana - Exchange of difficult houses".to_string(),
511            )
512        } else if d1 || d2 {
513            // One dusthana - Khala (Mixed)
514            (
515                YogaType::ParivartanaKhala,
516                "Khala Parivartana - Mixed exchange with one difficult house".to_string(),
517            )
518        } else if (is_kendra(h1) || is_trikona(h1)) && (is_kendra(h2) || is_trikona(h2)) {
519            // Both auspicious - Maha (Great)
520            (
521                YogaType::ParivartanaMaha,
522                "Maha Parivartana - Great exchange of auspicious houses".to_string(),
523            )
524        } else {
525            // Mixed but not involving dusthana - Khala
526            (
527                YogaType::ParivartanaKhala,
528                "Khala Parivartana - Mixed exchange".to_string(),
529            )
530        }
531    }

533    fn evaluate_condition(cond: &YogaCondition, chart: &VedicChart) -> Option<Vec<VedicPlanet>> {
534        match cond {
535            YogaCondition::Conjunction(p1, p2) => {
536                let pos1 = chart.planets.iter().find(|p| p.planet == *p1)?;
537                let pos2 = chart.planets.iter().find(|p| p.planet == *p2)?;
538                if pos1.rasi == pos2.rasi {
539                    Some(vec![*p1, *p2])
540                } else {
541                    None
542                }
543            }
544            YogaCondition::KendraRelationship(p1, p2) => {
545                let pos1 = chart.planets.iter().find(|p| p.planet == *p1)?;
546                let pos2 = chart.planets.iter().find(|p| p.planet == *p2)?;
547
548                // House from P2
549                let diff = pos1.rasi as i32 - pos2.rasi as i32;
550                let dist = if diff >= 0 { diff + 1 } else { diff + 13 };
551
552                if [1, 4, 7, 10].contains(&dist) {
553                    Some(vec![*p1, *p2])
554                } else {
555                    None
556                }
557            }
558            YogaCondition::Placement(planet, house) => {
559                let pos = chart.planets.iter().find(|p| p.planet == *planet)?;

1100    /// Returns the unique rasi numbers occupied by the 9 planets (Ascendant excluded).
1101    fn occupied_rasis(chart: &VedicChart) -> std::collections::HashSet<u8> {
1102        use std::collections::HashSet;
1103        chart
1104            .planets
1105            .iter()
1106            .filter(|p| {
1107                // Exclude Ascendant pseudo-planet (planet == Ketu is the mock ascendant sentinel
1108                // in tests, but real charts use VedicPlanet::Ascendant).
1109                p.planet != VedicPlanet::Ascendant
1110            })
1111            .map(|p| p.rasi)
1112            .collect::<HashSet<u8>>()
1113    }

1115    /// Returns true when ALL 9 non-Ascendant planets fall within rasis of the given quality.
1116    /// quality: 0 = Chara (0-indexed rasi % 3 == 0),
1117    ///          1 = Sthira (rasi % 3 == 1),
1118    ///          2 = Dvisvabhava (rasi % 3 == 2).
1119    /// Rasi numbering: 1-Aries, 2-Taurus … 12-Pisces.
1120    /// Chara  (Cardinal): Aries(1), Cancer(4), Libra(7), Capricorn(10)  → (rasi-1) % 3 == 0
1121    /// Sthira (Fixed):    Taurus(2), Leo(5), Scorpio(8), Aquarius(11)   → (rasi-1) % 3 == 1
1122    /// Dvísva (Mutable):  Gemini(3), Virgo(6), Sagittarius(9), Pisces(12)→ (rasi-1) % 3 == 2

1124        chart
1125            .planets
1126            .iter()
1127            .filter(|p| p.planet != VedicPlanet::Ascendant)
1128            .all(|p| (p.rasi - 1) % 3 == quality)
1129    }

1238    // Updated assess_quality to pass 'chart' to FunctionalNature::analyze
1239    fn assess_quality(chart: &VedicChart, planets: &[VedicPlanet]) -> YogaQuality {
1240        let mut score = 0;
1241        let mut reasons = Vec::new();
1242        let mut is_combust = false;
1243
1244        for &pl in planets {
1245            if let Some(pos) = chart.planets.iter().find(|p| p.planet == pl) {
1246                if pos.is_combust {
1247                    is_combust = true;
1248                    reasons.push(format!("{:?} is Combust", pl));
1249                }
1250
1251                // CHANGED: Passing 'chart' instead of just 'rasi'
1252                match FunctionalNature::analyze(chart, pl) {
1253                    FunctionalStatus::Yogakaraka => {
1254                        score += 3; // Powerful boost
1255                    }
1256                    FunctionalStatus::FunctionalBenefic => {
1257                        score += 1;
1258                    }
1259                    FunctionalStatus::FunctionalMalefic => {
1260                        score -= 2; // Significant penalty
1261                        reasons.push(format!("{:?} is Functional Malefic", pl));
1262                    }
1263                    FunctionalStatus::Maraka => {
1264                        score -= 1;
1265                        reasons.push(format!("{:?} is Maraka", pl));
1266                    }
1267                    FunctionalStatus::Neutral => {}
1268                }
1269            }
1270        }
1271
1272        if is_combust {
1273            return YogaQuality::Weak(format!("Combustion: {}", reasons.join(", ")));
1274        }
1275
1276        match score {
1277            s if s >= 3 => YogaQuality::VeryHigh,
1278            s if s >= 1 => YogaQuality::High,
1279            s if s >= -1 => YogaQuality::Medium,
1280            _ => YogaQuality::Weak(if reasons.is_empty() {
1281                "Malefic Influence".to_string()
1282            } else {
1283                reasons.join(", ")
1284            }),
1285        }
1286    }
