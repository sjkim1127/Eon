31    pub fn calculate_rasi(&self, longitude: f64) -> u8 {
32        // Normalize longitude to 0..360 range for robustness across all sub-functions
33        let longitude = (longitude % 360.0 + 360.0) % 360.0;
34        
35        match self {
36            Self::D1 => ((longitude / 30.0).floor() as u8 % 12) + 1,
37            Self::D2 => calculate_hora(longitude),
38            Self::D3 => calculate_drekkana(longitude),
39            Self::D4 => calculate_chaturthamsha(longitude),
40            Self::D5 => calculate_panchamsa(longitude),
41            Self::D6 => calculate_shashtamsa(longitude),
42            Self::D7 => calculate_saptamsa(longitude),
43            Self::D8 => calculate_ashtamsa(longitude),
44            Self::D9 => calculate_navamsa(longitude),
45            Self::D10 => calculate_dasamsa(longitude),
46            Self::D11 => calculate_rudramsa(longitude),
47            Self::D12 => calculate_dwadasamsa(longitude),
48            Self::D16 => calculate_shodashamsa(longitude),
49            Self::D20 => calculate_vimsamsa(longitude),
50            Self::D24 => calculate_chaturvimshamsa(longitude),
51            Self::D27 => calculate_saptavimsamsa(longitude),
52            Self::D30 => calculate_trimsamsa(longitude),
53            Self::D40 => calculate_khavedamsa(longitude),
54            Self::D45 => calculate_akshavedamsa(longitude),
55            Self::D60 => calculate_shashtyamsa(longitude),
56            Self::D81 => {
57                // D9 of D9 (Nava-Navamsa)
58                let sign_degree = longitude % 30.0;
59                let pada_idx = (sign_degree * 9.0 / 30.0).floor();
60                let r1 = calculate_navamsa(longitude);
61                let d_rem = sign_degree - (pada_idx * 30.0 / 9.0);
62                let d_scaled = d_rem * 9.0;
63                calculate_navamsa(((r1 as f64 - 1.0) * 30.0 + d_scaled) % 360.0)
64            },
65            Self::D108 => {
66                // D9 of D12 (Ashtottaramsa)
67                let sign_degree = longitude % 30.0;
68                let div_idx = (sign_degree * 12.0 / 30.0).floor();
69                let r1 = calculate_dwadasamsa(longitude);
70                let d_rem = sign_degree - (div_idx * 30.0 / 12.0);
71                let d_scaled = d_rem * 12.0;
72                calculate_navamsa(((r1 as f64 - 1.0) * 30.0 + d_scaled) % 360.0)
73            },
74            Self::D144 => {
75                // D12 of D12 (Dwadas-Dwadasamsa)
76                let sign_degree = longitude % 30.0;
77                let div_idx = (sign_degree * 12.0 / 30.0).floor();
78                let r1 = calculate_dwadasamsa(longitude);
79                let d_rem = sign_degree - (div_idx * 30.0 / 12.0);
80                let d_scaled = d_rem * 12.0;
81                calculate_dwadasamsa(((r1 as f64 - 1.0) * 30.0 + d_scaled) % 360.0)
82            },
83        }
84    }

86    pub fn division_count(&self) -> u32 {
87        match self {
88            Self::D1 => 1, Self::D2 => 2, Self::D3 => 3, Self::D4 => 4,
89            Self::D5 => 5, Self::D6 => 6, Self::D7 => 7, Self::D8 => 8, Self::D9 => 9,
90            Self::D10 => 10, Self::D11 => 11, Self::D12 => 12, Self::D16 => 16,
91            Self::D20 => 20, Self::D24 => 24, Self::D27 => 27, Self::D30 => 30,
92            Self::D40 => 40, Self::D45 => 45, Self::D60 => 60,
93            Self::D81 => 81, Self::D108 => 108, Self::D144 => 144,
94        }
95    }

97    /// Effective longitude (0..360) for nakshatra calculation in a varga chart.
98    /// D1: use sidereal directly. Other vargas: project degree within division onto full sign.
99    pub fn effective_longitude_for_nakshatra(
100        &self,
101        sidereal_deg: f64,
102        varga_rasi: u8,
103    ) -> f64 {
104        if self.division_count() <= 1 {
105            return (sidereal_deg % 360.0 + 360.0) % 360.0;
106        }
107        let deg = (sidereal_deg % 360.0 + 360.0) % 360.0;
108        let sign_degree = deg % 30.0;
109        let sign_idx = get_sign_idx(deg);
110
111        let (scaled_degree, _div_size) = if matches!(self, Self::D30) {
112            let (start, width) = if is_odd_sign(sign_idx) {
113                if sign_degree < 5.0 { (0.0, 5.0) }
114                else if sign_degree < 10.0 { (5.0, 5.0) }
115                else if sign_degree < 18.0 { (10.0, 8.0) }
116                else if sign_degree < 25.0 { (18.0, 7.0) }
117                else { (25.0, 5.0) }
118            } else {
119                if sign_degree < 5.0 { (0.0, 5.0) }
120                else if sign_degree < 12.0 { (5.0, 7.0) }
121                else if sign_degree < 20.0 { (12.0, 8.0) }
122                else if sign_degree < 25.0 { (20.0, 5.0) }
123                else { (25.0, 5.0) }
124            };
125            let degree_in_division = sign_degree - start;
126            ((degree_in_division / width) * 30.0, width)
127        } else {
128            let division_size = 30.0 / self.division_count() as f64;
129            let degree_in_division = sign_degree % division_size;
130            (degree_in_division * self.division_count() as f64, division_size)
131        };
132
133        ((varga_rasi as f64 - 1.0) * 30.0 + scaled_degree) % 360.0
134    }
135

138/// Helper: Get 0-based sign index (0=Aries, 11=Pisces)
139fn get_sign_idx(longitude: f64) -> u8 {
140    (longitude / 30.0).floor() as u8
141}

143/// Helper: Is odd sign?
144fn is_odd_sign(sign_idx: u8) -> bool {
145    (sign_idx + 1) % 2 != 0
146}

148fn calculate_hora(longitude: f64) -> u8 {
149    let sign_degree = longitude % 30.0;
150    let sign_idx = get_sign_idx(longitude);
151    let is_first_half = sign_degree < 15.0;
152    
153    if is_odd_sign(sign_idx) {
154        if is_first_half { 5 } else { 4 } // Leo / Cancer
155    } else if is_first_half { 4 } else { 5 }
156}

158fn calculate_drekkana(longitude: f64) -> u8 {
159    let sign_degree = longitude % 30.0;
160    let sign_idx = get_sign_idx(longitude);
161    let offset = if sign_degree < 10.0 { 0 } else if sign_degree < 20.0 { 4 } else { 8 };
162    (sign_idx + offset) % 12 + 1
163}
164
165fn calculate_chaturthamsha(longitude: f64) -> u8 {
166    let sign_degree = longitude % 30.0;
167    let sign_idx = get_sign_idx(longitude);
168    let part = (sign_degree / 7.5).floor() as u8;
169    let offset = match part {
170        0 => 0, 1 => 3, 2 => 6, 3 => 9, _ => 0,
171    };
172    (sign_idx + offset) % 12 + 1
173}

175/// D6 (Shashtamsa) - Disease, Enemies
176/// Odd signs: Count from sign itself
177/// Even signs: Count from the 6th sign from itself
178fn calculate_shashtamsa(longitude: f64) -> u8 {
179    let sign_degree = longitude % 30.0;
180    let sign_idx = get_sign_idx(longitude);
181    let part = (sign_degree / 5.0).floor() as u8; // 0..5
182    let start_sign = if is_odd_sign(sign_idx) { sign_idx } else { (sign_idx + 5) % 12 };
183    (start_sign + part) % 12 + 1
184}

186fn calculate_saptamsa(longitude: f64) -> u8 {
187    let sign_degree = longitude % 30.0;
188    let sign_idx = get_sign_idx(longitude);
189    let part = (sign_degree / (30.0 / 7.0)).floor() as u8;
190    let start_sign_idx = if is_odd_sign(sign_idx) { sign_idx } else { (sign_idx + 6) % 12 };
191    (start_sign_idx + part) % 12 + 1
192}

194fn calculate_navamsa(longitude: f64) -> u8 {
195    let d1_rasi_idx = get_sign_idx(longitude);
196    let sign_degree = longitude % 30.0;
197    let pada_idx = (sign_degree / (30.0 / 9.0)).floor() as u8;
198    let start_sign = match d1_rasi_idx % 4 {
199        0 => 0, 1 => 9, 2 => 6, 3 => 3, _ => unreachable!(),
200    };
201    (start_sign + pada_idx) % 12 + 1
202}

208    let start_sign_idx = if is_odd_sign(sign_idx) { sign_idx } else { (sign_idx + 8) % 12 };
209    (start_sign_idx + part) % 12 + 1

228    let sign_degree = longitude % 30.0;
229    let sign_idx = get_sign_idx(longitude);
230    let part = (sign_degree / 3.75).floor() as u8;
231    let offset = match sign_idx % 3 {

237fn calculate_rudramsa(longitude: f64) -> u8 {
238    let sign_degree = longitude % 30.0;
239    let sign_idx = get_sign_idx(longitude);
240    let part = (sign_degree / (30.0 / 11.0)).floor() as i32;
241    let mut target = sign_idx as i32 - part;
242    while target < 0 { target += 12; }
243    (target as u8) % 12 + 1
244}

419    fn test_projection_consistency() {
420        let vargas = [VargaType::D9, VargaType::D12, VargaType::D60];
421        for v in vargas {
422            let div_size = 30.0 / v.division_count() as f64;
423            // Start of a division
424            let eff_start = v.effective_longitude_for_nakshatra(0.01, v.calculate_rasi(0.01));
425            let r_idx = v.calculate_rasi(0.01) as f64 - 1.0;
426            assert!(eff_start >= r_idx * 30.0 && eff_start < r_idx * 30.0 + 5.0);
427
428            // End of a division
429            let end_val = div_size - 0.0001;
430            let eff_end = v.effective_longitude_for_nakshatra(end_val, v.calculate_rasi(end_val));
431            assert!(eff_end > r_idx * 30.0 + 29.9);
432            assert!(eff_end < r_idx * 30.0 + 30.0);
433            
434            // Across boundary
435            let next_val = div_size + 0.0001;
436            let eff_next = v.effective_longitude_for_nakshatra(next_val, v.calculate_rasi(next_val));
437            let next_r_idx = v.calculate_rasi(next_val) as f64 - 1.0;
438            assert!(eff_next >= next_r_idx * 30.0 && eff_next < next_r_idx * 30.0 + 1.0);

506        // Point A (1.0): D9 is Aries. Degree in D9 = 1.0.
