39fn get_nakshatra_lord(nakshatra: u8) -> VedicPlanet {
40    let lords = [
41        VedicPlanet::Ketu,
42        VedicPlanet::Venus,
43        VedicPlanet::Sun,
44        VedicPlanet::Moon,
45        VedicPlanet::Mars,
46        VedicPlanet::Rahu,
47        VedicPlanet::Jupiter,
48        VedicPlanet::Saturn,
49        VedicPlanet::Mercury,
50    ];

54fn get_dagdha_rashis(tithi: u8) -> Vec<u8> {
55    let t = if tithi > 15 { tithi - 15 } else { tithi };
56    match t {
57        1 => vec![1, 7],        // Aries, Libra
58        2 => vec![2, 5],        // Taurus, Leo
59        3 => vec![3, 6],        // Gemini, Virgo
60        4 => vec![2, 11],       // Taurus, Aquarius
61        5 => vec![3, 9],        // Gemini, Sagittarius
62        6 => vec![1, 8],        // Aries, Scorpio
63        7 => vec![3, 12],       // Gemini, Pisces
64        8 => vec![3, 6],        // Gemini, Virgo
65        9 => vec![5, 8],        // Leo, Scorpio
66        10 => vec![5, 9],       // Leo, Sagittarius
67        11 => vec![9, 12],      // Sagittarius, Pisces
68        12 => vec![7, 10],      // Libra, Capricorn
69        13 => vec![2, 3],       // Taurus, Gemini
70        14 => vec![2, 3, 6, 9], // Taurus, Gemini, Virgo, Sagittarius
71        _ => vec![],
72    }

78    pub fn calculate(
79        sun_deg: f64,
80        moon_deg: f64,
81        time: DateTime<Utc>,
82        latitude: f64,
83        longitude: f64,
84    ) -> Panchanga {
85        let sun = sun_deg;
86        let moon = moon_deg;
87
88        // 1. Fetch sunrise/sunset for the current LOCAL date
89        let local_offset_mins = (longitude * 4.0) as i64;
90        let local_time = time + chrono::Duration::minutes(local_offset_mins);
91
92        let (mut sunrise, mut sunset) =
93            Self::calculate_sunrise_sunset(local_time, latitude, longitude);
94
95        // If current time is before the sunrise of its local date,
96        // the astrological day started at yesterday's sunrise
97        if time < sunrise {
98            let (prev_rise, prev_set) = Self::calculate_sunrise_sunset(
99                local_time - chrono::Duration::days(1),
100                latitude,
101                longitude,
102            );
103            sunrise = prev_rise;
104            sunset = prev_set;
105        }
106
107        let (next_sunrise, _) = Self::calculate_sunrise_sunset(
108            sunrise + chrono::Duration::days(1),
109            latitude,
110            longitude,
111        );
112
113        // 2. Vara (Weekday) - Vedic Day starts at Sunrise
114        let is_day_birth = time >= sunrise && time < sunset;
115        let is_night_birth = !is_day_birth;
116
117        // Use the actual sunrise time to determine the local weekday
118        let local_offset_mins = (longitude * 4.0) as i64;
119        let sunrise_local = sunrise + chrono::Duration::minutes(local_offset_mins);
120        let vara_date = sunrise_local.date_naive();
121        let vara = match vara_date.weekday() {
122            chrono::Weekday::Sun => "Sunday",
123            chrono::Weekday::Mon => "Monday",
124            chrono::Weekday::Tue => "Tuesday",
125            chrono::Weekday::Wed => "Wednesday",
126            chrono::Weekday::Thu => "Thursday",
127            chrono::Weekday::Fri => "Friday",
128            chrono::Weekday::Sat => "Saturday",
129        }
130        .to_string();
131
132        let day_lord = match vara_date.weekday() {
133            chrono::Weekday::Sun => VedicPlanet::Sun,
134            chrono::Weekday::Mon => VedicPlanet::Moon,
135            chrono::Weekday::Tue => VedicPlanet::Mars,
136            chrono::Weekday::Wed => VedicPlanet::Mercury,
137            chrono::Weekday::Thu => VedicPlanet::Jupiter,
138            chrono::Weekday::Fri => VedicPlanet::Venus,
139            chrono::Weekday::Sat => VedicPlanet::Saturn,
140        };
141
142        // 3. Hourly Lord (Hora)
143        let hour_lord = Self::calculate_hora(time, sunrise, sunset, next_sunrise, day_lord);
144        let daily_parts = Self::calculate_tribhaga_lords(day_lord); // Simplified for Tribhaga
145
146        // 4. Tithi (Moon - Sun) / 12
147        let tithi_deg = (moon - sun + 360.0) % 360.0;
148        let tithi = (tithi_deg / 12.0).floor() as u8 + 1;
149        let tithi_name = Self::get_tithi_name(tithi);
150
151        // 5. Nakshatra (Moon / 13.333)
152        let nakshatra = (moon / (360.0 / 27.0)).floor() as u8 + 1;
153
154        // 6. Yoga (Sun + Moon) / 13.333
155        let yoga_deg = (sun + moon) % 360.0;
156        let yoga = (yoga_deg / (360.0 / 27.0)).floor() as u8 + 1;
157
158        // 7. Karana (Tithi_deg / 6)
159        // Karana is half of a Tithi.
160        let karana_idx = (tithi_deg / 6.0).floor() as u16 + 1;
161        let karana_name = Self::get_karana_name(karana_idx);
162
163        // Yogi, Avayogi & Dagdha Rashi
164        let yogi_point = (sun + moon + 93.3333333) % 360.0;
165        let yogi_nak = (yogi_point / (360.0 / 27.0)).floor() as u8 + 1;
166        let yogi_planet = get_nakshatra_lord(yogi_nak);
167
168        let avayogi_point = (yogi_point + 186.6666667) % 360.0;
169        let avayogi_nak = (avayogi_point / (360.0 / 27.0)).floor() as u8 + 1;
170        let avayogi_planet = get_nakshatra_lord(avayogi_nak);
171
172        let dagdha_rashis = get_dagdha_rashis(tithi);
173
174        // Rahu Kalam, Yamaganda, Gulika calculations (daytime divided into 8 parts)
175        let rahu_part = match vara_date.weekday() {
176            chrono::Weekday::Sun => 8,
177            chrono::Weekday::Mon => 2,
178            chrono::Weekday::Tue => 7,
179            chrono::Weekday::Wed => 5,
180            chrono::Weekday::Thu => 6,
181            chrono::Weekday::Fri => 4,
182            chrono::Weekday::Sat => 3,
183        };
184
185        let yama_part = match vara_date.weekday() {
186            chrono::Weekday::Sun => 5,
187            chrono::Weekday::Mon => 4,
188            chrono::Weekday::Tue => 3,
189            chrono::Weekday::Wed => 2,
190            chrono::Weekday::Thu => 1,
191            chrono::Weekday::Fri => 7,
192            chrono::Weekday::Sat => 6,
193        };
194
195        let guli_part = match vara_date.weekday() {
196            chrono::Weekday::Sun => 7,
197            chrono::Weekday::Mon => 6,
198            chrono::Weekday::Tue => 5,
199            chrono::Weekday::Wed => 4,
200            chrono::Weekday::Thu => 3,
201            chrono::Weekday::Fri => 2,
202            chrono::Weekday::Sat => 1,
203        };
204
205        let day_duration_ms = sunset.signed_duration_since(sunrise).num_milliseconds() as f64;
206        let part_ms = day_duration_ms / 8.0;
207
208        let get_time_range = |part_idx: u8| -> (DateTime<Utc>, DateTime<Utc>) {
209            let start_offset_ms = (part_idx - 1) as f64 * part_ms;
210            let end_offset_ms = part_idx as f64 * part_ms;
211
212            let start_time = sunrise + chrono::Duration::milliseconds(start_offset_ms as i64);
213            let end_time = sunrise + chrono::Duration::milliseconds(end_offset_ms as i64);
214            (start_time, end_time)
215        };
216
217        let rahu_kalam = get_time_range(rahu_part);
218        let yamaganda = get_time_range(yama_part);
219        let gulika = get_time_range(guli_part);
220
221        Panchanga {
222            vara,
223            tithi,
224            tithi_name,
225            nakshatra,
226            yoga,
227            karana: Self::get_karana_type(karana_idx),
228            karana_name,
229            current_time: time,
230            sunrise,
231            sunset,
232            next_sunrise,
233            is_day_birth,
234            is_night_birth,
235            day_lord,
236            hour_lord,
237            daily_parts,
238            yogi_point,
239            yogi_planet,
240            avayogi_planet,
241            dagdha_rashis,
242            rahu_kalam,
243            yamaganda,
244            gulika,
245        }
246    }

248    /// Calculate Hora Lord
249    fn calculate_hora(
250        time: DateTime<Utc>,
251        sunrise: DateTime<Utc>,
252        sunset: DateTime<Utc>,
253        next_sunrise: DateTime<Utc>,
254        day_lord: VedicPlanet,
255    ) -> VedicPlanet {
256        // Hora order: Sun, Venus, Mercury, Moon, Saturn, Jupiter, Mars sequence repeats
257        // Standard Sequence starting from Day Lord is:
258        // 1st Hora = Day Lord
259        // 2nd Hora = 6th weekday from current...
260
261        // Correct Sequence for Hora starting from Sunday: Sun, Venus, Mercury, Moon, Saturn, Jupiter, Mars.
262        let hora_seq = [
263            VedicPlanet::Sun,
264            VedicPlanet::Venus,
265            VedicPlanet::Mercury,
266            VedicPlanet::Moon,
267            VedicPlanet::Saturn,
268            VedicPlanet::Jupiter,
269            VedicPlanet::Mars,
270        ];
271
272        // Find index of day lord in the sequence
273        let start_idx = hora_seq.iter().position(|&p| p == day_lord).unwrap_or(0);
274
275        let is_day = time >= sunrise && time < sunset;
276
277        let (duration, start_time) = if is_day {
278            let diff = sunset.signed_duration_since(sunrise).num_seconds() as f64;
279            (diff / 12.0, sunrise)
280        } else {
281            let diff = next_sunrise.signed_duration_since(sunset).num_seconds() as f64;
282            (diff / 12.0, sunset)
283        };
284
285        if duration == 0.0 {
286            return day_lord;
287        } // Safety
288
289        // Calculate offset from start_time
290        let elapsed = if time >= start_time {
291            time.signed_duration_since(start_time).num_seconds() as f64
292        } else {
293            0.0
294        };
295
296        let hora_idx = (elapsed / duration).floor() as usize;
297        let offset = if is_day { hora_idx } else { hora_idx + 12 };
298
299        let final_idx = (start_idx + offset) % 7;
300        hora_seq[final_idx]
301    }

303    /// Calculate Tribhaga Lords (BPHS Standard)
304    /// Day is divided into 3 parts: Mercury (1st), Sun (2nd), Saturn (3rd)
305    /// Night is divided into 3 parts: Moon (1st), Venus (2nd), Mars (3rd)
306    /// Returns array of 8 lords (3 day + 3 night + 2 padding for Kala Bala compatibility)
307    fn calculate_tribhaga_lords(day_lord: VedicPlanet) -> [VedicPlanet; 8] {
308        // BPHS Standard Tribhaga Lords:
309        // Day Parts (from sunrise to sunset): Mercury, Sun, Saturn
310        // Night Parts (from sunset to next sunrise): Moon, Venus, Mars
311
312        // Return standard pattern (first 3 for day, next 3 for night)
313        [
314            // Day lords (used when is_day_birth == true)
315            VedicPlanet::Mercury, // 1st third of day
316            VedicPlanet::Sun,     // 2nd third of day
317            VedicPlanet::Saturn,  // 3rd third of day
318            // Night lords (used when is_day_birth == false)
319            VedicPlanet::Moon,  // 1st third of night
320            VedicPlanet::Venus, // 2nd third of night
321            VedicPlanet::Mars,  // 3rd third of night
322            // Padding for compatibility
323            day_lord,
324            day_lord,
325        ]
326    }

328    /// NOAA Sunrise/Sunset Algorithm (Simplified)
329    fn calculate_sunrise_sunset(
330        date: DateTime<Utc>,
331        lat: f64,
332        lon: f64,
333    ) -> (DateTime<Utc>, DateTime<Utc>) {
334        // Convert date to Julian Day
335        let timestamp = date.timestamp();
336        let julian_day = (timestamp as f64 / 86400.0) + 2440587.5;
337        let julian_century = (julian_day - 2451545.0) / 36525.0;
338
339        // Geom Mean Long Sun (deg)
340        let geom_mean_long_sun =
341            (280.46646 + julian_century * (36000.76983 + julian_century * 0.0003032)) % 360.0;
342
343        // Geom Mean Anom Sun (deg)
344        let geom_mean_anom_sun =
345            357.52911 + julian_century * (35999.05029 - 0.0001537 * julian_century);
346
347        // Eccent Earth Orbit
348        let eccent_earth_orbit =
349            0.016708634 - julian_century * (0.000042037 + 0.0000001267 * julian_century);
350
351        // Sun Eq of Ctr
352        let sun_eq_of_ctr = geom_mean_anom_sun.to_radians().sin()
353            * (1.914602 - julian_century * (0.004817 + 0.000014 * julian_century))
354            + (2.0 * geom_mean_anom_sun.to_radians()).sin()
355                * (0.019993 - 0.000101 * julian_century)
356            + (3.0 * geom_mean_anom_sun.to_radians()).sin() * 0.000289;
357
358        let sun_true_long = geom_mean_long_sun + sun_eq_of_ctr;
359
360        // Sun App Long
361        let sun_app_long = sun_true_long
362            - 0.00569
363            - 0.00478 * (125.04 - 1934.136 * julian_century).to_radians().sin();
364
365        // Mean Obliq Ecliptic
366        let mean_obliq_ecliptic = 23.0
367            + (26.0
368                + (21.448
369                    - julian_century
370                        * (46.815 + julian_century * (0.00059 - julian_century * 0.001813)))
371                    / 60.0)
372                / 60.0;
373        let obliq_corr =
374            mean_obliq_ecliptic + 0.00256 * (125.04 - 1934.136 * julian_century).to_radians().cos();
375
376        // Sun Declin
377        let sun_declin = (obliq_corr.to_radians().sin() * sun_app_long.to_radians().sin()).asin();
378
379        // Var y
380        let var_y = (obliq_corr.to_radians() / 2.0).tan().powi(2);
381
382        // Eq of Time (minutes)
383        let eq_of_time = 4.0
384            * (var_y * (2.0 * geom_mean_long_sun.to_radians()).sin()
385                - 2.0 * eccent_earth_orbit * geom_mean_anom_sun.to_radians().sin()
386                + 4.0
387                    * eccent_earth_orbit
388                    * var_y
389                    * geom_mean_anom_sun.to_radians().sin()
390                    * (2.0 * geom_mean_long_sun.to_radians()).cos()
391                - 0.5 * var_y * var_y * (4.0 * geom_mean_long_sun.to_radians()).sin()
392                - 1.25
393                    * eccent_earth_orbit
394                    * eccent_earth_orbit
395                    * (2.0 * geom_mean_anom_sun.to_radians()).sin())
396            .to_degrees();
397
398        // HA Sunrise (deg)
399        // cos(HA) = cos(90.833) / (cos(lat)*cos(declin)) - tan(lat)*tan(declin)
400        let zenith = 90.833f64.to_radians(); // Official sunrise zenith
401        let cos_ha = (zenith.cos() / (lat.to_radians().cos() * sun_declin.cos()))
402            - (lat.to_radians().tan() * sun_declin.tan());
403
404        // Check for polar day/night (simplification: clamp)
405        let ha_deg = if cos_ha > 1.0 {
406            0.0
407        } else if cos_ha < -1.0 {
408            180.0
409        } else {
410            cos_ha.acos().to_degrees()
411        };
412
413        let solar_noon = 720.0 - 4.0 * lon - eq_of_time;
414
415        let sunrise_min = solar_noon - 4.0 * ha_deg;
416        let sunset_min = solar_noon + 4.0 * ha_deg;
417
418        // Use UTC midnight of the given date reference
419        let midnight = date.date_naive().and_hms_opt(0, 0, 0).unwrap().and_utc();
420
421        let rise_secs = (sunrise_min * 60.0) as i64;
422        let set_secs = (sunset_min * 60.0) as i64;
423
424        (
425            midnight + chrono::Duration::seconds(rise_secs),
426            midnight + chrono::Duration::seconds(set_secs),
427        )
428    }

469    /// Karana type mapping (BPHS rules)
470    /// Sequential index 1-60 → Type 1-11
471    /// 1: Kimstughna (fixed, only first karana)
472    /// 2-8: Bava, Balava, Kaulava, Taitila, Gara, Vanija, Vishti (rotating, 8 cycles)
473    /// 9-11: Shakuni, Chatushpada, Naga (fixed, last 3 karanas)
474    fn get_karana_type(karana_idx: u16) -> u8 {
475        if karana_idx == 1 {
476            1 // Kimstughna (fixed)
477        } else if karana_idx >= 58 {
478            // Last 3: Shakuni(9), Chatushpada(10), Naga(11)
479            (karana_idx - 58 + 9) as u8
480        } else {
481            // Rotating 7 movable karanas: Bava(2)..Vishti(8)
482            (((karana_idx - 2) % 7) + 2) as u8
483        }
484    }
485
486    fn get_karana_name(idx: u16) -> String {
487        if idx == 1 {
488            return "Kimstughna".to_string();
489        }
490        if idx >= 58 {
491            match idx {
492                58 => return "Shakuni".to_string(),
493                59 => return "Chatushpada".to_string(),
494                60 => return "Naga".to_string(),
495                _ => {}
496            }
497        }
498        let movables = [
499            "Bava", "Balava", "Kaulava", "Taitila", "Gara", "Vanija", "Vishti",
500        ];
501        let m_idx = ((idx.saturating_sub(2)) % 7) as usize;
502        movables[m_idx].to_string()
503    }
