1use crate::core::config::AyanamsaSystem;
2use chrono::{DateTime, Utc};
3use eon_astro::AstroEngine;
4
5/// Calculate Ayanamsa (Precession correction) using Swiss Ephemeris
6pub fn get_ayanamsa(astro: &AstroEngine, time: DateTime<Utc>, method: AyanamsaSystem) -> f64 {
7    // 1. Set Sidereal Mode based on config
8    let method_id = match method {
9        AyanamsaSystem::Lahiri => 1,       // SE_SIDM_LAHIRI
10        AyanamsaSystem::FaganBradley => 0, // SE_SIDM_FAGAN_BRADLEY
11        AyanamsaSystem::Raman => 3,        // SE_SIDM_RAMAN
12        AyanamsaSystem::Krishnamurti => 5, // SE_SIDM_KRISHNAMURTI
13    };
14
15    // t0=0, ayan_t0=0 for standard methods
16    astro.set_sidereal_mode(method_id, 0.0, 0.0);
17
18    // 2. Calculate
19    astro.get_ayanamsa_ut(time)
20}
21
22/// Helper for default Lahiri Ayanamsa
23pub fn get_lahiri_ayanamsa(astro: &AstroEngine, time: DateTime<Utc>) -> f64 {
24    get_ayanamsa(astro, time, AyanamsaSystem::Lahiri)
25}
