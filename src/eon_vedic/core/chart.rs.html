120    fn calculate_nakshatra_and_pada(sidereal: f64) -> (u8, u8) {
121        let nak_pos = sidereal / (360.0 / 27.0);
122        let nakshatra = (nak_pos.floor() as u8) + 1;
123
124        let pada_pos = (sidereal % (360.0 / 27.0)) / (360.0 / 108.0);
125        let pada = (pada_pos.floor() as u8) + 1;
126
127        (nakshatra, pada)
128    }

144    pub fn calculate(
145        &self,
146        time: DateTime<Utc>,
147        latitude: f64,
148        longitude: f64,
149    ) -> Result<VedicChart, VedicError> {
150        let ayanamsa =
151            crate::calc::ayanamsa::get_ayanamsa(&self.engine, time, self.config.ayanamsa);
152
153        let hsys = match self.config.house_system {
154            crate::core::config::HouseSystem::WholeSign => b'W' as i32,
155            crate::core::config::HouseSystem::Sripati => b'S' as i32,
156        };
157
158        let (cusps, ascmc) = self.engine.get_houses(time, latitude, longitude, hsys)?;
159
160        let sidereal_cusps: Vec<f64> = cusps
161            .iter()
162            .map(|c| (c - ayanamsa + 360.0) % 360.0)
163            .collect();
164        let asc_sidereal = (ascmc[0] - ayanamsa + 360.0) % 360.0;
165
166        let mut sandhis = Vec::new();
167        if self.config.house_system == crate::core::config::HouseSystem::Sripati
168            && !sidereal_cusps.is_empty()
169        {
170            for i in 0..12 {
171                let c1 = sidereal_cusps[i];
172                let c2 = sidereal_cusps[(i + 1) % 12];
173                let mut diff = c2 - c1;
174                if diff < 0.0 {
175                    diff += 360.0;
176                }
177                sandhis.push((c1 + diff / 2.0) % 360.0);
178            }
179        }
180
181        let asc_rasi = (asc_sidereal / 30.0).floor() as u8 + 1;
182
183        let create_position = |planet: VedicPlanet,
184                               sidereal: f64,
185                               tropical: f64,
186                               speed: f64,
187                               declination: f64,
188                               sun_sidereal: Option<f64>|
189         -> VedicPosition {
190            let (nakshatra, pada) = Self::calculate_nakshatra_and_pada(sidereal);
191            let rasi = (sidereal / 30.0).floor() as u8 + 1;
192
193            let house_index = match self.config.house_system {
194                crate::core::config::HouseSystem::WholeSign => {
195                    ((rasi as i32 - asc_rasi as i32 + 12) % 12) as u8 + 1
196                }
197                crate::core::config::HouseSystem::Sripati => {
198                    let mut h = 0;
199                    for i in 0..12 {
200                        let s_start = sandhis[(i + 11) % 12];
201                        let s_end = sandhis[i];
202
203                        let in_house = if s_start < s_end {
204                            sidereal >= s_start && sidereal < s_end
205                        } else {
206                            sidereal >= s_start || sidereal < s_end
207                        };
208
209                        if in_house {
210                            h = i + 1;
211                            break;
212                        }
213                    }
214                    h as u8
215                }
216            };
217
218            VedicPosition {
219                planet,
220                tropical_deg: tropical,
221                sidereal_deg: sidereal,
222                nakshatra,
223                pada,
224                rasi,
225                house_index,
226                speed,
227                is_retrograde: speed < 0.0,
228                is_combust: if let Some(sun_long) = sun_sidereal {
229                    if planet == VedicPlanet::Sun {
230                        false
231                    } else {
232                        let diff = (sidereal - sun_long).abs();
233                        let d = if diff > 180.0 { 360.0 - diff } else { diff };
234                        let limit = match planet {
235                            VedicPlanet::Moon => 12.0,
236                            VedicPlanet::Mars => 17.0,
237                            VedicPlanet::Mercury => {
238                                if speed < 0.0 {
239                                    12.0
240                                } else {
241                                    14.0
242                                }
243                            }
244                            VedicPlanet::Jupiter => 11.0,
245                            VedicPlanet::Venus => 10.0,
246                            VedicPlanet::Saturn => 15.0,
247                            _ => 0.0,
248                        };
249                        d < limit
250                    }
251                } else {
252                    false
253                },
254                declination,
255                hora_rasi: crate::calc::varga::VargaType::D2.calculate_rasi(sidereal),
256                drekkana_rasi: crate::calc::varga::VargaType::D3.calculate_rasi(sidereal),
257                chaturthamsha_rasi: crate::calc::varga::VargaType::D4.calculate_rasi(sidereal),
258                panchamsa_rasi: crate::calc::varga::VargaType::D5.calculate_rasi(sidereal),
259                saptamsa_rasi: crate::calc::varga::VargaType::D7.calculate_rasi(sidereal),
260                ashtamsa_rasi: crate::calc::varga::VargaType::D8.calculate_rasi(sidereal),
261                navamsa_rasi: crate::calc::varga::VargaType::D9.calculate_rasi(sidereal),
262                dasamsa_rasi: crate::calc::varga::VargaType::D10.calculate_rasi(sidereal),
263                shashtamsa_rasi: crate::calc::varga::VargaType::D6.calculate_rasi(sidereal),
264                rudramsa_rasi: crate::calc::varga::VargaType::D11.calculate_rasi(sidereal),
265                dwadasamsa_rasi: crate::calc::varga::VargaType::D12.calculate_rasi(sidereal),
266                shodashamsa_rasi: crate::calc::varga::VargaType::D16.calculate_rasi(sidereal),
267                vimsamsa_rasi: crate::calc::varga::VargaType::D20.calculate_rasi(sidereal),
268                chaturvimshamsa_rasi: crate::calc::varga::VargaType::D24.calculate_rasi(sidereal),
269                saptavimsamsa_rasi: crate::calc::varga::VargaType::D27.calculate_rasi(sidereal),
270                trimsamsa_rasi: crate::calc::varga::VargaType::D30.calculate_rasi(sidereal),
271                khavedamsa_rasi: crate::calc::varga::VargaType::D40.calculate_rasi(sidereal),
272                akshavedamsa_rasi: crate::calc::varga::VargaType::D45.calculate_rasi(sidereal),
273                shashtyamsa_rasi: crate::calc::varga::VargaType::D60.calculate_rasi(sidereal),
274                navanavamsa_rasi: crate::calc::varga::VargaType::D81.calculate_rasi(sidereal),
275                ashtottaramsa_rasi: crate::calc::varga::VargaType::D108.calculate_rasi(sidereal),
276                dwadasdwadasamsa_rasi: crate::calc::varga::VargaType::D144.calculate_rasi(sidereal),
277            }
278        };
279
280        let (sun_trop, _) = self
281            .engine
282            .get_planet_full(time, VedicPlanet::Sun.se_id(), 256 | 2)?;
283        let (_, _sun_dec) = self
284            .engine
285            .get_planet_equatorial(time, VedicPlanet::Sun.se_id())?;
286        let sun_sidereal = (sun_trop - ayanamsa + 360.0) % 360.0;
287
288        let asc_position = create_position(
289            VedicPlanet::Ascendant,
290            asc_sidereal,
291            ascmc[0],
292            0.0,
293            0.0,
294            Some(sun_sidereal),
295        );
296
297        let planets_names = [
298            VedicPlanet::Sun,
299            VedicPlanet::Moon,
300            VedicPlanet::Mars,
301            VedicPlanet::Mercury,
302            VedicPlanet::Jupiter,
303            VedicPlanet::Venus,
304            VedicPlanet::Saturn,
305            VedicPlanet::Rahu,
306        ];
307
308        let mut planets = Vec::new();
309        for p in &planets_names {
310            let se_id = if *p == VedicPlanet::Rahu {
311                match self.config.node_calc {
312                    NodeCalculation::MeanNode => 10,
313                    NodeCalculation::TrueNode => 11,
314                }
315            } else {
316                p.se_id()
317            };
318            let flag = 256 | 2;
319            let (trop, speed) = self.engine.get_planet_full(time, se_id, flag)?;
320            let (_, dec) = self.engine.get_planet_equatorial(time, se_id)?;
321            let sidereal = (trop - ayanamsa + 360.0) % 360.0;
322            planets.push(create_position(
323                *p,
324                sidereal,
325                trop,
326                speed,
327                dec,
328                Some(sun_sidereal),
329            ));
330
331            if *p == VedicPlanet::Rahu {
332                let ketu_sidereal = (sidereal + 180.0) % 360.0;
333                let ketu_tropical = (trop + 180.0) % 360.0;
334                let ketu_dec = -dec;
335                planets.push(create_position(
336                    VedicPlanet::Ketu,
337                    ketu_sidereal,
338                    ketu_tropical,
339                    speed,
340                    ketu_dec,
341                    Some(sun_sidereal),
342                ));
343            }
344        }
345
346        let sun_deg = planets
347            .iter()
348            .find(|p| p.planet == VedicPlanet::Sun)
349            .map(|p| p.sidereal_deg)
350            .unwrap_or(0.0);
351        let moon_deg = planets
352            .iter()
353            .find(|p| p.planet == VedicPlanet::Moon)
354            .map(|p| p.sidereal_deg)
355            .unwrap_or(0.0);
356
357        let panchanga = crate::panchanga::PanchangaEngine::calculate(
358            sun_deg, moon_deg, time, latitude, longitude,
359        );
360
361        let mut chart = VedicChart {
362            ascendant: asc_position,
363            planets,
364            aspects: Vec::new(),
365            sav: crate::analysis::ashtakavarga::Sarvashtakavarga { points: [0; 12] },
366            bav: Vec::new(),
367            house_cusps: sidereal_cusps,
368            karakas: Vec::new(),
369            arudha_padas: Vec::new(),
370            special_lagnas: Vec::new(),
371            bhava_strengths: Vec::new(),
372            vimshopaka_scores: Vec::new(),
373            avasthas: Vec::new(),
374            panchanga,
375            analysis_report: None,
376            ayanamsa,
377            shadbalas: Vec::new(),
378        };
379
380        chart.aspects = crate::analysis::aspects::AspectEngine::calculate_aspects(&chart);
381        chart.sav = crate::analysis::ashtakavarga::AshtakavargaEngine::calculate_sav(&chart);
382
383        let bav_planets = [
384            VedicPlanet::Sun,
385            VedicPlanet::Moon,
386            VedicPlanet::Mars,
387            VedicPlanet::Mercury,
388            VedicPlanet::Jupiter,
389            VedicPlanet::Venus,
390            VedicPlanet::Saturn,
391        ];
392        chart.bav = bav_planets
393            .iter()
394            .map(|&p| crate::analysis::ashtakavarga::AshtakavargaEngine::calculate_bav(p, &chart))
395            .collect();
396
397        chart.karakas = crate::analysis::jaimini::JaiminiEngine::calculate_karakas(&chart, true);
398        chart.arudha_padas =
399            crate::analysis::jaimini::JaiminiEngine::calculate_arudha_padas(&chart);
400        chart.special_lagnas =
401            crate::analysis::jaimini::JaiminiEngine::calculate_special_lagnas(&chart);
402        chart.bhava_strengths = crate::analysis::bhava::BhavaEngine::calculate_all(&chart);
403
404        chart.avasthas = chart
405            .planets
406            .iter()
407            .map(|p| crate::analysis::avasthas::AvasthaEngine::calculate(p, &chart))
408            .collect();
409
410        let mut v_scores = Vec::new();
411        for p in &chart.planets {
412            v_scores.push((
413                p.planet,
414                crate::analysis::vimshopaka::VimshopakaEngine::calculate(p, &chart),
415            ));
416        }
417        chart.vimshopaka_scores = v_scores;
418
419        let mut shadbalas = Vec::new();
420        for p in &chart.planets {
421            shadbalas.push(crate::analysis::strength::StrengthEngine::calculate(
422                p, &chart,
423            ));
424        }
425        chart.shadbalas = shadbalas;
426
427        chart.analysis_report = Some(crate::analysis::report::VedicAnalysisReport::generate(
428            &chart,
429            time,
430            chart.ascendant.rasi,
431        ));
432
433        Ok(chart)
434    }

436    /// Calculate Solar Return (Varshaphala) chart for a specific year
437    pub fn calculate_solar_return(
438        &self,
439        birth_time: DateTime<Utc>,
440        latitude: f64,
441        longitude: f64,
442        target_year: i32,
443    ) -> Result<VedicChart, VedicError> {
444        // 1. Get birth Sun sidereal longitude
445        let birth_chart = self.calculate(birth_time, latitude, longitude)?;
446        let birth_sun_sidereal = birth_chart
447            .planets
448            .iter()
449            .find(|p| p.planet == VedicPlanet::Sun)
450            .map(|p| p.sidereal_deg)
451            .unwrap_or(0.0);
452
453        // 2. Find time when Sun returns to this sidereal longitude in the target year
454        // Handle Feb 29 leap year birthday for non-leap years
455        let (m, d) = if birth_time.month() == 2 && birth_time.day() == 29 {
456            let is_leap =
457                (target_year % 4 == 0 && target_year % 100 != 0) || (target_year % 400 == 0);
458            if is_leap {
459                (2, 29)
460            } else {
461                (3, 1)
462            }
463        } else {
464            (birth_time.month(), birth_time.day())
465        };
466
467        let approx_time = Utc
468            .with_ymd_and_hms(target_year, m, d, birth_time.hour(), birth_time.minute(), 0)
469            .single()
470            .ok_or_else(|| VedicError::CalculationError("Invalid approx time".to_string()))?;
471
472        let mut current_guess = approx_time;
473        for _ in 0..3 {
474            let ayanamsa = crate::calc::ayanamsa::get_ayanamsa(
475                &self.engine,
476                current_guess,
477                self.config.ayanamsa,
478            );
479            let target_tropical = (birth_sun_sidereal + ayanamsa) % 360.0;
480            if let Ok(exact_time) = self
481                .engine
482                .find_time_for_longitude(current_guess, target_tropical)
483            {
484                current_guess = exact_time;
485            }
486        }
487
488        // 3. Calculate full chart for the exact return time
489        let mut annual_chart = self.calculate(current_guess, latitude, longitude)?;
490
491        // Overwrite report with correct birth lagna context and correct return time
492        annual_chart.analysis_report =
493            Some(crate::analysis::report::VedicAnalysisReport::generate(
494                &annual_chart,
495                current_guess,
496                birth_chart.ascendant.rasi,
497            ));
498
499        Ok(annual_chart)
500    }

508    fn nakshatra_and_pada_start_boundary_is_correct() {
509        let (nak, pada) = VedicChartCalculator::calculate_nakshatra_and_pada(0.0);

515    fn nakshatra_and_pada_transition_boundary_is_correct() {
516        let segment = 360.0 / 27.0;
517
518        let (nak_before, pada_before) =
519            VedicChartCalculator::calculate_nakshatra_and_pada(segment - 1e-9);
520        assert_eq!(nak_before, 1);
521        assert_eq!(pada_before, 4);
522
523        let (nak_after, pada_after) = VedicChartCalculator::calculate_nakshatra_and_pada(segment);
524        assert_eq!(nak_after, 2);
525        assert_eq!(pada_after, 1);
