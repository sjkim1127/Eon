26    /// 모든 CS 엔진의 출력을 수집하여 LLM용 컨텍스트 데이터 생성
27    pub fn generate_context(
28        pillars: &eon_saju::FourPillars,
29        life_report: &LifePathReport,
30    ) -> String {
31        let entropy = DestinyEntropy::analyze(pillars);
32        let topology = QiTopology::analyze(pillars);
33        let load_diagnostics = KarmaLoadBalancer::diagnose(&life_report.frames);
34        let load_status = load_diagnostics
35            .first()
36            .map(|d| d.status.clone())
37            .unwrap_or(eon_saju::TrafficStatus::Normal);
38        let complexity = DestinyComplexity::analyze(&life_report.frames);
39
40        let vm = eon_saju::SajuVM::new(pillars.clone());
41        let fuzzer = DestinyFuzzer::new(vm);
42
43        // 현재 원국의 월주를 기준으로 기본 정적 감사 수행 (대운 컨텍스트 없이 세운만 60개 퍼징)
44        let fuzz_report = fuzzer.audit(pillars.month);
45
46        // TTD Backtrace (예시로 가장 낮은 지점 분석)
47        let valley_age = life_report.valley_age;
48        let root_cause = DestinyDebugger::backtrace(life_report, valley_age, "기신");
49
50        let mut context = String::new();
51        context.push_str("=== [DESTINY SYSTEM CORE DUMP] ===\n");
52        context.push_str(&format!("* Entropy/DIE Level: {:?}\n", entropy.level));
53        context.push_str(&format!(
54            "* Topology Trace: {:?}\n",
55            topology
56                .nodes
57                .iter()
58                .map(|n| (n.element.hangul(), n.output))
59                .collect::<Vec<_>>()
60        ));
61        context.push_str(&format!("* Current Load Status: {:?}\n", load_status));
62        context.push_str(&format!(
63            "* Cyclomatic Complexity (M): {}\n",
64            complexity.cyclomatic_complexity
65        ));
66        context.push_str(&format!(
67            "* Stability Grade: {}\n",
68            complexity.stability_grade
69        ));
70        context.push_str(&format!(
71            "* Fuzzing Audit: Found {} vulnerabilities in static scan\n",
72            fuzz_report.total_crashes
73        ));
74        if let Some(rc) = root_cause {
75            context.push_str(&format!("* TTD Backtrace Log: {}\n", rc.reason));
76        }
77
78        context
79    }

81    /// LLM에게 전달할 최종 프롬프트 생성 (Tool-Calling / Agentic Style)
82    pub fn build_agent_prompt(pillars: &eon_saju::FourPillars) -> Result<String, AiError> {
83        let manifest = EonToolbox::get_manifest();
84        let manifest_json = serde_json::to_string_pretty(&manifest)?;
85
86        let mut prompt = String::new();
87        prompt.push_str("당신은 'Eon Destiny Security Agency'의 수석 에이전트 분석관입니다.\n");
88        prompt.push_str("분석 대상 시스템의 기본 정보는 다음과 같습니다:\n");
89        prompt.push_str(&format!("* Target System GanZi: {}\n\n", pillars.hangul()));
90
91        prompt
92            .push_str("당신은 다음 도구들을 호출하여 시스템의 상세 데이터를 조사할 수 있습니다:\n");
93        prompt.push_str(&format!("{}\n\n", manifest_json));
94
95        prompt.push_str("[지침]\n");
96        prompt.push_str("1. 먼저 필요한 도구를 호출하여 시스템 상태를 파악하십시오.\n");
97        prompt.push_str("2. 수집된 데이터를 바탕으로 '시스템 보안 감사 보고서(Security Audit Report)'를 작성하십시오.\n");
98        prompt.push_str("3. 보고서에는 취약점, 병목 구간, 유지보수 복잡도, 해결책(Patch)을 기술 전문 용어로 포함하십시오.\n\n");
99
100        prompt.push_str("분석을 시작하기 위해 첫 번째 도구 호출을 수행하거나 계획을 세우십시오.");
101
102        Ok(prompt)
103    }
