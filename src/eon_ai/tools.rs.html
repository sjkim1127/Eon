55    /// 도구 호출 실행 (Dispatch)
56    pub fn call(
57        pillars: &FourPillars,
58        tool_name: &str,
59        params: Value,
60    ) -> Result<Value, crate::error::AiError> {
61        match tool_name {
62            "analyze_entropy" => {
63                let res = DestinyEntropy::analyze(pillars);
64                serde_json::to_value(res).map_err(|e| e.into())
65            }
66            "scan_topology" => {
67                let res = QiTopology::analyze(pillars);
68                serde_json::to_value(res).map_err(|e| e.into())
69            }
70            "fuzz_luck_vulnerabilities" => {
71                let idx = params["major_ganzi_index"].as_i64().unwrap_or(0) as i32;
72                let major = eon_saju::GanZi::from_index(idx);
73                let vm = eon_saju::SajuVM::new(pillars.clone());
74                let fuzzer = DestinyFuzzer::new(vm);
75                let res = fuzzer.audit(major);
76                serde_json::to_value(res).map_err(|e| e.into())
77            }
78            "backtrace_root_cause" => {
79                let age = params["target_age"].as_u64().unwrap_or(0) as u32;
80                let tag = params["target_tag"].as_str().unwrap_or("");
81
82                // 에뮬레이션 시뮬레이션 데이터 필요
83                let emulator = LifePathEmulator::new(pillars.clone(), Gender::Male, 2004); // 예시용 고정 데이터
84                match emulator.emulate() {
85                    Ok(report) => {
86                        let res = DestinyDebugger::backtrace(&report, age, tag);
87                        serde_json::to_value(res).map_err(|e| e.into())
88                    }
89                    Err(e) => Ok(serde_json::json!({"error": format!("에뮬레이션 실패: {}", e)})),
90                }
91            }
92            _ => Ok(serde_json::json!({"error": "Unknown tool"})),
93        }
94    }
95}
