7/// 린터 진단 레벨
8#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
9pub enum LintSeverity {
10    Error,
11    Warning,
12    Info,
13}
14
15/// 사주 린트 결과 (Diagnostic)
16#[derive(Debug, Clone, Serialize, Deserialize)]
17pub struct SajuLint {
18    pub code: String,
19    pub severity: LintSeverity,
20    pub message: String,
21    pub advice: String,
22}

26impl DestinyLinter {
27    pub fn lint(pillars: &FourPillars) -> Vec<SajuLint> {
28        let mut diagnostics = Vec::new();
29        let yongshin = pillars.yongshin();
30        let strength = pillars.strength();
31        let counts = pillars.element_counts();
32
33        // E404: 핵심 용신 부재
34        Self::check_missing_dependency(&counts, &yongshin, &mut diagnostics);
35        // E405: 일간 극약 (4득 전무)
36        Self::check_extreme_weakness(&strength.strength_type, &mut diagnostics);
37        // W105: 오행 억제 (한 오행이 다른 오행을 압도)
38        Self::check_unused_elements(&counts, &mut diagnostics);
39        // W202: 오행 편중
40        Self::check_infinite_loop(&counts, &mut diagnostics);
41        // W203: 오행 결핍 2종 이상
42        Self::check_missing_elements(&counts, &mut diagnostics);
43        // 신살 체크
44        Self::check_legacy_patterns(pillars, &mut diagnostics);
45
46        diagnostics
47    }

49    fn check_missing_dependency(
50        counts: &[(Element, u32); 5],
51        yongshin: &YongshinAnalysis,
52        lints: &mut Vec<SajuLint>,
53    ) {
54        let primary_count = counts
55            .iter()
56            .find(|(el, _)| *el == yongshin.primary)
57            .map(|(_, c)| *c)
58            .unwrap_or(0);
59
60        if primary_count == 0 {
61            lints.push(SajuLint {
62                code: "E404".to_string(),
63                severity: LintSeverity::Error,
64                message: format!("핵심 기운 부재: 필요한 '{}' 기운이 사주에 없습니다.", yongshin.primary.hangul()),
65                advice: format!("대운이나 세운에서 '{}' 기운이 들어올 때까지 핵심 역량 발휘가 제한될 수 있습니다.", yongshin.primary.hangul()),
66            });
67        }
68    }

70    fn check_extreme_weakness(strength_type: &StrengthType, lints: &mut Vec<SajuLint>) {
71        if *strength_type == StrengthType::Weak {
72            lints.push(SajuLint {
73                code: "E405".to_string(),
74                severity: LintSeverity::Error,
75                message: "일간 극약: 자신의 기운이 매우 약해 외부 환경에 쉽게 흔들릴 수 있습니다.".to_string(),
76                advice: "일간을 돕는 기운(인성·비겁)이 들어오는 대운·세운을 적극 활용하고, 평소 심신 안정에 집중하세요.".to_string(),
77            });
78        }
79    }

81    fn check_missing_elements(counts: &[(Element, u32); 5], lints: &mut Vec<SajuLint>) {
82        let missing: Vec<_> = counts
83            .iter()
84            .filter(|(_, c)| *c == 0)
85            .map(|(el, _)| el.hangul())
86            .collect();
87        if missing.len() >= 2 {
88            lints.push(SajuLint {
89                code: "W203".to_string(),
90                severity: LintSeverity::Warning,
91                message: format!("오행 결핍: '{}' 기운이 사주에 완전히 없어 흐름이 편향되어 있습니다.", missing.join("·")),
92                advice: "부족한 기운이 들어오는 시기(대운·세운)에 중요한 결정을 내리거나 해당 기운을 보완하는 활동을 이어가세요.".to_string(),
93            });
94        }
95    }

97    fn check_unused_elements(counts: &[(Element, u32); 5], lints: &mut Vec<SajuLint>) {
98        for (el, count) in counts.iter() {
99            if *count > 0 {
100                let controller = el.controlled_by();
101                let controller_count = counts
102                    .iter()
103                    .find(|(e, _)| *e == controller)
104                    .map(|(_, c)| *c)
105                    .unwrap_or(0);
106
107                if controller_count >= 3 {
108                    lints.push(SajuLint {
109                        code: "W105".to_string(),
110                        severity: LintSeverity::Warning,
111                        message: format!("기운 억제: '{}' 기운이 강한 '{}' 기운에 눌려 제대로 발휘되지 못하고 있습니다.", el.hangul(), controller.hangul()),
112                        advice: format!("'{}' 기운을 조절하거나 균형을 잡아줄 보완 요소가 필요합니다.", controller.hangul()),
113                    });
114                }
115            }
116        }
117    }

119    fn check_infinite_loop(counts: &[(Element, u32); 5], lints: &mut Vec<SajuLint>) {
120        for (el, count) in counts.iter() {
121            if *count >= 4 {
122                lints.push(SajuLint {
123                    code: "W202".to_string(),
124                    severity: LintSeverity::Warning,
125                    message: format!("기운 편중: '{}' 기운이 지나치게 강합니다.", el.hangul()),
126                    advice: "한 가지 기운에 치우치면 전체 흐름이 불균형해집니다. 부족한 기운을 보완하는 것이 중요합니다.".to_string(),
127                });
128            }
129        }
130    }
