5/// 인생 복잡도 등급
6#[derive(Debug, Clone, Serialize, Deserialize)]
7pub enum ObfuscationLevel {
8    #[serde(rename = "단순형")]
9    Plaintext, // 오행이 한곳에 집중, 인생 경로가 선명함
10    #[serde(rename = "보통형")]
11    Standard, // 고르게 분산된 균형 있는 사주
12    #[serde(rename = "복합형")]
13    Packed, // 다양한 기운이 얽혀 있어 복잡한 흐름
14    #[serde(rename = "복잡형")]
15    Encrypted, // 오행이 고도로 복잡하게 얽혀 있어 전문 분석 필요
16}

18/// 엔트로피 분석 결과
19#[derive(Debug, Clone, Serialize, Deserialize)]
20pub struct EntropyAnalysis {
21    pub score: f32, // Shannon Entropy Score (0.0 ~ 2.322)
22    pub level: ObfuscationLevel,
23    pub description: String,
24    pub unpacker_element: Option<Element>, // 복잡성을 해소해줄 '언패커' 오행
25}

29impl DestinyEntropy {
30    pub fn analyze(pillars: &FourPillars) -> EntropyAnalysis {
31        // 1. 모든 요소 수집 (천간 + 지지 + 지장간 포함하여 데이터 밀도 높임)
32        let mut element_counts = [0.0f32; 5];
33        let mut total_elements = 0.0f32;
34
35        // 원국 8글자 반영
36        for (el, count) in pillars.element_counts() {
37            element_counts[el.index() as usize] += count as f32;
38            total_elements += count as f32;
39        }
40
41        // 지장간(Hidden Stems) 데이터 추가 반영하여 엔트로피 정밀도 향상
42        // 지장간은 내부 로직(난독화 데이터)으로 간주
43        for branch in [
44            &pillars.year.branch,
45            &pillars.month.branch,
46            &pillars.day.branch,
47            &pillars.hour.branch,
48        ] {
49            for stem in branch.jijanggan() {
50                element_counts[stem.element().index() as usize] += 0.3;
51                total_elements += 0.3;
52            }
53        }
54
55        // 2. Shannon Entropy 계산: H = -sum(pi * log2(pi))
56        let mut entropy = 0.0f32;
57        for count in element_counts {
58            if count > 0.0 {
59                let p = count / total_elements;
60                entropy -= p * p.log2();
61            }
62        }
63
64        // 3. 등급 판별 (DIE 기준 차용)
65        // 최대 엔트로피는 log2(5) ≈ 2.32
66        let (level, description) = if entropy < 1.0 {
67            (ObfuscationLevel::Plaintext, "오행이 한곳에 집중되어 인생 경로가 매우 명확합니다. 목표가 뚜렷하지만 유연성이 부족할 수 있습니다.".to_string())
68        } else if entropy < 1.8 {
69            (
70                ObfuscationLevel::Standard,
71                "오행이 고르게 분포되어 균형 잡힌 삶의 흐름을 가집니다.".to_string(),
72            )
73        } else if entropy < 2.1 {
74            (
75                ObfuscationLevel::Packed,
76                "여러 기운이 복잡하게 얽혀 있어 특정 시점에 진정한 모습이 드러납니다.".to_string(),
77            )
78        } else {
79            (ObfuscationLevel::Encrypted, "오행이 매우 복잡하게 얽혀 있어 삶의 흐름을 파악하기 어렵습니다. 전문적인 분석이 도움이 됩니다.".to_string())
80        };
81
82        // 4. Unpacker 탐색: 가장 부족하거나 엔트로피를 낮춰줄(균형을 잡을) 핵심 오행
83        // 실제로는 용신과 유사하나 '복잡도 해소' 관점에서 접근
84        let unpacker = pillars.yongshin().primary;
85
86        EntropyAnalysis {
87            score: entropy,
88            level,
89            description,
90            unpacker_element: Some(unpacker),
91        }
92    }
93}

95impl std::fmt::Display for EntropyAnalysis {
96    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
97        writeln!(f, "【Destiny Entropy & Obfuscation Analysis】")?;
98        writeln!(f, "Entropy Score : {:.3} (DIE Standard)", self.score)?;
99
100        let level_str = match self.level {
101            ObfuscationLevel::Plaintext => "🟢 Plaintext (해석 용이)",
102            ObfuscationLevel::Standard => "🔵 Standard (일반)",
103            ObfuscationLevel::Packed => "🟡 Packed (압축/은폐)",
104            ObfuscationLevel::Encrypted => "🔴 Encrypted (난독화/난해)",
105        };
106        writeln!(f, "Analysis Level: {}", level_str)?;
107        writeln!(f, "Description   : {}", self.description)?;
108
109        if let Some(up) = self.unpacker_element {
110            writeln!(
111                f,
112                "Unpacker Hint : '{}' 에너지가 유입될 때 인생의 압축이 해제(Unpacking)됩니다.",
113                up.hangul()
114            )?;
115        }
116        Ok(())
117    }
