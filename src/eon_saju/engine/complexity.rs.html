4/// 인생 복잡도 분석 결과
5#[derive(Debug, Clone, Serialize, Deserialize)]
6pub struct ComplexityAnalysis {
7    /// 순환 복잡도 (Cyclomatic Complexity: M = P + 1)
8    /// P: 결정 지점(Decision Points)의 수
9    pub cyclomatic_complexity: u32,
10
11    /// 코드 안정성 등급 (시스템 복잡도 기준)
12    pub stability_grade: String,
13
14    /// 유지보수 엔트로피 (인생의 파란만장함 정도)
15    pub entropy: f32,
16
17    /// 주요 분기점(Decision Nodes) 위치
18    pub decision_nodes: Vec<u32>,
19}
20
21pub struct DestinyComplexity;
22
23impl DestinyComplexity {
24    pub fn analyze(frames: &[LifeFrame]) -> ComplexityAnalysis {
25        let mut decision_points = 0;
26        let mut decision_nodes = Vec::new();
27        let mut total_conflict_weight = 0.0;
28
29        for frame in frames {
30            let mut is_decision_point = false;
31
32            // 1. 충격 및 형살 (Exception Handling)
33            // 충(Clash)이나 형(Punishment)은 실행 흐름의 예외 상황인 분기점으로 간주
34            let conflict_count = frame
35                .tags
36                .iter()
37                .filter(|t| {
38                    t.contains_pattern("충") || t.contains_pattern("형") || t.contains_pattern("해")
39                })
40                .count();
41
42            if conflict_count > 0 {
43                is_decision_point = true;
44                total_conflict_weight += conflict_count as f32;
45            }
46
47            // 2. 특수 시그니처 (Conditional Branch)
48            // 천지덕합, 용신충극 등 강력한 시그니처는 시스템의 상태 변화를 유발하는 결정 지점
49            if !frame.signatures.is_empty() {
50                is_decision_point = true;
51                total_conflict_weight += frame.signatures.len() as f32 * 1.5;
52            }
53
54            if is_decision_point {
55                decision_points += 1;
56                decision_nodes.push(frame.age);
57            }
58        }
59
60        let m = decision_points + 1; // M = decision points + 1
61        let entropy = (total_conflict_weight / frames.len() as f32) * 10.0;
62
63        let stability_grade = match m {
64            0..=10 => "A (Linear & Stable)".to_string(),
65            11..=30 => "B (Structured)".to_string(),
66            31..=60 => "C (Complex/Spaghetti)".to_string(),
67            _ => "D (High Entropy/Unstable)".to_string(),
68        };
69
70        ComplexityAnalysis {
71            cyclomatic_complexity: m,
72            stability_grade,
73            entropy,
74            decision_nodes,
75        }
76    }
77}

79impl std::fmt::Display for ComplexityAnalysis {
80    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
81        writeln!(f, "【Destiny Cyclomatic Complexity】")?;
82        writeln!(
83            f,
84            "Complexity (M): {} (Decision Points + 1)",
85            self.cyclomatic_complexity
86        )?;
87        writeln!(f, "Stability Grade: {}", self.stability_grade)?;
88        writeln!(f, "Maintenance Entropy: {:.2}", self.entropy)?;
89
90        let path_type = if self.cyclomatic_complexity > 50 {
91            "스파게티 코드 (Spaghetti Path)"
92        } else if self.cyclomatic_complexity > 20 {
93            "구조적 분기 (Structured Path)"
94        } else {
95            "단순 수차적 흐름 (Linear Path)"
96        };
97
98        writeln!(f, "Estimated Path Type: {}", path_type)?;
99
100        if !self.decision_nodes.is_empty() {
101            let nodes_str = self
102                .decision_nodes
103                .iter()
104                .take(10)
105                .map(|age| format!("{}세", age))
106                .collect::<Vec<_>>()
107                .join(", ");
108            writeln!(f, "Main Decision Nodes: {} ...", nodes_str)?;
109        }
110
111        Ok(())
112    }
113}
