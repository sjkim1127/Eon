5/// 운세 패턴 시그니처
6#[derive(Debug, Clone, Serialize, Deserialize)]
7pub struct LuckSignature {
8    pub id: String,
9    pub name: String,
10    pub severity: SignatureSeverity,
11    pub description: String,
12}
13
14#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Serialize, Deserialize)]
15pub enum SignatureSeverity {
16    Info,
17    Success,
18    Warning,
19    Critical,
20}
21
22pub struct SignatureScanner;
23
24impl SignatureScanner {
25    pub fn scan(frame: &LifeFrame, dynamic: &DynamicLuckAnalysis) -> Vec<LuckSignature> {
26        let mut detected = Vec::new();
27
28        // 1. 천지덕합 (Heavenly & Earthly Double Harmony)
29        if Self::check_double_harmony(dynamic) {
30            detected.push(LuckSignature {
31                id: "DOUBLE_HARMONY".to_string(),
32                name: "천지덕합 (天地德合)".to_string(),
33                severity: SignatureSeverity::Success,
34                description: "천간과 지지가 모두 합이 되어 환경과 정신이 완벽한 조화를 이루는 시기입니다.".to_string(),
35            });
36        }
37
38        // 2. 용신 충극 (Fatal Yongshin Clash)
39        if frame.score < 30.0 && frame.tags.iter().any(|t| t.contains_pattern("충")) {
40            detected.push(LuckSignature {
41                id: "FATAL_LUCK_CLASH".to_string(),
42                name: "용신충극 (用神沖剋)".to_string(),
43                severity: SignatureSeverity::Critical,
44                description: "핵심 에너지가 강력한 충돌로 인해 기반이 크게 흔들리는 위기 상황입니다.".to_string(),
45            });
46        }
47
48        // 3. 재공귀인 (Gong-Gwi - 가상 패턴: 복합적인 합의 중첩)
49        let branch_hap_count = dynamic.combined_relations.six_combinations.iter().filter(|(_, p1, p2)| p1.contains("운") || p2.contains("운")).count() + 
50                               dynamic.combined_relations.dominant_semi_combinations.iter().filter(|(_, p1, p2)| p1.contains("운") || p2.contains("운")).count() +
51                               dynamic.combined_relations.weak_semi_combinations.iter().filter(|(_, p1, p2)| p1.contains("운") || p2.contains("운")).count();
52
53        if branch_hap_count >= 2 && frame.score > 65.0 {
54            detected.push(LuckSignature {
55                id: "GONG_GWI_RESONANCE".to_string(),
56                name: "재공귀인 (財官貴人) 공명".to_string(),
57                severity: SignatureSeverity::Success,
58                description: "보이지 않는 기운이 명예와 재물을 끌어오는 강력한 성취의 시기입니다.".to_string(),
59            });
60        }
61
62        // 4. 등용문 (The Dragon Gate)
63        if frame.score > 85.0 && frame.tags.iter().any(|t| t.contains_pattern("핵심운")) {
64           detected.push(LuckSignature {
65                id: "DRAGON_GATE".to_string(),
66                name: "등용문 (登龍門)".to_string(),
67                severity: SignatureSeverity::Success,
68                description: "인생의 큰 관문을 넘어 신분이 상승하거나 큰 조직의 수장이 되는 운세입니다.".to_string(),
69            });
70        }
71
72        detected
73    }
74
75    fn check_double_harmony(dynamic: &DynamicLuckAnalysis) -> bool {
76        // 대운이나 세운이 개입된 합만 필터링
77        let has_dynamic_stem_hap = dynamic.combined_relations.stem_combinations.iter()
78            .any(|(_, p1, p2)| p1.contains("운") || p2.contains("운"));
79        
80        let has_dynamic_branch_hap = dynamic.combined_relations.six_combinations.iter().any(|(_, p1, p2)| p1.contains("운") || p2.contains("운")) ||
81                                    dynamic.combined_relations.dominant_semi_combinations.iter().any(|(_, p1, p2)| p1.contains("운") || p2.contains("운")) ||
82                                    dynamic.combined_relations.weak_semi_combinations.iter().any(|(_, p1, p2)| p1.contains("운") || p2.contains("운")) ||
83                                    !dynamic.combined_relations.triple_combinations.is_empty() ||
84                                    !dynamic.combined_relations.seasonal_combinations.is_empty();
85        
86        has_dynamic_stem_hap && has_dynamic_branch_hap
87    }
