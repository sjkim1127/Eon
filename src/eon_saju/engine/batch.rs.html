1//! 하드웨어 가속 시뮬레이터
2//!
3//! 멀티코어 병렬 처리를 통해 수십만 개의 인생 프레임을 수 밀리초 내에 시뮬레이션합니다.
4//! WASM 환경에서는 순차 처리로 자동 폴백됩니다.
5
6#[cfg(feature = "parallel")]
7use rayon::prelude::*;
8
9use crate::analysis::major_luck::MajorLuckAnalysis;
10use crate::core::ganzi::GanZi;
11use crate::core::pillars::FourPillars;
12use crate::engine::vm::{LifeFrame, SajuVM};
13
14pub struct BatchSimulator {
15    vm: SajuVM,
16}
17
18impl BatchSimulator {
19    pub fn new(pillars: FourPillars) -> Self {
20        Self {
21            vm: SajuVM::new(pillars),
22        }
23    }
24
25    /// 1000년 전수 조사 시뮬레이션
26    pub fn simulate_1000_years(
27        &self,
28        start_year: i32,
29        major_luck: &MajorLuckAnalysis,
30    ) -> Vec<LifeFrame> {
31        let years: Vec<i32> = (start_year..start_year + 1000).collect();
32
33        let mapper = |&year: &i32| {
34            let age = (year - start_year).unsigned_abs();
35            let yearly_ganzi = GanZi::from_year(year);
36            let major_ganzi = major_luck
37                .at_age(age)
38                .map(|l| l.ganzi)
39                .unwrap_or(yearly_ganzi);
40            self.vm
41                .step(age, major_ganzi, yearly_ganzi, None, None, None)
42        };
43
44        #[cfg(feature = "parallel")]
45        {
46            years.par_iter().map(mapper).collect()
47        }
48
49        #[cfg(not(feature = "parallel"))]
50        {
51            years.iter().map(mapper).collect()
52        }
53    }
54
55    /// 특정 범위의 연도 시뮬레이션
56    pub fn simulate_range(
57        &self,
58        start_year: i32,
59        end_year: i32,
60        major_luck: &MajorLuckAnalysis,
61    ) -> Vec<LifeFrame> {
62        let years: Vec<i32> = (start_year..=end_year).collect();
63
64        let mapper = |&year: &i32| {
65            let age = (year - start_year).unsigned_abs();
66            let yearly_ganzi = GanZi::from_year(year);
67            let major_ganzi = major_luck
68                .at_age(age)
69                .map(|l| l.ganzi)
70                .unwrap_or(yearly_ganzi);
71            self.vm
72                .step(age, major_ganzi, yearly_ganzi, None, None, None)
73        };
74
75        #[cfg(feature = "parallel")]
76        {
77            years.par_iter().map(mapper).collect()
78        }
79
80        #[cfg(not(feature = "parallel"))]
81        {
82            years.iter().map(mapper).collect()
83        }
84    }
85}
