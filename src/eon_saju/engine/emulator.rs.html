12/// 1년 단위의 시계열 인생 점수 데이터
13#[derive(Debug, Clone, Serialize, Deserialize)]
14#[serde(rename_all = "camelCase")]
15pub struct YearlyScore {
16    pub year: i32,
17    pub age: u32,
18
19    // 1. Base Score (기존 종합 점수)
20    pub total_score: f64,
21
22    // 2. 다차원 십성 점수
23    pub wealth_score: f64,
24    pub career_score: f64,
25    pub academic_score: f64,
26    pub health_score: f64,
27
28    // 3. 변동성 및 메타데이터
29    pub volatility_index: f64,      // 0.0 ~ 100.0
30    pub is_transition_period: bool, // 교운기 여부 플래그
31
32    // 4. 이동평균선 데이터 보관용
33    pub trend_ma_5yr: Option<f64>,
34}

36/// 100년 인생 시뮬레이션 결과 리포트
37#[derive(Debug, Clone, Serialize, Deserialize)]
38#[serde(rename_all = "camelCase")]
39pub struct LifePathReport {
40    /// 연도별 시계열 점수 데이터
41    pub timeline: Vec<YearlyScore>,
42    /// 연도별 프레임 목록
43    pub frames: Vec<LifeFrame>,
44    /// 인생의 최정점 연도 (나이)
45    pub peak_age: u32,
46    /// 인생의 최저점 연도 (나이)
47    pub valley_age: u32,
48}
49
50pub struct LifePathEmulator {
51    pub vm: SajuVM,
52    pub gender: Gender,
53    pub birth_year: i32,
54}
55
56impl LifePathEmulator {
57    pub fn new(natal: FourPillars, gender: Gender, birth_year: i32) -> Self {
58        Self {
59            vm: SajuVM::new(natal),
60            gender,
61            birth_year,
62        }
63    }
64
65    /// 100년 인생 경로를 에뮬레이션합니다.
66    pub fn emulate(&self) -> Result<LifePathReport, SajuError> {
67        let mut timeline = Vec::with_capacity(101);
68        let input = &self.vm.natal.raw_input;
69        let major_luck = self.vm.natal.major_luck(
70            self.gender,
71            input.year,
72            input.month,
73            input.day,
74            input.hour,
75            input.minute,
76        )?;
77
78        // 원국 기준 필요 데이터
79        let dm_element = self.vm.natal.day_master_element();
80        let base_dist = self.vm.natal.element_distribution();
81        let primary = self.vm.yongshin.primary;
82        let assistant = self.vm.yongshin.assistant;
83        let thermal =
84            crate::analysis::yongshin::calculate_thermal_index(&self.vm.natal, &self.vm.config);
85
86        let mut frames_buffer = Vec::new();
87
88        for age in 0..=100 {
89            let year = self.birth_year + age as i32;
90            let yearly_ganzi = GanZi::from_year(year);
91
92            // 현재 대운 및 주변 대운 정보 탐색
93            // Precondition: major_luck.cycles is non-empty and sorted ascending by start_age
94            // (guaranteed by MajorLuckAnalysis::calculate_astro). Using partition_point gives
95            // O(log n) lookup and lets us borrow cycles instead of cloning them.
96            let cycle_pos = major_luck.cycles.partition_point(|c| c.start_age <= age);
97            let current_major_cycle = if cycle_pos > 0 {
98                Some(&major_luck.cycles[cycle_pos - 1])
99            } else {
100                None
101            };
102            let start_age_of_major = current_major_cycle.map_or(0, |c| c.start_age);
103            let major_ganzi = current_major_cycle
104                .map(|c| c.ganzi)
105                .unwrap_or(self.vm.natal.month);
106
107            // 볼라틸리티(변동성, Volatility) 인덱스 계산 - 교운기
108            let mut dt = (age as i32 - start_age_of_major as i32).abs();
109            if dt > 5 {
110                // 교운기는 대운 시작 전후이므로 나중 대운과의 거리도 고려
111                // 다음 대운까지 남은 시간 (cycle_pos is the index of the first cycle with start_age > age)
112                if let Some(next_c) = major_luck.cycles.get(cycle_pos) {
113                    dt = (next_c.start_age as i32 - age as i32).abs();
114                }
115            }
116            let is_transition_period = dt <= 2;
117
118            // T(t) = max(0, 1.0 - (dt / 2.0))
119            let t_val = (1.0 - (dt as f64 / 2.0)).max(0.0);
120
121            let mut v_val = 0.0;
122            // 이전 대운과 현재 대운(또는 새 대운) 사이의 벡터 거리
123            if is_transition_period {
124                // 이전 대운 또는 다음 대운 찾기
125                let mut alt_ganzi = self.vm.natal.month;
126                if age >= start_age_of_major && dt <= 2 && age < start_age_of_major + 3 {
127                    // 방금 시작한 교운기 -> 이전 대운과의 차이
128                    // current cycle is at cycle_pos - 1; previous is at cycle_pos - 2
129                    if cycle_pos >= 2 {
130                        alt_ganzi = major_luck.cycles[cycle_pos - 2].ganzi;
131                    }
132                } else if dt <= 2 {
133                    // 곧 시작할 교운기 -> 다음 대운과의 차이
134                    if let Some(next_c) = major_luck.cycles.get(cycle_pos) {
135                        alt_ganzi = next_c.ganzi;
136                    }
137                }
138
139                // V 계산: 두 간지의 오행 차이 유클리디안 거리
140                let mut vec_current = [0.0; 5];
141                let mut vec_alt = [0.0; 5];
142                vec_current[major_ganzi.stem.element() as usize] += 1.0;
143                vec_current[major_ganzi.branch.element() as usize] += 1.0;
144                vec_alt[alt_ganzi.stem.element() as usize] += 1.0;
145                vec_alt[alt_ganzi.branch.element() as usize] += 1.0;
146
147                let mut sq_sum: f64 = 0.0;
148                for i in 0..5 {
149                    let diff = vec_current[i] - vec_alt[i];
150                    sq_sum += diff * diff;
151                }
152                v_val = sq_sum.sqrt();
153            }
154
155            // LifeFrame 추출
156            let frame = self
157                .vm
158                .step(age, major_ganzi, yearly_ganzi, None, None, None);
159
160            // 다이나믹 분석 재생성 (M(X,t)와 C(t) 계산을 위함)
161            let dynamic = DynamicLuckAnalysis::analyze(
162                &self.vm.natal,
163                Some(major_ganzi),
164                Some(yearly_ganzi),
165                None,
166                None,
167                None,
168            );
169
170            // 세운 충돌 계수 C(t)
171            let mut c_val = 1.0;
172            for (_clash, p1, p2) in &dynamic.combined_relations.branch_clashes {
173                if p1.contains("일지")
174                    || p2.contains("일지")
175                    || p1.contains("월지")
176                    || p2.contains("월지")
177                {
178                    c_val = 1.5;
179                }
180            }
181            if !dynamic.combined_relations.stem_clashes.is_empty() && c_val > 1.0 {
182                c_val = 2.0; // 천극지충
183            }
184
185            let volatility_index = (10.0 + (t_val * v_val * c_val * 20.0)).clamp(0.0, 100.0);
186
187            // --- 다차원 십성 점수(Ten Gods Multi-dimensional Score) 계산 ---
188            let mut e_scores: [f64; 5] = [0.0; 5];
189
190            for el_idx in 0..5 {
191                let el = Element::from_index(el_idx);
192
193                // S(X): 원국 + 대운 + 세운의 기본 세력
194                let mut s_val = 0.0;
195                s_val += base_dist[el_idx as usize].1 as f64;
196                if major_ganzi.stem.element() == el {
197                    s_val += 1.0;
198                }
199                if major_ganzi.branch.element() == el {
200                    s_val += 1.0;
201                }
202                if yearly_ganzi.stem.element() == el {
203                    s_val += 1.0;
204                }
205                if yearly_ganzi.branch.element() == el {
206                    s_val += 1.0;
207                }
208
209                // F(X): 기본 우호도 (-1.0 ~ 1.5)
210                let priority = self
211                    .vm
212                    .get_element_priority(el, primary, assistant, thermal);
213                // 0.0 기준을 1.0(보통)으로 맞추고 용신일때 증폭, 기신일때 축소
214                // get_element_priority는 1.0, 0.5, -0.8 등을 반환
215                // 변동성 강화를 위해 priority 가중치를 1.5로 더 증폭
216                let mut f_val = 1.0 + (priority as f64 * 1.5);
217                if f_val < 0.2 {
218                    f_val = 0.2;
219                } // 최소값 보장
220
221                // M(X,t): 합충형해파 변동
222                let mut m_val = 1.0;
223                // 세운 지지가 충당하는 원국/대운 요소의 오행일 때 세력 감소
224                for (_clash, _p1, _p2) in &dynamic.combined_relations.branch_clashes {
225                    // 충 발생 시 해당되는 오행(지장간 본기 기준)에 영향
226                    // 간략화를 위해 세운 지지의 오행과 충하는 오행이 el인지 확인
227                    if yearly_ganzi.branch.element().controls() == el
228                        || yearly_ganzi.branch.element().controlled_by() == el
229                    {
230                        m_val -= 0.3;
231                    }
232                }
233                for (_six, _p1, _p2) in &dynamic.combined_relations.six_combinations {
234                    if yearly_ganzi.branch.element() == el {
235                        m_val += 0.3;
236                    } // 합으로 세력 강화
237                }
238                if m_val < 0.4 {
239                    m_val = 0.4;
240                }
241                if m_val > 1.8 {
242                    m_val = 1.8;
243                }
244
245                e_scores[el_idx as usize] = s_val * f_val * m_val;
246            }
247
248            // 카테고리별 핵심 십성 매핑
249            // 변동성 확대를 위해 기본점수(base)를 0점으로 낮추고,
250            // 원점수 대비 증폭 계수(scale_factor)를 25.0으로 2배 더 높임
251            let base = 0.0;
252            let scale_factor = 25.0; // 점수 증폭 극대화
253
254            // 재물(Wealth) = 재성 0.6 + 식상 0.4
255            let wealth_el = dm_element.controls();
256            let shik_el = dm_element.generates();
257            let wealth_raw =
258                (e_scores[wealth_el as usize] * 0.6) + (e_scores[shik_el as usize] * 0.4);
259            let wealth_score = (base + (wealth_raw * scale_factor)).clamp(0.0, 100.0);
260
261            // 직업/명예(Career) = 관성 0.7 + 인성 0.3
262            let career_el = dm_element.controlled_by();
263            let yin_el = dm_element.generated_by();
264            let career_raw =
265                (e_scores[career_el as usize] * 0.7) + (e_scores[yin_el as usize] * 0.3);
266            let career_score = (base + (career_raw * scale_factor)).clamp(0.0, 100.0);
267
268            // 학업/문서(Academic) = 인성 0.7 + 식상 0.3
269            let academic_raw =
270                (e_scores[yin_el as usize] * 0.7) + (e_scores[shik_el as usize] * 0.3);
271            let academic_score = (base + (academic_raw * scale_factor)).clamp(0.0, 100.0);
272
273            // 건강/독립(Health) = 비겁 0.5 + 식상 0.5
274            let health_raw =
275                (e_scores[dm_element as usize] * 0.5) + (e_scores[shik_el as usize] * 0.5);
276            let health_score = (base + (health_raw * scale_factor)).clamp(0.0, 100.0);
277
278            // 통합 총점: VM 시뮬레이션 점수 + 4개 세부 점수의 균등 평균
279            let vm_score = frame.score as f64;
280            let integrated_total =
281                ((vm_score + wealth_score + career_score + academic_score + health_score) / 5.0)
282                    .clamp(0.0, 100.0);
283
284            timeline.push(YearlyScore {
285                year,
286                age,
287                total_score: integrated_total,
288                wealth_score,
289                career_score,
290                academic_score,
291                health_score,
292                volatility_index,
293                is_transition_period,
294                trend_ma_5yr: None, // 뒤에서 일괄 계산
295            });
296            frames_buffer.push(frame);
297        }
298
299        if timeline.is_empty() {
300            return Err(SajuError::CalculationError(
301                "No emulation frames generated".to_string(),
302            ));
303        }
304
305        // --- 4. 시계열 점수 보정 (Smoothing & Momentum) ---
306        // 5년 이동평균(MA) 선 (과거 2년 + 현재 + 미래 2년)
307        for i in 0..timeline.len() {
308            let start = i.saturating_sub(2);
309            let end = (i + 2).min(timeline.len() - 1);
310            let mut sum = 0.0;
311            let mut count = 0.0;
312
313            for j in start..=end {
314                sum += timeline[j].total_score;
315                count += 1.0;
316            }
317
318            timeline[i].trend_ma_5yr = Some(sum / count);
319        }
320
321        let peak_age = timeline
322            .iter()
323            .max_by(|a, b| {
324                a.total_score
325                    .partial_cmp(&b.total_score)
326                    .unwrap_or(std::cmp::Ordering::Equal)
327            })
328            .map(|t| t.age)
329            .unwrap_or(0);
330
331        let valley_age = timeline
332            .iter()
333            .min_by(|a, b| {
334                a.total_score
335                    .partial_cmp(&b.total_score)
336                    .unwrap_or(std::cmp::Ordering::Equal)
337            })
338            .map(|t| t.age)
339            .unwrap_or(0);
340
341        Ok(LifePathReport {
342            timeline,
343            frames: frames_buffer,
344            peak_age,
345            valley_age,
346        })
347    }
348}

350#[cfg(test)]
351mod tests {
352    use super::*;
353    use crate::core::pillars::SajuInput;
354
355    #[test]
356    fn test_multi_dimensional_scoring() {
357        let input = SajuInput::new_solar(1990, 5, 10, 12, 0);
358        let natal = FourPillars::calculate(&input).unwrap();
359        let emulator = LifePathEmulator::new(natal, Gender::Male, 1990);
360
361        let report = emulator.emulate().unwrap();
362        assert_eq!(report.timeline.len(), 101);
363        assert_eq!(report.frames.len(), 101);
364
365        let yr = report.timeline.iter().find(|t| t.age == 30).unwrap();
366
367        println!("YearlyScore at age 30: {:#?}", yr);
368        assert!(yr.wealth_score >= 0.0);
369        assert!(yr.career_score >= 0.0);
370        assert!(yr.academic_score >= 0.0);
371        assert!(yr.health_score >= 0.0);
372        assert!(yr.volatility_index >= 0.0);
373    }
