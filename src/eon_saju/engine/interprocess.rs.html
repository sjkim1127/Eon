1//! Inter-Process Compatibility (IPC) Audit Module
2//! 
3//! 두 개의 사주 시스템(SajuVM)이 상호작용할 때 발생하는
4//! 에너지 간섭, 자원 경합(상극), 시너지(합)를 분석합니다.
5
6use serde::{Deserialize, Serialize};
7use crate::engine::vm::SajuVM;
8
9/// IPC 분석 결과 리포트
10#[derive(Debug, Clone, Serialize, Deserialize)]
11pub struct CompatibilityAudit {
12    /// 종합 상호작용 점수 (0~100)
13    pub sync_score: f32,
14    /// 시스템 간 합(Synergy)
15    pub synergies: Vec<String>,
16    /// 시스템 간 충돌(Race Conditions/Conflicts)
17    pub conflicts: Vec<String>,
18    /// 자원 잠금 현상 (Deadlocks)
19    pub deadlocks: Vec<String>,
20    /// 두 시스템 결합 시의 ESIL 트레이스 요약
21    pub merged_esil_trace: String,
22}
23
24pub struct CompatibilityAuditor;
25
26impl CompatibilityAuditor {
27    /// 두 사주 시스템 간의 상호작용을 감사(Audit)합니다.
28    pub fn audit(system_a: &SajuVM, system_b: &SajuVM) -> CompatibilityAudit {
29        let mut synergies = Vec::new();
30        let mut conflicts = Vec::new();
31        let mut deadlocks = Vec::new();
32        let mut merged_trace = String::new();
33        
34        // 1. 천간 교차 결합 분석 (Interface Mapping)
35        let a_stems = [
36            ("A_년간", system_a.natal.year.stem),
37            ("A_월간", system_a.natal.month.stem),
38            ("A_일간", system_a.natal.day.stem),
39            ("A_시간", system_a.natal.hour.stem),
40        ];
41        let b_stems = [
42            ("B_년간", system_b.natal.year.stem),
43            ("B_월간", system_b.natal.month.stem),
44            ("B_일간", system_b.natal.day.stem),
45            ("B_시간", system_b.natal.hour.stem),
46        ];
47
48        // 2. 지지 교차 결합 분석 (Resource Contention)
49        let a_branches = [
50            ("A_년지", system_a.natal.year.branch),
51            ("A_월지", system_a.natal.month.branch),
52            ("A_일지", system_a.natal.day.branch),
53            ("A_시지", system_a.natal.hour.branch),
54        ];
55        let b_branches = [
56            ("B_년지", system_b.natal.year.branch),
57            ("B_월지", system_b.natal.month.branch),
58            ("B_일지", system_b.natal.day.branch),
59            ("B_시지", system_b.natal.hour.branch),
60        ];
61
62        // 3. 교차 합충 스캔
63        use crate::analysis::relationships::*;
64        let mut sync_score: f32 = 50.0;
65
66        for (p1, s1) in &a_stems {
67            for (p2, s2) in &b_stems {
68                if let Some(c) = StemCombination::check(*s1, *s2) {
69                    synergies.push(format!("천간합: {} <-> {} ({})", p1, p2, c.transformed_element().hangul()));
70                    sync_score += 5.0;
71                    merged_trace.push_str(&format!("ipc_joint:{}-{}_hap; ", p1, p2));
72                }
73                if let Some(_c) = StemClash::check(*s1, *s2) {
74                    conflicts.push(format!("천간충: {} <-> {}", p1, p2));
75                    sync_score -= 3.0;
76                    merged_trace.push_str(&format!("ipc_collision:{}-{}_clash; ", p1, p2));
77                }
78            }
79        }
80
81        for (p1, b1) in &a_branches {
82            for (p2, b2) in &b_branches {
83                if let Some(_c) = BranchClash::check(*b1, *b2) {
84                    deadlocks.push(format!("지지충: {} <-> {}", p1, p2));
85                    sync_score -= 5.0;
86                    merged_trace.push_str(&format!("resource_lock:{}-{}_deadlock; ", p1, p2));
87                }
88                if let Some(_s) = SixCombination::check(*b1, *b2) {
89                    synergies.push(format!("육합: {} <-> {}", p1, p2));
90                    sync_score += 4.0;
91                    merged_trace.push_str(&format!("memory_bridge:{}-{}_shared; ", p1, p2));
92                }
93            }
94        }
95
96        CompatibilityAudit {
97            sync_score: sync_score.clamp(0.0, 100.0),
98            synergies,
99            conflicts,
100            deadlocks,
101            merged_esil_trace: merged_trace,
102        }
103    }
104}
