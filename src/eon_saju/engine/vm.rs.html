23/// Saju Interrupt (하드웨어 예외/인터럽트)
24#[derive(Debug, Clone, Serialize, Deserialize)]
25#[serde(rename_all = "camelCase")]
26pub enum SajuInterrupt {
27    /// 0x01: 고우선순위 시스템 충돌 (백호살 등)
28    CriticalException,
29    /// 0x02: 리소스 오버플로우 (괴강살 등)
30    ResourceOverflow,
31    /// 0x03: 시스템 정지/고독 (고신/과숙 등)
32    SystemStall,
33    /// 0x04: 임시 서비스 중단 (망신/겁살 등)
34    ServiceInterrupt,
35}

37/// 오행 레지스터 (R0~R4)
38#[derive(Debug, Clone, Copy, Serialize, Deserialize)]
39#[serde(rename_all = "camelCase")]
40pub struct QiRegisters {
41    /// R0: 목 (Wood)
42    pub r0_wood: f32,
43    /// R1: 화 (Fire)
44    pub r1_fire: f32,
45    /// R2: 토 (Earth)
46    pub r2_earth: f32,
47    /// R3: 금 (Metal)
48    pub r3_metal: f32,
49    /// R4: 수 (Water)
50    pub r4_water: f32,
51}
52
53impl Default for QiRegisters {
54    fn default() -> Self {
55        Self::new()
56    }
57}
58
59impl QiRegisters {
60    pub fn new() -> Self {
61        Self {
62            r0_wood: 0.0,
63            r1_fire: 0.0,
64            r2_earth: 0.0,
65            r3_metal: 0.0,
66            r4_water: 0.0,
67        }
68    }
69
70    pub fn update(&mut self, el: Element, value: f32) {
71        match el {
72            Element::Wood => self.r0_wood += value,
73            Element::Fire => self.r1_fire += value,
74            Element::Earth => self.r2_earth += value,
75            Element::Metal => self.r3_metal += value,
76            Element::Water => self.r4_water += value,
77        }
78    }
79
80    /// 기운의 감쇠 (시간 흐름에 따른 기전 상실 표현)
81    pub fn apply_decay(&mut self, factor: f32) {
82        self.r0_wood *= factor;
83        self.r1_fire *= factor;
84        self.r2_earth *= factor;
85        self.r3_metal *= factor;
86        self.r4_water *= factor;
87    }
88
89    /// 기운의 정규화 (전체 합을 100%로 유지하여 상대적 균형 분석)
90    pub fn normalize(&mut self) {
91        let total = self.r0_wood + self.r1_fire + self.r2_earth + self.r3_metal + self.r4_water;
92        if total > 0.1 {
93            self.r0_wood = (self.r0_wood / total) * 100.0;
94            self.r1_fire = (self.r1_fire / total) * 100.0;
95            self.r2_earth = (self.r2_earth / total) * 100.0;
96            self.r3_metal = (self.r3_metal / total) * 100.0;
97            self.r4_water = (self.r4_water / total) * 100.0;
98        }
99    }
100}

159    pub fn step(
160        &self,
161        age: u32,
162        major_ganzi: GanZi,
163        yearly_ganzi: GanZi,
164        monthly_ganzi: Option<GanZi>,
165        daily_ganzi: Option<GanZi>,
166        hourly_ganzi: Option<GanZi>,
167    ) -> LifeFrame {
168        let dynamic = DynamicLuckAnalysis::analyze(
169            &self.natal,
170            Some(major_ganzi),
171            Some(yearly_ganzi),
172            monthly_ganzi,
173            daily_ganzi,
174            hourly_ganzi,
175        );
176
177        let mut tags = Vec::new();
178        let mut esil_trace = String::new();
179        let mut registers = QiRegisters::new();
180
181        let score = self.evaluate_score(&dynamic, &mut tags, &mut esil_trace, &mut registers);
182
183        let mut frame = LifeFrame {
184            age,
185            ganzi: yearly_ganzi,
186            major_ganzi,
187            score,
188            tags,
189            signatures: Vec::new(),
190            esil_trace,
191            register_state: registers,
192        };
193
194        // 시그니처 스캔
195        frame.signatures = SignatureScanner::scan(&frame, &dynamic);
196
197        frame
198    }

200    /// 점수 평가 로직 (Precision Tuning)
201    /// - Data Flow Analysis처럼 원국의 상태와 운의 상호작용을 정밀 추적
202    fn evaluate_score(
203        &self,
204        dynamic: &DynamicLuckAnalysis,
205        tags: &mut Vec<TraceTag>,
206        esil_trace: &mut String,
207        registers: &mut QiRegisters,
208    ) -> f32 {
209        let mut score = self.config.vm.base_score;
210
211        // 0. 하드웨어 인터럽트 체크 (Interrupt Vector Table Scan)
212        self.handle_interrupts(dynamic, tags, esil_trace, registers, &mut score);
213
214        // 1. 파이프라인 에뮬레이션 (Instruction Pipeline Execution)
215        // 대운(Fetch) -> 세운(Decode) -> 월운(Execute) -> 일운(Memory) -> 시운(WriteBack)
216        self.execute_pipeline(dynamic, tags, esil_trace, registers, &mut score);
217
218        let primary_yongshin = self.yongshin.primary;
219        let assistant_yongshin = self.yongshin.assistant;
220        let thermal_index =
221            crate::analysis::yongshin::calculate_thermal_index(&self.natal, &self.config);
222
223        // 2. 공망(Void) 동적 감지 및 탈공(Escaping Void) 분석
224        // 운에서 들어온 지지가 공망인지 확인
225        for (luck_branch, label) in [
226            (
227                dynamic.major_influence.as_ref().map(|i| i.ganzi.branch),
228                "대운",
229            ),
230            (
231                dynamic.yearly_influence.as_ref().map(|i| i.ganzi.branch),
232                "세운",
233            ),
234        ] {
235            if let Some(b) = luck_branch {
236                if self.void.void_branches.contains(&b) {
237                    // 공망 발생! 기본적으로 흉(비효율, 헛됨)으로 간주
238                    // 하지만 합(육합, 삼합, 방합)이나 충이 발생하면 '탈공(전실)'되어 공망이 해소됨
239
240                    let mut is_escaped = false;
241                    let _b_hangul = b.hangul();
242
243                    // 탈공 조건 1: 지지 충
244                    for (_, p1, p2) in &dynamic.combined_relations.branch_clashes {
245                        if (p1.contains(label) || p2.contains(label))
246                            && ((p1.contains(label)
247                                && self.get_branch_by_path(p1, dynamic) == Some(b))
248                                || (p2.contains(label)
249                                    && self.get_branch_by_path(p2, dynamic) == Some(b)))
250                        {
251                            is_escaped = true;
252                            let period = LuckPeriod::from_label(label);
253                            tags.push(TraceTag::EscapedVoidClash { period });
254                            esil_trace
255                                .push_str(&format!("void_escape:{}_clash,restore:10.0; ", label));
256                            break;
257                        }
258                    }
259
260                    // 탈공 조건 2: 합 (육합, 삼합, 방합)
261                    if !is_escaped {
262                        // 육합 체크
263                        for (_, p1, p2) in &dynamic.combined_relations.six_combinations {
264                            if p1.contains(label) || p2.contains(label) {
265                                is_escaped = true;
266                                let period = LuckPeriod::from_label(label);
267                                tags.push(TraceTag::EscapedVoidSixCombo { period });
268                                break;
269                            }
270                        }
271                        // 삼합/방합 체크 (이미 combined_relations에 포함됨)
272                        if !is_escaped {
273                            for trip in &dynamic.combined_relations.triple_combinations {
274                                if trip.branches().contains(&b) {
275                                    is_escaped = true;
276                                    let period = LuckPeriod::from_label(label);
277                                    tags.push(TraceTag::EscapedVoidTriple { period });
278                                    break;
279                                }
280                            }
281                        }
282                        if !is_escaped {
283                            for season in &dynamic.combined_relations.seasonal_combinations {
284                                if season.branches().contains(&b) {
285                                    is_escaped = true;
286                                    let period = LuckPeriod::from_label(label);
287                                    tags.push(TraceTag::EscapedVoidSeasonal { period });
288                                    break;
289                                }
290                            }
291                        }
292                    }
293
294                    if is_escaped {
295                        // 탈공됨 (전실): 전화위복
296                        score += 5.0; // 오히려 좋아질 수 있음
297                    } else {
298                        // 진공(眞空): 공망의 흉의가 그대로 작용
299                        let penalty = 10.0;
300                        score -= penalty;
301                        tags.push(TraceTag::VoidLuck {
302                            period: LuckPeriod::from_label(label),
303                        });
304                        esil_trace
305                            .push_str(&format!("void_luck:{},penalty:-{:.1}; ", label, penalty));
306                    }
307                }
308            }
309        }
310
311        // 2.5 신살(Shinsal) 동적 분석 (Phase 12)
312        // 12신살, 길신(천을/문창), 흉살(원진/귀문) 체크
313
314        let natal_day_branch = self.natal.day.branch;
315        let natal_year_branch = self.natal.year.branch;
316        let day_master = self.natal.day_master();
317
318        for (luck_branch, label) in [
319            (
320                dynamic.major_influence.as_ref().map(|i| i.ganzi.branch),
321                "대운",
322            ),
323            (
324                dynamic.yearly_influence.as_ref().map(|i| i.ganzi.branch),
325                "세운",
326            ),
327        ] {
328            if let Some(b) = luck_branch {
329                // A. 12신살 (일지 기준)
330                let shinsal_day = TwelveShinsal::calculate(natal_day_branch, b);
331                let (s_score, _s_tag) = match shinsal_day {
332                    TwelveShinsal::Jangseongsal | TwelveShinsal::Banansal => (5.0, "길신"),
333                    TwelveShinsal::Geopsal | TwelveShinsal::Jaesal | TwelveShinsal::Cheonsal => {
334                        (-5.0, "흉신")
335                    }
336                    TwelveShinsal::Yeokmasal | TwelveShinsal::Jisal => (2.0, "이동"), // 역마/지살은 활동성 증가
337                    _ => (0.0, "평"),
338                };
339                if s_score != 0.0 {
340                    score += s_score;
341                    tags.push(TraceTag::Custom(format!(
342                        "신살:{}({})",
343                        shinsal_day.hangul(),
344                        label
345                    )));
346                    esil_trace.push_str(&format!(
347                        "shinsal:{},score:{:.1}; ",
348                        shinsal_day.hangul(),
349                        s_score
350                    ));
351                }
352
353                // B. 길신 (천을귀인, 문창귀인)
354                let cheoneul = Gilsin::cheoneul_branches(day_master);
355                if cheoneul.contains(&b) {
356                    let bonus = 15.0;
357                    score += bonus;
358                    tags.push(TraceTag::AuspiciousSpirit {
359                        name: AuspiciousSpiritName::Cheoneul,
360                        period: LuckPeriod::from_label(label),
361                    });
362                    esil_trace.push_str(&format!("gilsin:cheoneul,bonus:{:.1}; ", bonus));
363                }
364
365                // C. 흉살 (원진, 귀문) - 일지/년지와 관계
366                for (target, target_name) in
367                    [(natal_day_branch, "일지"), (natal_year_branch, "년지")]
368                {
369                    if let Some(wonjin) = EvilSpirit::check_wonjin(target, b) {
370                        score -= 5.0;
371                        tags.push(TraceTag::Custom(format!(
372                            "흉살:{}({}-{})",
373                            wonjin.hangul(),
374                            target_name,
375                            label
376                        )));
377                    }
378                    if let Some(gwimun) = EvilSpirit::check_gwimun(target, b) {
379                        score -= 3.0;
380                        tags.push(TraceTag::Custom(format!(
381                            "흉살:{}({}-{})",
382                            gwimun.hangul(),
383                            target_name,
384                            label
385                        )));
386                    }
387                }
388            }
389        }
390
391        // 2.6 동적 12운성 (Lifecycle Analysis) - Phase 13
392        // 일간이 운에서 만나는 지지에 따라 에너지 상태 변화
393        for (luck_branch, label) in [
394            (
395                dynamic.major_influence.as_ref().map(|i| i.ganzi.branch),
396                "대운",
397            ),
398            (
399                dynamic.yearly_influence.as_ref().map(|i| i.ganzi.branch),
400                "세운",
401            ),
402        ] {
403            if let Some(b) = luck_branch {
404                let stage = crate::core::twelve_stages::calculate_twelve_stage(day_master, b);
405
406                // 왕상휴수사(旺相休囚死)와 유사한 에너지 레벨
407                let (stage_score, _description) = match stage {
408                    TwelveStage::Changsheng
409                    | TwelveStage::Guandai
410                    | TwelveStage::Jianlu
411                    | TwelveStage::Diwang => (10.0, "왕성"),
412                    TwelveStage::Shuai | TwelveStage::Bing | TwelveStage::Si | TwelveStage::Mu => {
413                        (-5.0, "쇠퇴")
414                    }
415                    TwelveStage::Jue | TwelveStage::Tai | TwelveStage::Yang => (-2.0, "불안/잉태"),
416                    TwelveStage::Muyu => (-2.0, "목욕/불안"),
417                };
418
419                if stage_score != 0.0 {
420                    score += stage_score;
421                    esil_trace.push_str(&format!(
422                        "lifecycle:{}({}),score:{:.1}; ",
423                        stage.hangul(),
424                        label,
425                        stage_score
426                    ));
427                    tags.push(TraceTag::Custom(format!(
428                        "운성:{}({})",
429                        stage.hangul(),
430                        label
431                    )));
432                }
433            }
434        }
435
436        // 2.7 십신 복합 패턴 (Combinatorial Ten Gods) - Phase 13
437        // 예: 상관견관(Hurting Officer + Direct Officer) -> 위화백단
438        for (luck_stem, label) in [
439            (
440                dynamic.major_influence.as_ref().map(|i| i.ganzi.stem),
441                "대운",
442            ),
443            (
444                dynamic.yearly_influence.as_ref().map(|i| i.ganzi.stem),
445                "세운",
446            ),
447        ] {
448            if let Some(s) = luck_stem {
449                let ten_god = TenGod::from_stems(day_master, s);
450
451                // A. 상관견관 (상관 운 + 원국 정관)
452                if ten_god == TenGod::Shangguan {
453                    // 원국 천간에 정관이 있는지 확인
454                    let has_zhengguan = [
455                        self.natal.year.stem,
456                        self.natal.month.stem,
457                        self.natal.hour.stem,
458                    ]
459                    .iter()
460                    .any(|&hz| TenGod::from_stems(day_master, hz) == TenGod::Zhengguan);
461
462                    if has_zhengguan {
463                        let penalty = 15.0;
464                        score -= penalty;
465                        tags.push(TraceTag::HurtingOfficerMeetsOfficer {
466                            period: LuckPeriod::from_label(label),
467                        });
468                        esil_trace.push_str(&format!(
469                            "pattern:shangguan_gyeongwan,penalty:-{:.1}; ",
470                            penalty
471                        ));
472                    }
473                }
474
475                // B. 식신생재 (식신 운 + 원국 재성)
476                if ten_god == TenGod::Shishen {
477                    let has_wealth = [
478                        self.natal.year.stem,
479                        self.natal.month.stem,
480                        self.natal.hour.stem,
481                    ]
482                    .iter()
483                    .any(|&hz| {
484                        let god = TenGod::from_stems(day_master, hz);
485                        god == TenGod::Zhengcai || god == TenGod::Piancai
486                    });
487
488                    if has_wealth {
489                        let bonus = 10.0;
490                        score += bonus;
491                        tags.push(TraceTag::EatingGodProducesWealth {
492                            period: LuckPeriod::from_label(label),
493                        });
494                        esil_trace
495                            .push_str(&format!("pattern:shishen_saengjae,bonus:{:.1}; ", bonus));
496                    }
497                }
498            }
499        }
500
501        // 3. 지지 합충 분석 (Structural Stability)
502        for (clash, p1, p2) in &dynamic.combined_relations.branch_clashes {
503            if p1.contains("세운")
504                || p2.contains("세운")
505                || p1.contains("대운")
506                || p2.contains("대운")
507            {
508                let b1 = self.get_branch_by_path(p1, dynamic);
509                let b2 = self.get_branch_by_path(p2, dynamic);
510
511                if let (Some(b1_obj), Some(b2_obj)) = (b1, b2) {
512                    let e1 = b1_obj.element();
513                    let e2 = b2_obj.element();
514
515                    let p1_priority = self.get_element_priority(
516                        e1,
517                        primary_yongshin,
518                        assistant_yongshin,
519                        thermal_index,
520                    );
521                    let p2_priority = self.get_element_priority(
522                        e2,
523                        primary_yongshin,
524                        assistant_yongshin,
525                        thermal_index,
526                    );
527
528                    // 정통 명리학: 희신(좋은 오행)을 충하면 흉, 기신(나쁜 오행)을 충하면 길
529                    // 가중치 합산: 우선순위가 높을수록(용신) 충돌 시 감점이 큼
530                    let total_impact = p1_priority + p2_priority;
531                    let score_change = if total_impact > 1.0 {
532                        // 희신/용신 충돌 (감점)
533                        self.config.vm.clash_bad
534                    } else if total_impact < -0.5 {
535                        // 기신/구신 충돌 (가점 - 개고 효과 등)
536                        self.config.vm.clash_good
537                    } else {
538                        // 일반적인 충돌 (약한 감점)
539                        self.config.vm.clash_default
540                    };
541
542                    score += score_change;
543                    esil_trace
544                        .push_str(&format!("clash:{}-{},impact:{:.1}; ", p1, p2, score_change));
545                }
546
547                // 지장간 메모리 덤프 (Memory Dump of Hidden Stems)
548                // 충돌이 발생한 지지의 지장간 기운을 해방하여 시스템에 반영
549                let b1_obj = self.get_branch_by_path(p1, dynamic);
550                let b2_obj = self.get_branch_by_path(p2, dynamic);
551
552                for b in [b1_obj, b2_obj].iter().flatten() {
553                    let hidden_stems = b.jijanggan();
554                    for stem in hidden_stems {
555                        let el = stem.element();
556                        let weight = self.get_element_priority(
557                            el,
558                            primary_yongshin,
559                            assistant_yongshin,
560                            thermal_index,
561                        );
562
563                        // 지장간 기운은 겉으로 드러난 기운보다 영향력은 작지만(가중치 0.3)
564                        // 충에 의해 해방될 때 시스템에 변수를 제공함
565                        let bonus = 10.0 * self.config.vm.memory_dump_weight * weight;
566                        score += bonus;
567                        registers.update(el, bonus);
568
569                        if weight.abs() > 0.1 {
570                            esil_trace.push_str(&format!(
571                                "mem_dump:{}({}),bonus:{:.1}; ",
572                                b.hangul(),
573                                stem.hangul(),
574                                bonus
575                            ));
576                        }
577                    }
578                }
579                tags.push(TraceTag::BranchClash {
580                    clash_type: clash.hangul().to_string(),
581                });
582            }
583        }
584
585        // 4. 지지 형 (刑) - Memory Corruption
586        for (pun, p1, p2) in &dynamic.combined_relations.branch_punishments {
587            let penalty = 5.0;
588            score -= penalty;
589            esil_trace.push_str(&format!(
590                "mem_corrupt:{}-{},penalty:-{:.1}; ",
591                p1, p2, penalty
592            ));
593            tags.push(TraceTag::Punishment {
594                punishment_type: pun.hangul().to_string(),
595            });
596        }
597
598        // 5. 지지 해 (害) - Race Condition
599        for (harm, p1, p2) in &dynamic.combined_relations.branch_harms {
600            let penalty = 3.0;
601            score -= penalty;
602            esil_trace.push_str(&format!(
603                "race_cond:{}-{},penalty:-{:.1}; ",
604                p1, p2, penalty
605            ));
606            tags.push(TraceTag::Harm {
607                harm_type: harm.hangul().to_string(),
608            });
609        }
610
611        // 6. 지지 파 (破) - IO Error
612        for (dest, p1, p2) in &dynamic.combined_relations.branch_destructions {
613            let penalty = 2.0;
614            score -= penalty;
615            esil_trace.push_str(&format!("io_error:{}-{},penalty:-{:.1}; ", p1, p2, penalty));
616            tags.push(TraceTag::Destruction {
617                destruction_type: dest.hangul().to_string(),
618            });
619        }
620
621        // 6.5 육합 (Six Combinations) - Stable Connection
622        for (six, p1, p2) in &dynamic.combined_relations.six_combinations {
623            // 원국 내의 육합은 이미 정적 분석에 포함됨, 여기서는 운과의 결합을 중시
624            if p1.contains("운") || p2.contains("운") {
625                let bonus = 8.0;
626                score += bonus;
627
628                // 합화 오행 기운 상승
629                let element = six.transformed_element(); // 육합의 화기(예: 자축합토)
630                if let Some(transformed_el) = element {
631                    let weight = self.get_element_priority(
632                        transformed_el,
633                        primary_yongshin,
634                        assistant_yongshin,
635                        thermal_index,
636                    );
637                    score += weight * 3.0;
638                    registers.update(transformed_el, weight * 3.0);
639                }
640
641                esil_trace.push_str(&format!("six_combo:{}-{},bonus:{:.1}; ", p1, p2, bonus));
642                tags.push(TraceTag::SixCombination {
643                    combo_type: six.hangul().to_string(),
644                });
645            }
646        }
647
648        // 6.6 천간충 (Stem Clashes) - Mental Stress
649        for (clash, p1, p2) in &dynamic.combined_relations.stem_clashes {
650            let penalty = 5.0;
651            score -= penalty;
652            esil_trace.push_str(&format!(
653                "stem_clash:{}-{},penalty:-{:.1}; ",
654                p1, p2, penalty
655            ));
656            tags.push(TraceTag::StemClash {
657                clash_type: clash.hangul().to_string(),
658            });
659        }
660
661        // 7. 동적 회합(Triple/Seasonal) 완성 분석 - Dynamic Combination Completion
662        // 원국에 없던 삼합/방합이 대운/세운에 의해 완성될 때 강력한 시너지 보너스 부여
663        self.evaluate_dynamic_combinations(dynamic, tags, esil_trace, registers, &mut score);
664
665        // 8. 신강/신약 태깅
666        let strength = self.natal.strength();
667        let strength_name = match strength.strength_type {
668            crate::analysis::strength::StrengthType::Strong => StrengthTypeName::Strong,
669            crate::analysis::strength::StrengthType::Weak => StrengthTypeName::Weak,
670            crate::analysis::strength::StrengthType::Balanced => StrengthTypeName::Balanced,
671        };
672        tags.push(TraceTag::StrengthType {
673            strength: strength_name,
674        });
675        if strength.deuk_ryeong.acquired {
676            tags.push(TraceTag::DeukRyeong);
677        }
678        if strength.deuk_ji.acquired {
679            tags.push(TraceTag::DeukJi);
680        }
681        if strength.deuk_si.acquired {
682            tags.push(TraceTag::DeukSi);
683        }
684        if strength.deuk_se.acquired {
685            tags.push(TraceTag::DeukSe);
686        }
687
688        self.apply_dynamic_transformation(registers, esil_trace);
689
690        let mut final_score = score;
691        if final_score > 100.0 {
692            final_score = 100.0;
693        } else if final_score < 15.0 {
694            // Asymptotic decay below 15.0 to prevent hard 0.0 clamp
695            final_score = 15.0 * ((final_score - 15.0) / 15.0).exp();
696        }
697
698        final_score
699    }

701    /// 동적 오행 변환 (에너지 보존 법칙)
702    fn apply_dynamic_transformation(&self, registers: &mut QiRegisters, esil_trace: &mut String) {
703        // 생(生) 상생 사이클 시뮬레이션
704        // Wood -> Fire -> Earth -> Metal -> Water -> Wood
705        // 소모되는 에너지는 줄이고 생성되는 에너지는 증가 (보존)
706
707        let transfer_rate = 0.1; // 10% 에너지 전이
708
709        let wood_transfer = registers.r0_wood * transfer_rate;
710        let fire_transfer = registers.r1_fire * transfer_rate;
711        let earth_transfer = registers.r2_earth * transfer_rate;
712        let metal_transfer = registers.r3_metal * transfer_rate;
713        let water_transfer = registers.r4_water * transfer_rate;
714
715        // Wood -> Fire
716        registers.r0_wood -= wood_transfer;
717        registers.r1_fire += wood_transfer;
718
719        // Fire -> Earth
720        registers.r1_fire -= fire_transfer;
721        registers.r2_earth += fire_transfer;
722
723        // Earth -> Metal
724        registers.r2_earth -= earth_transfer;
725        registers.r3_metal += earth_transfer;
726
727        // Metal -> Water
728        registers.r3_metal -= metal_transfer;
729        registers.r4_water += metal_transfer;
730
731        // Water -> Wood
732        registers.r4_water -= water_transfer;
733        registers.r0_wood += water_transfer;
734
735        registers.normalize();
736        esil_trace.push_str("qi_conservation:applied; ");
737    }

739    /// 인터럽트 벡터 테이블(IVT) 및 핸들링 로직
740    fn handle_interrupts(
741        &self,
742        dynamic: &DynamicLuckAnalysis,
743        tags: &mut Vec<TraceTag>,
744        esil_trace: &mut String,
745        registers: &mut QiRegisters,
746        score: &mut f32,
747    ) {
748        // 현재 활성화된 신살 스캔 (원국 + 운의 상호작용)
749        // 100년 전수 조사를 위해 가벼운 체크 로직 적용
750
751        let yearly_ganzi = dynamic.yearly_influence.as_ref().map(|i| i.ganzi);
752
753        if let Some(y) = yearly_ganzi {

755            // 지지 충돌 시 백호대살이 겹치면 '심각한 하드웨어 예외' 발생
756            let el = y.stem.element();
757            let thermal_index =
758                crate::analysis::yongshin::calculate_thermal_index(&self.natal, &self.config);
759            let p = self.get_element_priority(

766            if crate::analysis::spirit_markers::SpiritMarkerAnalysis::is_baihu(y) {
767                let mut irq = SajuInterrupt::CriticalException;
768                if p > 0.5 {
769                    irq = SajuInterrupt::ServiceInterrupt; // 용신 제화(완화)
770                    esil_trace.push_str("irq_mitigate:yongshin_protect; ");
771                }
772                self.trigger_irq(irq, "백호대살", tags, esil_trace, registers, score);
773            }
774
775            // 2. 리소스 오버플로우: 괴강살 (Kuigang)
776            if crate::analysis::spirit_markers::SpiritMarkerAnalysis::is_kuigang(y) {
777                let mut irq = SajuInterrupt::ResourceOverflow;
778                if p > 0.5 {
779                    irq = SajuInterrupt::ServiceInterrupt; // 용신 제화(완화)
780                    esil_trace.push_str("irq_mitigate:yongshin_protect; ");
781                }
782                self.trigger_irq(irq, "괴강특수운", tags, esil_trace, registers, score);
783            }
784        }

787    /// 인터럽트 발생 (Handler Execution)
788    fn trigger_irq(
789        &self,
790        irq: SajuInterrupt,
791        marker_name: &str,
792        tags: &mut Vec<TraceTag>,
793        esil_trace: &mut String,
794        registers: &mut QiRegisters,
795        score: &mut f32,
796    ) {
797        let (penalty, irq_type) = match irq {
798            SajuInterrupt::CriticalException => (
799                self.config.vm.irq_critical,
800                InterruptType::CriticalException,
801            ),
802            SajuInterrupt::ResourceOverflow => {
803                (self.config.vm.irq_overflow, InterruptType::ResourceOverflow)
804            }
805            SajuInterrupt::SystemStall => (self.config.vm.irq_stall, InterruptType::SystemStall),
806            SajuInterrupt::ServiceInterrupt => {
807                (self.config.vm.irq_service, InterruptType::ServiceInterrupt)
808            }
809        };
810
811        *score -= penalty;
812        esil_trace.push_str(&format!("irq_handle:{:?},impact:-{:.1}; ", irq, penalty));
813        tags.push(TraceTag::Interrupt {
814            irq_type,
815            marker: marker_name.to_string(),
816        });
817
818        // 인터럽트 발생 시 특정 레지스터 강제 변조 (Kernel Panic 효과)
819        if matches!(irq, SajuInterrupt::CriticalException) {
820            registers.r2_earth += self.config.vm.irq_critical; // 토(Earth) 기운 급증으로 인한 시스템 정체
821            esil_trace.push_str("kernel_panic:earth_overflow; ");
822        }
823    }

825    /// 파이프라인 명령어 실행 루프
826    fn execute_pipeline(
827        &self,
828        dynamic: &DynamicLuckAnalysis,
829        tags: &mut Vec<TraceTag>,
830        esil_trace: &mut String,
831        registers: &mut QiRegisters,
832        score: &mut f32,
833    ) {
834        let primary_yongshin = self.yongshin.primary;
835        let assistant_yongshin = self.yongshin.assistant;
836        let thermal_index =
837            crate::analysis::yongshin::calculate_thermal_index(&self.natal, &self.config);
838
839        // 스테이지 정의
840        let stages = [
841            (
842                dynamic.major_influence.as_ref().map(|i| i.ganzi),
843                "대운",
844                self.config.vm.pipeline_major,
845            ),
846            (
847                dynamic.yearly_influence.as_ref().map(|i| i.ganzi),
848                "세운",
849                self.config.vm.pipeline_yearly,
850            ),
851            (
852                dynamic.monthly_influence.as_ref().map(|i| i.ganzi),
853                "월운",
854                self.config.vm.pipeline_monthly,
855            ),
856            (
857                dynamic.daily_influence.as_ref().map(|i| i.ganzi),
858                "일운",
859                self.config.vm.pipeline_daily,
860            ),
861            (
862                dynamic.hourly_influence.as_ref().map(|i| i.ganzi),
863                "시운",
864                self.config.vm.pipeline_hourly,
865            ),
866        ];
867
868        let mut prev_el: Option<Element> = None;
869        let mut forwarding_buffer: Option<Element> = None;
870
871        for (opt_ganzi, label, base_weight) in stages {
872            if let Some(ganzi) = opt_ganzi {
873                // 천간 합에 의한 변환된 오행 확인 (Forwarding 체크)
874                let stem_label = format!("{}천간", label);
875                let eff_el = self
876                    .get_effective_stem(dynamic, &stem_label)
877                    .unwrap_or(ganzi.stem.element());
878
879                let mut weight = self.get_element_priority(
880                    eff_el,
881                    primary_yongshin,
882                    assistant_yongshin,
883                    thermal_index,
884                );
885
886                // 1. Data Hazard (Stall): 앞선 단계와 현재 단계가 상극일 경우
887                if let Some(prev) = prev_el {
888                    if prev.controls() == eff_el {
889                        let stall_penalty = self.config.vm.stall_penalty;
890                        weight *= stall_penalty;
891                        esil_trace.push_str(&format!(
892                            "pipeline_stall:{}_by_{}; ",
893                            label,
894                            prev.hangul()
895                        ));
896                    }
897                }
898
899                // 2. Bypassing (Forwarding): 합에 의해 생성된 기운이 용신일 경우 보너스 가중치
900                if let Some(fwd) = forwarding_buffer {
901                    if fwd == eff_el || fwd.generates() == eff_el {
902                        let fwd_bonus = self.config.vm.forwarding_bonus;
903                        weight *= fwd_bonus;
904                        esil_trace.push_str(&format!(
905                            "pipeline_forwarding:{}_aided_by_{}; ",
906                            label,
907                            fwd.hangul()
908                        ));
909                    }
910                }
911
912                let increment = base_weight * weight;
913                *score += increment;
914
915                // VM 레지스터 업데이트 전 감쇠 및 후 정규화
916                registers.apply_decay(self.config.vm.decay_factor);
917                registers.update(eff_el, increment);
918                registers.normalize();
919
920                esil_trace.push_str(&format!(
921                    "{}_infl:{},weight:{:.1},score+={:.1}; ",
922                    label.to_lowercase(),
923                    eff_el.hangul(),
924                    weight,
925                    increment
926                ));
927
928                // 태깅
929                if label == "대운" || label == "세운" {
930                    let period = LuckPeriod::from_label(label);
931                    if weight >= 1.5 {
932                        tags.push(TraceTag::CoreLuck { period });
933                    } else if weight >= 1.0 {
934                        tags.push(TraceTag::GoodLuck { period });
935                    } else if weight <= -0.5 {
936                        tags.push(TraceTag::BadLuck { period });
937                    }
938                }
939
940                // 다음 단계를 위해 현재 기운 저장
941                prev_el = Some(eff_el);
942
943                // 합화된 기운이 있다면 포워딩 버퍼에 저장
944                if self.get_effective_stem(dynamic, &stem_label).is_some() {
945                    forwarding_buffer = Some(eff_el);
946                } else {
947                    forwarding_buffer = None;
948                }
949            }
950        }
951    }

954    fn get_effective_stem(&self, dynamic: &DynamicLuckAnalysis, target: &str) -> Option<Element> {
955        for (combo, p1, p2) in &dynamic.combined_relations.stem_combinations {
956            if p1 == target || p2 == target {
957                let transformed = combo.transformed_element();
958                let month_el = self.natal.month.branch.element();
959                if month_el == transformed || month_el.generates() == transformed {
960                    return Some(transformed);
961                }
962            }
963        }
964        None
965    }

967    /// 오행별 우선순위 가중치 (Thermal + Yongshin Matrix)
968    pub fn get_element_priority(
969        &self,
970        el: Element,
971        primary: Element,
972        assistant: Element,
973        thermal: i32,
974    ) -> f32 {
975        let mut multiplier = 0.0;
976
977        if el == primary {
978            multiplier += 1.0;
979        } else if el == assistant {
980            multiplier += 0.5;
981        } else if el == primary.controlled_by() {
982            multiplier -= 0.8;
983        }
984
985        if thermal <= -30 && el == Element::Fire {
986            multiplier *= 2.0;
987            if multiplier == 0.0 {
988                multiplier = 1.2;
989            }
990        }
991        if thermal >= 30 && el == Element::Water {
992            multiplier *= 2.0;
993            if multiplier == 0.0 {
994                multiplier = 1.2;
995            }
996        }
997
998        multiplier
999    }

1021    /// 동적 회합(삼합/방합) 완성 평가
1022    fn evaluate_dynamic_combinations(
1023        &self,
1024        dynamic: &DynamicLuckAnalysis,
1025        tags: &mut Vec<TraceTag>,
1026        esil_trace: &mut String,
1027        registers: &mut QiRegisters,
1028        score: &mut f32,
1029    ) {
1030        let natal_triples = &dynamic.natal_relations.triple_combinations;
1031        let natal_seasonals = &dynamic.natal_relations.seasonal_combinations;
1032
1033        // 1. 삼합 완성 체크
1034        for triple in &dynamic.combined_relations.triple_combinations {
1035            if !natal_triples.contains(triple) {
1036                // 원국에는 없던 삼합이 완성됨!
1037                let element = triple.element();
1038                let priority = self.get_element_priority(
1039                    element,
1040                    self.yongshin.primary,
1041                    self.yongshin.assistant,
1042                    0,
1043                );
1044
1045                let bonus = 15.0 * priority;
1046                *score += bonus;
1047                registers.update(element, bonus.abs());
1048
1049                let is_beneficial = priority > 0.0;
1050                tags.push(TraceTag::TripleCombination {
1051                    element,
1052                    is_beneficial,
1053                });
1054                esil_trace.push_str(&format!(
1055                    "dynamic_triple:{},bonus:{:.1}; ",
1056                    triple.hangul(),
1057                    bonus
1058                ));
1059            }
1060        }
1061
1062        // 2. 방합 완성 체크
1063        for seasonal in &dynamic.combined_relations.seasonal_combinations {
1064            if !natal_seasonals.contains(seasonal) {
1065                let element = seasonal.element();
1066                let priority = self.get_element_priority(
1067                    element,
1068                    self.yongshin.primary,
1069                    self.yongshin.assistant,
1070                    0,
1071                );
1072
1073                let bonus = 20.0 * priority; // 방합은 세력이 더 강력함
1074                *score += bonus;
1075                registers.update(element, bonus.abs());
1076
1077                let is_beneficial = priority > 0.0;
1078                tags.push(TraceTag::SeasonalCombination {
1079                    element,
1080                    is_beneficial,
1081                });
1082                esil_trace.push_str(&format!(
1083                    "dynamic_seasonal:{},bonus:{:.1}; ",
1084                    seasonal.hangul(),
1085                    bonus
1086                ));
1087            }
1088        }
1089    }

1099            self.natal.raw_input.day,
1100            self.natal.raw_input.hour,
1101            self.natal.raw_input.minute,
1102        )
1103        .ok();
1104
1105        // 출생 연도 (세운 계산 기준)
1106        let birth_year = self.natal.raw_input.year;
1107
1108        let mapper = |age: u32| {
1109            // 2. 해당 나이의 대운 간지 추출
1110            let major_ganzi = luck_analysis
1111                .as_ref()
1112                .and_then(|a| a.at_age(age))
1113                .map(|m| m.ganzi)
1114                .unwrap_or_else(|| GanZi::from_index(0));
1115
1116            // 3. 해당 나이의 세운 간지 추출
1117            let target_year = birth_year + (age as i32) - 1;
1118            let yearly_ganzi = YearlyLuck::calculate(target_year, &self.natal).ganzi;
1119
1120            // 4. 정밀한 step 수행
1121            self.step(age, major_ganzi, yearly_ganzi, None, None, None)
1122        };
1123
1124        #[cfg(feature = "parallel")]
1125        {
1126            use rayon::prelude::*;
1127            (start_age..=end_age).into_par_iter().map(mapper).collect()
1128        }
1129
1130        #[cfg(not(feature = "parallel"))]
1131        {
1132            (start_age..=end_age).map(mapper).collect()
1133        }
1134    }
