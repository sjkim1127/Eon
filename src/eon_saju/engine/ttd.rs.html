25    /// Destiny Backtrace: 특정 시점의 태그/상태의 근본 원인을 역방향 탐색
26    pub fn backtrace(report: &LifePathReport, age: u32, target_tag: &str) -> Option<RootCause> {
27        let frames = &report.frames;
28        if age as usize >= frames.len() {
29            return None;
30        }
31
32        // 현재 시점에 해당 태그가 없으면 분석 불가
33        let current_frame = &frames[age as usize];
34        let exists = current_frame
35            .tags
36            .iter()
37            .any(|t| t.contains_pattern(target_tag))
38            || current_frame
39                .signatures
40                .iter()
41                .any(|s| s.name.contains(target_tag) || s.id.contains(target_tag));
42
43        if !exists {
44            return None;
45        }
46
47        let mut root_cause_age = age;
48        let mut current_age = age as i32;
49
50        // 역방향으로 탐색하며 해당 태그가 처음 나타난 시점(Entry Point)을 찾음
51        while current_age >= 0 {
52            let frame = &frames[current_age as usize];
53            let has_tag = frame.tags.iter().any(|t| t.contains_pattern(target_tag))
54                || frame
55                    .signatures
56                    .iter()
57                    .any(|s| s.name.contains(target_tag) || s.id.contains(target_tag));
58
59            if has_tag {
60                root_cause_age = current_age as u32;
61            } else {
62                // 태그가 더 이상 발견되지 않으면 직전 시점이 진입점
63                break;
64            }
65            current_age -= 1;
66        }
67
68        // 해당 시점의 대운(Major Luck)이 바뀐 시점인지 확인 (보통 대운이 근본 원인인 경우가 많음)
69        let mut major_change_age = root_cause_age;
70        let root_major = frames[root_cause_age as usize].major_ganzi;
71        let mut check_age = root_cause_age as i32;
72        while check_age >= 0 {
73            if frames[check_age as usize].major_ganzi != root_major {
74                major_change_age = (check_age + 1) as u32;
75                break;
76            }
77            check_age -= 1;
78        }
79
80        Some(RootCause {
81            target_age: age,
82            root_cause_age: major_change_age.min(root_cause_age),
83            reason: format!(
84                "'{}' 현상이 포착된 지점({}세)으로부터 역추적한 결과, {}세 기점으로 로직이 변동되었습니다.", 
85                target_tag, root_cause_age, major_change_age.min(root_cause_age)
86            ),
87        })
88    }

90    /// State Snapshoting & Diffing: 두 인생 경로의 차이 분석
91    pub fn diff(report_a: &LifePathReport, report_b: &LifePathReport) -> Vec<LifeDiff> {
92        let mut diffs = Vec::new();
93        let max_len = report_a.frames.len().min(report_b.frames.len());
94
95        for i in 0..max_len {
96            let fa = &report_a.frames[i];
97            let fb = &report_b.frames[i];
98
99            let score_delta = fb.score - fa.score;
100
101            let mut added_tags = Vec::new();
102            for tag in &fb.tags {
103                if !fa.tags.iter().any(|t| t == tag) {
104                    added_tags.push(tag.to_string());
105                }
106            }
107
108            let mut removed_tags = Vec::new();
109            for tag in &fa.tags {
110                if !fb.tags.iter().any(|t| t == tag) {
111                    removed_tags.push(tag.to_string());
112                }
113            }
114
115            // 유의미한 차이가 있는 경우에만 기록 (델타 분석)
116            if score_delta.abs() > 3.0 || !added_tags.is_empty() || !removed_tags.is_empty() {
117                diffs.push(LifeDiff {
118                    age: fa.age,
119                    score_delta,
120                    added_tags,
121                    removed_tags,
122                });
123            }
124        }
125
126        diffs
127    }

129    /// Time-Travel Breakpoints: 특정 조건을 만족하는 시점들 탐색
130    pub fn find_breakpoints<F>(report: &LifePathReport, condition: F) -> Vec<u32>
131    where
132        F: Fn(&LifeFrame) -> bool,
133    {
134        report
135            .frames
136            .iter()
137            .filter(|f| condition(f))
138            .map(|f| f.age)
139            .collect()
140    }
