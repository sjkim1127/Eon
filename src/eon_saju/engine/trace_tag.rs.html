10/// 분석 태그 - 문자열 대신 구조화된 Enum 사용
11#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
12pub enum TraceTag {
13    // === 운성 태그 (대운/세운) ===
14    /// 핵심운 (용신 적중)
15    CoreLuck { period: LuckPeriod },
16    /// 길운
17    GoodLuck { period: LuckPeriod },
18    /// 기신운
19    BadLuck { period: LuckPeriod },
20
21    // === 합충 태그 ===
22    /// 지지 충
23    BranchClash { clash_type: String },
24    /// 천간 충
25    StemClash { clash_type: String },
26    /// 육합
27    SixCombination { combo_type: String },
28    /// 삼합 완성
29    TripleCombination { element: Element, is_beneficial: bool },
30    /// 방합 완성
31    SeasonalCombination { element: Element, is_beneficial: bool },
32    /// 합화 (오행 변화)
33    Transformation { from: Element, to: Element },
34
35    // === 지지 관계 태그 ===
36    /// 형 (刑)
37    Punishment { punishment_type: String },
38    /// 해 (害)
39    Harm { harm_type: String },
40    /// 파 (破)
41    Destruction { destruction_type: String },
42
43    // === 공망 태그 ===
44    /// 운성 공망 (진공)
45    VoidLuck { period: LuckPeriod },
46    /// 탈공 (충에 의한)
47    EscapedVoidClash { period: LuckPeriod },
48    /// 탈공 (육합에 의한)
49    EscapedVoidSixCombo { period: LuckPeriod },
50    /// 탈공 (삼합에 의한)
51    EscapedVoidTriple { period: LuckPeriod },
52    /// 탈공 (방합에 의한)
53    EscapedVoidSeasonal { period: LuckPeriod },
54
55    // === 신살 태그 ===
56    /// 12신살
57    TwelveShinsal { name: ShinsalName, period: LuckPeriod },
58    /// 길신 (천을귀인, 문창귀인 등)
59    AuspiciousSpirit { name: AuspiciousSpiritName, period: LuckPeriod },
60    /// 흉살 (원진, 귀문 등)
61    InauspiciousSpirit { name: InauspiciousSpiritName, position: PillarPosition, period: LuckPeriod },
62
63    // === 12운성 태그 ===
64    /// 12운성 (장생, 건록, 제왕 등)
65    LifeStage { stage: LifeStageName, period: LuckPeriod },
66
67    // === 십신 패턴 태그 ===
68    /// 상관견관 (상관 운 + 원국 정관)
69    HurtingOfficerMeetsOfficer { period: LuckPeriod },
70    /// 식신생재 (식신 운 + 원국 재성)
71    EatingGodProducesWealth { period: LuckPeriod },
72
73    // === 신강/신약 태그 ===
74    /// 신강신약 타입
75    StrengthType { strength: StrengthTypeName },
76    /// 득령
77    DeukRyeong,
78    /// 득지
79    DeukJi,
80    /// 득시
81    DeukSi,
82    /// 득세
83    DeukSe,
84
85    // === 인터럽트 태그 ===
86    /// 하드웨어 인터럽트 (백호살, 괴강살 등)
87    Interrupt { irq_type: InterruptType, marker: String },
88
89    // === 기타 ===
90    /// 커스텀 태그 (레거시 호환용)
91    Custom(String),
92}

94/// 운의 종류 (대운, 세운, 월운 등)
95#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
96pub enum LuckPeriod {
97    Major,   // 대운
98    Yearly,  // 세운
99    Monthly, // 월운
100    Daily,   // 일운
101    Hourly,  // 시운
102}
103
104impl LuckPeriod {
105    pub const fn hangul(&self) -> &'static str {
106        match self {
107            LuckPeriod::Major => "대운",
108            LuckPeriod::Yearly => "세운",
109            LuckPeriod::Monthly => "월운",
110            LuckPeriod::Daily => "일운",
111            LuckPeriod::Hourly => "시운",
112        }
113    }
114
115    /// 문자열로부터 LuckPeriod 변환
116    pub fn from_label(label: &str) -> Self {
117        match label {
118            "대운" => LuckPeriod::Major,
119            "세운" => LuckPeriod::Yearly,
120            "월운" => LuckPeriod::Monthly,
121            "일운" => LuckPeriod::Daily,
122            "시운" => LuckPeriod::Hourly,
123            _ => LuckPeriod::Yearly, // fallback
124        }
125    }
126}

280/// 인터럽트 타입
281#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
282pub enum InterruptType {
283    CriticalException, // 0x01: 백호살 등
284    ResourceOverflow,  // 0x02: 괴강살 등
285    SystemStall,       // 0x03: 고신/과숙 등
286    ServiceInterrupt,  // 0x04: 망신/겁살 등
287}
288
289impl InterruptType {
290    pub const fn code(&self) -> &'static str {
291        match self {
292            InterruptType::CriticalException => "IRQ_0x01",
293            InterruptType::ResourceOverflow => "IRQ_0x02",
294            InterruptType::SystemStall => "IRQ_0x03",
295            InterruptType::ServiceInterrupt => "IRQ_0x04",
296        }
297    }
298}

300impl fmt::Display for TraceTag {
301    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
302        match self {
303            // 운성 태그
304            TraceTag::CoreLuck { period } => write!(f, "{}핵심운", period.hangul()),
305            TraceTag::GoodLuck { period } => write!(f, "{}길운", period.hangul()),
306            TraceTag::BadLuck { period } => write!(f, "{}기신운", period.hangul()),
307            
308            // 합충 태그
309            TraceTag::BranchClash { clash_type } => write!(f, "{}", clash_type),
310            TraceTag::StemClash { clash_type } => write!(f, "천간충:{}", clash_type),
311            TraceTag::SixCombination { combo_type } => write!(f, "육합:{}", combo_type),
312            TraceTag::TripleCombination { element, is_beneficial } => {
313                let suffix = if *is_beneficial { "吉" } else { "凶" };
314                write!(f, "삼합완성({}-{})", element.hangul(), suffix)
315            }
316            TraceTag::SeasonalCombination { element, is_beneficial } => {
317                let suffix = if *is_beneficial { "吉" } else { "凶" };
318                write!(f, "방합완성({}-{})", element.hangul(), suffix)
319            }
320            TraceTag::Transformation { from, to } => {
321                write!(f, "합화:{}→{}", from.hangul(), to.hangul())
322            }
323            
324            // 지지 관계
325            TraceTag::Punishment { punishment_type } => write!(f, "{}", punishment_type),
326            TraceTag::Harm { harm_type } => write!(f, "{}", harm_type),
327            TraceTag::Destruction { destruction_type } => write!(f, "{}", destruction_type),
328            
329            // 공망
330            TraceTag::VoidLuck { period } => write!(f, "운성공망:{}", period.hangul()),
331            TraceTag::EscapedVoidClash { period } => write!(f, "탈공:충({})", period.hangul()),
332            TraceTag::EscapedVoidSixCombo { period } => write!(f, "탈공:육합({})", period.hangul()),
333            TraceTag::EscapedVoidTriple { period } => write!(f, "탈공:삼합({})", period.hangul()),
334            TraceTag::EscapedVoidSeasonal { period } => write!(f, "탈공:방합({})", period.hangul()),
335            
336            // 신살
337            TraceTag::TwelveShinsal { name, period } => {
338                write!(f, "신살:{}({})", name.hangul(), period.hangul())
339            }
340            TraceTag::AuspiciousSpirit { name, period } => {
341                write!(f, "길신:{}({})", name.hangul(), period.hangul())
342            }
343            TraceTag::InauspiciousSpirit { name, position, period } => {
344                write!(f, "흉살:{}({}-{})", name.hangul(), position.hangul(), period.hangul())
345            }
346            
347            // 12운성
348            TraceTag::LifeStage { stage, period } => {
349                write!(f, "운성:{}({})", stage.hangul(), period.hangul())
350            }
351            
352            // 십신 패턴
353            TraceTag::HurtingOfficerMeetsOfficer { period } => {
354                write!(f, "패턴:상관견관({})", period.hangul())
355            }
356            TraceTag::EatingGodProducesWealth { period } => {
357                write!(f, "패턴:식신생재({})", period.hangul())
358            }
359            
360            // 신강/신약
361            TraceTag::StrengthType { strength } => write!(f, "신강약:{}", strength.hangul()),
362            TraceTag::DeukRyeong => write!(f, "득령"),
363            TraceTag::DeukJi => write!(f, "득지"),
364            TraceTag::DeukSi => write!(f, "득시"),
365            TraceTag::DeukSe => write!(f, "득세"),
366            
367            // 인터럽트
368            TraceTag::Interrupt { irq_type, marker } => {
369                write!(f, "{}:{}", irq_type.code(), marker)
370            }
371            
372            // 커스텀
373            TraceTag::Custom(s) => write!(f, "{}", s),
374        }
375    }
376}

378impl TraceTag {
379    /// 문자열 패턴이 이 태그에 포함되는지 확인 (레거시 contains() 호환용)
380    pub fn contains_pattern(&self, pattern: &str) -> bool {
381        self.to_string().contains(pattern)
382    }
383}
384
385/// TraceTag 컬렉션을 Vec<String>으로 변환 (레거시 호환용)
386pub fn tags_to_strings(tags: &[TraceTag]) -> Vec<String> {
387    tags.iter().map(|t| t.to_string()).collect()
388}
389
390/// Vec<String>에서 TraceTag 컬렉션으로 변환 (레거시 호환용)
391pub fn strings_to_tags(strings: &[String]) -> Vec<TraceTag> {
392    strings.iter().map(|s| TraceTag::Custom(s.clone())).collect()
393}

414            TraceTag::DeukRyeong,

416            TraceTag::StrengthType { strength: StrengthTypeName::Strong },
417        ];
418        let strings = tags_to_strings(&tags);
419        assert_eq!(strings, vec!["득령", "득지", "신강약:신강"]);
