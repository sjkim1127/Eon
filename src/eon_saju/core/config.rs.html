101/// VM 시뮬레이션 파라미터
102///
103/// SajuVM이 인생 시뮬레이션을 실행할 때 사용하는 가중치와 페널티입니다.
104/// ML 기반 최적화나 사용자 커스터마이징을 위해 외부에서 조정 가능합니다.
105///
106/// ## 튜닝 가이드
107///
108/// ### 기본 점수 (`base_score`)
109/// - 기본값: 50.0
110/// - 시뮬레이션 시작 점수. 중립적인 운세 상태를 나타냅니다.
111///
112/// ### 충돌 점수 (`clash_*`)
113/// - `clash_bad`: 희신/용신 충돌 시 감점 (기본: -20.0)
114/// - `clash_good`: 기신/구신 충돌 시 가점 (기본: +10.0, 개고 효과)
115/// - `clash_default`: 일반적인 충돌 (기본: -5.0)
116///
117/// ### 파이프라인 가중치 (`pipeline_*`)
118/// - 각 운(대운, 세운, 월운, 일운, 시운)이 점수에 미치는 기본 영향력
119/// - 대운이 가장 크고, 세운, 월운, 일운, 시운 순으로 감소
120///
121/// ### 인터럽트 페널티 (`irq_*`)
122/// - 흉살(백호살, 괴강살 등)에 의한 시스템 인터럽트 발생 시 감점
123/// - `irq_critical`: 가장 심각한 예외 (백호살 등, 기본: -20.0)
124/// - `irq_overflow`: 리소스 오버플로우 (괴강살 등, 기본: -15.0)
125/// - `irq_stall`: 시스템 정지 (고신/과숙 등, 기본: -10.0)
126/// - `irq_service`: 서비스 중단 (망신/겁살 등, 기본: -7.0)
127///
128/// ### 파이프라인 제어 (`stall_penalty`, `forwarding_bonus`, etc.)
129/// - `stall_penalty`: 앞 단계와 상극 시 효율 감소 배율 (기본: 0.5, 절반)
130/// - `forwarding_bonus`: 합화된 기운이 용신일 때 효율 증가 배율 (기본: 1.2)
131/// - `memory_dump_weight`: 충에 의해 해방된 지장간 기운의 영향력 (기본: 0.3)
132/// - `decay_factor`: 시간 흐름에 따른 기운 감쇠율 (기본: 0.95, 5% 감쇠)
133#[derive(Debug, Clone, Serialize, Deserialize)]
134pub struct VmConfig {
135    /// 시뮬레이션 시작 점수 (0-100 범위의 중앙값)
136    pub base_score: f32,
137    
138    /// 희신/용신 충돌 시 감점 (음수)
139    pub clash_bad: f32,
140    /// 기신/구신 충돌 시 가점 (양수, 개고 효과)
141    pub clash_good: f32,
142    /// 일반적인 충돌 시 감점
143    pub clash_default: f32,
144
145    // 파이프라인 가중치 (각 운의 기본 영향력)
146    /// 대운(大運) 가중치 - 10년 단위 대 흐름
147    pub pipeline_major: f32,
148    /// 세운(歲運) 가중치 - 1년 단위 운세
149    pub pipeline_yearly: f32,
150    /// 월운(月運) 가중치 - 월별 변동
151    pub pipeline_monthly: f32,
152    /// 일운(日運) 가중치 - 일별 변동
153    pub pipeline_daily: f32,
154    /// 시운(時運) 가중치 - 시간별 변동
155    pub pipeline_hourly: f32,
156
157    // 인터럽트 페널티 (흉살에 의한 시스템 인터럽트)
158    /// 0x01: 고우선순위 시스템 충돌 (백호살 등)
159    pub irq_critical: f32,
160    /// 0x02: 리소스 오버플로우 (괴강살 등)
161    pub irq_overflow: f32,
162    /// 0x03: 시스템 정지/고독 (고신/과숙 등)
163    pub irq_stall: f32,
164    /// 0x04: 임시 서비스 중단 (망신/겁살 등)
165    pub irq_service: f32,
166
167    // 파이프라인 제어 파라미터
168    /// 파이프라인 스톨 발생 시 효율 감소 배율 (0.0-1.0)
169    pub stall_penalty: f32,
170    /// 데이터 포워딩(합화) 성공 시 효율 증가 배율 (1.0+)
171    pub forwarding_bonus: f32,
172    /// 충에 의한 지장간 메모리 덤프 영향력 (0.0-1.0)
173    pub memory_dump_weight: f32,
174    /// 시간 흐름에 따른 기운 감쇠율 (0.0-1.0)
175    pub decay_factor: f32,
176}
177

179    fn default() -> Self {
180        Self {
181            base_score: 50.0,
182            clash_bad: -20.0,
183            clash_good: 10.0,
184            clash_default: -5.0,
185
186            pipeline_major: 10.0,
187            pipeline_yearly: 15.0,
188            pipeline_monthly: 5.0,
189            pipeline_daily: 2.0,
190            pipeline_hourly: 1.0,
191
192            irq_critical: 20.0,
193            irq_overflow: 15.0,
194            irq_stall: 10.0,
195            irq_service: 7.0,
196
197            stall_penalty: 0.5,
198            forwarding_bonus: 1.2,
199            memory_dump_weight: 0.3,
200            decay_factor: 0.95,
201        }
202    }
