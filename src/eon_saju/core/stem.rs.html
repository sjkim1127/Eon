1//! 천간(天干, Heavenly Stems) - 10개의 천간
2//!
3//! 甲乙丙丁戊己庚辛壬癸
4
5use crate::core::element::{Element, Polarity};
6use serde::{Deserialize, Serialize};
7
8/// 천간(天干) - 10개
9#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
10#[repr(u8)]
11pub enum HeavenlyStem {
12    /// 甲 (갑) - 양목
13    Jia = 0,
14    /// 乙 (을) - 음목
15    Yi = 1,
16    /// 丙 (병) - 양화
17    Bing = 2,
18    /// 丁 (정) - 음화
19    Ding = 3,
20    /// 戊 (무) - 양토
21    Wu = 4,
22    /// 己 (기) - 음토
23    Ji = 5,
24    /// 庚 (경) - 양금
25    Geng = 6,
26    /// 辛 (신) - 음금
27    Xin = 7,
28    /// 壬 (임) - 양수
29    Ren = 8,
30    /// 癸 (계) - 음수
31    Gui = 9,
32}
33
34impl HeavenlyStem {
35    /// 모든 천간 배열
36    pub const ALL: [HeavenlyStem; 10] = [
37        Self::Jia,
38        Self::Yi,
39        Self::Bing,
40        Self::Ding,
41        Self::Wu,
42        Self::Ji,
43        Self::Geng,
44        Self::Xin,
45        Self::Ren,
46        Self::Gui,
47    ];
48
49    /// 한자 표기
50    pub const HANJA: [&'static str; 10] =
51        ["甲", "乙", "丙", "丁", "戊", "己", "庚", "辛", "壬", "癸"];
52
53    /// 한글 표기
54    pub const HANGUL: [&'static str; 10] =
55        ["갑", "을", "병", "정", "무", "기", "경", "신", "임", "계"];
56
57    /// 인덱스 (0-9)
58    #[inline]
59    pub const fn index(self) -> u8 {
60        self as u8
61    }
62
63    /// 인덱스로부터 천간 생성 (0-9, 그 외는 모듈로 연산)
64    #[inline]
65    pub const fn from_index(idx: i32) -> Self {
66        Self::ALL[idx.rem_euclid(10) as usize]
67    }
68
69    /// 오행 반환
70    #[inline]
71    pub const fn element(self) -> Element {
72        match self {
73            Self::Jia | Self::Yi => Element::Wood,
74            Self::Bing | Self::Ding => Element::Fire,
75            Self::Wu | Self::Ji => Element::Earth,
76            Self::Geng | Self::Xin => Element::Metal,
77            Self::Ren | Self::Gui => Element::Water,
78        }
79    }
80
81    /// 음양 반환
82    #[inline]
83    pub const fn polarity(self) -> Polarity {
84        match self.index() % 2 {
85            0 => Polarity::Yang,
86            _ => Polarity::Yin,
87        }
88    }
89
90    /// 한자 표기
91    #[inline]
92    pub const fn hanja(self) -> &'static str {
93        Self::HANJA[self.index() as usize]
94    }
95
96    /// 한글 표기
97    #[inline]
98    pub const fn hangul(self) -> &'static str {
99        Self::HANGUL[self.index() as usize]
100    }
101
102    /// 다음 천간
103    #[inline]
104    pub const fn next(self) -> Self {
105        Self::from_index(self.index() as i32 + 1)
106    }
107
108    /// 이전 천간
109    #[inline]
110    pub const fn prev(self) -> Self {
111        Self::from_index(self.index() as i32 - 1)
112    }
113}
114
115impl std::fmt::Display for HeavenlyStem {
116    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
117        write!(f, "{}", self.hanja())
118    }
119}

121#[cfg(test)]
122mod tests {
123    use super::*;
124
125    #[test]
126    fn test_stem_index() {
127        assert_eq!(HeavenlyStem::Jia.index(), 0);
128        assert_eq!(HeavenlyStem::Gui.index(), 9);
129    }
130
131    #[test]
132    fn test_stem_from_index() {
133        assert_eq!(HeavenlyStem::from_index(0), HeavenlyStem::Jia);
134        assert_eq!(HeavenlyStem::from_index(10), HeavenlyStem::Jia); // 순환
135        assert_eq!(HeavenlyStem::from_index(-1), HeavenlyStem::Gui); // 음수
136    }
137
138    #[test]
139    fn test_stem_element() {
140        assert_eq!(HeavenlyStem::Jia.element(), Element::Wood);
141        assert_eq!(HeavenlyStem::Bing.element(), Element::Fire);
142        assert_eq!(HeavenlyStem::Wu.element(), Element::Earth);
143        assert_eq!(HeavenlyStem::Geng.element(), Element::Metal);
144        assert_eq!(HeavenlyStem::Ren.element(), Element::Water);
145    }
146
147    #[test]
148    fn test_stem_polarity() {
149        assert_eq!(HeavenlyStem::Jia.polarity(), Polarity::Yang);
150        assert_eq!(HeavenlyStem::Yi.polarity(), Polarity::Yin);
