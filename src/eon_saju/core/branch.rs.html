8/// 지지(地支) - 12개
9#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
10#[repr(u8)]
11pub enum EarthlyBranch {
12    /// 子 (자) - 쥐 - 양수
13    Zi = 0,
14    /// 丑 (축) - 소 - 음토
15    Chou = 1,
16    /// 寅 (인) - 호랑이 - 양목
17    Yin = 2,
18    /// 卯 (묘) - 토끼 - 음목
19    Mao = 3,
20    /// 辰 (진) - 용 - 양토
21    Chen = 4,
22    /// 巳 (사) - 뱀 - 음화
23    Si = 5,
24    /// 午 (오) - 말 - 양화
25    Wu = 6,
26    /// 未 (미) - 양 - 음토
27    Wei = 7,
28    /// 申 (신) - 원숭이 - 양금
29    Shen = 8,
30    /// 酉 (유) - 닭 - 음금
31    You = 9,
32    /// 戌 (술) - 개 - 양토
33    Xu = 10,
34    /// 亥 (해) - 돼지 - 음수
35    Hai = 11,
36}

96        match self {
97            Self::Yin | Self::Mao => Element::Wood,
98            Self::Si | Self::Wu => Element::Fire,
99            Self::Chou | Self::Chen | Self::Wei | Self::Xu => Element::Earth,
100            Self::Shen | Self::You => Element::Metal,
101            Self::Hai | Self::Zi => Element::Water,
102        }
103    }

132    /// 시주 계산용: 시간(0-23)으로부터 지지 반환
133    ///
134    /// 자시(子時): 23:00-00:59 → Zi
135    /// 축시(丑時): 01:00-02:59 → Chou
136    /// ...
137    #[inline]
138    pub const fn from_hour(hour: u8) -> Self {
139        // 23시-00시는 자시, 1-2시는 축시, ...
140        let adjusted = (hour + 1) % 24;
141        Self::from_index((adjusted / 2) as i32)
142    }
143
144    /// 월주 계산용: 월(1-12, 음력 기준)으로부터 지지 반환
145    ///
146    /// 인월(1월) → 寅, 묘월(2월) → 卯, ...
147    #[inline]
148    pub const fn from_month(month: u8) -> Self {
149        // 음력 1월 = 인월(寅)
150        Self::from_index((month as i32 + 1) % 12)
151    }

160    #[inline]

162        Self::from_index(self.index() as i32 - 1)
163    }

165    /// 지장간(地藏干, Hidden Stems) 정보를 반환합니다.
166    /// 여기(餘氣), 중기(中氣), 정기(正氣) 순서로 반환됩니다.
167    pub fn jijanggan(self) -> Vec<crate::core::stem::HeavenlyStem> {
168        use crate::core::stem::HeavenlyStem::*;
169        match self {
170            Self::Zi => vec![Ren, Gui],        // 壬, 癸 (여기, 정기)
171            Self::Chou => vec![Gui, Xin, Ji],  // 癸, 辛, 己 (여기, 중기, 정기)
172            Self::Yin => vec![Wu, Bing, Jia],  // 戊, 丙, 甲 (여기, 중기, 정기)
173            Self::Mao => vec![Jia, Yi],        // 甲, 乙 (여기, 정기)
174            Self::Chen => vec![Yi, Gui, Wu],   // 乙, 癸, 戊 (여기, 중기, 정기)
175            Self::Si => vec![Wu, Geng, Bing],  // 戊, 庚, 丙 (여기, 중기, 정기)
176            Self::Wu => vec![Bing, Ji, Ding],  // 丙, 己, 丁 (여기, 중기, 정기)
177            Self::Wei => vec![Ding, Yi, Ji],   // 丁, 乙, 己 (여기, 중기, 정기)
178            Self::Shen => vec![Wu, Ren, Geng], // 戊, 壬, 庚 (여기, 중기, 정기)
179            Self::You => vec![Geng, Xin],      // 庚, Xin (여기, 정기)
180            Self::Xu => vec![Xin, Ding, Wu],   // 辛, 丁, 戊 (여기, 중기, 정기)
181            Self::Hai => vec![Wu, Jia, Ren],   // 戊, 甲, 壬 (여기, 중기, 정기)
182        }
183    }
184
185    /// 지장간(地藏干) 정보를 반환합니다. (Alias of jijanggan)
186    pub fn hidden_stems(self) -> Vec<crate::core::stem::HeavenlyStem> {
187        self.jijanggan()
188    }
189
190    /// 지장간의 여기(餘氣, Remnant Stem)를 반환합니다.
191    pub fn yeogi(self) -> crate::core::stem::HeavenlyStem {
192        self.jijanggan()[0]
193    }
194
195    /// 지장간의 중기(中氣, Middle Stem)를 반환합니다.
196    /// 중기가 없는 경우(자, 묘, 유) 정기를 반환하거나 처리 필요.
197    pub fn junggi(self) -> crate::core::stem::HeavenlyStem {
198        let stems = self.jijanggan();
199        if stems.len() == 3 {
200            stems[1]
201        } else {
202            stems[stems.len() - 1] // 사실상 정기
203        }
204    }
205
206    /// 지장간의 정기(正氣, Main Stem)를 반환합니다.
207    pub fn jeonggi(self) -> crate::core::stem::HeavenlyStem {
208        let stems = self.jijanggan();
209        stems[stems.len() - 1]
210    }
211
212    /// 지장간의 정기(正氣, Main Stem)를 반환합니다. (Alias of jeonggi)
213    pub fn primary_stem(self) -> crate::core::stem::HeavenlyStem {
214        self.jeonggi()
215    }

217    /// 계절(Season) 반환
218    pub fn season(self) -> Season {
219        match self {
220            Self::Yin | Self::Mao | Self::Chen => Season::Spring,
221            Self::Si | Self::Wu | Self::Wei => Season::Summer,
222            Self::Shen | Self::You | Self::Xu => Season::Autumn,
223            Self::Hai | Self::Zi | Self::Chou => Season::Winter,
224        }
225    }
226}
