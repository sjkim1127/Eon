80    pub fn element(&self) -> Element {
81        match self {
82            Self::SeaGold
83            | Self::SwordGold
84            | Self::WaxGold
85            | Self::SandGold

87            | Self::JewelryGold => Element::Metal,

286/// GanZi에서 납음 오행을 계산
287pub fn get_nayin(ganzi: &GanZi) -> NayinType {
288    let idx = ganzi.index() as usize;
289    NAYIN_TABLE[idx]
290}
291
292impl GanZi {
293    /// 이 간지의 납음 오행을 반환
294    pub fn nayin(&self) -> NayinType {
295        get_nayin(self)
296    }
297}

299#[cfg(test)]
300mod tests {
301    use super::*;
302    use crate::core::branch::EarthlyBranch;
303    use crate::core::stem::HeavenlyStem;
304
305    #[test]
306    fn test_nayin_basic() {
307        // 갑자 = 해중금
308        let jiazi = GanZi::new(HeavenlyStem::Jia, EarthlyBranch::Zi);
309        assert_eq!(jiazi.nayin(), NayinType::SeaGold);
310        assert_eq!(jiazi.nayin().element(), Element::Metal);
311
312        // 병인 = 노중화
313        let bingyin = GanZi::new(HeavenlyStem::Bing, EarthlyBranch::Yin);
314        assert_eq!(bingyin.nayin(), NayinType::FurnaceFire);
315        assert_eq!(bingyin.nayin().element(), Element::Fire);
316
317        // 경술 = 차천금 (김성주 일주)
318        let gengxu = GanZi::new(HeavenlyStem::Geng, EarthlyBranch::Xu);
319        assert_eq!(gengxu.nayin(), NayinType::JewelryGold);
320        assert_eq!(gengxu.nayin().element(), Element::Metal);
321    }
322
323    #[test]
324    fn test_nayin_pairs() {
325        // 연속된 두 간지는 같은 납음을 공유
326        let jiazi = GanZi::new(HeavenlyStem::Jia, EarthlyBranch::Zi);
327        let yichou = GanZi::new(HeavenlyStem::Yi, EarthlyBranch::Chou);
328        assert_eq!(jiazi.nayin(), yichou.nayin());
329    }
330
331    #[test]
332    fn test_nayin_hangul() {
333        let jiazi = GanZi::new(HeavenlyStem::Jia, EarthlyBranch::Zi);
334        assert_eq!(jiazi.nayin().hangul(), "해중금(海中金)");
335    }
