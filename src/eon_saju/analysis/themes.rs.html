55    fn analyze(pillars: &FourPillars, config: &AnalysisConfig) -> Self::Output {
56        let strength = pillars.strength_with_config(config);
57        let deuk_se = &strength.deuk_se;
58
59        // 1. Career (직업/커리어)
60        let career_score = ((60
61            + (deuk_se.guanxing_count as u32 * 8)
62            + (deuk_se.shishang_count as u32 * 6)
63            + (deuk_se.bijie_count as u32 * 4))
64            .min(95))
65        .max(50);
66
67        let career_keywords = if deuk_se.guanxing_count >= 2 {
68            vec!["조직 리더십".to_string(), "체계적 역량".to_string()]
69        } else if deuk_se.shishang_count >= 2 {
70            vec!["창의적 전문성".to_string(), "기술/기획력".to_string()]
71        } else if deuk_se.bijie_count >= 2 {
72            vec![
73                "독립 자영/프리랜서".to_string(),
74                "주도적 실행력".to_string(),
75            ]
76        } else {
77            vec!["수용적 협업".to_string(), "안정적 기획".to_string()]
78        };
79
80        let career_summary = if deuk_se.guanxing_count >= 2 {
81            "조직 및 가치 체계 내에서 신뢰와 리더십을 바탕으로 성장하는 스타일입니다.".to_string()
82        } else if deuk_se.shishang_count >= 2 {
83            "독창적인 기술이나 아이디어를 바탕으로 전문성을 인정받을 때 가장 빛납니다.".to_string()
84        } else {
85            "자율성과 독자적인 판단력을 존중받는 업무 환경에서 높은 성과를 냅니다.".to_string()
86        };
87
88        let career_recommendation = if deuk_se.guanxing_count >= 2 {
89            "공공기관, 대기업, 체계화된 조직 내 핵심 보직이나 전문 관리직이 적합합니다.".to_string()
90        } else if deuk_se.shishang_count >= 2 {
91            "연구개발, 디자이너, 콘텐츠 크리에이터, 전문 자격 기술직이 유망합니다.".to_string()
92        } else {
93            "주도적으로 프로젝트를 이끌 수 있는 전문 프론티어 분야나 스타트업이 적합합니다."
94                .to_string()
95        };
96
97        // 2. Wealth (재물운)
98        let wealth_score = ((50
99            + (deuk_se.caisheng_count as u32 * 12)
100            + (deuk_se.shishang_count as u32 * 6)
101            + (deuk_se.yinxing_count as u32 * 4))
102            .min(98))
103        .max(45);
104
105        let wealth_keywords = if deuk_se.caisheng_count >= 2 {
106            vec!["풍부한 재물 기반".to_string(), "자산 운용".to_string()]
107        } else if deuk_se.shishang_count >= 1 {
108            vec!["식상생재".to_string(), "기술/성과 수익".to_string()]
109        } else {
110            vec!["안정적 저축".to_string(), "지출 관리 필요".to_string()]
111        };
112
113        let wealth_summary = if deuk_se.caisheng_count >= 2 {
114            "재물 인연이 깊으며 실속 있게 자산을 증식할 수 있는 사주 구조입니다.".to_string()
115        } else if deuk_se.shishang_count >= 1 {
116            "자신의 재능과 활동이 곧 직간접적인 수익으로 이어지는 재물 흐름을 가집니다.".to_string()
117        } else {
118            "무리한 투기보다 꾸준한 저축과 계획적인 리스크 관리가 재물운을 지켜줍니다.".to_string()
119        };
120
121        let wealth_flow = if deuk_se.caisheng_count >= 2 {
122            "다각화된 투자나 부동산/금융 자산을 통해 안정적 부를 구축하는 흐름입니다.".to_string()
123        } else {
124            "한 번에 큰 이익을 노리기보다 꾸준한 파이프라인 확장을 통한 장기 성장이 유리합니다."
125                .to_string()
126        };
127
128        // 3. Romance (연애/인연)
129        let romance_score = ((55
130            + (deuk_se.caisheng_count as u32 * 7)
131            + (deuk_se.guanxing_count as u32 * 7)
132            + (deuk_se.yinxing_count as u32 * 5))
133            .min(95))
134        .max(50);
135
136        let romance_keywords = if deuk_se.caisheng_count >= 1 || deuk_se.guanxing_count >= 1 {
137            vec!["적극적 매력".to_string(), "활발한 이성운".to_string()]
138        } else if deuk_se.yinxing_count >= 2 {
139            vec!["신중함".to_string(), "정서적 깊이".to_string()]
140        } else {
141            vec!["자연스러운 인연".to_string(), "상호 존중".to_string()]
142        };
143
144        let romance_summary = if deuk_se.caisheng_count >= 1 || deuk_se.guanxing_count >= 1 {
145            "이성과의 인연이 활발하며 배려와 이끌림을 동시에 줄 수 있는 인연운을 지닙니다."
146                .to_string()
147        } else {
148            "신중하게 상대를 탐색하고 서로의 깊은 정서적 안정감을 중요하게 생각합니다.".to_string()
149        };
150
151        let romance_advice =
152            "상대방의 입장을 먼저 공감해 주고 사소한 대화를 자주 나누는 것이 깊은 관계 발전에 도움이 됩니다."
153                .to_string();
154
155        // 4. Health (건강운 & 장기 취약도)
156        let elements = [
157            pillars.year.stem.element(),
158            pillars.year.branch.element(),
159            pillars.month.stem.element(),
160            pillars.month.branch.element(),
161            pillars.day.stem.element(),
162            pillars.day.branch.element(),
163            pillars.hour.stem.element(),
164            pillars.hour.branch.element(),
165        ];
166
167        let mut wood_cnt = 0;
168        let mut fire_cnt = 0;
169        let mut earth_cnt = 0;
170        let mut metal_cnt = 0;
171        let mut water_cnt = 0;
172
173        for e in elements {
174            match e {
175                Element::Wood => wood_cnt += 1,
176                Element::Fire => fire_cnt += 1,
177                Element::Earth => earth_cnt += 1,
178                Element::Metal => metal_cnt += 1,
179                Element::Water => water_cnt += 1,
180            }
181        }
182
183        let mut vulnerable_organs = Vec::new();
184        if wood_cnt == 0 || wood_cnt >= 4 {
185            vulnerable_organs.push("간/담(Liver/Gallbladder)".to_string());
186            vulnerable_organs.push("신경계".to_string());
187        }
188        if fire_cnt == 0 || fire_cnt >= 4 {
189            vulnerable_organs.push("심장/소장".to_string());
190            vulnerable_organs.push("혈관계".to_string());
191        }
192        if earth_cnt == 0 || earth_cnt >= 4 {
193            vulnerable_organs.push("위장/비장(Stomach)".to_string());
194            vulnerable_organs.push("소화기계".to_string());
195        }
196        if metal_cnt == 0 || metal_cnt >= 4 {
197            vulnerable_organs.push("폐/대장(Lungs)".to_string());
198            vulnerable_organs.push("호흡기계".to_string());
199        }
200        if water_cnt == 0 || water_cnt >= 4 {
201            vulnerable_organs.push("신장/방광(Kidney)".to_string());
202            vulnerable_organs.push("비뇨생식계".to_string());
203        }
204
205        if vulnerable_organs.is_empty() {
206            vulnerable_organs.push("위장계(스트레스 주의)".to_string());
207            vulnerable_organs.push("자율신경계".to_string());
208        }
209
210        let health_score = 90 - (vulnerable_organs.len() as u32 * 5);
211
212        let health_keywords = vec!["체질 밸런스".to_string(), "스트레스 케어".to_string()];
213        let health_summary = format!(
214            "오행 균형 분석에 따라 {} 관련 장기 관리 및 일상 스트레스 완화에 유의하는 것이 좋습니다.",
215            vulnerable_organs.first().cloned().unwrap_or_default()
216        );
217
218        ThemedAnalysis {
219            career: CareerAnalysis {
220                score: career_score,
221                keywords: career_keywords,
222                summary: career_summary,
223                recommendation: career_recommendation,
224            },
225            wealth: WealthAnalysis {
226                score: wealth_score,
227                keywords: wealth_keywords,
228                summary: wealth_summary,
229                flow: wealth_flow,
230            },
231            romance: RomanceAnalysis {
232                score: romance_score,
233                keywords: romance_keywords,
234                summary: romance_summary,
235                advice: romance_advice,
236            },
237            health: HealthAnalysis {
238                score: health_score,
239                keywords: health_keywords,
240                summary: health_summary,
241                vulnerable_organs,
242            },
243        }
244    }
