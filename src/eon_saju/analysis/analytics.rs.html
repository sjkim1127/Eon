1//! 인생 통계 및 골든 타임 분석 엔진
2//!
3//! 시뮬레이션 데이터를 바탕으로 인생의 전성기(Peak)를 추출합니다.
4
5use crate::engine::emulator::YearlyScore;
6use serde::{Deserialize, Serialize};
7
8#[derive(Debug, Clone, Serialize, Deserialize)]
9#[serde(rename_all = "camelCase")]
10pub struct GoldenTime {
11    pub start_age: u32,
12    pub end_age: u32,
13    pub average_score: f32,
14    pub description: String,
15}
16
17pub struct Analyzer;
18
19impl Analyzer {
20    /// 시뮬레이션 데이터에서 골든 타임 추출 (Sliding Window)
21    pub fn find_golden_time(timeline: &[YearlyScore], window_size: usize) -> Option<GoldenTime> {
22        if timeline.len() < window_size {
23            return None;
24        }
25
26        let mut max_avg = -1.0;
27        let mut best_start = 0;
28
29        // Sliding Window Average 계산
30        for i in 0..=(timeline.len() - window_size) {
31            let window = &timeline[i..i + window_size];
32            let sum: f64 = window.iter().map(|f| f.total_score).sum();
33            let avg = sum as f32 / window_size as f32;
34
35            if avg > max_avg {
36                max_avg = avg;
37                best_start = i;
38            }
39        }
40
41        Some(GoldenTime {
42            start_age: timeline[best_start].age,
43            end_age: timeline[best_start + window_size - 1].age,
44            average_score: max_avg,
45            description: format!(
46                "{}세부터 {}세까지 가장 운의 밀도가 높은 골든 타임입니다.",
47                timeline[best_start].age,
48                timeline[best_start + window_size - 1].age
49            ),
50        })
51    }
52
53    /// 전체 시뮬레이션 기간 중 득점 분포 분석
54    pub fn analyze_distribution(timeline: &[YearlyScore]) -> Vec<(u32, f32)> {
55        timeline
56            .iter()
57            .map(|f| (f.age, f.total_score as f32))
58            .collect()
59    }
60}
