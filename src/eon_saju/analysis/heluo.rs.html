89/// 천간의 낙서 수리 배정 (후천팔괘 납갑수)
90fn stem_to_num(stem: HeavenlyStem) -> u32 {
91    match stem {
92        HeavenlyStem::Jia => 6,  // 乾
93        HeavenlyStem::Yi => 2,   // 坤
94        HeavenlyStem::Bing => 8, // 艮
95        HeavenlyStem::Ding => 7, // 兌
96        HeavenlyStem::Wu => 1,   // 坎
97        HeavenlyStem::Ji => 9,   // 離
98        HeavenlyStem::Geng => 3, // 震
99        HeavenlyStem::Xin => 4,  // 巽
100        HeavenlyStem::Ren => 6,  // 乾
101        HeavenlyStem::Gui => 2,  // 坤
102    }
103}

118/// 5의 기궁(寄宮) 처리
119fn get_gigung_trigram(era: Era, is_male: bool, is_yang_male_or_yin_female: bool) -> u8 {
120    match era {
121        Era::SangWon => {
122            if is_male {
123                8
124            } else {
125                2
126            } // 남기간(8), 여기관(2)
127        }
128        Era::JungWon => {
129            if is_yang_male_or_yin_female {
130                8
131            } else {
132                2
133            } // 양남음녀기간(8), 음남양녀기관(2)
134        }
135        Era::HaWon => {
136            if is_male {
137                9
138            } else {
139                7
140            } // 남기리(9), 여기태(7)
141        }
142    }
143}

145fn reduce_heaven(sum: u32, is_male: bool, is_yang_male_or_yin_female: bool, era: Era) -> u8 {
146    let mut remainder = sum;
147    while remainder > 25 {
148        remainder -= 25;
149    }
150    if remainder == 0 {
151        remainder = 25;
152    }
153
154    // 去十不用 (십의 자리는 버림)
155    let val = if remainder == 10 {
156        1
157    } else if remainder == 20 {
158        2
159    } else {
160        (remainder % 10) as u8
161    };
162
163    if val == 5 {
164        get_gigung_trigram(era, is_male, is_yang_male_or_yin_female)
165    } else {
166        val
167    }
168}

170fn reduce_earth(sum: u32, is_male: bool, is_yang_male_or_yin_female: bool, era: Era) -> u8 {
171    let mut remainder = sum;
172    while remainder > 30 {
173        remainder -= 30;
174    }
175    if remainder == 0 {
176        remainder = 30;
177    }
178
179    // 去十不用 (십의 자리는 버림)
180    let val = if remainder == 10 {
181        1
182    } else if remainder == 20 {
183        2
184    } else if remainder == 30 {
185        3
186    } else {
187        (remainder % 10) as u8
188    };
189
190    if val == 5 {
191        get_gigung_trigram(era, is_male, is_yang_male_or_yin_female)
192    } else {
193        val
194    }
195}

281pub fn calculate_heluo(
282    birth_year: i32,
283    is_male: bool,
284    year_ganzi: &GanZi,
285    month_ganzi: &GanZi,
286    day_ganzi: &GanZi,
287    hour_ganzi: &GanZi,
288) -> HeLuoResult {
289    let era = Era::from_year(birth_year);
290
291    // 1. 천간 수리 치환 및 합산
292    let stem_nums = vec![
293        stem_to_num(year_ganzi.stem),
294        stem_to_num(month_ganzi.stem),
295        stem_to_num(day_ganzi.stem),
296        stem_to_num(hour_ganzi.stem),
297    ];
298
299    // 2. 지지 수리 치환 및 합산
300    let mut branch_nums = Vec::new();
301    branch_nums.extend(branch_to_nums(year_ganzi.branch));
302    branch_nums.extend(branch_to_nums(month_ganzi.branch));
303    branch_nums.extend(branch_to_nums(day_ganzi.branch));
304    branch_nums.extend(branch_to_nums(hour_ganzi.branch));
305
306    let mut all_nums = Vec::new();
307    all_nums.extend(stem_nums);
308    all_nums.extend(branch_nums);
309
310    // 천수(홀수) 및 지수(짝수) 분류 및 합산
311    let mut sum_odd = 0;
312    let mut sum_even = 0;
313    for num in all_nums {
314        if num % 2 == 1 {
315            sum_odd += num;
316        } else {
317            sum_even += num;
318        }
319    }
320
321    // 양남음녀 판단
322    let is_year_yang = matches!(
323        year_ganzi.stem,
324        HeavenlyStem::Jia
325            | HeavenlyStem::Bing
326            | HeavenlyStem::Wu
327            | HeavenlyStem::Geng
328            | HeavenlyStem::Ren
329    );
330    let is_yang_male_or_yin_female = (is_year_yang && is_male) || (!is_year_yang && !is_male);
331
332    // 상괘 / 하괘 번호 도출
333    let odd_trigram_num = reduce_heaven(sum_odd, is_male, is_yang_male_or_yin_female, era);
334    let even_trigram_num = reduce_earth(sum_even, is_male, is_yang_male_or_yin_female, era);
335
336    let (pre_upper, pre_lower) = if is_yang_male_or_yin_female {
337        (odd_trigram_num, even_trigram_num)
338    } else {
339        (even_trigram_num, odd_trigram_num)
340    };
341
342    let pre_natal_hexagram = get_king_wen_index(pre_upper, pre_lower);
343
344    // 선천괘의 6효 구조 만들기
345    let upper_lines = get_trigram_lines(pre_upper);
346    let lower_lines = get_trigram_lines(pre_lower);
347    let mut pre_lines = [false; 6];
348    pre_lines[0] = lower_lines[0]; // 초효
349    pre_lines[1] = lower_lines[1]; // 2효
350    pre_lines[2] = lower_lines[2]; // 3효
351    pre_lines[3] = upper_lines[0]; // 4효
352    pre_lines[4] = upper_lines[1]; // 5효
353    pre_lines[5] = upper_lines[2]; // 상효
354
355    // 태어난 시의 음양 및 인덱스 (자~사=양시(0..5), 오~해=음시(6..11))
356    let hour_branch_idx = hour_ganzi.branch.index() as usize;
357    let is_yang_hour = hour_branch_idx <= 5;
358    let hi = hour_branch_idx % 6;
359
360    // 원당효 계산을 위해 매칭/타겟 라인 분리
361    let mut matching_indices = Vec::new();
362    let mut other_indices = Vec::new();
363    for i in 0..6 {
364        if pre_lines[i] == is_yang_hour {
365            matching_indices.push(i);
366        } else {
367            other_indices.push(i);
368        }
369    }
370
371    let k = matching_indices.len();
372    let yuan_dang_idx = if k == 0 || k == 6 {
373        // 순수 괘상(전부 양이거나 전부 음)일 때의 순수 원당 규칙
374        let base_line = if is_yang_hour {
375            (hi % 3) + 1
376        } else {
377            (hi % 3) + 4
378        };
379
380        let is_yang_ling = month_ganzi.branch.index() <= 5;
381        let mut reverse = false;
382        if pre_natal_hexagram == 1 {
383            // 乾
384            if !is_male && is_yang_ling {
385                reverse = true;
386            }
387        } else if pre_natal_hexagram == 2 {
388            // 坤
389            if is_male && !is_yang_ling {
390                reverse = true;
391            }
392        }
393
394        if reverse {
395            6 - base_line
396        } else {
397            base_line - 1
398        }
399    } else {
400        // 일반적인 괘상
401        let mut slots = Vec::new();
402        if k <= 3 {
403            slots.extend(&matching_indices);
404            slots.extend(&matching_indices);
405            slots.extend(&other_indices);
406        } else {
407            slots.extend(&matching_indices);
408            slots.extend(&other_indices);
409        }
410        slots[hi]
411    };
412
413    let yuan_dang_yao = (yuan_dang_idx + 1) as u8;
414
415    // 후천괘 만들기
416    let is_yang_ling = month_ganzi.branch.index() <= 5;
417
418    // 삼지존괘 (坎 29, 屯 3, 蹇 39) 조건 체크
419    let is_three_sovereign =
420        pre_natal_hexagram == 29 || pre_natal_hexagram == 3 || pre_natal_hexagram == 39;
421    let is_special_sovereign_case = is_three_sovereign
422        && (
423            (yuan_dang_yao == 5 && !is_yang_ling) || // 9五 이며 음령일 때
424        (yuan_dang_yao == 6 && is_yang_ling)
425            // 上六 이며 양령일 때
426        );
427
428    let mut post_lines = pre_lines;
429    post_lines[yuan_dang_idx] = !pre_lines[yuan_dang_idx];
430
431    let (post_upper, post_lower, post_yuan_dang_idx) = if is_special_sovereign_case {
432        // 변이불이 (변하되 바뀌지 않음): 내외괘(상하괘)를 서로 교환하지 않음, 원당효 위치 유지
433        let post_lower_lines = [post_lines[0], post_lines[1], post_lines[2]];
434        let post_upper_lines = [post_lines[3], post_lines[4], post_lines[5]];
435        let pl = lines_to_trigram_num(post_lower_lines);
436        let pu = lines_to_trigram_num(post_upper_lines);
437        (pu, pl, yuan_dang_idx)
438    } else {
439        // 일반 케이스: 내외괘(상하괘) 교환 및 원당효 위치 3칸 시프트
440        let post_lower_lines = [post_lines[3], post_lines[4], post_lines[5]];
441        let post_upper_lines = [post_lines[0], post_lines[1], post_lines[2]];
442        let pl = lines_to_trigram_num(post_lower_lines);
443        let pu = lines_to_trigram_num(post_upper_lines);
444
445        let pyd = if yuan_dang_idx < 3 {
446            yuan_dang_idx + 3
447        } else {
448            yuan_dang_idx - 3
449        };
450
451        (pu, pl, pyd)
452    };
453
454    let post_natal_hexagram = get_king_wen_index(post_upper, post_lower);
455
456    // 평생 대운 타임라인 생성
457    let mut lifetime_cycles = Vec::new();
458    let mut current_age = 1;
459
460    // 1. 선천괘 주기 (원당효부터 항상 위로 순행)
461    let mut line_ptr = yuan_dang_idx;
462    for _ in 0..6 {
463        let is_yang = pre_lines[line_ptr];
464        let duration = if is_yang { 9 } else { 6 };
465        let end_age = current_age + duration - 1;
466
467        lifetime_cycles.push(HeLuoCycle {
468            start_age: current_age,
469            end_age,
470            hexagram_index: pre_natal_hexagram,
471            line_index: (line_ptr + 1) as u8,
472            is_pre_natal: true,
473        });
474
475        current_age += duration;
476        line_ptr = (line_ptr + 1) % 6; // 항상 위쪽 방향(초->상)으로 진행
477    }
478
479    // 2. 후천괘 주기 (후천괘 원당효부터 항상 위로 순행)
480    let mut post_line_ptr = post_yuan_dang_idx;
481    for _ in 0..6 {
482        let is_yang = post_lines[post_line_ptr];
483        let duration = if is_yang { 9 } else { 6 };
484        let end_age = current_age + duration - 1;
485
486        lifetime_cycles.push(HeLuoCycle {
487            start_age: current_age,
488            end_age,
489            hexagram_index: post_natal_hexagram,
490            line_index: (post_line_ptr + 1) as u8,
491            is_pre_natal: false,
492        });
493
494        current_age += duration;
495        post_line_ptr = (post_line_ptr + 1) % 6; // 항상 위쪽 방향으로 진행
496    }
497
498    // 확장 필드 연산
499    let (se_yao, ying_yao) = get_se_ying_yao(pre_natal_hexagram);
500    let ti_yong = analyze_ti_yong(pre_upper, pre_lower, yuan_dang_yao);
501
502    // 납갑(NaJia) 6효 계산
503    let mut najia_lines = Vec::new();
504    for i in 1..=6 {
505        let (stem, branch) = get_najia_for_line(pre_upper, pre_lower, i);
506        najia_lines.push(NaJiaYao { stem, branch });
507    }
508
509    // 신살(Shinsal) 6효 계산
510    let (void_branches, _) = crate::analysis::void::calculate_void_branches(*day_ganzi);
511    let mut shinsal_lines = Vec::new();
512    for i in 0..6 {
513        let branch = najia_lines[i].branch;
514        let is_noble = matches!(
515            (day_ganzi.stem, branch),
516            (
517                HeavenlyStem::Jia | HeavenlyStem::Wu | HeavenlyStem::Geng,
518                EarthlyBranch::Chou | EarthlyBranch::Wei
519            ) | (
520                HeavenlyStem::Yi | HeavenlyStem::Ji,
521                EarthlyBranch::Zi | EarthlyBranch::Shen
522            ) | (
523                HeavenlyStem::Bing | HeavenlyStem::Ding,
524                EarthlyBranch::Hai | EarthlyBranch::You
525            ) | (
526                HeavenlyStem::Ren | HeavenlyStem::Gui,
527                EarthlyBranch::Si | EarthlyBranch::Mao
528            ) | (HeavenlyStem::Xin, EarthlyBranch::Wu | EarthlyBranch::Yin)
529        );
530        let is_void = void_branches.contains(&branch);
531        let is_rok = matches!(
532            (day_ganzi.stem, branch),
533            (HeavenlyStem::Jia, EarthlyBranch::Yin)
534                | (HeavenlyStem::Yi, EarthlyBranch::Mao)
535                | (HeavenlyStem::Bing | HeavenlyStem::Wu, EarthlyBranch::Si)
536                | (HeavenlyStem::Ding | HeavenlyStem::Ji, EarthlyBranch::Wu)
537                | (HeavenlyStem::Geng, EarthlyBranch::Shen)
538                | (HeavenlyStem::Xin, EarthlyBranch::You)
539                | (HeavenlyStem::Ren, EarthlyBranch::Hai)
540                | (HeavenlyStem::Gui, EarthlyBranch::Zi)
541        );
542        let is_horse = matches!(
543            (year_ganzi.branch, branch),
544            (
545                EarthlyBranch::Shen | EarthlyBranch::Zi | EarthlyBranch::Chen,
546                EarthlyBranch::Yin
547            ) | (
548                EarthlyBranch::Yin | EarthlyBranch::Wu | EarthlyBranch::Xu,
549                EarthlyBranch::Shen
550            ) | (
551                EarthlyBranch::Si | EarthlyBranch::You | EarthlyBranch::Chou,
552                EarthlyBranch::Hai
553            ) | (
554                EarthlyBranch::Hai | EarthlyBranch::Mao | EarthlyBranch::Wei,
555                EarthlyBranch::Si
556            )
557        );
558        shinsal_lines.push(YaoShinsal {
559            is_noble,
560            is_void,
561            is_rok,
562            is_horse,
563        });
564    }
565
566    // 원기(YuanQi) / 화공(HuaGong) 판정
567    let yd_idx = (yuan_dang_yao - 1) as usize;
568    let yd_branch = najia_lines[yd_idx].branch;
569    let yd_element = yd_branch.element();
570
571    let nayin_el = year_ganzi.nayin().element();
572    let yuan_qi = matches!(
573        nayin_el.relation_to(yd_element),
574        ElementRelation::Same | ElementRelation::Generates | ElementRelation::GeneratedBy
575    );
576
577    let month_el = month_ganzi.branch.element();
578    let hua_gong = matches!(
579        month_el.relation_to(yd_element),
580        ElementRelation::Same | ElementRelation::GeneratedBy
581    );
582
583    let yearly_hexagrams = calculate_yearly_hexagrams(
584        pre_natal_hexagram,
585        post_natal_hexagram,
586        &lifetime_cycles,
587        pre_lines,
588        post_lines,
589    );
590
591    HeLuoResult {
592        pre_natal_hexagram,
593        post_natal_hexagram,
594        yuan_dang_yao,
595        lifetime_cycles,
596        yuan_qi,
597        hua_gong,
598        se_yao,
599        ying_yao,
600        najia_lines,
601        shinsal_lines,
602        ti_yong,
603        yearly_hexagrams,
604    }
605}

713pub fn analyze_ti_yong(upper: u8, lower: u8, yuan_dang_yao: u8) -> TiYongRelation {
714    let (ti_trigram, yong_trigram) = if yuan_dang_yao <= 3 {
715        (upper, lower)
716    } else {
717        (lower, upper)
718    };
719
720    let ti_element = get_trigram_element(ti_trigram);
721    let yong_element = get_trigram_element(yong_trigram);

724    TiYongRelation {
725        ti_element,
726        yong_element,
727        relationship,
728    }

731pub fn calculate_yearly_hexagrams(
732    _pre_natal_hexagram: u8,
733    _post_natal_hexagram: u8,
734    lifetime_cycles: &[HeLuoCycle],
735    pre_lines: [bool; 6],
736    post_lines: [bool; 6],
737) -> Vec<YearlyHexagramResult> {
738    let mut results = Vec::new();
739
740    for age in 1..=100 {
741        if let Some(cycle) = lifetime_cycles
742            .iter()
743            .find(|c| age >= c.start_age && age <= c.end_age)
744        {
745            let offset = age - cycle.start_age;
746            let start_line = cycle.line_index;
747
748            let yearly_line_idx = ((start_line - 1) as u32 + offset) % 6;
749            let yearly_line = (yearly_line_idx + 1) as u8;
750
751            let base_lines = if cycle.is_pre_natal {
752                pre_lines
753            } else {
754                post_lines
755            };
756
757            let mut changed_lines = base_lines;
758            changed_lines[yearly_line_idx as usize] = !base_lines[yearly_line_idx as usize];
759
760            let lower_lines = [changed_lines[0], changed_lines[1], changed_lines[2]];
761            let upper_lines = [changed_lines[3], changed_lines[4], changed_lines[5]];
762            let pl = lines_to_trigram_num(lower_lines);
763            let pu = lines_to_trigram_num(upper_lines);
764            let yearly_hex_idx = get_king_wen_index(pu, pl);
765
766            let mut monthly_hexagrams = Vec::new();
767            for m in 1..=12 {
768                let m_line_idx = (yearly_line_idx + (m - 1) as u32) % 6;
769                let mut m_lines = changed_lines;
770                m_lines[m_line_idx as usize] = !changed_lines[m_line_idx as usize];
771
772                let m_lower = [m_lines[0], m_lines[1], m_lines[2]];
773                let m_upper = [m_lines[3], m_lines[4], m_lines[5]];
774                let ml = lines_to_trigram_num(m_lower);
775                let mu = lines_to_trigram_num(m_upper);
776                monthly_hexagrams.push(get_king_wen_index(mu, ml));
777            }
778
779            results.push(YearlyHexagramResult {
780                age,
781                hexagram_index: yearly_hex_idx,
782                yearly_line,
783                monthly_hexagrams,
784            });
785        }
786    }
787
788    results
789}

791fn lines_to_trigram_num(lines: [bool; 3]) -> u8 {
792    match lines {
793        [false, true, false] => 1,  // 坎
794        [false, false, false] => 2, // 坤
795        [true, false, false] => 3,  // 震
796        [false, true, true] => 4,   // 巽
797        [true, true, true] => 6,    // 乾
798        [true, true, false] => 7,   // 兌
799        [false, false, true] => 8,  // 艮
800        [true, false, true] => 9,   // 離
801    }
802}
