1//! 대운(大運, Major Luck Cycles) 계산
2//!
3//! 10년 단위의 운세 흐름을 계산합니다.
4//!
5//! ## 대운 계산 원리
6//!
7//! 1. **순행/역행** 결정
8//!    - 양년생 남자 or 음년생 여자 → 순행 (월주에서 앞으로)
9//!    - 음년생 남자 or 양년생 여자 → 역행 (월주에서 뒤로)
10//!
11//! 2. **대운 시작 나이** 계산
12//!    - 생일부터 가장 가까운 절기까지의 일수 ÷ 3 = 시작 나이
13//!    - 순행: 다음 절기까지의 거리
14//!    - 역행: 이전 절기까지의 거리
15//!
16//! 3. **대운 간지** 계산
17//!    - 월주를 기준으로 순행 또는 역행하여 간지 결정
18//! 
19

30/// 대운 진행 방향
31#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
32#[serde(rename_all = "camelCase")]
33pub enum LuckDirection {
34    /// 순행 (월주에서 앞으로)
35    Forward,
36    /// 역행 (월주에서 뒤로)
37    Reverse,
38}
39
40impl LuckDirection {
41    /// 년간과 성별로 대운 방향 결정
42    /// 
43    /// - 양년생 남자 or 음년생 여자 → 순행
44    /// - 음년생 남자 or 양년생 여자 → 역행
45    pub fn from_year_and_gender(year_stem: HeavenlyStem, gender: Gender) -> Self {
46        let is_yang_year = year_stem.polarity() == Polarity::Yang;
47        let is_male = gender.is_male();
48
49        if (is_yang_year && is_male) || (!is_yang_year && !is_male) {
50            Self::Forward
51        } else {
52            Self::Reverse
53        }
54    }
55}
56
57impl std::fmt::Display for LuckDirection {
58    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
59        match self {
60            Self::Forward => write!(f, "순행"),
61            Self::Reverse => write!(f, "역행"),
62        }
63    }
64}

66#[derive(Debug, Clone, Serialize, Deserialize)]
67#[serde(rename_all = "camelCase")]
68pub struct MajorLuck {
69    /// 대운 간지
70    pub ganzi: GanZi,
71    /// 시작 나이
72    pub start_age: u32,
73    /// 종료 나이
74    pub end_age: u32,
75    /// 천간 십성
76    pub stem_god: TenGod,
77    /// 지지 십성 (정기 기준)
78    pub branch_god: TenGod,
79    /// 실제 대운 시작 날짜 (초정밀 교운기)
80    pub start_date: chrono::DateTime<chrono::Utc>,
81}

83impl MajorLuck {
84    /// 새 대운 생성
85    pub fn new(
86        ganzi: GanZi, 
87        start_age: u32, 
88        day_master: HeavenlyStem,
89        start_date: chrono::DateTime<chrono::Utc>,
90    ) -> Self {
91        Self {
92            ganzi,
93            start_age,
94            end_age: start_age + 9, // 10년 주기
95            stem_god: TenGod::from_stems(day_master, ganzi.stem),
96            branch_god: TenGod::from_stem_and_branch(day_master, ganzi.branch),
97            start_date,
98        }
99    }
100
101    /// 특정 나이가 이 대운에 해당하는지 확인
102    pub fn contains_age(&self, age: u32) -> bool {
103        age >= self.start_age && age <= self.end_age
104    }
105
106    /// 한국식 나이 (태어나자마자 1세)
107    pub fn korean_age(&self, birth_year: i32) -> u32 {
108        // 대운 시작 날짜의 연도 - 출생 연도 + 1
109        (self.start_date.naive_utc().year() - birth_year + 1) as u32
110    }
111}

125/// 대운 분석 결과
126#[derive(Debug, Clone, Serialize, Deserialize)]
127#[serde(rename_all = "camelCase")]
128pub struct MajorLuckAnalysis {
129    /// 대운 방향
130    pub direction: LuckDirection,
131    /// 대운 시작 나이 (세)
132    pub start_age: u32,
133    /// 대운 시작 개월 (0~11)
134    pub start_months: u32,
135    /// 대운 시작 일 (0~30)
136    pub start_days: u32,
137    /// 실제 대운 시작 날짜 (교운기 연월일시)
138    pub start_date: chrono::DateTime<chrono::Utc>,
139    /// 대운 목록 (보통 8~10개)
140    pub cycles: Vec<MajorLuck>,
141    /// 일간 (기준)
142    pub day_master: HeavenlyStem,
143}

145impl MajorLuckAnalysis {
146    /// 천문 엔진을 사용한 대운 분석 생성
147    pub fn calculate_astro(
148        pillars: &FourPillars,
149        gender: Gender,
150        birth_year: i32,
151        birth_month: u32,
152        birth_day: u32,
153        birth_hour: u32,
154        birth_min: u32,
155    ) -> Result<Self, SajuError> {
156        use eon_astro::AstroEngine;
157
158        let day_master = pillars.day_master();
159        let direction = LuckDirection::from_year_and_gender(pillars.year.stem, gender);
160        let astro = AstroEngine::new();
161        
162        // KST 9시간 차이 가정하여 UTC로 변환하여 천문 계산 수행
163        let dt_local = NaiveDate::from_ymd_opt(birth_year, birth_month, birth_day)
164            .and_then(|d| d.and_hms_opt(birth_hour, birth_min, 0))
165            .ok_or_else(|| SajuError::InvalidDateTime(format!(
166                "대운 계산용 날짜 오류: {}-{}-{} {}:{}", birth_year, birth_month, birth_day, birth_hour, birth_min
167            )))?;
168        
169        let tz_offset = chrono::Duration::minutes((pillars.raw_input.timezone_offset_h * 60.0) as i64);
170        let birth_time = Utc.from_utc_datetime(&(dt_local - tz_offset));
171        
172        // 현재 24절기 인덱스 확인 (0: 입춘 ~ 23: 대한)
173        let term_idx = astro.get_solar_term_index(birth_time);
174        
175        // 대운은 '절기(Jieqi, 짝수 인덱스)'를 기준으로 함
176        let target_term_idx = match direction {
177            LuckDirection::Forward => {
178                // 순행: 다음 절입 시각 (현재보다 큰 가장 가까운 짝수 인덱스)
179                if term_idx.is_multiple_of(2) { (term_idx + 2) % 24 } else { (term_idx + 1) % 24 }
180            },
181            LuckDirection::Reverse => {
182                // 역행: 이전 절입 시각 (현재 또는 이전의 가장 가까운 짝수 인덱스)
183                if term_idx.is_multiple_of(2) { term_idx } else { term_idx - 1 }
184            }
185        };
186
187        let target_term_time = astro.find_solar_term_time(birth_time, target_term_idx)
188            .map_err(|e| SajuError::CalculationError(format!("절기 시각 계산 실패: {}", e)))?;
189
190        // 대운 시작 정밀 계산 (년, 월, 일 단위)
191        let (start_age, start_months, start_days) = Self::calculate_precise_start_with_times(
192            birth_time, target_term_time, direction
193        );
194        
195        // 실제 대운 시작 날짜 (교운기 확정)
196        // 3일 = 1년 법칙: (초)로 환산하면 (10일 diff = 1217일 실제 시간)
197        let diff = if direction == LuckDirection::Forward {
198            target_term_time - birth_time
199        } else {
200            birth_time - target_term_time
201        };
202        let offset_seconds = (diff.num_seconds() as f64 * (365.2425 / 3.0)) as i64;
203        let start_date = birth_time + chrono::Duration::seconds(offset_seconds);
204
205        // 대운 간지 계산 (월주 기준으로 순행/역행)
206        let mut cycles = Vec::new();
207        let mut current_ganzi = pillars.month;
208        
209        for i in 0..10 {
210            let age = start_age + (i * 10);
211            let cycle_start_date = start_date + chrono::Duration::seconds((i as f64 * 10.0 * 365.2425 * 86400.0) as i64);
212            
213            current_ganzi = match direction {
214                LuckDirection::Forward => current_ganzi.next(),
215                LuckDirection::Reverse => current_ganzi.prev(),
216            };
217            cycles.push(MajorLuck::new(current_ganzi, age, day_master, cycle_start_date));
218        }
219
220        Ok(Self {
221            direction,
222            start_age,
223            start_months,
224            start_days,
225            start_date,
226            cycles,
227            day_master,
228        })
229    }

231    pub fn calculate(
232        pillars: &FourPillars,
233        gender: Gender,
234        birth_year: i32,
235        birth_month: u32,
236        birth_day: u32,
237        birth_hour: u32,
238        birth_min: u32,
239        term_year: i32,
240        term_month: u32,
241        term_day: u32,
242        term_hour: u32,
243        term_min: u32,
244    ) -> Result<Self, SajuError> {
245        let day_master = pillars.day_master();
246        let direction = LuckDirection::from_year_and_gender(pillars.year.stem, gender);
247        
248        let birth_time = Utc.with_ymd_and_hms(birth_year, birth_month, birth_day, birth_hour, birth_min, 0)
249            .earliest()
250            .ok_or_else(|| SajuError::InvalidDateTime(format!("생년월일시 오류: {}-{}-{} {}:{}", birth_year, birth_month, birth_day, birth_hour, birth_min)))?;
251        let term_time = Utc.with_ymd_and_hms(term_year, term_month, term_day, term_hour, term_min, 0)
252            .earliest()
253            .ok_or_else(|| SajuError::InvalidDateTime(format!("절기 날짜시간 오류: {}-{}-{} {}:{}", term_year, term_month, term_day, term_hour, term_min)))?;
254
255        // 대운 시작 정밀 계산 (년, 월, 일 단위)
256        let (start_age, start_months, start_days) = Self::calculate_precise_start_with_times(
257            birth_time, term_time, direction
258        );
259        
260        let diff = if direction == LuckDirection::Forward {
261            term_time - birth_time
262        } else {
263            birth_time - term_time
264        };
265        let offset_seconds = (diff.num_seconds() as f64 * (365.2425 / 3.0)) as i64;
266        let start_date = birth_time + chrono::Duration::seconds(offset_seconds);
267
268        // 대운 간지 계산 (월주 기준으로 순행/역행)
269        let mut cycles = Vec::new();
270        let mut current_ganzi = pillars.month;
271        
272        for i in 0..10 {
273            let age = start_age + (i * 10);
274            let cycle_start_date = start_date + chrono::Duration::seconds((i as f64 * 10.0 * 365.2425 * 86400.0) as i64);
275
276            current_ganzi = match direction {
277                LuckDirection::Forward => current_ganzi.next(),
278                LuckDirection::Reverse => current_ganzi.prev(),
279            };
280            
281            cycles.push(MajorLuck::new(current_ganzi, age, day_master, cycle_start_date));
282        }
283
284        Ok(Self {
285            direction,
286            start_age,
287            start_months,
288            start_days,
289            start_date,
290            cycles,
291            day_master,
292        })
293    }

295    fn calculate_precise_start_with_times(
296        birth_time: chrono::DateTime<chrono::Utc>,
297        term_time: chrono::DateTime<chrono::Utc>,
298        direction: LuckDirection
299    ) -> (u32, u32, u32) {
300        let diff = if direction == LuckDirection::Forward {
301            term_time - birth_time
302        } else {
303            birth_time - term_time
304        };
305
306        let total_minutes = diff.num_minutes().abs();
307
308        // 3일(4320분) = 1년
309        let years = total_minutes / 4320;
310        let rem_years = total_minutes % 4320;
311
312        // 6시간(360분) = 1개월
313        let months = rem_years / 360;
314        let rem_months = rem_years % 360;
315
316        // 12분 = 1일
317        let days = rem_months / 12;
318
319        (years as u32, months as u32, days as u32)
320    }

322    pub fn at_age(&self, age: u32) -> Option<&MajorLuck> {
323        self.cycles.iter().find(|luck| luck.contains_age(age))
324    }
325
326    pub fn current(&self, current_age: u32) -> Option<&MajorLuck> {
327        self.at_age(current_age)
328    }

331impl std::fmt::Display for MajorLuckAnalysis {
332    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
333        writeln!(f, "대운 ({}, {}년 {}개월 {}일 뒤 시작)", 
334            self.direction, self.start_age, self.start_months, self.start_days)?;
335        writeln!(f, "──────────────────────────────────────")?;
336        
337        for (i, luck) in self.cycles.iter().enumerate() {
338            write!(f, "{:2}. ", i + 1)?;
339            writeln!(f, "{}", luck)?;
340        }
341        
342        Ok(())
343    }
