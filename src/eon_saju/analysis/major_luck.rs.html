1//! 대운(大運, Major Luck Cycles) 계산
2//!
3//! 10년 단위의 운세 흐름을 계산합니다.
4//!
5//! ## 대운 계산 원리
6//!
7//! 1. **순행/역행** 결정
8//!    - 양년생 남자 or 음년생 여자 → 순행 (월주에서 앞으로)
9//!    - 음년생 남자 or 양년생 여자 → 역행 (월주에서 뒤로)
10//!
11//! 2. **대운 시작 나이** 계산
12//!    - 생일부터 가장 가까운 절기까지의 일수 ÷ 3 = 시작 나이
13//!    - 순행: 다음 절기까지의 거리
14//!    - 역행: 이전 절기까지의 거리
15//!
16//! 3. **대운 간지** 계산
17//!    - 월주를 기준으로 순행 또는 역행하여 간지 결정
18//!
19

30/// 대운 진행 방향
31#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
32#[serde(rename_all = "camelCase")]
33pub enum LuckDirection {
34    /// 순행 (월주에서 앞으로)
35    Forward,
36    /// 역행 (월주에서 뒤로)
37    Reverse,
38}
39
40impl LuckDirection {
41    /// 년간과 성별로 대운 방향 결정
42    ///
43    /// - 양년생 남자 or 음년생 여자 → 순행
44    /// - 음년생 남자 or 양년생 여자 → 역행
45    pub fn from_year_and_gender(year_stem: HeavenlyStem, gender: Gender) -> Self {
46        let is_yang_year = year_stem.polarity() == Polarity::Yang;
47        let is_male = gender.is_male();
48
49        if (is_yang_year && is_male) || (!is_yang_year && !is_male) {
50            Self::Forward
51        } else {
52            Self::Reverse
53        }
54    }
55}
56
57impl std::fmt::Display for LuckDirection {
58    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
59        match self {
60            Self::Forward => write!(f, "순행"),
61            Self::Reverse => write!(f, "역행"),
62        }
63    }
64}

70    pub ganzi: GanZi,
71    /// 시작 나이
72    pub start_age: u32,
73    /// 종료 나이

77    /// 지지 십성 (정기 기준)
78    pub branch_god: TenGod,
79    /// 실제 대운 시작 날짜 (초정밀 교운기)
80    pub start_date: chrono::DateTime<chrono::Utc>,

83impl MajorLuck {
84    /// 새 대운 생성
85    pub fn new(
86        ganzi: GanZi,
87        start_age: u32,
88        day_master: HeavenlyStem,
89        start_date: chrono::DateTime<chrono::Utc>,
90    ) -> Self {
91        Self {
92            ganzi,
93            start_age,
94            end_age: start_age + 9, // 10년 주기
95            stem_god: TenGod::from_stems(day_master, ganzi.stem),
96            branch_god: TenGod::from_stem_and_branch(day_master, ganzi.branch),
97            start_date,
98        }
99    }

101    /// 특정 나이가 이 대운에 해당하는지 확인
102    pub fn contains_age(&self, age: u32) -> bool {
103        age >= self.start_age && age <= self.end_age
104    }
105
106    /// 한국식 나이 (태어나자마자 1세)
107    pub fn korean_age(&self, birth_year: i32) -> u32 {
108        // 대운 시작 날짜의 연도 - 출생 연도 + 1
109        (self.start_date.naive_utc().year() - birth_year + 1) as u32
110    }
111}

128/// 대운 분석 결과
129#[derive(Debug, Clone, Serialize, Deserialize)]
130#[serde(rename_all = "camelCase")]
131pub struct MajorLuckAnalysis {
132    /// 대운 방향
133    pub direction: LuckDirection,
134    /// 대운 시작 나이 (세)
135    pub start_age: u32,
136    /// 대운 시작 개월 (0~11)
137    pub start_months: u32,
138    /// 대운 시작 일 (0~30)
139    pub start_days: u32,
140    /// 실제 대운 시작 날짜 (교운기 연월일시)
141    pub start_date: chrono::DateTime<chrono::Utc>,
142    /// 대운 목록 (보통 8~10개)
143    pub cycles: Vec<MajorLuck>,
144    /// 일간 (기준)
145    pub day_master: HeavenlyStem,
146}

148impl MajorLuckAnalysis {
149    /// 천문 엔진을 사용한 대운 분석 생성
150    pub fn calculate_astro(
151        pillars: &FourPillars,
152        gender: Gender,
153        birth_year: i32,
154        birth_month: u32,
155        birth_day: u32,
156        birth_hour: u32,
157        birth_min: u32,
158    ) -> Result<Self, SajuError> {
159        use eon_astro::AstroEngine;
160
161        let day_master = pillars.day_master();
162        let direction = LuckDirection::from_year_and_gender(pillars.year.stem, gender);
163        let astro = AstroEngine::new();
164
165        // KST 9시간 차이 가정하여 UTC로 변환하여 천문 계산 수행
166        let dt_local = NaiveDate::from_ymd_opt(birth_year, birth_month, birth_day)
167            .and_then(|d| d.and_hms_opt(birth_hour, birth_min, 0))
168            .ok_or_else(|| {
169                SajuError::InvalidDateTime(format!(
170                    "대운 계산용 날짜 오류: {}-{}-{} {}:{}",
171                    birth_year, birth_month, birth_day, birth_hour, birth_min
172                ))
173            })?;
174
175        let tz_offset =
176            chrono::Duration::minutes((pillars.raw_input.timezone_offset_h * 60.0) as i64);
177        let birth_time = Utc.from_utc_datetime(&(dt_local - tz_offset));
178
179        // 현재 24절기 인덱스 확인 (0: 입춘 ~ 23: 대한)
180        let term_idx = astro.get_solar_term_index(birth_time);
181
182        // 대운은 '절기(Jieqi, 짝수 인덱스)'를 기준으로 함
183        let target_term_idx = match direction {
184            LuckDirection::Forward => {
185                // 순행: 다음 절입 시각 (현재보다 큰 가장 가까운 짝수 인덱스)
186                if term_idx.is_multiple_of(2) {
187                    (term_idx + 2) % 24
188                } else {
189                    (term_idx + 1) % 24
190                }
191            }
192            LuckDirection::Reverse => {
193                // 역행: 이전 절입 시각 (현재 또는 이전의 가장 가까운 짝수 인덱스)
194                if term_idx.is_multiple_of(2) {
195                    term_idx
196                } else {
197                    term_idx - 1
198                }
199            }
200        };
201
202        let target_term_time = astro
203            .find_solar_term_time(birth_time, target_term_idx)
204            .map_err(|e| SajuError::CalculationError(format!("절기 시각 계산 실패: {}", e)))?;
205
206        // 대운 시작 정밀 계산 (년, 월, 일 단위)
207        let (start_age, start_months, start_days) =
208            Self::calculate_precise_start_with_times(birth_time, target_term_time, direction);
209
210        // 실제 대운 시작 날짜 (교운기 확정)
211        // 3일 = 1년 법칙: (초)로 환산하면 (10일 diff = 1217일 실제 시간)
212        let diff = if direction == LuckDirection::Forward {
213            target_term_time - birth_time
214        } else {
215            birth_time - target_term_time
216        };
217        let offset_seconds = (diff.num_seconds() as f64 * (365.2425 / 3.0)) as i64;
218        let start_date = birth_time + chrono::Duration::seconds(offset_seconds);
219
220        // 대운 간지 계산 (월주 기준으로 순행/역행)
221        let mut cycles = Vec::new();
222        let mut current_ganzi = pillars.month;
223
224        for i in 0..10 {
225            let age = start_age + (i * 10);
226            let cycle_start_date = start_date
227                + chrono::Duration::seconds((i as f64 * 10.0 * 365.2425 * 86400.0) as i64);
228
229            current_ganzi = match direction {
230                LuckDirection::Forward => current_ganzi.next(),
231                LuckDirection::Reverse => current_ganzi.prev(),
232            };
233            cycles.push(MajorLuck::new(
234                current_ganzi,
235                age,
236                day_master,
237                cycle_start_date,
238            ));
239        }
240
241        Ok(Self {
242            direction,
243            start_age,
244            start_months,
245            start_days,
246            start_date,
247            cycles,
248            day_master,
249        })
250    }

252    pub fn calculate(
253        pillars: &FourPillars,
254        gender: Gender,
255        birth_year: i32,
256        birth_month: u32,
257        birth_day: u32,
258        birth_hour: u32,
259        birth_min: u32,
260        term_year: i32,
261        term_month: u32,
262        term_day: u32,
263        term_hour: u32,
264        term_min: u32,
265    ) -> Result<Self, SajuError> {
266        let day_master = pillars.day_master();
267        let direction = LuckDirection::from_year_and_gender(pillars.year.stem, gender);
268
269        let birth_time = Utc
270            .with_ymd_and_hms(birth_year, birth_month, birth_day, birth_hour, birth_min, 0)
271            .earliest()
272            .ok_or_else(|| {
273                SajuError::InvalidDateTime(format!(
274                    "생년월일시 오류: {}-{}-{} {}:{}",
275                    birth_year, birth_month, birth_day, birth_hour, birth_min
276                ))
277            })?;
278        let term_time = Utc
279            .with_ymd_and_hms(term_year, term_month, term_day, term_hour, term_min, 0)
280            .earliest()
281            .ok_or_else(|| {
282                SajuError::InvalidDateTime(format!(
283                    "절기 날짜시간 오류: {}-{}-{} {}:{}",
284                    term_year, term_month, term_day, term_hour, term_min
285                ))
286            })?;
287
288        // 대운 시작 정밀 계산 (년, 월, 일 단위)
289        let (start_age, start_months, start_days) =
290            Self::calculate_precise_start_with_times(birth_time, term_time, direction);
291
292        let diff = if direction == LuckDirection::Forward {
293            term_time - birth_time
294        } else {
295            birth_time - term_time
296        };
297        let offset_seconds = (diff.num_seconds() as f64 * (365.2425 / 3.0)) as i64;
298        let start_date = birth_time + chrono::Duration::seconds(offset_seconds);
299
300        // 대운 간지 계산 (월주 기준으로 순행/역행)
301        let mut cycles = Vec::new();
302        let mut current_ganzi = pillars.month;
303
304        for i in 0..10 {
305            let age = start_age + (i * 10);
306            let cycle_start_date = start_date
307                + chrono::Duration::seconds((i as f64 * 10.0 * 365.2425 * 86400.0) as i64);
308
309            current_ganzi = match direction {
310                LuckDirection::Forward => current_ganzi.next(),
311                LuckDirection::Reverse => current_ganzi.prev(),
312            };
313
314            cycles.push(MajorLuck::new(
315                current_ganzi,
316                age,
317                day_master,
318                cycle_start_date,
319            ));
320        }
321
322        Ok(Self {
323            direction,
324            start_age,
325            start_months,
326            start_days,
327            start_date,
328            cycles,
329            day_master,
330        })
331    }

333    fn calculate_precise_start_with_times(
334        birth_time: chrono::DateTime<chrono::Utc>,
335        term_time: chrono::DateTime<chrono::Utc>,
336        direction: LuckDirection,
337    ) -> (u32, u32, u32) {
338        let diff = if direction == LuckDirection::Forward {
339            term_time - birth_time
340        } else {
341            birth_time - term_time
342        };
343
344        let total_minutes = diff.num_minutes().abs();
345
346        // 3일(4320분) = 1년
347        let years = total_minutes / 4320;
348        let rem_years = total_minutes % 4320;
349
350        // 6시간(360분) = 1개월
351        let months = rem_years / 360;
352        let rem_months = rem_years % 360;
353
354        // 12분 = 1일
355        let days = rem_months / 12;
356
357        (years as u32, months as u32, days as u32)
358    }

360    pub fn at_age(&self, age: u32) -> Option<&MajorLuck> {
361        self.cycles.iter().find(|luck| luck.contains_age(age))
362    }
363
364    pub fn current(&self, current_age: u32) -> Option<&MajorLuck> {
365        self.at_age(current_age)
366    }

369impl std::fmt::Display for MajorLuckAnalysis {
370    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
371        writeln!(
372            f,
373            "대운 ({}, {}년 {}개월 {}일 뒤 시작)",
374            self.direction, self.start_age, self.start_months, self.start_days
375        )?;
376        writeln!(f, "──────────────────────────────────────")?;
377
378        for (i, luck) in self.cycles.iter().enumerate() {
379            write!(f, "{:2}. ", i + 1)?;
380            writeln!(f, "{}", luck)?;
381        }
382
383        Ok(())
384    }
