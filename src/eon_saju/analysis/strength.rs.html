1//! 신강신약(身强身弱, Day Master Strength) 판정
2//!
3//! 일간(日干)의 강약을 판단합니다.
4//!
5//! ## 판정 기준
6//!
7//! 1. **득령(得令)** - 월지가 일간을 생(生) 또는 비(比)하는가
8//! 2. **득지(得地)** - 지지에 일간의 뿌리(통근)가 있는가
9//! 3. **득시(得時)** - 시지가 일간을 생(生) 또는 비(比)하는가
10//! 4. **득세(得勢)** - 비겁(比劫)/인성(印星)이 많은가
11//!
12//! ## 강약 판정
13//!
14//! - 4가지 중 2가지 이상 만족 → 신강(身强)
15//! - 4가지 중 1가지 이하 만족 → 신약(身弱)
16

27/// 강약 유형
28#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
29#[serde(rename_all = "camelCase")]
30pub enum StrengthType {
31    /// 신강(身强) - 강한 사주
32    Strong,
33    /// 신약(身弱) - 약한 사주
34    Weak,
35    /// 중화(中和) - 균형 잡힌 사주 (드묾)
36    Balanced,
37}
38
39impl StrengthType {
40    pub const fn hangul(&self) -> &'static str {
41        match self {
42            Self::Strong => "신강",
43            Self::Weak => "신약",
44            Self::Balanced => "중화",
45        }
46    }
47
48    pub const fn hanja(&self) -> &'static str {
49        match self {
50            Self::Strong => "身强",
51            Self::Weak => "身弱",
52            Self::Balanced => "中和",
53        }
54    }
55
56    pub const fn description(&self) -> &'static str {
57        match self {
58            Self::Strong => "일간의 힘이 강하므로 식상/재성/관성이 필요합니다.",
59            Self::Weak => "일간의 힘이 약하므로 인성/비겁이 필요합니다.",
60            Self::Balanced => "일간과 다른 기운이 균형을 이루고 있습니다.",
61        }
62    }
63}
64
65impl std::fmt::Display for StrengthType {
66    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
67        write!(f, "{}", self.hangul())
68    }
69}

71/// 득령(得令) 판정 - 월지와 일간의 관계
72#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
73#[serde(rename_all = "camelCase")]
74pub struct DeukRyeong {
75    /// 득령 여부
76    pub acquired: bool,
77    /// 월지
78    pub month_branch: EarthlyBranch,
79    /// 월지 오행
80    pub month_element: Element,
81    /// 일간 오행
82    pub day_master_element: Element,
83}
84
85impl DeukRyeong {
86    /// 득령 판정
87    /// 월지 오행이 일간을 생하거나 같으면 득령
88    pub fn check(day_master: HeavenlyStem, month_branch: EarthlyBranch) -> Self {
89        let day_element = day_master.element();
90        let month_element = month_branch.element();
91
92        // 월지가 일간을 생(生)하거나 비(比)하면 득령
93        let is_generating = month_element.generates() == day_element;
94        let is_same = month_element == day_element;
95        let acquired = is_generating || is_same;
96
97        Self {
98            acquired,
99            month_branch,
100            month_element,
101            day_master_element: day_element,
102        }
103    }
104
105    /// 실질 오행을 반영한 득령 판정
106    pub fn check_effective(
107        day_master: HeavenlyStem,
108        month_branch: EarthlyBranch,
109        month_eff: Element,
110    ) -> Self {
111        let day_element = day_master.element();
112
113        // 월지의 실질 오행이 일간을 생(生)하거나 비(比)하면 득령
114        let is_generating = month_eff.generates() == day_element;
115        let is_same = month_eff == day_element;
116        let acquired = is_generating || is_same;
117
118        Self {
119            acquired,
120            month_branch,
121            month_element: month_eff,
122            day_master_element: day_element,
123        }
124    }
125
126    /// 관계 설명
127    pub fn relation(&self) -> &'static str {
128        let is_generating = self.month_element.generates() == self.day_master_element;
129        let is_same = self.month_element == self.day_master_element;
130
131        if is_generating {
132            "월지가 일간을 생함 (生)"
133        } else if is_same {
134            "월지와 일간이 같은 오행 (比)"
135        } else if self.day_master_element.generates() == self.month_element {
136            "일간이 월지를 생함 (泄)"
137        } else if self.day_master_element.controls() == self.month_element {
138            "일간이 월지를 극함 (剋)"
139        } else {
140            "월지가 일간을 극함 (剋)"
141        }
142    }
143}

145/// 득지(得地) 판정 - 지지에 통근 여부
146#[derive(Debug, Clone, Serialize, Deserialize)]
147#[serde(rename_all = "camelCase")]
148pub struct DeukJi {
149    /// 득지 여부
150    pub acquired: bool,
151    /// 통근(通根) 개수
152    pub root_count: u8,
153    /// 통근(通根) 가중치 점수 합계
154    pub root_score: f32,
155    /// 통근 위치들
156    pub root_positions: Vec<String>,
157    /// 강한 12운성 개수 (A급 또는 B급)
158    pub strong_stage_count: u8,
159    /// 12운성 가중치 합계 (A급=1.0, B급=0.5, C급=0.0)
160    pub stage_weight_sum: f32,
161}
162
163impl DeukJi {
164    /// 득지 판정 (기존 호환성 유지)
165    pub fn check(pillars: &FourPillars, config: &AnalysisConfig) -> Self {
166        let relations = RelationshipAnalysis::from_pillars(pillars);
167        Self::check_with_relations(pillars, &relations, config)
168    }
169
170    /// 득지 판정 (합충 관계 반영)
171    pub fn check_with_relations(
172        pillars: &FourPillars,
173        relations: &RelationshipAnalysis,
174        config: &AnalysisConfig,
175    ) -> Self {
176        let day_element = pillars.day_master().element();
177
178        let branches = [
179            ("년지", pillars.year.branch),
180            ("월지", pillars.month.branch),
181            ("일지", pillars.day.branch),
182            ("시지", pillars.hour.branch),
183        ];
184
185        let mut root_count = 0;
186        let mut root_score = 0.0;
187        let mut root_positions = Vec::new();
188        let mut strong_stage_count = 0;
189        let mut stage_weight_sum = 0.0_f32;
190
191        // 충돌 정보를 위치별로 매핑 (충 종류 포함)
192        use std::collections::HashMap;
193        let mut clash_info: HashMap<&str, &crate::analysis::relationships::BranchClash> =
194            HashMap::new();
195        for (clash, pos1, pos2) in &relations.branch_clashes {
196            clash_info.insert(pos1.as_str(), clash);
197            clash_info.insert(pos2.as_str(), clash);
198        }
199
200        for (name, branch) in &branches {
201            // 위치별 기본 가중치
202            let base_weight = match *name {
203                "월지" => config.weights.month_branch,
204                "일지" => config.weights.day_branch,
205                _ => config.weights.other_branch, // 년지, 시지
206            };
207
208            // 충(Clash) 발생 시 충 종류에 따른 차등 감산율 적용
209            let (weight, clash_label) = if let Some(clash) = clash_info.get(*name) {
210                let damage = clash.damage_ratio();
211                let adjusted = base_weight * (1.0 - damage);
212                (
213                    adjusted,
214                    Some(format!("{} ({})", name, clash.clash_type().hangul())),
215                )
216            } else {
217                (base_weight, None)
218            };
219
220            // 지장간(Hidden Stems) 전체를 확인하여 통근 여부 및 세기 분석
221            let hidden_stems = branch.hidden_stems();
222            let mut branch_root_score = 0.0;
223            let saryeong = if *name == "월지" {
224                pillars.saryeong().ok()
225            } else {
226                None
227            };
228
229            for (idx, stem) in hidden_stems.iter().enumerate() {
230                if stem.element() == day_element {
231                    // 기운의 종류에 따른 가중치 (정기: 1.0, 중기: 0.6, 여기: 0.3)
232                    let part_weight = if idx == hidden_stems.len() - 1 {
233                        config.root.main_root
234                    } else if hidden_stems.len() == 3 && idx == 1 {
235                        config.root.middle_root
236                    } else {
237                        config.root.remain_root
238                    };
239
240                    // 사령(Commanding) 여부 확인 (월지만 해당, 가중치 20% 보정)
241                    let saryeong_multiplier = if let Some(ref s) = saryeong {
242                        if s.commanding_stem == *stem {
243                            config.root.saryeong_bonus
244                        } else {
245                            1.0
246                        }
247                    } else {
248                        1.0
249                    };
250
251                    branch_root_score += (weight * part_weight * saryeong_multiplier).min(weight);
252                }
253            }
254
255            if branch_root_score > 0.0 {
256                root_count += 1;
257                root_score += branch_root_score;
258                if let Some(label) = clash_label {
259                    root_positions.push(label);
260                } else {
261                    root_positions.push(name.to_string());
262                }
263            }
264
265            // 12운성 확인 및 가중치 계산
266            let stage =
267                crate::core::twelve_stages::calculate_twelve_stage(pillars.day_master(), *branch);
268
269            // 12운성 가중치 누적 (A급=1.0, B급=0.5, C급=0.0)
270            stage_weight_sum += stage.root_weight();
271
272            // 강한 12운성 개수 (A급 또는 B급)
273            if stage.is_strong() {
274                strong_stage_count += 1;
275            }
276        }
277
278        // 득지 판정:
279        // 1) 12운성 가중치 합이 1.0 이상 (A급 1개 또는 B급 2개 이상)
280        // 2) 또는 통근 점수가 설정된 최소 점수 이상
281        let acquired = stage_weight_sum >= 1.0 || root_score >= config.root.min_deuk_ji_score;
282
283        Self {
284            acquired,
285            root_count,
286            root_score,
287            root_positions,
288            strong_stage_count,
289            stage_weight_sum,
290        }
291    }
292}

368/// 득세(得勢) 판정 - 비겁/인성의 숫자
369#[derive(Debug, Clone, Serialize, Deserialize)]
370#[serde(rename_all = "camelCase")]
371pub struct DeukSe {
372    /// 득세 여부
373    pub acquired: bool,
374    /// 비겁(比肩+劫財) 개수
375    pub bijie_count: u8,
376    /// 인성(正印+偏印) 개수
377    pub yinxing_count: u8,
378    /// 식상(食神+傷官) 개수
379    pub shishang_count: u8,
380    /// 재성(正財+偏財) 개수
381    pub caisheng_count: u8,
382    /// 관성(正官+偏官) 개수
383    pub guanxing_count: u8,
384    /// 비겁+인성 비율 (%)
385    pub support_ratio: f32,
386}
387
388impl DeukSe {
389    /// 득세 판정
390    /// 비겁+인성이 식상+재성+관성보다 많으면 득세
391    pub fn check(pillars: &FourPillars, config: &AnalysisConfig) -> Self {
392        Self::check_with_options(pillars, false, config)
393    }
394
395    /// 옵션에 따른 득세 판정 (Input vs Output 에너지량 비교 추상화)
396    pub fn check_with_options(
397        pillars: &FourPillars,
398        apply_transform: bool,
399        config: &AnalysisConfig,
400    ) -> Self {
401        use crate::analysis::power::{AnalysisOptions, IntegratedAnalysis};
402        let options = AnalysisOptions {
403            apply_transform,
404            apply_correction: true,
405        };
406        let integrated = IntegratedAnalysis::calculate(pillars, options, config);
407
408        let mut bijie_count = 0;
409        let mut yinxing_count = 0;
410        let mut shishang_count = 0;
411        let mut caisheng_count = 0;
412        let mut guanxing_count = 0;
413
414        let mut support_ratio = 0.0;
415
416        for (ten_god, percentage, _) in integrated.ten_god_scores {
417            match ten_god {
418                TenGod::Bijian | TenGod::Jiecai => {
419                    bijie_count += 1;
420                    support_ratio += percentage;
421                }
422                TenGod::Zhengyin | TenGod::Pianyin => {
423                    yinxing_count += 1;
424                    support_ratio += percentage;
425                }
426                TenGod::Shishen | TenGod::Shangguan => shishang_count += 1,
427                TenGod::Zhengcai | TenGod::Piancai => caisheng_count += 1,
428                TenGod::Zhengguan | TenGod::Pianguan => guanxing_count += 1,
429            }
430        }
431
432        let acquired = support_ratio >= 50.0;
433
434        Self {
435            acquired,
436            bijie_count,
437            yinxing_count,
438            shishang_count,
439            caisheng_count,
440            guanxing_count,
441            support_ratio,
442        }
443    }
444}

446/// 신강신약 분석 결과
447#[derive(Debug, Clone, Serialize, Deserialize)]
448#[serde(rename_all = "camelCase")]
449pub struct StrengthAnalysis {
450    /// 일간
451    pub day_master: HeavenlyStem,
452    /// 강약 유형
453    pub strength_type: StrengthType,
454    /// 득령 판정
455    pub deuk_ryeong: DeukRyeong,
456    /// 득지 판정
457    pub deuk_ji: DeukJi,
458    /// 득시 판정
459    pub deuk_si: DeukSi,
460    /// 득세 판정
461    pub deuk_se: DeukSe,
462    /// 만족 조건 개수 (0-4)
463    pub acquired_count: u8,
464    /// 신강/신약 점수 (0-100, 50 기준)
465    pub strength_score: f32,
466}
467
468impl StrengthAnalysis {
469    /// 사주로부터 신강신약 분석
470    pub fn from_pillars(pillars: &FourPillars) -> Self {
471        // 기본적으로 합화 반영
472        Self::from_pillars_with_options(pillars, true, &AnalysisConfig::default())
473    }
474
475    /// 옵션 및 설정을 기반으로 한 신강신약 분석
476    pub fn from_pillars_with_options(
477        pillars: &FourPillars,
478        apply_transform: bool,
479        config: &AnalysisConfig,
480    ) -> Self {
481        let day_master = pillars.day_master();
482        let relations = RelationshipAnalysis::from_pillars(pillars);
483        let eff_map = if apply_transform {
484            Some(pillars.effective_elements())
485        } else {
486            None
487        };
488
489        let deuk_ryeong = if let Some(map) = eff_map {
490            DeukRyeong::check_effective(day_master, pillars.month.branch, map[5].1)
491        } else {
492            DeukRyeong::check(day_master, pillars.month.branch)
493        };
494
495        let deuk_ji = DeukJi::check_with_relations(pillars, &relations, config);
496
497        let deuk_si = if let Some(map) = eff_map {
498            DeukSi::check_effective(day_master, pillars.hour.branch, map[7].1)
499        } else {
500            DeukSi::check(day_master, pillars.hour.branch)
501        };
502
503        let deuk_se = DeukSe::check_with_options(pillars, apply_transform, config);
504
505        // 만족 조건 개수 (4가지)
506        let acquired_count = (deuk_ryeong.acquired as u8)
507            + (deuk_ji.acquired as u8)
508            + (deuk_si.acquired as u8)
509            + (deuk_se.acquired as u8);
510
511        // 강약 판정
512        let strength_type = match acquired_count {
513            2..=4 => StrengthType::Strong,
514            1 | 0 => StrengthType::Weak,
515            _ => StrengthType::Balanced,
516        };
517
518        // 종합 점수 계산
519        let score_ryeong = if deuk_ryeong.acquired {
520            config.strength.criteria_score
521        } else {
522            0.0
523        };
524        let score_ji = if deuk_ji.acquired {
525            config.strength.criteria_score
526        } else {
527            0.0
528        };
529        let score_si = if deuk_si.acquired {
530            config.strength.criteria_score
531        } else {
532            0.0
533        };
534        let score_se = deuk_se.support_ratio * config.strength.deuk_se_weight;
535        let strength_score = score_ryeong + score_ji + score_si + score_se;
536
537        Self {
538            day_master,
539            strength_type,
540            deuk_ryeong,
541            deuk_ji,
542            deuk_si,
543            deuk_se,
544            acquired_count,
545            strength_score,
546        }
547    }
548
549    /// 용신(用神) 추천 (오행 그룹 반환)
550    pub fn recommend_yongshin(&self) -> Vec<Element> {
551        let dm_element = self.day_master.element();
552        match self.strength_type {
553            StrengthType::Strong => {
554                // 식상 / 재성 / 관성 오행들
555                vec![
556                    dm_element.generates(),     // 식상
557                    dm_element.controls(),      // 재성
558                    dm_element.controlled_by(), // 관성
559                ]
560            }
561            StrengthType::Weak => {
562                // 인성 / 비겁 오행들
563                vec![
564                    dm_element.generated_by(), // 인성
565                    dm_element,                // 비겁
566                ]
567            }
568            StrengthType::Balanced => {
569                vec![]
570            }
571        }
572    }

575impl std::fmt::Display for StrengthAnalysis {
576    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
577        writeln!(f, "【신강/신약 분석】")?;
578        writeln!(f, "─────────────────────────────────")?;
579        writeln!(
580            f,
581            "일간: {} ({})",
582            self.day_master.hanja(),
583            self.day_master.element().hangul()
584        )?;
585        writeln!(f)?;
586
587        // 득령
588        let ryeong_mark = if self.deuk_ryeong.acquired {
589            "○"
590        } else {
591            "✗"
592        };
593        writeln!(
594            f,
595            "득령(得令): {} - {}",
596            ryeong_mark,
597            self.deuk_ryeong.relation()
598        )?;
599
600        // 득지
601        let ji_mark = if self.deuk_ji.acquired { "○" } else { "✗" };
602        writeln!(
603            f,
604            "득지(得地): {} - 통근 {}개, 강한 운성 {}개",
605            ji_mark, self.deuk_ji.root_count, self.deuk_ji.strong_stage_count
606        )?;
607
608        // 득시
609        let si_mark = if self.deuk_si.acquired { "○" } else { "✗" };
610        writeln!(f, "득시(得時): {} - {}", si_mark, self.deuk_si.relation())?;
611
612        // 득세
613        let se_mark = if self.deuk_se.acquired { "○" } else { "✗" };
614        writeln!(
615            f,
616            "득세(得勢): {} - 비겁+인성 {:.1}%",
617            se_mark, self.deuk_se.support_ratio
618        )?;
619
620        writeln!(f)?;
621        writeln!(f, "【판정 결과】")?;
622        writeln!(f, "조건 충족: {}/4", self.acquired_count)?;
623        writeln!(f, "종합 점수: {:.1}%", self.strength_score)?;
624        writeln!(f)?;
625        writeln!(
626            f,
627            "▶ {} ({})",
628            self.strength_type.hangul(),
629            self.strength_type.hanja()
630        )?;
631        writeln!(f, "  {}", self.strength_type.description())?;
632
633        writeln!(f)?;
634        let yongshin_names: Vec<String> = self
635            .recommend_yongshin()
636            .iter()
637            .map(|e| e.hangul().to_string())
638            .collect();
639        writeln!(f, "용신(추천 오행): {}", yongshin_names.join(", "))?;
640
641        Ok(())
642    }
643}
