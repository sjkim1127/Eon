97    /// 기준 지지와 대상 지지 간의 12신살 계산
98    /// 보통 기준은 일지(Day Branch) 또는 년지(Year Branch)를 사용함.
99    pub fn calculate(criteria: EarthlyBranch, target: EarthlyBranch) -> Self {
100        use EarthlyBranch::*;
101
102        // 삼합 국(Frame)의 첫 글자(생지) 찾기
103        // 인오술(火) -> 인
104        // 신자진(水) -> 신
105        // 사유축(金) -> 사
106        // 해묘미(木) -> 해
107        let start_branch = match criteria {
108            Yin | Wu | Xu => Yin,
109            Shen | Zi | Chen => Shen,
110            Si | You | Chou => Si,
111            Hai | Mao | Wei => Hai,
112        };
113
114        // 생지와의 거리 계산 (순행)
115        let diff = (target.index() as i32 - start_branch.index() as i32).rem_euclid(12);
116
117        match diff {
118            0 => Self::Jisal,
119            1 => Self::Yeonsal,
120            2 => Self::Wolsal,
121            3 => Self::Mangshinsal,
122            4 => Self::Jangseongsal,
123            5 => Self::Banansal,
124            6 => Self::Yeokmasal,
125            7 => Self::Yukhaesal,
126            8 => Self::Hwagaesal,
127            9 => Self::Geopsal,
128            10 => Self::Jaesal,
129            11 => Self::Cheonsal,
130            _ => unreachable!(),
131        }
132    }
133}

250#[derive(Debug, Clone, Serialize, Deserialize)]
251pub struct ShinsalAnalysis {
252    /// 12신살 (일지 기준)
253    pub twelve_shinsal_day: Vec<(String, TwelveShinsal)>,
254    /// 12신살 (년지 기준)
255    pub twelve_shinsal_year: Vec<(String, TwelveShinsal)>,
256    /// 원진/귀문 등 특수 신살
257    pub special_shinsals: Vec<(String, String, EvilSpirit)>,
258    /// 길신 (천을귀인 등)
259    pub gilsin: Vec<(String, Gilsin)>,
260}
261
262impl ShinsalAnalysis {
263    pub fn from_pillars(pillars: &FourPillars) -> Self {
264        let day_stem = pillars.day_master();
265        let branches = [
266            ("년지", pillars.year.branch),
267            ("월지", pillars.month.branch),
268            ("일지", pillars.day.branch),
269            ("시지", pillars.hour.branch),
270        ];
271
272        // 1. 12신살 (일지 기준)
273        let mut twelve_shinsal_day = Vec::new();
274        for (name, branch) in &branches {
275            // 일지는 자기 자신이므로 제외하거나 장성살이나 지살이 됨. 포함시킴.
276            let shinsal = TwelveShinsal::calculate(pillars.day.branch, *branch);
277            twelve_shinsal_day.push((name.to_string(), shinsal));
278        }
279
280        // 2. 12신살 (년지 기준)
281        let mut twelve_shinsal_year = Vec::new();
282        for (name, branch) in &branches {
283            let shinsal = TwelveShinsal::calculate(pillars.year.branch, *branch);
284            twelve_shinsal_year.push((name.to_string(), shinsal));
285        }
286
287        // 3. 원진/귀문 등 흉살 (모든 지지 조합 체크)
288        let mut special_shinsals = Vec::new();
289        for i in 0..branches.len() {
290            for j in (i + 1)..branches.len() {
291                let (n1, b1) = branches[i];
292                let (n2, b2) = branches[j];
293
294                if let Some(w) = EvilSpirit::check_wonjin(b1, b2) {
295                    special_shinsals.push((n1.to_string(), n2.to_string(), w));
296                }
297                if let Some(g) = EvilSpirit::check_gwimun(b1, b2) {
298                    special_shinsals.push((n1.to_string(), n2.to_string(), g));
299                }
300            }
301        }
302
303        // 4. 길신 (천을귀인 등)
304        let mut gilsin = Vec::new();
305        let cheoneul = Gilsin::cheoneul_branches(day_stem);
306        for (name, b) in &branches {
307            if cheoneul.contains(b) {
308                gilsin.push((name.to_string(), Gilsin::CheoneulGwiin));
309            }
310        }
311
312        Self {
313            twelve_shinsal_day,
314            twelve_shinsal_year,
315            special_shinsals,
316            gilsin,
317        }
318    }
319
320    /// 특정 운(대운/세운 등)에 대한 신살 분석
321    pub fn calculate_for_luck(
322        luck_ganzi: crate::core::ganzi::GanZi,
323        pillars: &FourPillars,
324    ) -> Vec<String> {
325        let mut results = Vec::new();
326        let day_master = pillars.day_master();
327
328        // 1. 12신살 (일지 기준)
329        let s_day = TwelveShinsal::calculate(pillars.day.branch, luck_ganzi.branch);
330        results.push(format!("(일) {}", s_day.hangul()));
331
332        // 2. 12신살 (년지 기준)
333        let s_year = TwelveShinsal::calculate(pillars.year.branch, luck_ganzi.branch);
334        results.push(format!("(년) {}", s_year.hangul()));
335
336        // 3. 천을귀인 체크
337        let cheoneul = Gilsin::cheoneul_branches(day_master);
338        if cheoneul.contains(&luck_ganzi.branch) {
339            results.push("천을귀인".to_string());
340        }
341
342        // 4. 백호/괴강 등 특수 지지
343        use crate::core::branch::EarthlyBranch as B;
344        use crate::core::stem::HeavenlyStem as S;
345
346        // 백호살 (Luck Ganzi 자체가 백호인 경우)
347        match (luck_ganzi.stem, luck_ganzi.branch) {
348            (S::Jia, B::Chen)
349            | (S::Yi, B::Wei)
350            | (S::Bing, B::Xu)
351            | (S::Ding, B::Chou)
352            | (S::Wu, B::Chen)
353            | (S::Ren, B::Xu)
354            | (S::Gui, B::Chou) => {
355                results.push("백호살".to_string());
356            }
357            _ => {}
358        }
359
360        // 괴강살
361        match (luck_ganzi.stem, luck_ganzi.branch) {
362            (S::Wu, B::Xu)
363            | (S::Wu, B::Chen)
364            | (S::Geng, B::Xu)
365            | (S::Geng, B::Chen)
366            | (S::Ren, B::Xu)
367            | (S::Ren, B::Chen) => {
368                results.push("괴강살".to_string());
369            }
370            _ => {}
371        }
372
373        results
374    }
375}

377impl std::fmt::Display for ShinsalAnalysis {
378    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
379        writeln!(f, "【신살(神殺) 분석】")?;
380        writeln!(f, "─────────────────────────────────")?;
381
382        writeln!(f, "[12신살 - 일지 기준]")?;
383        for (pos, shinsal) in &self.twelve_shinsal_day {
384            write!(f, "  {}: {} ", pos, shinsal.hangul())?;
385        }
386        writeln!(f)?;
387
388        writeln!(f, "\n[12신살 - 년지 기준]")?;
389        for (pos, shinsal) in &self.twelve_shinsal_year {
390            write!(f, "  {}: {} ", pos, shinsal.hangul())?;
391        }
392        writeln!(f)?;
393
394        if !self.special_shinsals.is_empty() {
395            writeln!(f, "\n[특수 신살 (원진/귀문)]")?;
396            for (p1, p2, spirit) in &self.special_shinsals {
397                writeln!(f, "  {} - {}: {}", p1, p2, spirit.hangul())?;
398            }
399        }
400
401        if !self.gilsin.is_empty() {
402            writeln!(f, "\n[길신 (Auspicious Spirits)]")?;
403            for (pos, spirit) in &self.gilsin {
404                writeln!(f, "  {}: {}", pos, spirit.hangul())?;
405            }
406        }
407
408        Ok(())
409    }
410}

462        let analysis = pillars.shinsal();
463        println!("{}", analysis);
464
465        // 각 위치별 신살 확인
466        // 년지(신): 인(0) ~ 신(6) -> 역마살
467        let year_shinsal = analysis
468            .twelve_shinsal_day
469            .iter()
470            .find(|(p, _)| p == "년지")
471            .unwrap();
472        assert_eq!(year_shinsal.1, TwelveShinsal::Yeokmasal);
473
474        // 월지(해): 인(0) ~ 해(9) -> 겁살
475        let month_shinsal = analysis
476            .twelve_shinsal_day
477            .iter()
478            .find(|(p, _)| p == "월지")
479            .unwrap();
480        assert_eq!(month_shinsal.1, TwelveShinsal::Geopsal);
