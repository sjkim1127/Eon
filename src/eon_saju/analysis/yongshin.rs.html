1//! 용신(用神, Useful God) 정밀 분석
2//!
3//! 억부(抑扶), 조후(調候), 통관(通關), 병약(病藥)의 관점에서
4//! 사주의 균형을 맞추는 최적의 오행을 찾습니다.

14/// 용신의 종류
15#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
16#[serde(rename_all = "camelCase")]
17pub enum YongshinType {
18    /// 억부용신 (일간의 강약을 조절)
19    Eokbu,
20    /// 조후용신 (기후와 온도를 조절)
21    Johu,
22    /// 통관용신 (대립하는 기운을 소통)
23    Tonggwan,
24    /// 병약용신 (병이 되는 기운을 치유)
25    Byeongyak,
26}
27
28impl YongshinType {
29    pub const fn hangul(&self) -> &'static str {
30        match self {
31            Self::Eokbu => "억부용신",
32            Self::Johu => "조후용신",
33            Self::Tonggwan => "통관용신",
34            Self::Byeongyak => "병약용신",
35        }
36    }
37}

50/// 용신 분석 종합 결과
51#[derive(Debug, Clone, Serialize, Deserialize)]
52#[serde(rename_all = "camelCase")]
53pub struct YongshinAnalysis {
54    /// 추천된 용신 목록
55    pub recommendations: Vec<RecommendedYongshin>,
56    /// 가장 우선시되는 제1용신

58    /// 이를 돕는 희신(喜神)
59    pub assistant: Element,

62impl YongshinAnalysis {
63    pub fn from_pillars(pillars: &FourPillars) -> Self {
64        Self::from_pillars_with_config(pillars, &AnalysisConfig::default())

67    pub fn from_pillars_with_config(pillars: &FourPillars, config: &AnalysisConfig) -> Self {
68        let mut recommendations = Vec::new();
69        let strength = pillars.strength_with_config(config);
70        let day_master_el = pillars.day_master_element();
71        let structure_analysis = pillars.structure();
72
73        let eokbu_element = match structure_analysis.structure {
74            StructureType::JongAh => day_master_el.generates(),
75            StructureType::JongJae => day_master_el.generates().generates(), // 재성
76            StructureType::JongSal => day_master_el.controlled_by(),
77            StructureType::JongGang => day_master_el.generated_by(),
78            StructureType::JongWang => day_master_el,
79            StructureType::Follower => pillars.month.branch.element(),
80            StructureType::SpecialTransformation => day_master_el,
81            _ => {
82                match strength.strength_type {
83                    StrengthType::Weak => {
84                        // 신약(身弱)은 항상 인성(印星)이 억부용신
85                        // - 인성이 원국에 없으면: 인성 보충 필요
86                        // - 인성이 원국에 있어도: 그 힘을 더 강화해야 함
87                        day_master_el.generated_by() // 인성
88                    }
89                    StrengthType::Strong => {
90                        // 신강의 원인을 분석하여 용신 세분화
91                        let yinxing = strength.deuk_se.yinxing_count as f32;
92                        let bijie = strength.deuk_se.bijie_count as f32;
93
94                        if yinxing > bijie * 1.5 {
95                            // 인성 과다로 신강: 재성으로 인성 극복 (용재파인)
96                            day_master_el.generates() // 재성
97                        } else if bijie > yinxing * 1.5 {
98                            // 비겁 과다로 신강: 관성으로 비겁 제어 (관살제겁)
99                            day_master_el.controlled_by() // 관성
100                        } else {
101                            // 인성/비겁 균형: 식상으로 설기 (설기생재)
102                            day_master_el.generates() // 식상
103                        }
104                    }
105                    StrengthType::Balanced => day_master_el,
106                }
107            }
108        };
109
110        // 종격/전왕격인 경우 억부(격국) 용신을 최우선으로 배치하고 조후는 참고로만 제시
111        let is_polarized = matches!(
112            structure_analysis.structure,
113            StructureType::JongAh
114                | StructureType::JongJae
115                | StructureType::JongSal
116                | StructureType::JongGang
117                | StructureType::JongWang
118                | StructureType::Follower
119                | StructureType::SpecialTransformation
120        );
121
122        if is_polarized {
123            let reasons = vec![
124                format!("격국: {}", structure_analysis.structure.hangul()),
125                format!("세력비율: {:.1}%", strength.deuk_se.support_ratio),
126                format!(
127                    "판정기준: {}",
128                    if strength.deuk_se.support_ratio >= config.strength.polarized_high {
129                        "전왕(專旺)"
130                    } else {
131                        "종(從)"
132                    }
133                ),
134            ];
135            recommendations.push(RecommendedYongshin {
136                yongshin_type: YongshinType::Eokbu,
137                element: eokbu_element,
138                summary: format!("격국의 기세를 따르는 {} 용신", eokbu_element.hangul()),
139                description: "강한 세력에 대항하기보다 그 흐름을 따르는 것이 운의 흐름을 원활하게 합니다.".to_string(),
140                reasons,
141            });
142
143            // 조후(調候) 판단 (종격에서는 조후보다 격국이 우선임)
144            let thermal_index = calculate_thermal_index(pillars, config);
145            if let Some(mut johu) = get_johu_analysis(pillars, thermal_index, config) {
146                johu.description = format!(
147                    "{} 단, 종격 사주이므로 조후보다 격국의 기세를 따르는 것이 안전함",
148                    johu.description
149                );
150                recommendations.push(johu);
151            }
152        } else {
153            // 일반적인 경우 조후가 급하면 조후 우선
154            let thermal_index = calculate_thermal_index(pillars, config);
155            if let Some(johu) = get_johu_analysis(pillars, thermal_index, config) {
156                recommendations.push(johu);
157            }
158
159            let (summary, description, reasons) =
160                get_eokbu_explainability(&strength, eokbu_element);
161            recommendations.push(RecommendedYongshin {
162                yongshin_type: YongshinType::Eokbu,
163                element: eokbu_element,
164                summary,
165                description,
166                reasons,
167            });
168        }
169
170        // 3. 통관용신(通關) 판단 (대립 해소)
171        if let Some(tonggwan) = get_tonggwan_analysis(pillars, config) {
172            recommendations.push(tonggwan);
173        }
174
175        // 4. 병약용신(病藥) 판단 (최악의 기운 제어)
176        if let Some(byeongyak) = get_byeongyak_analysis(pillars, &strength) {
177            recommendations.push(byeongyak);
178        }
179
180        // 제1용신 결정 로직
181        // 조후가 극단적이거나(절기 영향) 억부 균형보다 시급할 때 조후 우선
182        let thermal_index_for_primary = calculate_thermal_index(pillars, config);
183        let is_extreme_thermal = thermal_index_for_primary.abs() >= config.thermal.extreme
184            || (thermal_index_for_primary.abs() >= config.thermal.moderate
185                && strength.strength_score.abs() < 10.0);
186        let primary = if is_extreme_thermal
187            && recommendations
188                .iter()
189                .any(|r| r.yongshin_type == YongshinType::Johu)
190        {
191            recommendations
192                .iter()
193                .find(|r| r.yongshin_type == YongshinType::Johu)
194                .map(|r| r.element)
195                .unwrap_or(recommendations[0].element)
196        } else if recommendations
197            .iter()
198            .any(|r| r.yongshin_type == YongshinType::Byeongyak)
199        {
200            // 병약용신이 있으면 약을 우선으로 쓰는 경우가 많음
201            recommendations
202                .iter()
203                .find(|r| r.yongshin_type == YongshinType::Byeongyak)
204                .map(|r| r.element)
205                .unwrap_or(recommendations[0].element)
206        } else {
207            recommendations
208                .iter()
209                .find(|r| r.yongshin_type == YongshinType::Eokbu)
210                .map(|r| r.element)
211                .unwrap_or(recommendations[0].element)
212        };
213
214        // 희신은 용신을 생하거나 돕는 오행
215        let assistant = primary.generated_by();
216
217        Self {
218            recommendations,
219            primary,
220            assistant,
221        }
222    }
223}

225/// 조후 지수 계산 (-100 ~ 100)
226/// - 음수: 춥고 습함 (Water, Metal, Yin-Earth)
227/// - 양수: 덥고 건조함 (Fire, Wood, Yang-Earth)
228pub fn calculate_thermal_index(pillars: &FourPillars, _config: &AnalysisConfig) -> i32 {
229    let mut score = 0;
230
231    // 월지의 영향력이 가장 큼 (40%)
232    match pillars.month.branch {
233        EarthlyBranch::Hai | EarthlyBranch::Zi | EarthlyBranch::Chou => score -= 40,
234        EarthlyBranch::Si | EarthlyBranch::Wu | EarthlyBranch::Wei => score += 40,
235        _ => {}
236    }
237
238    // 전체 오행 분포 확인
239    let stems = [
240        pillars.year.stem,
241        pillars.month.stem,
242        pillars.day.stem,
243        pillars.hour.stem,
244    ];
245    let branches = [
246        pillars.year.branch,
247        pillars.month.branch,
248        pillars.day.branch,
249        pillars.hour.branch,
250    ];
251
252    for s in stems {
253        use crate::core::stem::HeavenlyStem as S;
254        match s {
255            S::Bing => score += 15, // 태양 (가장 뜨거움)
256            S::Ding => score += 10, // 등촉 (따뜻함)
257            S::Ren => score -= 15,  // 강수 (가장 차가움)
258            S::Gui => score -= 10,  // 우로 (차갑고 습함)
259            S::Jia | S::Yi => score += 5,
260            S::Geng | S::Xin => score -= 5,
261            _ => {}
262        }
263    }
264
265    for b in branches {
266        use EarthlyBranch as EB;
267        match b {
268            EB::Si | EB::Wu => score += 15,
269            EB::Wei | EB::Xu => score += 10, // 마른 흙
270            EB::Hai | EB::Zi => score -= 15,
271            EB::Chou | EB::Chen => score -= 10, // 습한 흙
272            EB::Yin | EB::Mao => score += 5,
273            EB::Shen | EB::You => score -= 5,
274        }
275    }
276
277    score.clamp(-100, 100)
278}

280/// 조후 분석
281fn get_johu_analysis(
282    pillars: &FourPillars,
283    index: i32,
284    _config: &AnalysisConfig,
285) -> Option<RecommendedYongshin> {
286    let month = pillars.month.branch;
287
288    if index <= -30
289        || matches!(
290            month,
291            EarthlyBranch::Hai | EarthlyBranch::Zi | EarthlyBranch::Chou
292        )
293    {
294        let mut reasons = vec![format!("조후 지수: {}", index)];
295        if matches!(
296            month,
297            EarthlyBranch::Hai | EarthlyBranch::Zi | EarthlyBranch::Chou
298        ) {
299            reasons.push(format!("동절기({}) 출생", month.hangul()));
300        }
301        Some(RecommendedYongshin {
302            yongshin_type: YongshinType::Johu,
303            element: Element::Fire,
304            summary: "한랭한 사주를 따뜻하게 하는 火 용신".to_string(),
305            description:
306                "사주의 기운이 차갑고 습하므로 불(火)의 기운으로 온도를 조절해야 발복합니다."
307                    .to_string(),
308            reasons,
309        })
310    } else if index >= 30
311        || matches!(
312            month,
313            EarthlyBranch::Si | EarthlyBranch::Wu | EarthlyBranch::Wei
314        )
315    {
316        let mut reasons = vec![format!("조후 지수: {}", index)];
317        if matches!(
318            month,
319            EarthlyBranch::Si | EarthlyBranch::Wu | EarthlyBranch::Wei
320        ) {
321            reasons.push(format!("하절기({}) 출생", month.hangul()));
322        }
323        Some(RecommendedYongshin {
324            yongshin_type: YongshinType::Johu,
325            element: Element::Water,
326            summary: "조열한 사주를 시원하게 하는 水 용신".to_string(),
327            description:
328                "사주의 기운이 뜨겁고 건조하므로 물(水)의 기운으로 온도를 낮추어야 발복합니다."
329                    .to_string(),
330            reasons,
331        })
332    } else {
333        None
334    }
335}

337/// 통관 분석 (서로 싸우는 강한 두 기운 중재)
338///
339/// 개수가 아닌 **세력 점수**를 기준으로 판단합니다.
340/// 두 상충 오행의 세력이 비등할 때(차이 20% 이내)만 통관용신을 적용합니다.
341fn get_tonggwan_analysis(
342    pillars: &FourPillars,
343    config: &AnalysisConfig,
344) -> Option<RecommendedYongshin> {
345    use crate::analysis::power::{AnalysisOptions, IntegratedAnalysis};
346
347    // 세력 점수 계산 (기본 보정 적용)
348    let options = AnalysisOptions {
349        apply_transform: false, // 합화는 적용하지 않음 (원국 기준)
350        apply_correction: false,
351    };
352    let analysis = IntegratedAnalysis::calculate(pillars, options, config);
353
354    // 오행별 세력 점수 추출 (0~100)
355    let mut scores: [f32; 5] = [0.0; 5];
356    for (elem, pct, _) in &analysis.element_scores {
357        scores[elem.index() as usize] = *pct;
358    }
359
360    // 상충 관계 쌍과 통관 오행 정의
361    // (오행1, 오행2, 통관 오행, 설명)
362    let conflicts: [(Element, Element, Element, &str); 5] = [
363        (
364            Element::Metal,
365            Element::Wood,
366            Element::Water,
367            "금(金)과 목(木)이 대립하고 있어 이를 유통시키는 수(水)가 필요함",
368        ),
369        (
370            Element::Water,
371            Element::Fire,
372            Element::Wood,
373            "수(水)와 화(火)가 대립하고 있어 이를 유통시키는 목(木)이 필요함",
374        ),
375        (
376            Element::Wood,
377            Element::Earth,
378            Element::Fire,
379            "목(木)과 토(土)가 대립하고 있어 이를 유통시키는 화(火)가 필요함",
380        ),
381        (
382            Element::Fire,
383            Element::Metal,
384            Element::Earth,
385            "화(火)와 금(金)이 대립하고 있어 이를 유통시키는 토(土)가 필요함",
386        ),
387        (
388            Element::Earth,
389            Element::Water,
390            Element::Metal,
391            "토(土)와 수(水)가 대립하고 있어 이를 유통시키는 금(金)이 필요함",
392        ),
393    ];
394
395    for (elem1, elem2, mediator, reason) in conflicts {
396        let score1 = scores[elem1.index() as usize];
397        let score2 = scores[elem2.index() as usize];
398
399        // 두 오행의 세력이 모두 15% 이상이고, 차이가 20% 이내일 때 통관 적용
400        let min_threshold = 15.0; // 최소 세력
401        let max_diff = 20.0; // 세력 차이 허용 범위
402
403        if score1 >= min_threshold && score2 >= min_threshold {
404            let diff = (score1 - score2).abs();
405            if diff <= max_diff {
406                return Some(RecommendedYongshin {
407                    yongshin_type: YongshinType::Tonggwan,
408                    element: mediator,
409                    summary: format!("{}와 {}의 대립을 해소하는 {} 용신", elem1.hangul(), elem2.hangul(), mediator.hangul()),
410                    description: "상극하는 두 기운이 팽팽하게 대립할 때는 이를 연결해 주는 오행이 행운의 열쇠가 됩니다.".to_string(),
411                    reasons: vec![
412                        reason.to_string(),
413                        format!("{}: {:.1}%", elem1.hangul(), score1),
414                        format!("{}: {:.1}%", elem2.hangul(), score2),
415                    ],
416                });
417            }
418        }
419    }
420
421    None
422}

424/// 병약 분석 (사주의 가장 큰 문제점 제어)
425fn get_byeongyak_analysis(
426    pillars: &FourPillars,
427    strength: &crate::analysis::strength::StrengthAnalysis,
428) -> Option<RecommendedYongshin> {
429    // 일간을 극하거나 설기하는 기운이 너무 강할 때 (병)
430    // 병을 제어하는 기운 (약)
431    if strength.strength_type == StrengthType::Weak {
432        if strength.deuk_se.guanxing_count >= 3 {
433            return Some(RecommendedYongshin {
434                 yongshin_type: YongshinType::Byeongyak,
435                 element: pillars.day_master_element().generated_by(),
436                 summary: "과도한 관성을 제어하는 병약용신".to_string(),
437                 description: "일간을 극하는 관성이 너무 강해 병이 되었으므로, 이를 화(化)해주는 인성이 약이 됩니다.".to_string(),
438                 reasons: vec![format!("관성 개수: {}개", strength.deuk_se.guanxing_count)],
439             });
440        }
441        if strength.deuk_se.shishang_count >= 3 {
442            return Some(RecommendedYongshin {
443                yongshin_type: YongshinType::Byeongyak,
444                element: pillars.day_master_element().generated_by(),
445                summary: "과도한 식상을 제어하는 병약용신".to_string(),
446                description: "일간의 기운을 빼앗는 식상이 너무 강해 병이 되었으므로, 이를 제어하는 인성이 약이 됩니다.".to_string(),
447                reasons: vec![format!("식상 개수: {}개", strength.deuk_se.shishang_count)],
448            });
449        }
450    }
451    None
452}

454impl std::fmt::Display for YongshinAnalysis {
455    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
456        writeln!(f, "【용신(用神) 정밀 판정】")?;
457        writeln!(f, "─────────────────────────────────")?;
458
459        for rec in &self.recommendations {
460            writeln!(
461                f,
462                "● {} : {} ({})",
463                rec.yongshin_type.hangul(),
464                rec.element.hangul(),
465                rec.element.hanja()
466            )?;
467            writeln!(f, "   - 요약: {}", rec.summary)?;
468            writeln!(f, "   - 설명: {}", rec.description)?;
469            if !rec.reasons.is_empty() {
470                writeln!(f, "   - 근거: {}", rec.reasons.join(", "))?;
471            }
472        }
473
474        writeln!(f)?;
475        writeln!(
476            f,
477            "▶ 제1용신: {} ({})",
478            self.primary.hangul(),
479            self.primary.hanja()
480        )?;
481        writeln!(
482            f,
483            "▶ 희신(喜神): {} ({})",
484            self.assistant.hangul(),
485            self.assistant.hanja()
486        )?;
487
488        Ok(())
489    }
490}

513/// 억부용신 결정 이유 상세화
514fn get_eokbu_explainability(
515    strength: &StrengthAnalysis,
516    eokbu_element: Element,
517) -> (String, String, Vec<String>) {
518    let mut reasons = vec![
519        format!("신강약점수: {:.1}", strength.strength_score),
520        format!("현재상태: {}", strength.strength_type.hangul()),
521    ];
522
523    match strength.strength_type {
524        StrengthType::Weak => {
525            (
526                format!("일간을 돕는 {} 억부용신", eokbu_element.hangul()),
527                "일간이 신약하여 기운이 부족하므로 이를 비추거나 생조해주는 오행이 행운을 가져옵니다.".to_string(),
528                reasons
529            )
530        },
531        StrengthType::Strong => {
532            let yinxing = strength.deuk_se.yinxing_count as f32;
533            let bijie = strength.deuk_se.bijie_count as f32;
534
535            if yinxing > bijie * 1.5 {
536                reasons.push(format!("인성({}) 과다", yinxing));
537                (
538                    format!("인성을 제어하는 {} 용재파인 용신", eokbu_element.hangul()),
539                    "인성이 너무 많아 신강해진 경우, 부작용을 막기 위해 재성으로 인성을 적절히 견제해야 합니다.".to_string(),
540                    reasons
541                )
542            } else if bijie > yinxing * 1.5 {
543                reasons.push(format!("비겁({}) 과다", bijie));
544                (
545                    format!("비겁을 제어하는 {} 관살제겁 용신", eokbu_element.hangul()),
546                    "자아가 너무 강해져 주변과 충돌하기 쉬운 경우, 관성으로 스스로를 다스리는 힘을 길러야 합니다.".to_string(),
547                    reasons
548                )
549            } else {
550                (
551                    format!("기운을 유통시키는 {} 설기생재 용신", eokbu_element.hangul()),
552                    "일간이 신강하여 기운이 옹색해지기 쉬우므로 식상으로 기운을 빼서 재성으로 연결해야 합니다.".to_string(),
553                    reasons
554                )
555            }
556        },
557        StrengthType::Balanced => {
558            (
559                format!("균형을 유지하는 {} 중화용신", eokbu_element.hangul()),
560                "이미 기운이 중화되어 안정적이므로, 현재의 균형을 유지해 주는 오행을 용신으로 삼습니다.".to_string(),
561                reasons
562            )
563        }
564    }
565}
