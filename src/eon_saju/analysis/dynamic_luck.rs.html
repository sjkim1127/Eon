1//! 동적 운세 분석 (Dynamic Luck Analysis)
2//! 
3//! 원국(Four Pillars)과 대운, 세운의 상호작용을 분석합니다.
4//! 특히 대운/세운이 들어오면서 완성되는 합충(合沖)을 포착합니다.
5

8use crate::core::pillars::FourPillars;
9use crate::analysis::relationships::{RelationshipAnalysis, TripleCombination, SeasonalCombination};
10use crate::core::branch::EarthlyBranch;

13/// 동적 분석 결과
14#[derive(Debug, Clone, Serialize, Deserialize)]
15pub struct DynamicLuckAnalysis {
16    /// 기준 원국
17    pub natal_relations: RelationshipAnalysis,
18    /// 대운 영향
19    pub major_influence: Option<LuckInfluence>,
20    /// 세운 영향
21    pub yearly_influence: Option<LuckInfluence>,
22    /// 월운 영향
23    pub monthly_influence: Option<LuckInfluence>,
24    /// 일운 영향
25    pub daily_influence: Option<LuckInfluence>,
26    /// 시운 영향
27    pub hourly_influence: Option<LuckInfluence>,
28    /// 원국 + 모든 운이 결합된 종합 합충
29    pub combined_relations: RelationshipAnalysis,
30}

32/// 특정 운(대운/세운)이 원국에 미치는 영향
33#[derive(Debug, Clone, Serialize, Deserialize)]
34pub struct LuckInfluence {
35    pub ganzi: GanZi,
36    pub label: String, // "대운", "세운" 등
37    pub relations_with_natal: Vec<String>,
38}

40impl DynamicLuckAnalysis {
41    /// 원국, 대운, 세운을 통합하여 분석합니다.
42    pub fn analyze(
43        natal: &FourPillars,
44        major: Option<GanZi>,
45        yearly: Option<GanZi>,
46        monthly: Option<GanZi>,
47        daily: Option<GanZi>,
48        hourly: Option<GanZi>,
49    ) -> Self {
50        let natal_relations = RelationshipAnalysis::from_pillars(natal);
51        
52        // 종합 분석을 위한 확장된 구성 요소들
53        let mut stems = vec![
54            ("년간", natal.year.stem),
55            ("월간", natal.month.stem),
56            ("일간", natal.day.stem),
57            ("시간", natal.hour.stem),
58        ];
59        let mut branches = vec![
60            ("년지", natal.year.branch),
61            ("월지", natal.month.branch),
62            ("일지", natal.day.branch),
63            ("시지", natal.hour.branch),
64        ];
65
66        if let Some(m) = major {
67            stems.push(("대운천간", m.stem));
68            branches.push(("대운지지", m.branch));
69        }
70        if let Some(y) = yearly {
71            stems.push(("세운천간", y.stem));
72            branches.push(("세운지지", y.branch));
73        }
74        if let Some(m) = monthly {
75            stems.push(("월운천간", m.stem));
76            branches.push(("월운지지", m.branch));
77        }
78        if let Some(d) = daily {
79            stems.push(("일운천간", d.stem));
80            branches.push(("일운지지", d.branch));
81        }
82        if let Some(h) = hourly {
83            stems.push(("시운천간", h.stem));
84            branches.push(("시운지지", h.branch));
85        }
86
87        // 확장된 구성을 기반으로 종합 합충 분석
88        let combined_relations = Self::analyze_expanded(&stems, &branches);
89
90        let major_influence = major.map(|m| Self::get_influence(m, "대운", natal));
91        let yearly_influence = yearly.map(|y| Self::get_influence(y, "세운", natal));
92        let monthly_influence = monthly.map(|m| Self::get_influence(m, "월운", natal));
93        let daily_influence = daily.map(|d| Self::get_influence(d, "일운", natal));
94        let hourly_influence = hourly.map(|h| Self::get_influence(h, "시운", natal));
95
96        Self {
97            natal_relations,
98            major_influence,
99            yearly_influence,
100            monthly_influence,
101            daily_influence,
102            hourly_influence,
103            combined_relations,
104        }
105    }

107    /// 확장된 천간/지지 목록을 바탕으로 합충 분석
108    fn analyze_expanded(
109        stems: &[(&'static str, HeavenlyStem)],
110        branches: &[(&'static str, EarthlyBranch)],
111    ) -> RelationshipAnalysis {
112        let mut analysis = RelationshipAnalysis {
113            stem_combinations: Vec::new(),
114            stem_clashes: Vec::new(),
115            triple_combinations: Vec::new(),
116            seasonal_combinations: Vec::new(),
117            dominant_semi_combinations: Vec::new(),
118            weak_semi_combinations: Vec::new(),
119            six_combinations: Vec::new(),
120            branch_clashes: Vec::new(),
121            branch_punishments: Vec::new(),
122            branch_harms: Vec::new(),
123            branch_destructions: Vec::new(),
124            am_combinations: Vec::new(),
125            myung_am_combinations: Vec::new(),
126            mapped_relationships: Vec::new(),
127        };
128
129        // 모든 쌍에 대한 분석 로직 호출
130        use crate::analysis::relationships::{
131            StemCombination, StemClash, SemiCombination, SixCombination, 
132            BranchClash, BranchPunishment, BranchHarm, BranchDestruction,
133            Amhap, MyungAmHap
134        };
135
136        // 천간 분석
137        for i in 0..stems.len() {
138            for j in (i + 1)..stems.len() {
139                let (p1, s1) = stems[i];
140                let (p2, s2) = stems[j];
141                if let Some(c) = StemCombination::check(s1, s2) {
142                    analysis.stem_combinations.push((c, p1.to_string(), p2.to_string()));
143                }
144                if let Some(c) = StemClash::check(s1, s2) {
145                    analysis.stem_clashes.push((c, p1.to_string(), p2.to_string()));
146                }
147            }
148        }
149
150        // 지지 분석
151        for i in 0..branches.len() {
152            for j in (i + 1)..branches.len() {
153                let (p1, b1) = branches[i];
154                let (p2, b2) = branches[j];
155                if let Some(s) = SemiCombination::check(b1, b2) {
156                    if s.is_dominant() {
157                        analysis.dominant_semi_combinations.push((s, p1.to_string(), p2.to_string()));
158                    } else {
159                        analysis.weak_semi_combinations.push((s, p1.to_string(), p2.to_string()));
160                    }
161                }
162                if let Some(s) = SixCombination::check(b1, b2) {
163                    analysis.six_combinations.push((s, p1.to_string(), p2.to_string()));
164                }
165                if let Some(c) = BranchClash::check(b1, b2) {
166                    analysis.branch_clashes.push((c, p1.to_string(), p2.to_string()));
167                }
168                if let Some(h) = BranchHarm::check(b1, b2) {
169                    analysis.branch_harms.push((h, p1.to_string(), p2.to_string()));
170                }
171                if let Some(d) = BranchDestruction::check(b1, b2) {
172                    analysis.branch_destructions.push((d, p1.to_string(), p2.to_string()));
173                }
174                if let Some(p) = BranchPunishment::check_self(b1, b2) {
175                    analysis.branch_punishments.push((p, p1.to_string(), p2.to_string()));
176                }
177                
178                // 암합
179                let ams = RelationshipAnalysis::check_am_combinations(b1, b2);
180                for am in ams {
181                    analysis.am_combinations.push((Amhap { branches: (b1, b2), combination: am }, p1.to_string(), p2.to_string()));
182                }
183            }
184        }
185
186        // 명암합
187        for (sp, s) in stems {
188            for (bp, b) in branches {
189                let mas = RelationshipAnalysis::check_myung_am_combinations(*s, *b);
190                for ma in mas {
191                    analysis.myung_am_combinations.push((MyungAmHap { stem: *s, branch: *b, combination: ma }, sp.to_string(), bp.to_string()));
192                }
193            }
194        }
195
196        // 삼합/방합 완성 체크
197        let all_b: Vec<_> = branches.iter().map(|(_, b)| *b).collect();
198        analysis.triple_combinations = TripleCombination::check(&all_b);
199        analysis.seasonal_combinations = SeasonalCombination::check(&all_b);
200
201        let is_luck = |p: &str| p == "대운" || p == "세운" || p == "월운" || p == "일운" || p == "시운";
202
203        let mut mapped = Vec::new();
204        let make_detail = |rel_type: &str, name: &str, p1: &str, p2: &str| crate::analysis::relationships::RelationshipDetail {
205            relation_type: rel_type.to_string(),
206            name: name.to_string(),
207            positions: vec![p1.to_string(), p2.to_string()],
208            level: crate::analysis::supplementary_pillars::InterpretationLevel::Neutral,
209            summary: format!("{}과 {}의 {}", p1, p2, name),
210            description: "".to_string(),
211            reasons: vec![],
212            transformed_element: None,
213        };
214
215        for (_, p1, p2) in &analysis.stem_combinations {
216            if is_luck(p1) && is_luck(p2) { mapped.push(make_detail("합", "천간합", p1, p2)); }
217        }
218        for (_, p1, p2) in &analysis.stem_clashes {
219            if is_luck(p1) && is_luck(p2) { mapped.push(make_detail("충", "천간충", p1, p2)); }
220        }
221        for (_, p1, p2) in &analysis.dominant_semi_combinations {
222            if is_luck(p1) && is_luck(p2) { mapped.push(make_detail("합", "반합(주도)", p1, p2)); }
223        }
224        for (_, p1, p2) in &analysis.weak_semi_combinations {
225            if is_luck(p1) && is_luck(p2) { mapped.push(make_detail("합", "반합(보조)", p1, p2)); }
226        }
227        for (_, p1, p2) in &analysis.six_combinations {
228            if is_luck(p1) && is_luck(p2) { mapped.push(make_detail("합", "육합", p1, p2)); }
229        }
230        for (_, p1, p2) in &analysis.branch_clashes {
231            if is_luck(p1) && is_luck(p2) { mapped.push(make_detail("충", "지지충", p1, p2)); }
232        }
233        for (_, p1, p2) in &analysis.branch_punishments {
234            if is_luck(p1) && is_luck(p2) { mapped.push(make_detail("형", "형", p1, p2)); }
235        }
236        for (_, p1, p2) in &analysis.branch_harms {
237            if is_luck(p1) && is_luck(p2) { mapped.push(make_detail("해", "해", p1, p2)); }
238        }
239        for (_, p1, p2) in &analysis.branch_destructions {
240            if is_luck(p1) && is_luck(p2) { mapped.push(make_detail("파", "파", p1, p2)); }
241        }
242
243        analysis.mapped_relationships = mapped;
244
245        analysis
246    }

248    pub fn get_influence(luck: GanZi, label: &str, natal: &FourPillars) -> LuckInfluence {
249        let mut relations = Vec::new();
250        // 원국과의 관계 정리
251        let n_stems = [natal.year.stem, natal.month.stem, natal.day.stem, natal.hour.stem];
252        let n_branches = [natal.year.branch, natal.month.branch, natal.day.branch, natal.hour.branch];
253        
254        use crate::analysis::relationships::*;
255        
256        // 1. 천간 관계
257        for s in &n_stems {
258            if let Some(_c) = StemCombination::check(luck.stem, *s) {
259                relations.push(format!("천간합: {} - {}", luck.stem.hanja(), s.hanja()));
260            }
261            if let Some(_c) = StemClash::check(luck.stem, *s) {
262                relations.push(format!("천간충: {} - {}", luck.stem.hanja(), s.hanja()));
263            }
264        }
265        
266        // 2. 지지 관계
267        for b in &n_branches {
268            // 육합
269            if let Some(_c) = SixCombination::check(luck.branch, *b) {
270                relations.push(format!("육합: {} - {}", luck.branch.hanja(), b.hanja()));
271            }
272            // 반합
273            if let Some(semi) = SemiCombination::check(luck.branch, *b) {
274                relations.push(format!("반합: {} ({}-{})", semi.hangul(), luck.branch.hanja(), b.hanja()));
275            }
276            // 충
277            if let Some(_c) = BranchClash::check(luck.branch, *b) {
278                relations.push(format!("지지충: {} - {}", luck.branch.hanja(), b.hanja()));
279            }
280            // 형
281            if let Some(p) = BranchPunishment::check_self(luck.branch, *b) {
282                relations.push(format!("지지형: {}", p.hangul()));
283            }
284            // 해
285            if let Some(_h) = BranchHarm::check(luck.branch, *b) {
286                relations.push(format!("지지해: {} - {}", luck.branch.hanja(), b.hanja()));
287            }
288            // 파
289            if let Some(_d) = BranchDestruction::check(luck.branch, *b) {
290                relations.push(format!("지지파: {} - {}", luck.branch.hanja(), b.hanja()));
291            }
292        }
293
294        LuckInfluence {
295            ganzi: luck,
296            label: label.to_string(),
297            relations_with_natal: relations,
298        }
299    }

302impl std::fmt::Display for DynamicLuckAnalysis {
303    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
304        writeln!(f, "【동적 종합 분석 (원국 + 대운 + 세운)】")?;
305        writeln!(f, "─────────────────────────────────")?;
306        
307        if let Some(infl) = &self.major_influence {
308            writeln!(f, "▶ 대운 영향 ({}): {}", infl.ganzi, infl.relations_with_natal.join(", "))?;
309        }
310        if let Some(infl) = &self.yearly_influence {
311            writeln!(f, "▶ 세운 영향 ({}): {}", infl.ganzi, infl.relations_with_natal.join(", "))?;
312        }
313        
314        writeln!(f, "\n[종합 합충 결과]")?;
315        write!(f, "{}", self.combined_relations)?;
316        
317        Ok(())
318    }
