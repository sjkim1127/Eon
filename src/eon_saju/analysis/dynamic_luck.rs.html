1//! 동적 운세 분석 (Dynamic Luck Analysis)
2//!
3//! 원국(Four Pillars)과 대운, 세운의 상호작용을 분석합니다.
4//! 특히 대운/세운이 들어오면서 완성되는 합충(合沖)을 포착합니다.
5

7    RelationshipAnalysis, SeasonalCombination, TripleCombination,
8};
9use crate::core::branch::EarthlyBranch;

11use crate::core::pillars::FourPillars;

15/// 동적 분석 결과
16#[derive(Debug, Clone, Serialize, Deserialize)]
17pub struct DynamicLuckAnalysis {
18    /// 기준 원국
19    pub natal_relations: RelationshipAnalysis,
20    /// 대운 영향
21    pub major_influence: Option<LuckInfluence>,
22    /// 세운 영향
23    pub yearly_influence: Option<LuckInfluence>,
24    /// 월운 영향
25    pub monthly_influence: Option<LuckInfluence>,
26    /// 일운 영향
27    pub daily_influence: Option<LuckInfluence>,
28    /// 시운 영향
29    pub hourly_influence: Option<LuckInfluence>,
30    /// 원국 + 모든 운이 결합된 종합 합충
31    pub combined_relations: RelationshipAnalysis,
32}

34/// 특정 운(대운/세운)이 원국에 미치는 영향
35#[derive(Debug, Clone, Serialize, Deserialize)]
36pub struct LuckInfluence {
37    pub ganzi: GanZi,
38    pub label: String, // "대운", "세운" 등
39    pub relations_with_natal: Vec<String>,
40}

42impl DynamicLuckAnalysis {
43    /// 원국, 대운, 세운을 통합하여 분석합니다.
44    pub fn analyze(
45        natal: &FourPillars,
46        major: Option<GanZi>,
47        yearly: Option<GanZi>,
48        monthly: Option<GanZi>,
49        daily: Option<GanZi>,
50        hourly: Option<GanZi>,
51    ) -> Self {
52        let natal_relations = RelationshipAnalysis::from_pillars(natal);
53
54        // 종합 분석을 위한 확장된 구성 요소들
55        let mut stems = vec![
56            ("년간", natal.year.stem),
57            ("월간", natal.month.stem),
58            ("일간", natal.day.stem),
59            ("시간", natal.hour.stem),
60        ];
61        let mut branches = vec![
62            ("년지", natal.year.branch),
63            ("월지", natal.month.branch),
64            ("일지", natal.day.branch),
65            ("시지", natal.hour.branch),
66        ];
67
68        if let Some(m) = major {
69            stems.push(("대운천간", m.stem));
70            branches.push(("대운지지", m.branch));
71        }
72        if let Some(y) = yearly {
73            stems.push(("세운천간", y.stem));
74            branches.push(("세운지지", y.branch));
75        }
76        if let Some(m) = monthly {
77            stems.push(("월운천간", m.stem));
78            branches.push(("월운지지", m.branch));
79        }
80        if let Some(d) = daily {
81            stems.push(("일운천간", d.stem));
82            branches.push(("일운지지", d.branch));
83        }
84        if let Some(h) = hourly {
85            stems.push(("시운천간", h.stem));
86            branches.push(("시운지지", h.branch));
87        }
88
89        // 확장된 구성을 기반으로 종합 합충 분석
90        let combined_relations = Self::analyze_expanded(&stems, &branches);
91
92        let major_influence = major.map(|m| Self::get_influence(m, "대운", natal));
93        let yearly_influence = yearly.map(|y| Self::get_influence(y, "세운", natal));
94        let monthly_influence = monthly.map(|m| Self::get_influence(m, "월운", natal));
95        let daily_influence = daily.map(|d| Self::get_influence(d, "일운", natal));
96        let hourly_influence = hourly.map(|h| Self::get_influence(h, "시운", natal));
97
98        Self {
99            natal_relations,
100            major_influence,
101            yearly_influence,
102            monthly_influence,
103            daily_influence,
104            hourly_influence,
105            combined_relations,
106        }
107    }

109    /// 확장된 천간/지지 목록을 바탕으로 합충 분석
110    fn analyze_expanded(
111        stems: &[(&'static str, HeavenlyStem)],
112        branches: &[(&'static str, EarthlyBranch)],
113    ) -> RelationshipAnalysis {
114        let mut analysis = RelationshipAnalysis {
115            stem_combinations: Vec::new(),
116            stem_clashes: Vec::new(),
117            triple_combinations: Vec::new(),
118            seasonal_combinations: Vec::new(),
119            dominant_semi_combinations: Vec::new(),
120            weak_semi_combinations: Vec::new(),
121            six_combinations: Vec::new(),
122            branch_clashes: Vec::new(),
123            branch_punishments: Vec::new(),
124            branch_harms: Vec::new(),
125            branch_destructions: Vec::new(),
126            am_combinations: Vec::new(),
127            myung_am_combinations: Vec::new(),
128            mapped_relationships: Vec::new(),
129        };
130
131        // 모든 쌍에 대한 분석 로직 호출
132        use crate::analysis::relationships::{
133            Amhap, BranchClash, BranchDestruction, BranchHarm, BranchPunishment, MyungAmHap,
134            SemiCombination, SixCombination, StemClash, StemCombination,
135        };
136
137        // 천간 분석
138        for i in 0..stems.len() {
139            for j in (i + 1)..stems.len() {
140                let (p1, s1) = stems[i];
141                let (p2, s2) = stems[j];
142                if let Some(c) = StemCombination::check(s1, s2) {
143                    analysis
144                        .stem_combinations
145                        .push((c, p1.to_string(), p2.to_string()));
146                }
147                if let Some(c) = StemClash::check(s1, s2) {
148                    analysis
149                        .stem_clashes
150                        .push((c, p1.to_string(), p2.to_string()));
151                }
152            }
153        }
154
155        // 지지 분석
156        for i in 0..branches.len() {
157            for j in (i + 1)..branches.len() {
158                let (p1, b1) = branches[i];
159                let (p2, b2) = branches[j];
160                if let Some(s) = SemiCombination::check(b1, b2) {
161                    if s.is_dominant() {
162                        analysis.dominant_semi_combinations.push((
163                            s,
164                            p1.to_string(),
165                            p2.to_string(),
166                        ));
167                    } else {
168                        analysis
169                            .weak_semi_combinations
170                            .push((s, p1.to_string(), p2.to_string()));
171                    }
172                }
173                if let Some(s) = SixCombination::check(b1, b2) {
174                    analysis
175                        .six_combinations
176                        .push((s, p1.to_string(), p2.to_string()));
177                }
178                if let Some(c) = BranchClash::check(b1, b2) {
179                    analysis
180                        .branch_clashes
181                        .push((c, p1.to_string(), p2.to_string()));
182                }
183                if let Some(h) = BranchHarm::check(b1, b2) {
184                    analysis
185                        .branch_harms
186                        .push((h, p1.to_string(), p2.to_string()));
187                }
188                if let Some(d) = BranchDestruction::check(b1, b2) {
189                    analysis
190                        .branch_destructions
191                        .push((d, p1.to_string(), p2.to_string()));
192                }
193                if let Some(p) = BranchPunishment::check_self(b1, b2) {
194                    analysis
195                        .branch_punishments
196                        .push((p, p1.to_string(), p2.to_string()));
197                }
198
199                // 암합
200                let ams = RelationshipAnalysis::check_am_combinations(b1, b2);
201                for am in ams {
202                    analysis.am_combinations.push((
203                        Amhap {
204                            branches: (b1, b2),
205                            combination: am,
206                        },
207                        p1.to_string(),
208                        p2.to_string(),
209                    ));
210                }
211            }
212        }
213
214        // 명암합
215        for (sp, s) in stems {
216            for (bp, b) in branches {
217                let mas = RelationshipAnalysis::check_myung_am_combinations(*s, *b);
218                for ma in mas {
219                    analysis.myung_am_combinations.push((
220                        MyungAmHap {
221                            stem: *s,
222                            branch: *b,
223                            combination: ma,
224                        },
225                        sp.to_string(),
226                        bp.to_string(),
227                    ));
228                }
229            }
230        }
231
232        // 삼합/방합 완성 체크
233        let all_b: Vec<_> = branches.iter().map(|(_, b)| *b).collect();
234        analysis.triple_combinations = TripleCombination::check(&all_b);
235        analysis.seasonal_combinations = SeasonalCombination::check(&all_b);
236
237        let is_luck =
238            |p: &str| p == "대운" || p == "세운" || p == "월운" || p == "일운" || p == "시운";
239
240        let mut mapped = Vec::new();
241        let make_detail = |rel_type: &str, name: &str, p1: &str, p2: &str| {
242            crate::analysis::relationships::RelationshipDetail {
243                relation_type: rel_type.to_string(),
244                name: name.to_string(),
245                positions: vec![p1.to_string(), p2.to_string()],
246                level: crate::analysis::supplementary_pillars::InterpretationLevel::Neutral,
247                summary: format!("{}과 {}의 {}", p1, p2, name),
248                description: "".to_string(),
249                reasons: vec![],
250                transformed_element: None,
251            }
252        };
253
254        for (_, p1, p2) in &analysis.stem_combinations {
255            if is_luck(p1) && is_luck(p2) {
256                mapped.push(make_detail("합", "천간합", p1, p2));
257            }
258        }
259        for (_, p1, p2) in &analysis.stem_clashes {
260            if is_luck(p1) && is_luck(p2) {
261                mapped.push(make_detail("충", "천간충", p1, p2));
262            }
263        }
264        for (_, p1, p2) in &analysis.dominant_semi_combinations {
265            if is_luck(p1) && is_luck(p2) {
266                mapped.push(make_detail("합", "반합(주도)", p1, p2));
267            }
268        }
269        for (_, p1, p2) in &analysis.weak_semi_combinations {
270            if is_luck(p1) && is_luck(p2) {
271                mapped.push(make_detail("합", "반합(보조)", p1, p2));
272            }
273        }
274        for (_, p1, p2) in &analysis.six_combinations {
275            if is_luck(p1) && is_luck(p2) {
276                mapped.push(make_detail("합", "육합", p1, p2));
277            }
278        }
279        for (_, p1, p2) in &analysis.branch_clashes {
280            if is_luck(p1) && is_luck(p2) {
281                mapped.push(make_detail("충", "지지충", p1, p2));
282            }
283        }
284        for (_, p1, p2) in &analysis.branch_punishments {
285            if is_luck(p1) && is_luck(p2) {
286                mapped.push(make_detail("형", "형", p1, p2));
287            }
288        }
289        for (_, p1, p2) in &analysis.branch_harms {
290            if is_luck(p1) && is_luck(p2) {
291                mapped.push(make_detail("해", "해", p1, p2));
292            }
293        }
294        for (_, p1, p2) in &analysis.branch_destructions {
295            if is_luck(p1) && is_luck(p2) {
296                mapped.push(make_detail("파", "파", p1, p2));
297            }
298        }
299
300        analysis.mapped_relationships = mapped;
301
302        analysis
303    }

305    pub fn get_influence(luck: GanZi, label: &str, natal: &FourPillars) -> LuckInfluence {
306        let mut relations = Vec::new();
307        // 원국과의 관계 정리
308        let n_stems = [
309            natal.year.stem,
310            natal.month.stem,
311            natal.day.stem,
312            natal.hour.stem,
313        ];
314        let n_branches = [
315            natal.year.branch,
316            natal.month.branch,
317            natal.day.branch,
318            natal.hour.branch,
319        ];
320
321        use crate::analysis::relationships::*;
322
323        // 1. 천간 관계
324        for s in &n_stems {
325            if let Some(_c) = StemCombination::check(luck.stem, *s) {
326                relations.push(format!("천간합: {} - {}", luck.stem.hanja(), s.hanja()));
327            }
328            if let Some(_c) = StemClash::check(luck.stem, *s) {
329                relations.push(format!("천간충: {} - {}", luck.stem.hanja(), s.hanja()));
330            }
331        }
332
333        // 2. 지지 관계
334        for b in &n_branches {
335            // 육합
336            if let Some(_c) = SixCombination::check(luck.branch, *b) {
337                relations.push(format!("육합: {} - {}", luck.branch.hanja(), b.hanja()));
338            }
339            // 반합
340            if let Some(semi) = SemiCombination::check(luck.branch, *b) {
341                relations.push(format!(
342                    "반합: {} ({}-{})",
343                    semi.hangul(),
344                    luck.branch.hanja(),
345                    b.hanja()
346                ));
347            }
348            // 충
349            if let Some(_c) = BranchClash::check(luck.branch, *b) {
350                relations.push(format!("지지충: {} - {}", luck.branch.hanja(), b.hanja()));
351            }
352            // 형
353            if let Some(p) = BranchPunishment::check_self(luck.branch, *b) {
354                relations.push(format!("지지형: {}", p.hangul()));
355            }
356            // 해
357            if let Some(_h) = BranchHarm::check(luck.branch, *b) {
358                relations.push(format!("지지해: {} - {}", luck.branch.hanja(), b.hanja()));
359            }
360            // 파
361            if let Some(_d) = BranchDestruction::check(luck.branch, *b) {
362                relations.push(format!("지지파: {} - {}", luck.branch.hanja(), b.hanja()));
363            }
364        }
365
366        LuckInfluence {
367            ganzi: luck,
368            label: label.to_string(),
369            relations_with_natal: relations,
370        }
371    }

374impl std::fmt::Display for DynamicLuckAnalysis {
375    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
376        writeln!(f, "【동적 종합 분석 (원국 + 대운 + 세운)】")?;
377        writeln!(f, "─────────────────────────────────")?;
378
379        if let Some(infl) = &self.major_influence {
380            writeln!(
381                f,
382                "▶ 대운 영향 ({}): {}",
383                infl.ganzi,
384                infl.relations_with_natal.join(", ")
385            )?;
386        }
387        if let Some(infl) = &self.yearly_influence {
388            writeln!(
389                f,
390                "▶ 세운 영향 ({}): {}",
391                infl.ganzi,
392                infl.relations_with_natal.join(", ")
393            )?;
394        }
395
396        writeln!(f, "\n[종합 합충 결과]")?;
397        write!(f, "{}", self.combined_relations)?;
398
399        Ok(())
400    }
