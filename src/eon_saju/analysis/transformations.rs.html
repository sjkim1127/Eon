14/// 합화 결과
15#[derive(Debug, Clone, Serialize, Deserialize)]
16pub struct EffectiveElement {
17    /// 원래 오행
18    pub original: Element,
19    /// 변한 오행 (합화된 경우)
20    pub effective: Element,
21    /// 합화 원인 (있을 경우)
22    pub reason: Option<String>,
23}

25/// 사주 전체의 실질적 오행 통계
26#[derive(Debug, Clone, Serialize, Deserialize)]
27pub struct TransformationAnalysis {
28    /// 년간 실질 오행
29    pub year_stem: EffectiveElement,
30    /// 월간 실질 오행
31    pub month_stem: EffectiveElement,
32    /// 일간 실질 오행
33    pub day_stem: EffectiveElement,
34    /// 시간 실질 오행
35    pub hour_stem: EffectiveElement,
36    /// 지지 4곳 포함 (지지는 주로 삼합 반영)
37    pub year_branch: EffectiveElement,
38    pub month_branch: EffectiveElement,
39    pub day_branch: EffectiveElement,
40    pub hour_branch: EffectiveElement,
41}
42
43impl TransformationAnalysis {
44    pub fn from_pillars(pillars: &FourPillars) -> Self {
45        let rel = pillars.relationships();
46        let month_branch_element = pillars.month.branch.element();
47
48        let mut year_stem = EffectiveElement::new(pillars.year.stem.element());
49        let mut month_stem = EffectiveElement::new(pillars.month.stem.element());
50        let mut day_stem = EffectiveElement::new(pillars.day.stem.element());
51        let mut hour_stem = EffectiveElement::new(pillars.hour.stem.element());
52
53        // 1. 천간 합화 처리
54        // 쟁합(争合) 및 투합(妬合) 분석: 한 천간이 여러 합에 관여하는지 확인
55        let mut stem_usage = [0u8; 4]; // 년, 월, 일, 시 순서
56        for (_combo, p1, p2) in &rel.stem_combinations {
57            let idx1 = match p1.as_str() {
58                "년간" => 0,
59                "월간" => 1,
60                "일간" => 2,
61                "시간" => 3,
62                _ => 99,
63            };
64            let idx2 = match p2.as_str() {
65                "년간" => 0,
66                "월간" => 1,
67                "일간" => 2,
68                "시간" => 3,
69                _ => 99,
70            };
71            if idx1 < 4 {
72                stem_usage[idx1] += 1;
73            }
74            if idx2 < 4 {
75                stem_usage[idx2] += 1;
76            }
77        }
78
79        for (combo, p1, p2) in &rel.stem_combinations {
80            let idx1 = match p1.as_str() {
81                "년간" => 0,
82                "월간" => 1,
83                "일간" => 2,
84                "시간" => 3,
85                _ => 99,
86            };
87            let idx2 = match p2.as_str() {
88                "년간" => 0,
89                "월간" => 1,
90                "일간" => 2,
91                "시간" => 3,
92                _ => 99,
93            };
94
95            // 쟁합/투합 발생 시 합화 불성립 (또는 약화되나 보통 불성립으로 간주)
96            let is_competed =
97                (idx1 < 4 && stem_usage[idx1] > 1) || (idx2 < 4 && stem_usage[idx2] > 1);
98
99            if is_competed {
100                let reason = format!(
101                    "{} 발생으로 합화 불성립",
102                    if stem_usage[idx1.min(3)] > 1 {
103                        "쟁합(争合)"
104                    } else {
105                        "투합(妬合)"
106                    }
107                );
108                // 실질 오행에 기록은 하되 원인은 "불성립"으로 명시
109                apply_stem_transform_failure(
110                    pillars,
111                    combo,
112                    &reason,
113                    &mut year_stem,
114                    &mut month_stem,
115                    &mut day_stem,
116                    &mut hour_stem,
117                );
118                continue;
119            }
120
121            let transformed_element = combo.transformed_element();
122
123            // 월지가 합화되는 오행을 돕거나 같을 때만 화(化)하는 것으로 간주
124            let is_supported = month_branch_element == transformed_element
125                || transformed_element.generated_by() == month_branch_element;
126
127            if is_supported {
128                let reason = format!("{}에 의한 합화", combo.hangul());
129                apply_stem_transform(
130                    pillars,
131                    combo,
132                    transformed_element,
133                    &reason,
134                    &mut year_stem,
135                    &mut month_stem,
136                    &mut day_stem,
137                    &mut hour_stem,
138                );
139            }
140        }
141
142        // 2. 지지 합화 처리 (삼합 우선)
143        let mut year_branch = EffectiveElement::new(pillars.year.branch.element());
144        let mut month_branch = EffectiveElement::new(pillars.month.branch.element());
145        let mut day_branch = EffectiveElement::new(pillars.day.branch.element());
146        let mut hour_branch = EffectiveElement::new(pillars.hour.branch.element());
147
148        for tri in &rel.triple_combinations {
149            let transformed = tri.element();
150            let reason = format!("{}에 의한 합화", tri.hangul());
151            apply_triple_transform(
152                pillars,
153                tri,
154                transformed,
155                &reason,
156                &mut year_branch,
157                &mut month_branch,
158                &mut day_branch,
159                &mut hour_branch,
160            );
161        }
162
163        // 3. 지지 방합 처리
164        for sea in &rel.seasonal_combinations {
165            let transformed = sea.element();
166            let reason = format!("{}에 의한 합화", sea.hangul());
167            apply_seasonal_transform(
168                pillars,
169                sea,
170                transformed,
171                &reason,
172                &mut year_branch,
173                &mut month_branch,
174                &mut day_branch,
175                &mut hour_branch,
176            );
177        }
178
179        // 4. 지지 반합 처리
180        for (semi, _p1, _p2) in &rel.dominant_semi_combinations {
181            // 오행 변환 (삼합과 동일 오행)
182            let transformed = match semi {
183                SemiCombination::YinWu | SemiCombination::WuXu => Element::Fire,
184                SemiCombination::ShenZi | SemiCombination::ZiChen => Element::Water,
185                SemiCombination::SiYou | SemiCombination::YouChou => Element::Metal,
186                SemiCombination::HaiMao | SemiCombination::MaoWei => Element::Wood,
187                _ => continue, // 혹시 모를 안전장치
188            };
189
190            let reason = format!("{}에 의한 합화", semi.hangul());
191            apply_semi_transform(
192                pillars,
193                semi,
194                transformed,
195                &reason,
196                &mut year_branch,
197                &mut month_branch,
198                &mut day_branch,
199                &mut hour_branch,
200            );
201        }
202
203        Self {
204            year_stem,
205            month_stem,
206            day_stem,
207            hour_stem,
208            year_branch,
209            month_branch,
210            day_branch,
211            hour_branch,
212        }
213    }
214
215    /// 실질 오행 개수 집계
216    pub fn counts(&self) -> [(Element, f32); 5] {
217        let mut counts = [
218            (Element::Wood, 0.0),
219            (Element::Fire, 0.0),
220            (Element::Earth, 0.0),
221            (Element::Metal, 0.0),
222            (Element::Water, 0.0),
223        ];
224
225        let elements = [
226            &self.year_stem,
227            &self.month_stem,
228            &self.day_stem,
229            &self.hour_stem,
230            &self.year_branch,
231            &self.month_branch,
232            &self.day_branch,
233            &self.hour_branch,
234        ];
235
236        for eff in elements {
237            let idx = eff.effective.index() as usize;
238            counts[idx].1 += 1.0;
239        }
240
241        counts
242    }
243}

255// 헬퍼 함수들
256fn apply_stem_transform(
257    pillars: &FourPillars,
258    combo: &StemCombination,
259    target: Element,
260    reason: &str,
261    y: &mut EffectiveElement,
262    m: &mut EffectiveElement,
263    d: &mut EffectiveElement,
264    h: &mut EffectiveElement,
265) {
266    let (s1, s2) = match combo {
267        StemCombination::JiaJi => (HeavenlyStem::Jia, HeavenlyStem::Ji),
268        StemCombination::YiGeng => (HeavenlyStem::Yi, HeavenlyStem::Geng),
269        StemCombination::BingXin => (HeavenlyStem::Bing, HeavenlyStem::Xin),
270        StemCombination::DingRen => (HeavenlyStem::Ding, HeavenlyStem::Ren),
271        StemCombination::WuGui => (HeavenlyStem::Wu, HeavenlyStem::Gui),
272    };
273    if pillars.year.stem == s1 || pillars.year.stem == s2 {
274        y.effective = target;
275        y.reason = Some(reason.to_string());
276    }
277    if pillars.month.stem == s1 || pillars.month.stem == s2 {
278        m.effective = target;
279        m.reason = Some(reason.to_string());
280    }
281    if pillars.day.stem == s1 || pillars.day.stem == s2 {
282        d.effective = target;
283        d.reason = Some(reason.to_string());
284    }
285    if pillars.hour.stem == s1 || pillars.hour.stem == s2 {
286        h.effective = target;
287        h.reason = Some(reason.to_string());
288    }
289}

291fn apply_stem_transform_failure(
292    pillars: &FourPillars,
293    combo: &StemCombination,
294    reason: &str,
295    y: &mut EffectiveElement,
296    m: &mut EffectiveElement,
297    d: &mut EffectiveElement,
298    h: &mut EffectiveElement,
299) {
300    let (s1, s2) = match combo {
301        StemCombination::JiaJi => (HeavenlyStem::Jia, HeavenlyStem::Ji),
302        StemCombination::YiGeng => (HeavenlyStem::Yi, HeavenlyStem::Geng),
303        StemCombination::BingXin => (HeavenlyStem::Bing, HeavenlyStem::Xin),
304        StemCombination::DingRen => (HeavenlyStem::Ding, HeavenlyStem::Ren),
305        StemCombination::WuGui => (HeavenlyStem::Wu, HeavenlyStem::Gui),
306    };
307    // 원인은 남기되 effective element는 원래 오행 유지
308    if (pillars.year.stem == s1 || pillars.year.stem == s2) && y.reason.is_none() {
309        y.reason = Some(reason.to_string());
310    }
311    if (pillars.month.stem == s1 || pillars.month.stem == s2) && m.reason.is_none() {
312        m.reason = Some(reason.to_string());
313    }
314    if (pillars.day.stem == s1 || pillars.day.stem == s2) && d.reason.is_none() {
315        d.reason = Some(reason.to_string());
316    }
317    if (pillars.hour.stem == s1 || pillars.hour.stem == s2) && h.reason.is_none() {
318        h.reason = Some(reason.to_string());
319    }
320}

322fn apply_triple_transform(
323    pillars: &FourPillars,
324    tri: &TripleCombination,
325    target: Element,
326    reason: &str,
327    y: &mut EffectiveElement,
328    m: &mut EffectiveElement,
329    d: &mut EffectiveElement,
330    h: &mut EffectiveElement,
331) {
332    let branches = match tri {
333        TripleCombination::YinWuXu => {
334            vec![EarthlyBranch::Yin, EarthlyBranch::Wu, EarthlyBranch::Xu]
335        }
336        TripleCombination::ShenZiChen => {
337            vec![EarthlyBranch::Shen, EarthlyBranch::Zi, EarthlyBranch::Chen]
338        }
339        TripleCombination::SiYouChou => {
340            vec![EarthlyBranch::Si, EarthlyBranch::You, EarthlyBranch::Chou]
341        }
342        TripleCombination::HaiMaoWei => {
343            vec![EarthlyBranch::Hai, EarthlyBranch::Mao, EarthlyBranch::Wei]
344        }
345    };
346    if branches.contains(&pillars.year.branch) {
347        y.effective = target;
348        y.reason = Some(reason.to_string());
349    }
350    if branches.contains(&pillars.month.branch) {
351        m.effective = target;
352        m.reason = Some(reason.to_string());
353    }
354    if branches.contains(&pillars.day.branch) {
355        d.effective = target;
356        d.reason = Some(reason.to_string());
357    }
358    if branches.contains(&pillars.hour.branch) {
359        h.effective = target;
360        h.reason = Some(reason.to_string());
361    }
362}

428        y.effective = target;

445impl std::fmt::Display for TransformationAnalysis {
446    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
447        writeln!(f, "【오행 합화(合化) 분석】")?;
448        writeln!(f, "─────────────────────────────────")?;
449
450        let items = [
451            ("년간", &self.year_stem),
452            ("월간", &self.month_stem),
453            ("일간", &self.day_stem),
454            ("시간", &self.hour_stem),
455            ("년지", &self.year_branch),
456            ("월지", &self.month_branch),
457            ("일지", &self.day_branch),
458            ("시지", &self.hour_branch),
459        ];
460
461        let mut has_transform = false;
462        for (name, eff) in items {
463            if let Some(reason) = &eff.reason {
464                writeln!(
465                    f,
466                    "  {} ({} → {}): {}",
467                    name,
468                    eff.original.hangul(),
469                    eff.effective.hangul(),
470                    reason
471                )?;
472                has_transform = true;
473            }
474        }
475
476        if !has_transform {
477            writeln!(f, "  합화된 오행이 없습니다.")?;
478        }
479
480        writeln!(f, "\n[실질 오행 통계]")?;
481        for (el, count) in self.counts() {
482            if count > 0.0 {
483                writeln!(f, "  {}: {:.1}개", el.hangul(), count)?;
484            }
485        }
486        Ok(())
487    }
488}

490impl FourPillars {
491    /// 오행 합화 분석
492    pub fn transformations(&self) -> TransformationAnalysis {
493        TransformationAnalysis::from_pillars(self)
494    }
495
496    /// 실질 오행 맵 추출 (년/월/일/시 간/지 순서)
497    /// 반환: [(원본, 실질); 8]
498    pub fn effective_elements(&self) -> [(Element, Element); 8] {
499        let trans = self.transformations();
500        [
501            (trans.year_stem.original, trans.year_stem.effective),
502            (trans.month_stem.original, trans.month_stem.effective),
503            (trans.day_stem.original, trans.day_stem.effective),
504            (trans.hour_stem.original, trans.hour_stem.effective),
505            (trans.year_branch.original, trans.year_branch.effective),
506            (trans.month_branch.original, trans.month_branch.effective),
507            (trans.day_branch.original, trans.day_branch.effective),
508            (trans.hour_branch.original, trans.hour_branch.effective),
509        ]
510    }
