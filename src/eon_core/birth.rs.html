40/// 달력 유형
41#[derive(Debug, Clone, Copy, PartialEq, Eq, Default, Serialize, Deserialize)]
42pub enum CalendarType {
43    /// 양력 (그레고리력)
44    #[default]
45    Solar,
46    /// 음력
47    Lunar {
48        /// 윤달 여부
49        is_leap_month: bool,
50    },
51}

53impl CalendarType {
54    pub fn is_solar(&self) -> bool {
55        matches!(self, CalendarType::Solar)
56    }
57
58    pub fn is_lunar(&self) -> bool {
59        matches!(self, CalendarType::Lunar { .. })
60    }
61}

63/// 출생 정보
64///
65/// 모든 운명 시스템(사주, 베딕, 자미두수 등)에서 공통으로 사용됩니다.
66///
67/// ## 타임존 지원
68///
69/// IANA 타임존 데이터베이스를 사용하여 역사적 썸머타임(DST)을 자동 처리합니다.
70///
71/// ### 한국 썸머타임 역사
72/// - 1948-1951년: 일광절약시간 시행
73/// - 1955-1960년: 일광절약시간 시행
74/// - 1987-1988년: 서울 올림픽 전후 일광절약시간 시행
75///
76/// ## 예시
77/// ```rust
78/// use eon_core::{BirthInfo, Location};
79///
80/// // 1988년 올림픽 시절 (썸머타임 적용 기간)
81/// let birth = BirthInfo::solar(1988, 5, 15, 10, 0)
82///     .with_timezone("Asia/Seoul");
83///
84/// // chrono-tz가 자동으로 +1시간(KDT) 보정
85/// ```
86#[derive(Debug, Clone, Serialize, Deserialize)]
87pub struct BirthInfo {
88    /// 년도
89    pub year: i32,
90    /// 월 (1-12)
91    pub month: u32,
92    /// 일 (1-31)
93    pub day: u32,
94    /// 시 (0-23)
95    pub hour: u32,
96    /// 분 (0-59)
97    pub minute: u32,
98    /// 달력 유형 (양력/음력)
99    pub calendar: CalendarType,
100    /// 출생 위치 (선택적)
101    pub location: Option<Location>,
102    /// IANA 타임존 (예: "Asia/Seoul", "America/New_York")
103    pub timezone: Option<String>,
104    /// 진태양시 사용 여부 (경도 기반 보정)
105    pub use_true_solar_time: bool,
106    /// 성별 (선택적)
107    pub gender: Option<Gender>,
108}

174    /// 타임존 파싱
175    fn parsed_timezone(&self) -> Option<Tz> {
176        self.timezone.as_ref().and_then(|tz| tz.parse().ok())
177    }
178
179    /// 해당 시간대의 로컬 DateTime 생성
180    fn local_datetime(&self) -> Option<NaiveDateTime> {
181        NaiveDate::from_ymd_opt(self.year, self.month, self.day)
182            .and_then(|d| d.and_hms_opt(self.hour, self.minute, 0))
183    }
184
185    /// 썸머타임(DST)이 적용된 UTC 시간 반환
186    ///
187    /// IANA 타임존 데이터베이스를 사용하여 역사적 DST를 자동 처리합니다.
188    pub fn to_utc(&self) -> Result<DateTime<Utc>, crate::error::CoreError> {
189        let naive = self
190            .local_datetime()
191            .ok_or(crate::error::CoreError::InvalidDateTime)?;
192
193        let tz = if let Some(ref tz_str) = self.timezone {
194            tz_str
195                .parse::<chrono_tz::Tz>()
196                .map_err(|_| crate::error::CoreError::InvalidTimezone(tz_str.clone()))?
197        } else {
198            chrono_tz::Asia::Seoul
199        };
200
201        // Handle ambiguous times (DST fall-back) by picking the earlier (DST) variant
202        match tz.from_local_datetime(&naive) {
203            chrono::LocalResult::Single(dt) => Ok(dt.with_timezone(&Utc)),
204            chrono::LocalResult::Ambiguous(dt1, _dt2) => Ok(dt1.with_timezone(&Utc)),
205            chrono::LocalResult::None => Err(crate::error::CoreError::NonExistentLocalTime),
206        }
207    }
208
209    /// 썸머타임(DST) 적용 여부 확인
210    pub fn is_dst(&self) -> bool {
211        if let (Some(tz), Some(naive)) = (self.parsed_timezone(), self.local_datetime()) {
212            let local_dt = match tz.from_local_datetime(&naive) {
213                chrono::LocalResult::Single(dt) => Some(dt),
214                chrono::LocalResult::Ambiguous(dt1, _) => Some(dt1),
215                chrono::LocalResult::None => None,
216            };
217            if let Some(dt) = local_dt {
218                let dst = dt.offset().dst_offset();
219                return dst.num_hours() > 0;
220            }
221        }
222        false
223    }
224
225    /// DST 오프셋 정보 반환 (시간 단위)
226    ///
227    /// 전체 UTC 오프셋(base + dst)을 반환합니다.
228    pub fn dst_offset_hours(&self) -> Option<i32> {
229        if let (Some(tz), Some(naive)) = (self.parsed_timezone(), self.local_datetime()) {
230            let local_dt = match tz.from_local_datetime(&naive) {
231                chrono::LocalResult::Single(dt) => Some(dt),
232                chrono::LocalResult::Ambiguous(dt1, _) => Some(dt1),
233                chrono::LocalResult::None => None,
234            };
235            if let Some(dt) = local_dt {
236                let base = dt.offset().base_utc_offset();
237                let dst = dt.offset().dst_offset();
238                return Some((base.num_hours() + dst.num_hours()) as i32);
239            }
240        }
241        None
242    }
243
244    /// 지역시 보정값 (분 단위) - 경도 기반
245    pub fn longitude_offset_minutes(&self) -> i32 {
246        if self.use_true_solar_time {
247            self.location
248                .as_ref()
249                .map(|l| l.time_offset_minutes())
250                .unwrap_or(0)
251        } else {
252            0
253        }
254    }
255
256    /// 보정된 시간 (시, 분)
257    ///
258    /// 진태양시를 사용하는 경우, 경도 기반 보정이 적용됩니다.
259    /// 주의: DST는 `to_utc()` 메서드에서 자동 처리됩니다.
260    pub fn corrected_time(&self) -> (u32, u32) {
261        let offset = self.longitude_offset_minutes();
262        if offset == 0 {
263            return (self.hour, self.minute);
264        }
265
266        let total_minutes =
267            (self.hour as i32 * 60 + self.minute as i32 + offset).rem_euclid(24 * 60);
268        let corrected_hour = (total_minutes / 60) as u32;
269        let corrected_minute = (total_minutes % 60) as u32;
270
271        (corrected_hour, corrected_minute)
272    }
273
274    /// 보정된 날짜와 시간 (경도 기반 진태양시)
275    pub fn corrected_datetime(&self) -> (i32, u32, u32, u32, u32) {
276        let offset = self.longitude_offset_minutes();
277        if offset == 0 {
278            return (self.year, self.month, self.day, self.hour, self.minute);
279        }
280
281        let dt = NaiveDate::from_ymd_opt(self.year, self.month, self.day)
282            .and_then(|d| d.and_hms_opt(self.hour, self.minute, 0))
283            .map(|dt| dt + Duration::minutes(offset as i64));
284
285        match dt {
286            Some(corrected) => (
287                corrected.year(),
288                corrected.month(),
289                corrected.day(),
290                corrected.hour(),
291                corrected.minute(),
292            ),
293            None => (self.year, self.month, self.day, self.hour, self.minute),
294        }
295    }
296
297    /// 사주 계산용 데이터 반환 (년, 월, 일, 시)
298    ///
299    /// 진태양시 보정이 적용된 값을 반환합니다.
300    pub fn for_saju(&self) -> (i32, u32, u32, u32) {
301        let (year, month, day, hour, _) = self.corrected_datetime();
302        (year, month, day, hour)
303    }
304}

306impl std::fmt::Display for BirthInfo {
307    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
308        let cal = match self.calendar {
309            CalendarType::Solar => "양력",
310            CalendarType::Lunar {
311                is_leap_month: false,
312            } => "음력",
313            CalendarType::Lunar {
314                is_leap_month: true,
315            } => "음력(윤달)",
316        };
317
318        write!(
319            f,
320            "{} {}년 {}월 {}일 {:02}:{:02}",
321            cal, self.year, self.month, self.day, self.hour, self.minute
322        )?;
323
324        if let Some(ref loc) = self.location {
325            write!(f, " ({})", loc.name)?;
326        }
327
328        if let Some(ref tz) = self.timezone {
329            write!(f, " [{}]", tz)?;
330        }
331
332        if let Some(gender) = self.gender {
333            write!(f, " {}", gender)?;
334        }
335
336        if self.is_dst() {
337            write!(f, " (썸머타임)")?;
338        }
339
340        if self.use_true_solar_time {
341            let offset = self.longitude_offset_minutes();
342            if offset != 0 {
343                write!(f, " [지역시 {:+}분]", offset)?;
344            }
345        }
346
347        Ok(())
348    }
349}
