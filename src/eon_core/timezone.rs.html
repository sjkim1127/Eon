1//! 타임존 유틸리티
2//!
3//! IANA 타임존 문자열에서 표준 자오선(경도)을 계산합니다.
4
5use chrono::TimeZone;
6use chrono_tz::Tz;
7
8/// IANA 타임존 문자열로부터 표준 자오선(경도, 도)을 계산합니다.
9///
10/// 예: "Asia/Seoul" (UTC+9) → 135.0, "America/New_York" (UTC-5) → -75.0
11///
12/// 파싱 실패 시 한국 표준경도 135.0을 반환합니다.
13pub fn standard_meridian_from_tz(timezone: &str) -> f64 {
14    if let Ok(tz) = timezone.parse::<Tz>() {
15        let ref_dt = tz.with_ymd_and_hms(2024, 1, 15, 12, 0, 0).single();
16        if let Some(dt) = ref_dt {
17            use chrono_tz::OffsetComponents;
18            let base_offset_secs = dt.offset().base_utc_offset().num_seconds() as f64;
19            return (base_offset_secs / 3600.0) * 15.0;
20        }
21    }
22    135.0 // Fallback: Korean Standard Meridian
23}

29    #[test]
30    fn test_asia_seoul() {
31        let m = standard_meridian_from_tz("Asia/Seoul");
32        assert!((m - 135.0).abs() < 0.01);
33    }

36    fn test_america_new_york() {
37        let m = standard_meridian_from_tz("America/New_York");
38        assert!((m - (-75.0)).abs() < 0.01);
39    }
40
41    #[test]
42    fn test_invalid_fallback() {
43        let m = standard_meridian_from_tz("invalid/tz");
44        assert!((m - 135.0).abs() < 0.01);
45    }
