3/// 구궁 (Nine Palaces)
4#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize, Hash)]
5pub enum Palace {
6    Kan1,    // 1궁 (감궁, 北)
7    Kun2,    // 2궁 (곤궁, 西南)
8    Zhen3,   // 3궁 (진궁, 東)
9    Xun4,    // 4궁 (손궁, 東南)
10    Center5, // 5궁 (중궁, 中央)
11    Qian6,   // 6궁 (건궁, 西北)
12    Dui7,    // 7궁 (태궁, 西)
13    Gen8,    // 8궁 (간궁, 東北)
14    Li9,     // 9궁 (이궁, 南)
15}
16
17impl Palace {
18    pub fn as_u8(&self) -> u8 {
19        match self {
20            Palace::Kan1 => 1,
21            Palace::Kun2 => 2,
22            Palace::Zhen3 => 3,
23            Palace::Xun4 => 4,
24            Palace::Center5 => 5,
25            Palace::Qian6 => 6,
26            Palace::Dui7 => 7,
27            Palace::Gen8 => 8,
28            Palace::Li9 => 9,
29        }
30    }
31
32    pub fn from_u8(num: u8) -> Option<Self> {
33        match num {
34            1 => Some(Palace::Kan1),
35            2 => Some(Palace::Kun2),
36            3 => Some(Palace::Zhen3),
37            4 => Some(Palace::Xun4),
38            5 => Some(Palace::Center5),
39            6 => Some(Palace::Qian6),
40            7 => Some(Palace::Dui7),
41            8 => Some(Palace::Gen8),
42            9 => Some(Palace::Li9),
43            _ => None,
44        }
45    }
46}
