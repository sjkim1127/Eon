20/// 양둔/음둔과 국수(1~9)를 받아 지반(Earth Plate) 9궁의 간(Stem) 배치를 반환한다.
21pub fn build_earth_plate(is_yin_ju: bool, ju_number: u8) -> [Option<HeavenlyStem>; 9] {
22    // 인덱스는 0~8이며, 이는 1궁~9궁에 매핑된다.
23    let mut earth_stems = [None; 9];
24
25    // 무(戊)가 배치될 시작 궁 번호 (1~9)
26    let start_palace = ju_number;
27
28    for (i, &stem) in EARTH_PLATE_ORDER.iter().enumerate() {
29        // 궁의 번호 계산
30        let current_palace_num = if !is_yin_ju {
31            // 양둔: 순행 (+)
32            let mut num = start_palace as i32 + i as i32;
33            while num > 9 {
34                num -= 9;
35            }
36            num as u8
37        } else {
38            // 음둔: 역행 (-)
39            let mut num = start_palace as i32 - i as i32;
40            while num < 1 {
41                num += 9;
42            }
43            num as u8
44        };
45
46        earth_stems[(current_palace_num - 1) as usize] = Some(stem);
47    }
48
49    earth_stems
50}

52pub fn get_xun_shou(pillar: GanZi) -> HeavenlyStem {
53    let diff = (pillar.branch.index() as i32 - pillar.stem.index() as i32).rem_euclid(12);
54    match diff {
55        0 => HeavenlyStem::Wu,   // 갑자순
56        10 => HeavenlyStem::Ji,  // 갑술순
57        8 => HeavenlyStem::Geng, // 갑신순
58        6 => HeavenlyStem::Xin,  // 갑오순
59        4 => HeavenlyStem::Ren,  // 갑진순
60        2 => HeavenlyStem::Gui,  // 갑인순
61        _ => unreachable!(),
62    }
63}

65pub fn get_base_star(palace: u8) -> Star {
66    match palace {
67        1 => Star::Peng,
68        2 => Star::Rui,
69        3 => Star::Chong,
70        4 => Star::Fu,
71        5 => Star::Rui, // 중궁 기궁 (곤2궁)
72        6 => Star::Xin,
73        7 => Star::Zhu,
74        8 => Star::Ren,
75        9 => Star::Ying,
76        _ => unreachable!(),
77    }
78}

95/// 낙서구궁 외곽(Bagua ring)의 순환 순서 (1-8-3-4-9-2-7-6)
96pub const RING_PALACES: [u8; 8] = [1, 8, 3, 4, 9, 2, 7, 6];

98pub fn build_qimen_pan(
99    time: chrono::DateTime<chrono::Utc>,
100    year_pillar: GanZi,
101    month_pillar: GanZi,
102    day_pillar: GanZi,
103    hour_pillar: GanZi,
104    is_yin_ju: bool,
105    ju_number: u8,
106) -> QimenPan {
107    let mut palaces: Vec<PalaceState> = (1..=9)
108        .map(|num| PalaceState {
109            palace: Palace::from_u8(num).unwrap(),
110            earth_stem: None,
111            heaven_stem: None,
112            door: None,
113            star: None,
114            deity: None,
115        })
116        .collect();
117
118    // 1. 지반(Earth Plate) 포국
119    let earth_stems = build_earth_plate(is_yin_ju, ju_number);
120    for i in 0..9 {
121        palaces[i].earth_stem = earth_stems[i];
122    }
123
124    // 2. 직부(Value Chief), 직사(Value Envoy) 찾기
125    let xun_shou = get_xun_shou(hour_pillar);
126
127    // 순수(Xun Shou) 천간이 지반에서 어느 궁에 있는지 찾는다.
128    let mut xun_shou_palace = 1;
129    for i in 0..9 {
130        if palaces[i].earth_stem == Some(xun_shou) {
131            xun_shou_palace = (i + 1) as u8;
132            break;
133        }
134    }
135
136    // 중궁(5궁)에 있다면 기궁 법칙에 따라 2궁으로 간주.
137    let base_palace_for_chief = if xun_shou_palace == 5 {
138        2
139    } else {
140        xun_shou_palace
141    };
142
143    let value_chief_star = get_base_star(base_palace_for_chief);
144    let value_envoy_door = get_base_door(base_palace_for_chief);
145
146    // 3. 천반(Heaven Plate) 및 구성(Star Plate) 포국
147    // 직부(Star)는 시간(Hour Stem)이 지반에 있는 궁으로 이동한다. (단, 시간이 甲이면 순수와 같으므로 직부 궁에 그대로)
148    let hour_stem = hour_pillar.stem;
149    let target_palace_for_chief = if hour_stem == HeavenlyStem::Jia {
150        // 시간이 甲이면 은복되므로 순수와 일치.
151        base_palace_for_chief
152    } else {
153        let mut p = 1;
154        for i in 0..9 {
155            if palaces[i].earth_stem == Some(hour_stem) {
156                p = (i + 1) as u8;
157                break;
158            }
159        }
160        if p == 5 {
161            2
162        } else {
163            p
164        }
165    };
166
167    // 구궁의 외곽(Ring) 인덱스
168    // RING_PALACES = [1, 8, 3, 4, 9, 2, 7, 6]
169    let base_ring_idx = RING_PALACES
170        .iter()
171        .position(|&x| x == base_palace_for_chief)
172        .unwrap_or(0);
173    let target_ring_idx = RING_PALACES
174        .iter()
175        .position(|&x| x == target_palace_for_chief)
176        .unwrap_or(0);
177
178    // 외곽 회전 (얼마나 이동했는지)
179    let star_rotation = (target_ring_idx as i32 - base_ring_idx as i32).rem_euclid(8) as usize;
180
181    for i in 0..8 {
182        let original_palace = RING_PALACES[i];
183        let original_star = get_base_star(original_palace);
184
185        // 지반에 있던 천간이 천반으로 함께 이동
186        let original_earth_stem = if original_palace == 2 {
187            // 곤2궁은 중궁(5궁)의 지반간도 함께 가져올 수 있으나, 일반적으로는 본궁의 천간을 가져감.
188            // 기문둔갑에서 중궁 기궁 시 천반간 처리는 문파마다 다르나,
189            // 여기서는 본래 2궁의 천간과 5궁의 천간 중 어떤 것을 취할지에 대해 단순화: 원래 궁의 천간 이동
190            palaces[(original_palace - 1) as usize].earth_stem
191        } else {
192            palaces[(original_palace - 1) as usize].earth_stem
193        };
194
195        let new_ring_idx = (i + star_rotation) % 8;
196        let new_palace = RING_PALACES[new_ring_idx];
197
198        palaces[(new_palace - 1) as usize].star = Some(original_star);
199        palaces[(new_palace - 1) as usize].heaven_stem = original_earth_stem;
200    }
201
202    // 중궁(5궁)의 천반성/간은 기본적으로 비워두거나(None) 지반과 동일하게 처리.
203    palaces[4].star = Some(Star::Qin);
204    palaces[4].heaven_stem = palaces[4].earth_stem;
205
206    // 4. 팔문(Human Plate) 포국
207    // 직사(Door)는 시지(Hour Branch)에 맞게 이동.
208    // 직사의 원래 위치는 xun_shou_palace (또는 2궁).
209    // 여기서 순수(Xun Shou) 지지의 궁에서 출발해, 시지까지 양/음둔 순역 무관하게 양둔은 순행(1->2->3...), 음둔은 역행(9->8->7...)한다는 설도 있으나,
210    // 대다수 문파: 직사는 순수 지지(예: 甲子순이면 子) 궁에서 시작, 시간에 도달할 때까지 궁수를 양둔은 +1, 음둔은 -1 씩 추산하여 도달한 궁이 타겟 궁.
211    // 1~9궁 순서로 이동 (1,2,3...9) (중궁은 제외하고 1~9로 세거나, 혹은 포함하여 세고 5면 2로).
212    // 좀 더 표준적인 방법:
213    let xun_branch_idx = match get_xun_shou(hour_pillar) {
214        HeavenlyStem::Wu => 0,   // 자
215        HeavenlyStem::Ji => 10,  // 술
216        HeavenlyStem::Geng => 8, // 신
217        HeavenlyStem::Xin => 6,  // 오
218        HeavenlyStem::Ren => 4,  // 진
219        HeavenlyStem::Gui => 2,  // 인
220        _ => 0,
221    };
222    let hour_branch_idx = hour_pillar.branch.index();
223    let steps = (hour_branch_idx as i32 - xun_branch_idx).rem_euclid(12);
224
225    let mut door_target_palace = base_palace_for_chief as i32;
226    if !is_yin_ju {
227        door_target_palace += steps;
228        while door_target_palace > 9 {
229            door_target_palace -= 9;
230        }
231    } else {
232        door_target_palace -= steps;
233        while door_target_palace < 1 {
234            door_target_palace += 9;
235        }
236    }
237    if door_target_palace == 5 {
238        door_target_palace = 2;
239    }
240
241    let base_door_ring_idx = RING_PALACES
242        .iter()
243        .position(|&x| x == base_palace_for_chief)
244        .unwrap_or(0);
245    let target_door_ring_idx = RING_PALACES
246        .iter()
247        .position(|&x| x == door_target_palace as u8)
248        .unwrap_or(0);
249    let door_rotation =
250        (target_door_ring_idx as i32 - base_door_ring_idx as i32).rem_euclid(8) as usize;
251
252    for i in 0..8 {
253        let original_palace = RING_PALACES[i];
254        let original_door = get_base_door(original_palace);
255        let new_ring_idx = (i + door_rotation) % 8;
256        let new_palace = RING_PALACES[new_ring_idx];
257        palaces[(new_palace - 1) as usize].door = Some(original_door);
258    }
259
260    // 5. 팔신(Deity Plate) 포국
261    // 소위 천반 팔신 (가장 흔함): 직부가 천반 직부(Star)가 위치한 궁에 오고,
262    // 양둔이면 시계방향(1-8-3-4-9-2-7-6), 음둔이면 반시계방향으로 8신이 배치.
263    let deity_order_yang = [
264        Deity::ZhiFu,
265        Deity::TengShe,
266        Deity::TaiYin,
267        Deity::LiuHe,
268        Deity::BaiHu,
269        Deity::XuanWu,
270        Deity::JiuDi,
271        Deity::JiuTian,
272    ];
273    let deity_order_yin = [
274        Deity::ZhiFu,
275        Deity::JiuTian,
276        Deity::JiuDi,
277        Deity::XuanWu,
278        Deity::BaiHu,
279        Deity::LiuHe,
280        Deity::TaiYin,
281        Deity::TengShe,
282    ];
283    let deities = if !is_yin_ju {
284        &deity_order_yang
285    } else {
286        &deity_order_yin
287    };
288
289    for i in 0..8 {
290        let new_ring_idx = (target_ring_idx + i) % 8;
291        let new_palace = RING_PALACES[new_ring_idx];
292        palaces[(new_palace - 1) as usize].deity = Some(deities[i]);
293    }
294
295    QimenPan {
296        time,
297        year_pillar,
298        month_pillar,
299        day_pillar,
300        hour_pillar,
301        is_yin_ju,
302        ju_number,
303        palaces,
304        value_chief_star: Some(value_chief_star),
305        value_envoy_door: Some(value_envoy_door),
306    }
307}
