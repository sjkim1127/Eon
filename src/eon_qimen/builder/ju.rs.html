70/// 차보법(Chai Bu)으로 해당 일진(Day Pillar)의 원(Yuan)을 구한다.
71/// 반환값: 0 (상원), 1 (중원), 2 (하원)
72pub fn get_yuan_index(day_pillar: GanZi) -> usize {
73    // 기문둔갑 차보법에서는 甲, 己 일의 지지에 따라 원(상,중,하)이 바뀐다.
74    // 甲, 己 일까지 며칠 뒤/앞인지 계산하여 부두(符头)를 찾는다.
75    // 甲(0), 己(5). 현재 천간이 무엇이든 간에, 최근의 甲 또는 己일이 부두가 된다.
76    let stem_idx = day_pillar.stem.index();
77
78    // 부두까지의 거리는 현재 천간이 갑(0)~무(4)이면 stem_idx만큼 빼면 甲일.
79    // 기(5)~계(9)이면 stem_idx - 5만큼 빼면 己일.
80    let offset = stem_idx % 5;
81
82    // 부두의 지지 인덱스를 구한다.
83    let branch_idx = (day_pillar.branch.index() as i32 - offset as i32).rem_euclid(12) as u8;
84
85    // 부두의 지지에 따라 원이 결정됨
86    // 子(0), 午(6), 卯(3), 酉(9) -> 상원 (0)
87    // 寅(2), 申(8), 巳(5), 亥(11) -> 중원 (1)
88    // 辰(4), 戌(10), 丑(1), 未(7) -> 하원 (2)
89    match branch_idx {
90        0 | 3 | 6 | 9 => 0,  // 상원
91        2 | 5 | 8 | 11 => 1, // 중원
92        1 | 4 | 7 | 10 => 2, // 하원
93        _ => unreachable!(),
94    }
95}

97/// UTC 시각을 받아 24절기 인덱스를 반환 (0: 입춘, 1: 우수, ... 23: 대한)
98pub fn get_24_solar_term_index(dt: DateTime<Utc>) -> Result<u8, eon_astro::AstroError> {
99    let engine = AstroEngine::new();
100    let sun_long = engine.get_sun_longitude(dt)?;
101    let adjusted = (sun_long - 315.0 + 360.0) % 360.0;
102    Ok((adjusted / 15.0).floor() as u8 % 24)
103}

105/// 차보법 기준 국수와 음양둔 산출
106pub fn calculate_ju(
107    dt: DateTime<Utc>,
108    day_pillar: GanZi,
109) -> Result<(bool, u8), eon_astro::AstroError> {
110    let term_idx = get_24_solar_term_index(dt)?;
111    let yuan_idx = get_yuan_index(day_pillar);
112
113    let (ju_type, term_ju) = JU_TABLE[term_idx as usize];
114    let is_yin_ju = ju_type == JuType::Yin;
115
116    let ju_number = match yuan_idx {
117        0 => term_ju.0,
118        1 => term_ju.1,
119        2 => term_ju.2,
120        _ => unreachable!(),
121    };
122
123    Ok((is_yin_ju, ju_number))
124}
