24/// 만세력 바이너리 캐시 구조체
25#[derive(Debug, Clone, Serialize, Deserialize, Default)]
26pub struct ManseryukCache {
27    /// 연도별 절기 데이터
28    pub years: HashMap<i32, SolarTermTable>,
29    /// 연도별 음력 데이터 (해당 연도에 시작하는 음력 월 목록)
30    pub lunar_months: HashMap<i32, Vec<LunarMonthRecord>>,
31}
32
33impl ManseryukCache {
34    /// 바이너리 데이터로부터 캐시 로드
35    pub fn from_binary(bytes: &[u8]) -> Result<Self, crate::error::DataError> {
36        bincode::deserialize(bytes)
37            .map_err(|e| crate::error::DataError::Deserialization(e.to_string()))
38    }
39
40    /// 특정 연도/절기의 시각 조회
41    pub fn get_solar_term(&self, year: i32, term_idx: u8) -> Option<DateTime<Utc>> {
42        self.years.get(&year).and_then(|table| {
43            // Terms are stored in sorted order by index; use binary search for O(log n) lookup.
44            table
45                .terms
46                .binary_search_by_key(&term_idx, |(idx, _)| *idx)
47                .ok()
48                .map(|i| table.terms[i].1)
49        })
50    }
51
52    /// 특정 시점이 포함된 절기 인덱스 조회 (정적 데이터 기반 고속 검색)
53    pub fn get_term_index_at(&self, time: DateTime<Utc>) -> Option<u8> {
54        let year = time.year();
55        let table = self.years.get(&year)?;
56
57        // Terms are in chronological order; find the latest term that started at or before `time`
58        // using binary search via partition_point (O(log n) instead of O(n)).
59        let pos = table
60            .terms
61            .partition_point(|(_, term_time)| *term_time <= time);
62        if pos == 0 {
63            // `time` is before all solar terms of this year. This edge case is rare in practice;
64            // match the original behavior of returning index 0 (or the first term's index if stored
65            // differently) rather than None, since callers expect a valid term index.
66            Some(table.terms.first().map_or(0, |(idx, _)| *idx))
67        } else {
68            Some(table.terms[pos - 1].0)
69        }
70    }
71
72    /// 특정 양력 날짜에 대응하는 음력 데이터 조회
73    pub fn get_lunar_date(&self, solar_date: chrono::NaiveDate) -> Option<(i32, u32, u32, bool)> {
74        let year = solar_date.year();
75        // Current year and previous year data (lunar month 1 often starts in February).
76        let prev_months = self
77            .lunar_months
78            .get(&(year - 1))
79            .map(|v| v.as_slice())
80            .unwrap_or(&[]);
81        let curr_months = self
82            .lunar_months
83            .get(&year)
84            .map(|v| v.as_slice())
85            .unwrap_or(&[]);
86
87        // Each slice is sorted by new_moon_date; use binary search (partition_point) to find the
88        // latest record whose new_moon_date <= solar_date in each slice, then pick the most recent.
89        // This avoids allocating a combined Vec and runs in O(log n) per slice.
90        let prev_pos = prev_months.partition_point(|r| r.new_moon_date <= solar_date);
91        let curr_pos = curr_months.partition_point(|r| r.new_moon_date <= solar_date);
92
93        let prev_best = if prev_pos > 0 {
94            Some(&prev_months[prev_pos - 1])
95        } else {
96            None
97        };
98        let curr_best = if curr_pos > 0 {
99            Some(&curr_months[curr_pos - 1])
100        } else {
101            None
102        };
103
104        let record = match (prev_best, curr_best) {
105            (Some(a), Some(b)) => {
106                if b.new_moon_date >= a.new_moon_date {
107                    b
108                } else {
109                    a
110                }
111            }
112            (Some(a), None) => a,
113            (None, Some(b)) => b,
114            (None, None) => return None,
115        };
116
117        let day = (solar_date - record.new_moon_date).num_days() as u32 + 1;
118        Some((record.lunar_year, record.lunar_month, day, record.is_leap))
119    }
120}

122/// 전역 공유 캐시 인스턴스 (Lazy initialization)
123static GLOBAL_CACHE_INTERNAL: OnceLock<ManseryukCache> = OnceLock::new();
124
125pub struct GlobalCache;
126
127impl GlobalCache {
128    pub fn get() -> &'static ManseryukCache {
129        GLOBAL_CACHE_INTERNAL.get_or_init(|| {
130            // 빌드 시 포함된 바이너리 데이터 로드
131            let bytes = include_bytes!("manseryuk.bin");
132            ManseryukCache::from_binary(bytes).unwrap_or_else(|e| {
133                eprintln!("Failed to load manseryuk cache: {}", e);
134                ManseryukCache::default()
135            })
136        })
137    }
138
139    pub fn get_solar_term(year: i32, term_idx: u8) -> Option<DateTime<Utc>> {
140        Self::get().get_solar_term(year, term_idx)
141    }
142
143    pub fn get_lunar_date(solar_date: chrono::NaiveDate) -> Option<(i32, u32, u32, bool)> {
144        Self::get().get_lunar_date(solar_date)
145    }
