8/// 특정 날짜의 일진(日辰) 60갑자 인덱스 조회
9///
10/// 율리우스 적일을 사용하여 계산합니다.
11pub fn get_day_ganzi_index(date: NaiveDate) -> u8 {
12    let jdn = gregorian_to_jdn(date.year(), date.month() as i32, date.day() as i32);
13    // 2000-01-01 = JDN 2451545 = 戊午일(index=54)
14    ((jdn + 49) % 60) as u8
15}
16
17/// 특정 연도의 절기 시각 데이터 조회
18///
19/// 바이너리 캐시에서 먼저 조회하고, 없으면 eon-astro를 통해 실시간 계산합니다.
20pub fn get_solar_term_datetime(
21    year: i32,
22    term_index: u8,
23) -> Result<DateTime<Utc>, crate::error::DataError> {
24    // 1. 바이너리 캐시에서 먼저 조회
25    if let Some(dt) = GlobalCache::get_solar_term(year, term_index) {
26        return Ok(dt);
27    }
28
29    // 2. 캐시가 없으면 eon-astro를 사용하여 실시간 정밀 계산
30    let engine = eon_astro::AstroEngine::new();
31
32    // 절기별 대략적인 위치 (입춘=315도)
33    // 2월 초순(입춘)부터 약 15일 간격으로 24절기가 배치됨
34    // 대략적인 시작 시점을 2월 1일로 잡고 검색
35    let approx_start = Utc
36        .with_ymd_and_hms(year, 2, 1, 0, 0, 0)
37        .single()
38        .ok_or(crate::error::DataError::InvalidDate)?;
39
40    match engine.find_solar_term_time(approx_start, term_index) {
41        Ok(dt) => Ok(dt),
42        Err(_) => {
43            // 엔진 실패시 최후의 폴백 (근사값 계산)
44            // 입춘(0번)은 대략 2월 4일
45            // 각 절기/중기는 약 15.22일 간격
46            let base_date = Utc
47                .with_ymd_and_hms(year, 2, 4, 0, 0, 0)
48                .single()
49                .ok_or(crate::error::DataError::InvalidDate)?;
50            let offset_days = (term_index as f64 * 15.218).round() as i64;
51            Ok(base_date + chrono::Duration::days(offset_days))
52        }
53    }
54}

56/// 그레고리력 날짜를 율리우스 적일(JDN)로 변환
57fn gregorian_to_jdn(year: i32, month: i32, day: i32) -> i64 {
58    let a = (14 - month) / 12;
59    let y = year + 4800 - a;
60    let m = month + 12 * a - 3;
61
62    let jdn = day + (153 * m + 2) / 5 + 365 * y + y / 4 - y / 100 + y / 400 - 32045;
63    jdn as i64
64}

66/// 음력-양력 변환 데이터
67pub struct LunarCalendar;
68
69impl LunarCalendar {
70    /// 양력을 음력으로 변환 (KST 기준 정밀 계산) - 캐시 사용
71    pub fn from_solar(date: NaiveDate) -> Result<(i32, u32, u32, bool), crate::error::DataError> {
72        // 1. 캐시에서 먼저 조회
73        if let Some(res) = GlobalCache::get_lunar_date(date) {
74            return Ok(res);
75        }
76
77        // 2. 캐시가 없으면 실시간 정밀 계산
78        Self::from_solar_internal(date)
79    }
80
81    /// 양력을 음력으로 변환 (실시간 천문 계산 엔진 사용)
82    pub fn from_solar_internal(
83        date: NaiveDate,
84    ) -> Result<(i32, u32, u32, bool), crate::error::DataError> {
85        let engine = eon_astro::AstroEngine::new();
86        let kst_offset = chrono::Duration::hours(9);
87
88        // 1. KST 당일의 끝 시점의 UTC 변환
89        let dt_utc = Utc
90            .with_ymd_and_hms(date.year(), date.month(), date.day(), 23, 59, 59)
91            .single()
92            .ok_or(crate::error::DataError::InvalidDate)?
93            - kst_offset;
94
95        // 2. 현재 달 혹은 당일의 합삭(New Moon) 시각 찾기
96        let nm_t = engine.find_new_moon_before(dt_utc)?;
97        let nm_date_kst = (nm_t + kst_offset).date_naive();
98        let lunar_day = (date - nm_date_kst).num_days() as u32 + 1;
99
100        // 3. 월 및 연도 결정을 위한 기준점(동지) 찾기
101        // 현재 합삭 시점 이전 혹은 포함된 동지(Winter Solstice)를 찾음
102        let ws_t = engine.find_winter_solstice_before(nm_t + chrono::Duration::days(1))?;
103        let ws_nm_t = engine.find_new_moon_before(ws_t + chrono::Duration::minutes(1))?;
104
105        // 동지가 포함된 달은 무조건 음력 11월
106        let mut curr_nm = ws_nm_t;
107        let mut month_num = 11;
108        let mut is_leap = false;
109        let mut lunar_year = (ws_t + kst_offset).year();
110        // 동지가 12월 말에 있으므로, 이 달이 속한 음력 연도의 1월은 보통 다음 해로 넘어감
111        if (ws_t + kst_offset).month() == 12 {
112            // 동지가 12월이면 그해 1월은 이미 지났으므로, 동지 달(11월)은 해당 연도의 음력으로 침
113        }
114
115        // 4. 동지 달(11월)부터 현재 합삭 달까지 진행하며 월 번호 부여
116        while (nm_t - curr_nm).num_minutes() > 4320 {
117            let next_nm = engine.find_new_moon_before(curr_nm + chrono::Duration::days(45))?;
118
119            // 중기(Zhong-qi) 포함 여부 확인
120            let mut has_zq = false;
121            for i in 1..=23 {
122                if i % 2 == 1 {
123                    // 중기 인덱스 (1, 3, 5, ...)
124                    if let Ok(zq_t) = engine.find_solar_term_time(curr_nm, i) {
125                        // 중기가 합삭일 당일에 있어도 포함된 것으로 간주 (KST 기준)
126                        let zq_date = (zq_t + kst_offset).date_naive();
127                        let curr_nm_date = (curr_nm + kst_offset).date_naive();
128                        let next_nm_date = (next_nm + kst_offset).date_naive();
129
130                        if zq_date >= curr_nm_date && zq_date < next_nm_date {
131                            has_zq = true;
132                            break;
133                        }
134                    }
135                }
136            }
137
138            if !has_zq && !is_leap {
139                // 이번 달에 중기가 없고, 이번 무중치윤 구간에서 아직 윤달을 넣지 않았다면 윤달 지정
140                is_leap = true;
141            } else {
142                month_num += 1;
143                if month_num > 12 {
144                    month_num = 1;
145                    lunar_year += 1;
146                }
147                is_leap = false;
148            }
149            curr_nm = next_nm;
150        }
151
152        Ok((lunar_year, month_num as u32, lunar_day, is_leap))
153    }
154
155    /// 음력을 양력으로 변환
156    pub fn to_solar(
157        year: i32,
158        month: u32,
159        day: u32,
160        is_leap: bool,
161    ) -> Result<NaiveDate, crate::error::DataError> {
162        let test_date =
163            NaiveDate::from_ymd_opt(year, month, 1).ok_or(crate::error::DataError::InvalidDate)?;
164
165        // 해당 연도 전후 6개월 범위를 스캔 (음력 1월이 양력 2월경이므로 폭넓게 스캔)
166        for i in -60..360 {
167            let date = test_date + chrono::Duration::days(i);
168            if let Ok((ly, lm, ld, il)) = Self::from_solar(date) {
169                if ly == year && lm == month && ld == day && il == is_leap {
170                    return Ok(date);
171                }
172            }
173        }
174        Err(crate::error::DataError::InvalidDate)
175    }
176}

178#[cfg(test)]
179mod tests {
180    use super::*;
181
182    #[test]
183    fn test_get_day_ganzi_index() {
184        // 2000-01-01 = 戊午일 (index=54)
185        let date = NaiveDate::from_ymd_opt(2000, 1, 1).unwrap();
186        assert_eq!(get_day_ganzi_index(date), 54);
187
188        // 2000-01-02 = 己未일 (index=55)
189        let date = NaiveDate::from_ymd_opt(2000, 1, 2).unwrap();
190        assert_eq!(get_day_ganzi_index(date), 55);
191    }
192
193    #[test]
194    fn test_sixty_day_cycle() {
195        // 60일 후에는 같은 일진
196        let date1 = NaiveDate::from_ymd_opt(2024, 1, 1).unwrap();
197        let date2 = NaiveDate::from_ymd_opt(2024, 3, 1).unwrap(); // 60일 후
198        assert_eq!(get_day_ganzi_index(date1), get_day_ganzi_index(date2));
199    }
200
201    #[test]
202    fn test_lunar_conversion() {
203        // 김성주님 생일: 양력 2004-11-27 -> 음력 2004-10-16
204        let solar = NaiveDate::from_ymd_opt(2004, 11, 27).unwrap();
205        let lunar = LunarCalendar::from_solar(solar).unwrap();
206        println!("DEBUG: Lunar result for 2004-11-27: {:?}", lunar);
207
208        assert_eq!(lunar.0, 2004, "Year should be 2004");
209        assert_eq!(lunar.1, 10, "Month should be 10");
210        assert_eq!(lunar.2, 16, "Day should be 16");
211        assert!(!lunar.3, "Should not be a leap month");
212
213        // 역변환 확인
214        let solar_back = LunarCalendar::to_solar(2004, 10, 16, false).unwrap();
215        assert_eq!(solar_back, solar);
216    }
