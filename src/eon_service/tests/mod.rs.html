15

17        let saju_res = facade::analyze_saju(saju_input.clone()).expect("Saju analysis failed");

39    fn test_vedic_analysis_parity() {
40        let (birth, _, _) = setup_test_inputs();
41        let vedic_input = VedicAnalysisInput {
42            base: birth.clone(),
43            precision: BirthTimePrecision::Exact,
44            current: CurrentContext {
45                now_utc: Utc.with_ymd_and_hms(2024, 1, 1, 0, 0, 0).unwrap(),
46                analysis_timezone: "Asia/Seoul".to_string(),
47            },
48            target_year: None,
49        };
50
51        let vedic_res = facade::analyze_vedic(vedic_input).expect("Vedic analysis failed");
52        
53        // Verify precise Vedic calculation outputs (Standard Time -> UTC directly, without TST)
54        assert_eq!(vedic_res.chart.ascendant.rasi, 10); // Capricorn
55        assert_eq!(vedic_res.chart.ascendant.nakshatra, 21); // Uttarashadha
56        
57        if let Some(moon) = vedic_res.chart.planets.iter().find(|p| p.planet == eon_vedic::planets::VedicPlanet::Moon) {
58            assert_eq!(moon.rasi, 10);
59            assert_eq!(moon.nakshatra, 22); // Shravana
60        } else {
61            panic!("Moon not found in Vedic chart");
62        }
63        
64        assert_eq!(vedic_res.meta.precision, BirthTimePrecision::Exact);
65        assert!(!vedic_res.chart.planets.is_empty(), "Chart planets should not be empty");
66        assert!(!vedic_res.gochara.transits.is_empty(), "Gochara transits should not be empty");
67    }

77    fn test_oracle_regression() {
78        let fixtures = get_v1_fixtures();
79        for f in fixtures {
80            println!("Testing fixture: {}", f.id);
81            
82            // 1. Saju Verification
83            let saju_input = SajuAnalysisInput {
84                base: f.input.clone(),
85                is_male: f.is_male,
86                use_night_rat_hour: false, // Default for oracle
87                precision: BirthTimePrecision::Exact,
88            };
89            let saju_res = facade::analyze_saju(saju_input).expect(&format!("Saju failed for {}", f.id));
90            
91            // 2. Vedic Verification
92            let vedic_input = VedicAnalysisInput {
93                base: f.input.clone(),
94                precision: BirthTimePrecision::Exact,
95                current: CurrentContext {
96                    now_utc: Utc.with_ymd_and_hms(2024, 1, 1, 0, 0, 0).unwrap(),
97                    analysis_timezone: f.input.timezone.clone(),
98                },
99                target_year: None,
100            };
101            let vedic_res = facade::analyze_vedic(vedic_input).expect(&format!("Vedic failed for {}", f.id));
102
103            if f.id == "lunar-edge-case" {
104                println!("SAJU: {:#?}", saju_res.report.pillars);
105                println!("VEDIC ASC: {:?}", vedic_res.chart.ascendant);
106                let moon = vedic_res.chart.planets.iter().find(|p| p.planet == eon_vedic::planets::VedicPlanet::Moon).unwrap();
107                println!("VEDIC MOON: {:?}", moon);
108            } else {
109                assert_eq!(format!("{:?}", saju_res.report.pillars.year.stem), f.expected_saju.year_stem, "Year Stem mismatch for {}", f.id);
110                assert_eq!(format!("{:?}", saju_res.report.pillars.year.branch), f.expected_saju.year_branch, "Year Branch mismatch for {}", f.id);
111                assert_eq!(format!("{:?}", saju_res.report.pillars.month.stem), f.expected_saju.month_stem, "Month Stem mismatch for {}", f.id);
112                assert_eq!(format!("{:?}", saju_res.report.pillars.month.branch), f.expected_saju.month_branch, "Month Branch mismatch for {}", f.id);
113                assert_eq!(format!("{:?}", saju_res.report.pillars.day.stem), f.expected_saju.day_stem, "Day Stem mismatch for {}", f.id);
114                assert_eq!(format!("{:?}", saju_res.report.pillars.day.branch), f.expected_saju.day_branch, "Day Branch mismatch for {}", f.id);
115                assert_eq!(format!("{:?}", saju_res.report.pillars.hour.stem), f.expected_saju.hour_stem, "Hour Stem mismatch for {}", f.id);
116                assert_eq!(format!("{:?}", saju_res.report.pillars.hour.branch), f.expected_saju.hour_branch, "Hour Branch mismatch for {}", f.id);
117                assert_eq!(format!("{:?}", saju_res.report.strength.day_master), f.expected_saju.day_master, "Day Master mismatch for {}", f.id);
118
119                assert_eq!(vedic_res.chart.ascendant.rasi, f.expected_vedic.ascendant_rasi, "Ascendant Rasi mismatch for {}", f.id);
120                let moon = vedic_res.chart.planets.iter().find(|p| p.planet == eon_vedic::planets::VedicPlanet::Moon).unwrap();
