1use crate::birth::prepare_birth_context;
2use crate::dto::{AnalysisMeta, BirthTimePrecision, QimenAnalysisInput, QimenAnalysisOutput};
3use crate::error::ServiceError;
4use eon_qimen::analysis::report::QimenAnalysisReport;
5use eon_qimen::core::QimenPan;
6use eon_saju::core::ganzi::GanZi;
7use eon_core::Gender;
8
9use eon_saju::core::stem::HeavenlyStem;
10use eon_saju::core::branch::EarthlyBranch;
11
12pub fn analyze_qimen(
13    input: QimenAnalysisInput,
14) -> Result<QimenAnalysisOutput, ServiceError> {
15    let gender = if input.is_male { Gender::Male } else { Gender::Female };
16    let birth_ctx = prepare_birth_context(&input.base, Some(gender), true)?;
17    let dt = birth_ctx
18        .birth_info
19        .to_utc()
20        .map_err(|e| ServiceError::InvalidInput(e.to_string()))?;
21
22    // 스캐폴딩: 임시로 년주, 월주, 일주, 시주를 Dummy 값으로 사용하거나 saju-core를 불러와야 하지만,
23    // 현재는 뼈대만 구축하는 단계이므로 QimenPan을 더미 데이터로 채웁니다.
24    // 추후 eon-saju::core::calendar::LunarCalendar 등과 연동하여 제대로 된 시주(Hour Pillar)를 가져옵니다.
25
26    let pan = QimenPan {
27        time: dt,
28        hour_pillar: GanZi::new(HeavenlyStem::Jia, EarthlyBranch::Zi), // 갑자(JiaZi) 더미
29        is_yin_ju: true, // 둔국 더미
30        ju_number: 1,    // 국수 더미
31    };
32
33    let report = QimenAnalysisReport::generate(pan);
34
35    let meta = AnalysisMeta {
36        precision: BirthTimePrecision::Exact, // 기본값
37        input_time: dt.to_rfc3339(),
38        corrected_time: dt.to_rfc3339(), // 진태양시 보정 추후 적용
39        is_dst: false,
40        dst_offset_hours: None,
41        analysis_timezone: input.base.timezone,
42    };
43
44    Ok(QimenAnalysisOutput { meta, report })
45}
