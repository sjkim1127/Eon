1use crate::birth::prepare_birth_context;
2use crate::dto::{AnalysisMeta, BirthTimePrecision, QimenAnalysisInput, QimenAnalysisOutput};
3use crate::error::ServiceError;
4use eon_core::Gender;
5use eon_qimen::analysis::report::QimenAnalysisReport;
6use eon_qimen::builder::ju::calculate_ju;
7use eon_qimen::builder::pan::build_qimen_pan;
8use eon_saju::core::pillars::{FourPillars, SajuInput};
9
10pub fn analyze_qimen(input: QimenAnalysisInput) -> Result<QimenAnalysisOutput, ServiceError> {
11    let gender = if input.is_male {
12        Gender::Male
13    } else {
14        Gender::Female
15    };
16
17    // 1. 탄생 컨텍스트(진태양시 보정 등) 생성
18    let birth_ctx = prepare_birth_context(&input.base, Some(gender), true)?;
19    let dt = birth_ctx
20        .birth_info
21        .to_utc()
22        .map_err(|e| ServiceError::InvalidInput(e.to_string()))?;
23
24    // 2. 사주 명식(Four Pillars) 계산
25    let saju_input = SajuInput::new_solar(
26        birth_ctx.corrected_year,
27        birth_ctx.corrected_month,
28        birth_ctx.corrected_day,
29        birth_ctx.corrected_hour,
30        birth_ctx.corrected_minute,
31    )
32    .with_gender(gender)
33    .with_night_rat_hour(input.use_night_rat_hour);
34
35    let pillars = FourPillars::calculate(&saju_input)
36        .map_err(|e| ServiceError::Saju(format!("사주 계산 실패: {}", e)))?;
37
38    // 3. 기문둔갑 국수(Ju Number) 및 음양둔 산출
39    let (is_yin_ju, ju_number) = calculate_ju(dt, pillars.day)
40        .map_err(|e| ServiceError::InvalidInput(format!("기문둔갑 국수 계산 실패: {}", e)))?;
41
42    // 4. 기문둔갑 포국(Pan Building) 알고리즘 실행
43    let pan = build_qimen_pan(
44        dt,
45        pillars.year,
46        pillars.month,
47        pillars.day,
48        pillars.hour,
49        is_yin_ju,
50        ju_number,
51    );
52
53    // 5. 분석 리포트 생성
54    let report = QimenAnalysisReport::generate(pan);
55
56    let meta = AnalysisMeta {
57        precision: BirthTimePrecision::Exact,
58        input_time: dt.to_rfc3339(),
59        corrected_time: dt.to_rfc3339(),
60        is_dst: false,
61        dst_offset_hours: None,
62        analysis_timezone: input.base.timezone,
63    };
64
65    Ok(QimenAnalysisOutput { meta, report })
66}
