2use crate::dto::{ReportTheme, SajuAnalysisOutput, ThemedReportInput, ThemedReportOutput};

9pub fn generate(input: ThemedReportInput) -> Result<ThemedReportOutput, ServiceError> {
10    // 1. 사주 분석 수행
11    let saju_out = saju::analyze(input.base.clone())?;
12
13    // 2. 테마에 따른 보고서 제목 및 본문 조합
14    let (title, content) = match input.theme {
15        ReportTheme::WealthAndCareer => generate_wealth_report(&saju_out, &input.user_name),
16        ReportTheme::LoveAndMarriage => {
17            generate_love_report(&saju_out, &input.user_name, input.base.is_male)
18        }
19        ReportTheme::HealthAndVitality => generate_health_report(&saju_out, &input.user_name),
20    };
21
22    Ok(ThemedReportOutput {
23        meta: saju_out.meta,
24        theme: input.theme,
25        user_name: input.user_name,
26        title,
27        content,
28    })
29}

232fn generate_health_report(saju: &SajuAnalysisOutput, name: &str) -> (String, String) {
233    let title = format!("🌿 {}님의 건강 체질 & 마음 치유 리포트", name);
234    let mut s = String::new();
235
236    let day_stem = saju.report.pillars.day.stem;
237
238    // Section 1: 체질분석
239    s.push_str("### 1. 오행 배분으로 보는 타고난 건강 체질\n\n");
240
241    let day_stem_kor = HeavenlyStem::HANGUL[day_stem.index() as usize];
242
243    s.push_str(&format!(
244        "당신은 **{}** 일간의 체질을 바탕으로 태어났습니다. 사주 오행 분포도에 기반한 장부(臟腑) 건강 상태는 다음과 같습니다.\n\n",
245        day_stem_kor
246    ));
247
248    // 오행 강약 분석에 기반한 건강 조언
249    let mut fire_count = 0;
250    let mut water_count = 0;
251    for pillar in &[
252        &saju.report.pillars.year,
253        &saju.report.pillars.month,
254        &saju.report.pillars.day,
255        &saju.report.pillars.hour,
256    ] {
257        if pillar.stem.element() == Element::Fire {
258            fire_count += 1;
259        }
260        if pillar.branch.element() == Element::Fire {
261            fire_count += 1;
262        }
263        if pillar.stem.element() == Element::Water {
264            water_count += 1;
265        }
266        if pillar.branch.element() == Element::Water {
267            water_count += 1;
268        }
269    }
270
271    let health_desc = if fire_count > 3 {
272        "명식 내에 **불(火)의 기운이 과다**한 편으로 분류됩니다. 이 체질은 혈압 관리, 심혈관계 질환, 몸에 열이 몰려 생기는 피부 질환이나 안구 건조증을 예방하는 식습관이 중요합니다. 평소 충분한 수분 섭취와 규칙적인 수면을 적극 권장합니다."
273    } else if water_count > 3 {
274        "명식 내에 **물(水)의 기운이 과다**하거나 차가운 성질이 짙습니다. 이 경우 냉증, 신장 및 방광 건강, 순환계통의 둔화에 유의해야 합니다. 따뜻한 성질의 차를 즐겨 마시고 하체 근력 운동을 꾸준히 해 주는 것이 신체 밸런스 유지에 유리합니다."
275    } else {
276        "오행이 비교적 고르게 안배되어 있어, 특정 장기의 급격한 기능 저하 우려는 낮습니다. 다만 계절이 바뀌는 환절기에 피로가 누적되기 쉬우니 충분한 영양 공급과 스트레칭으로 신체 리듬을 정비하십시오."
277    };
278    s.push_str(health_desc);
279    s.push_str("\n\n");
280
281    // Section 2: 주의해야 할 신체 부위 및 에러 Lints
282    s.push_str("### 2. 신체 불균형 및 주요 경고 신호 (Destiny Lints)\n\n");

284
285    if saju.lints.is_empty() {

287    } else {
288        for lint in saju.lints.iter().take(2) {
289            s.push_str(&format!("- **[{:?}]**: {}\n", lint.severity, lint.message));
290        }
291        s.push('\n');
292    }
293
294    // Section 3: 마음 치유 및 에너지 관리법
295    s.push_str("### 3. 마음 치유 & 웰니스 추천 가이드\n\n");
296    s.push_str(
297        "신체 건강뿐만 아니라, 스트레스 관리를 위한 마음 치유 웰니스 처방 가이드라인입니다.\n\n",
298    );
299    s.push_str("- **추천 운동 요법**: 가벼운 유산소 산책과 호흡 명상 요법을 병행하여 체내 산소 순환율을 향상시키십시오.\n");
300    s.push_str("- **치유 색상 테라피**: 신체 리듬을 다스리기 위하여, 침실이나 자주 사용하는 소품의 색상을 부드러운 자연의 색조(그린, 소프트 우드 계열)로 선택하는 것을 권장합니다.\n");
301    s.push_str("- **스트레스 아웃 가이드**: 과다한 생각과 고민은 위장 기능을 떨어트리는 주원인이 되므로, 하루 10분 온전히 생각을 비우는 명상 훈련을 하십시오.\n");
302
303    (title, s)
304}
