1//! 자미두수 분석 서비스 모듈
2//!
3//! 입력된 데이터를 바탕으로 생년월일시 보정을 수행하고, 자미두수 엔진을 호출해 성반과 대한/유년을 연산합니다.
4
5use crate::dto::{AnalysisMeta, ZwdsAnalysisInput, ZwdsAnalysisOutput, BirthTimePrecision};
6use crate::error::ServiceError;
7use crate::birth::prepare_birth_context;
8use eon_core::Gender;
9use eon_zwds::build_chart;
10use eon_zwds::annual::calculate_liunian;
11
12/// 자미두수 분석을 수행합니다.
13pub fn analyze(input: ZwdsAnalysisInput) -> Result<ZwdsAnalysisOutput, ServiceError> {
14    let gender = if input.is_male {
15        Gender::Male
16    } else {
17        Gender::Female
18    };
19
20    // 자미두수는 통상적으로 경도 기반 진태양시 보정을 하지 않으므로 apply_tst = false
21    let birth_ctx = prepare_birth_context(&input.base, Some(gender), false)?;
22
23    // 1. 성반 빌드
24    let mut chart = build_chart(&birth_ctx.birth_info)
25        .map_err(|e| ServiceError::Zwds(format!("자미두수 성반 생성 실패: {}", e)))?;
26
27    // 2. 대상 연도 지정 (기본값은 현재 연도 2026년)
28    let target_year = input.target_year.unwrap_or(2026);
29
30    // 3. 대상 연도에 따른 유년(流年) 계산
31    let current_liu_nian = calculate_liunian(target_year);
32
33    // 성반 내 유년 궁 표시 업데이트
34    for palace in chart.palaces.iter_mut() {
35        if palace.index == current_liu_nian.palace_idx {
36            palace.is_current_liu_nian = true;
37        }
38    }
39
40    // 4. 나이 계산 및 현재 대한(大限) 식별
41    // 자미두수 대운(대한)은 태어난 해(1세)부터 시작해 만 나이/한국 나이가 아닌 오행국 나이(예: 2~11세) 기준입니다.
42    // 여기서는 간단하게 경과 연도(target_year - birth_year + 1)를 나이로 보고 현재 대한을 매핑합니다.
43    let age = (target_year - birth_ctx.birth_info.year).max(0) as u32 + 1; // 세는 나이 기준
44
45    let current_daxian = chart.daxian.iter()
46        .find(|d| age >= d.age_start && age <= d.age_end)
47        .cloned()
48        .unwrap_or_else(|| {
49            // 매핑되지 않으면 첫 번째 대한 리턴
50            chart.daxian.first().cloned().expect("대한 목록이 비어있음")
51        });
52
53    Ok(ZwdsAnalysisOutput {
54        meta: AnalysisMeta {
55            precision: BirthTimePrecision::Exact,
56            input_time: birth_ctx.input_time_string,
57            corrected_time: birth_ctx.corrected_time_string,
58            is_dst: birth_ctx.is_dst,
59            dst_offset_hours: birth_ctx.dst_offset_hours,
60            analysis_timezone: input.base.timezone,
61        },
62        chart,
63        current_daxian,
64        current_liu_nian,
65    })
66}
