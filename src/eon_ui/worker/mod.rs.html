3use eon_service::dto::{
4    SajuAnalysisInput, SajuAnalysisOutput, VedicAnalysisInput, VedicAnalysisOutput,
5};
6use eon_service::facade;
7use gloo_worker::{HandlerId, Worker, WorkerScope};
8use serde::{Deserialize, Serialize};
9
10#[derive(Serialize, Deserialize)]
11pub enum WorkerInput {
12    RunSaju(SajuAnalysisInput),
13    RunVedic(VedicAnalysisInput),
14}
15
16#[derive(Serialize, Deserialize)]
17pub enum WorkerOutput {
18    SajuResult(Result<SajuAnalysisOutput, String>),
19    VedicResult(Result<VedicAnalysisOutput, String>),
20}
21
22pub struct AnalysisWorker {}
23
24impl Worker for AnalysisWorker {
25    type Message = ();
26    type Input = WorkerInput;
27    type Output = WorkerOutput;
28
29    fn create(_scope: &WorkerScope<Self>) -> Self {
30        Self {}
31    }
32
33    fn update(&mut self, _scope: &WorkerScope<Self>, _msg: Self::Message) {}
34
35    fn received(&mut self, scope: &WorkerScope<Self>, msg: Self::Input, id: HandlerId) {
36        match msg {
37            WorkerInput::RunSaju(input) => {
38                let res = facade::analyze_saju(input).map_err(|e| format!("{:?}", e));
39                scope.respond(id, WorkerOutput::SajuResult(res));
40            }
41            WorkerInput::RunVedic(input) => {
42                let res = facade::analyze_vedic(input).map_err(|e| format!("{:?}", e));
43                scope.respond(id, WorkerOutput::VedicResult(res));
44            }
45        }
46    }
47}
