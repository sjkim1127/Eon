16/// Supported UI locales.
17#[derive(Clone, Copy, Debug, PartialEq, Eq, Default, Serialize, Deserialize)]
18pub enum Locale {
19    #[default]
20    Ko,
21    En,
22    Zh,
23    Ru,
24}
25
26impl Locale {
27    pub fn label(self) -> &'static str {
28        match self {
29            Locale::Ko => "🇰🇷 한국어",
30            Locale::En => "🇺🇸 English",
31            Locale::Zh => "🇨🇳 中文",
32            Locale::Ru => "🇷🇺 Русский",
33        }
34    }
35
36    pub fn flag(self) -> &'static str {
37        match self {
38            Locale::Ko => "🇰🇷",
39            Locale::En => "🇺🇸",
40            Locale::Zh => "🇨🇳",
41            Locale::Ru => "🇷🇺",
42        }
43    }
44
45    pub fn from_code(code: &str) -> Self {
46        match code {
47            "en" => Locale::En,
48            "zh" => Locale::Zh,
49            "ru" => Locale::Ru,
50            _ => Locale::Ko,
51        }
52    }
53
54    pub fn code(self) -> &'static str {
55        match self {
56            Locale::Ko => "ko",
57            Locale::En => "en",
58            Locale::Zh => "zh",
59            Locale::Ru => "ru",
60        }
61    }
62
63    pub fn all() -> &'static [Locale] {
64        &[Locale::Ko, Locale::En, Locale::Zh, Locale::Ru]
65    }
66}
