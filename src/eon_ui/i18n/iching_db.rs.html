18pub fn get_hexagram_info(index: u8) -> HexagramInfo {
19    let list = get_all_hexagrams();
20    if (1..=64).contains(&index) {
21        list[(index - 1) as usize].clone()
22    } else {
23        list[0].clone()
24    }
25}

27pub fn get_yao_name(line_idx: u8, is_yang: bool, locale: Locale) -> String {
28    match locale {
29        Locale::Ko => {
30            let prefix = if is_yang { "구" } else { "육" };
31            match line_idx {
32                1 => format!("초{}", prefix),
33                2 => format!("{}이", prefix),
34                3 => format!("{}삼", prefix),
35                4 => format!("{}사", prefix),
36                5 => format!("{}오", prefix),
37                _ => format!("상{}", prefix),
38            }
39        }
40        Locale::En => {
41            let num = if is_yang { "Nine" } else { "Six" };
42            match line_idx {
43                1 => format!("First {}", num),
44                2 => format!("Second {}", num),
45                3 => format!("Third {}", num),
46                4 => format!("Fourth {}", num),
47                5 => format!("Fifth {}", num),
48                _ => format!("Top {}", num),
49            }
50        }
51        Locale::Zh => {
52            let prefix = if is_yang { "九" } else { "六" };
53            match line_idx {
54                1 => format!("初{}", prefix),
55                2 => format!("{}二", prefix),
56                3 => format!("{}三", prefix),
57                4 => format!("{}四", prefix),
58                5 => format!("{}五", prefix),
59                _ => format!("上{}", prefix),
60            }
61        }
62        Locale::Ru => {
63            let num = if is_yang {
64                "Девятка"
65            } else {
66                "Шестерка"
67            };
68            match line_idx {
69                1 => format!("Начальная {}", num),
70                2 => format!("Вторая {}", num),
71                3 => format!("Третья {}", num),
72                4 => format!("Четвертая {}", num),
73                5 => format!("Пятая {}", num),
74                _ => format!("Верхняя {}", num),
75            }
76        }
77    }
78}
