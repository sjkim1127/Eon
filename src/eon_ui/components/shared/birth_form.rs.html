1use crate::i18n::{t, TK};
2use crate::store::db::{self, UserProfile};
3use crate::store::AnalysisState;
4use dioxus::prelude::*;
5use serde::Deserialize;

7use crate::utils::geocode::CityRecord;
8
9#[component]
10pub fn BirthForm() -> Element {
11    let mut state = use_context::<AnalysisState>();
12    let locale = *state.locale.read();
13
14    // Local state
15    let mut profiles = use_signal(Vec::<UserProfile>::new);
16    let mut new_profile_name = use_signal(String::new);
17    let mut city_input = use_signal(String::new);
18    let mut geo_status = use_signal(String::new);
19    let mut search_results = use_signal(Vec::<CityRecord>::new);
20
21    // Load profiles on mount
22    use_effect(move || {
23        spawn(async move {
24            if let Ok(loaded) = db::load_all_profiles().await {
25                profiles.set(loaded);
26            }
27        });
28    });
29
30    let on_save_profile = move |_| {
31        spawn(async move {
32            let name = new_profile_name.read().clone();
33            if !name.is_empty() {
34                let form_data = state.form.read().clone();
35                if let Ok(new_profile) = db::save_profile(name.clone(), form_data).await {
36                    profiles.write().insert(0, new_profile);
37                    new_profile_name.set(String::new());
38                }
39            }
40        });
41    };
42
43    let on_select_profile = move |evt: Event<FormData>| {
44        let selected_id = evt.value();
45        if let Some(profile) = profiles.read().iter().find(|p| p.id == selected_id) {
46            *state.form.write() = profile.form_state.clone();
47            let lat = profile.form_state.lat;
48            let lon = profile.form_state.lon;
49            geo_status.set(format!("📍 {:.4}°N, {:.4}°E", lat, lon));
50        }
51    };
52
53    let mut select_city = move |result: CityRecord| {
54        state.form.write().lat = result.lat;
55        state.form.write().lon = result.lon;
56        // Optionally save timezone if added to form state later
57        let display_name = if let Some(ko) = &result.name_ko {
58            format!("{} ({})", ko, result.name)
59        } else {
60            result.name.clone()
61        };
62        geo_status.set(format!("✅ {}, {}", display_name, result.country));
63        search_results.set(Vec::new());
64        city_input.set(String::new());
65    };
66
67    let geo_searching_str = t(locale, TK::GeoSearching);
68    let geo_no_result_str = t(locale, TK::GeoNoResult);
69    let geo_parse_err_str = t(locale, TK::GeoParseError);
70    let geo_net_err_str = t(locale, TK::GeoNetworkError);
71
72    // 도시 검색 (Local Offline DB)
73    let on_city_geocode = move |_| {
74        let query = city_input.read().clone();
75        if query.trim().is_empty() {
76            return;
77        }
78        geo_status.set(geo_searching_str.to_string());
79        
80        let results = crate::utils::geocode::search_city(&query, 15);
81        if results.is_empty() {
82            geo_status.set(geo_no_result_str.to_string());
83            search_results.set(Vec::new());
84        } else {
85            geo_status.set(format!("✅ ({} results)", results.len()));
86            search_results.set(results);
87        }
88    };
89
90    // Enter key search
91    let on_city_keydown = move |evt: Event<KeyboardData>| {
92        if evt.key() == Key::Enter {
93            let query = city_input.read().clone();
94            if query.trim().is_empty() {
95                return;
96            }
97            geo_status.set(geo_searching_str.to_string());
98            let results = crate::utils::geocode::search_city(&query, 15);
99            if results.is_empty() {
100                geo_status.set(geo_no_result_str.to_string());
101                search_results.set(Vec::new());
102            } else {
103                geo_status.set(format!("✅ ({} results)", results.len()));
104                search_results.set(results);
105            }
106        }
107    };
108
109    let lat_display = state.form.read().lat;
110    let lon_display = state.form.read().lon;
111
112    rsx! {
113        div { class: "glass-premium rounded-2xl p-5 flex flex-col gap-4 border border-white/5",
114
115            // Profile Row
116            div { class: "flex items-center gap-3 pb-3 border-b border-white/5",
117                div { class: "flex flex-col gap-1.5 flex-1 min-w-0",
118                    label { class: "text-xs text-slate-400 font-semibold tracking-wide", "{t(locale, TK::FormSavedProfiles)}" }
119                    select {
120                        class: "bg-white/5 border border-white/10 rounded-xl px-3 py-2.5 text-sm text-slate-200 focus-glow transition-all w-full cursor-pointer",
121                        onchange: on_select_profile,
122                        option { value: "", class: "bg-brand-950 text-slate-300", "{t(locale, TK::FormLoadProfile)}" }
123                        {profiles.read().iter().map(|p| rsx! {
124                            option { value: "{p.id}", class: "bg-brand-950 text-slate-300", "{p.name}" }
125                        })}
126                    }
127                }
128                div { class: "flex items-end gap-2 flex-1 min-w-0",
129                    div { class: "flex flex-col gap-1.5 flex-1 min-w-0",
130                        label { class: "text-xs text-slate-400 font-semibold tracking-wide", "{t(locale, TK::FormSaveProfile)}" }
131                        input {
132                            class: "w-full bg-white/5 border border-white/10 rounded-xl px-3 py-2.5 text-sm text-slate-200 focus-glow transition-all",
133                            placeholder: "{t(locale, TK::FormProfileNamePlaceholder)}",
134                            value: "{new_profile_name}",
135                            oninput: move |evt| new_profile_name.set(evt.value()),
136                        }
137                    }
138                    button {
139                        class: "shrink-0 bg-gradient-to-r from-violet-600 to-indigo-600 hover:from-violet-500 hover:to-indigo-500 text-white text-xs font-semibold px-4 py-2.5 rounded-xl transition-all shadow-md shadow-violet-950/20 active:scale-95 cursor-pointer",
140                        onclick: on_save_profile,
141                        "{t(locale, TK::FormSaveBtn)}"
142                    }
143                }
144            }
145
146            // Input Row
147            div { class: "flex flex-wrap gap-3.5 items-end",
148                // Year
149                div { class: "flex flex-col gap-1.5",
150                    label { class: "text-xs text-slate-400 font-semibold tracking-wide", "{t(locale, TK::FormYear)}" }
151                    input {
152                        class: "w-24 bg-white/5 border border-white/10 rounded-xl px-3 py-2 text-sm text-slate-200 focus-glow transition-all",
153                        r#type: "number", min: "1900", max: "2100",
154                        value: "{state.form.read().year}",
155                        oninput: move |evt| {
156                            if let Ok(v) = evt.value().parse::<i32>() { state.form.write().year = v; }
157                        },
158                    }
159                }
160                // Month
161                div { class: "flex flex-col gap-1.5",
162                    label { class: "text-xs text-slate-400 font-semibold tracking-wide", "{t(locale, TK::FormMonth)}" }
163                    input {
164                        class: "w-16 bg-white/5 border border-white/10 rounded-xl px-3 py-2 text-sm text-slate-200 focus-glow transition-all",
165                        r#type: "number", min: "1", max: "12",
166                        value: "{state.form.read().month}",
167                        oninput: move |evt| {
168                            if let Ok(v) = evt.value().parse::<u32>() { state.form.write().month = v; }
169                        },
170                    }
171                }
172                // Day
173                div { class: "flex flex-col gap-1.5",
174                    label { class: "text-xs text-slate-400 font-semibold tracking-wide", "{t(locale, TK::FormDay)}" }
175                    input {
176                        class: "w-16 bg-white/5 border border-white/10 rounded-xl px-3 py-2 text-sm text-slate-200 focus-glow transition-all",
177                        r#type: "number", min: "1", max: "31",
178                        value: "{state.form.read().day}",
179                        oninput: move |evt| {
180                            if let Ok(v) = evt.value().parse::<u32>() { state.form.write().day = v; }
181                        },
182                    }
183                }
184                // Hour
185                div { class: "flex flex-col gap-1.5",
186                    label { class: "text-xs text-slate-400 font-semibold tracking-wide", "{t(locale, TK::FormHour)}" }
187                    input {
188                        class: "w-16 bg-white/5 border border-white/10 rounded-xl px-3 py-2 text-sm text-slate-200 focus-glow transition-all",
189                        r#type: "number", min: "0", max: "23",
190                        value: "{state.form.read().hour}",
191                        oninput: move |evt| {
192                            if let Ok(v) = evt.value().parse::<u32>() { state.form.write().hour = v; }
193                        },
194                    }
195                }
196                // Minute
197                div { class: "flex flex-col gap-1.5",
198                    label { class: "text-xs text-slate-400 font-semibold tracking-wide", "{t(locale, TK::FormMinute)}" }
199                    input {
200                        class: "w-16 bg-white/5 border border-white/10 rounded-xl px-3 py-2 text-sm text-slate-200 focus-glow transition-all",
201                        r#type: "number", min: "0", max: "59",
202                        value: "{state.form.read().minute}",
203                        oninput: move |evt| {
204                            if let Ok(v) = evt.value().parse::<u32>() { state.form.write().minute = v; }
205                        },
206                    }
207                }
208                // Birthplace (text search)
209                div { class: "flex flex-col gap-1.5 min-w-0 relative flex-1 md:flex-initial",
210                    label { class: "text-xs text-slate-400 font-semibold tracking-wide",
211                        "{t(locale, TK::FormBirthplace)}"
212                    }
213                    div { class: "flex gap-1.5",
214                        input {
215                            class: "w-full md:w-44 bg-white/5 border border-white/10 rounded-xl px-3 py-2 text-sm text-slate-200 focus-glow transition-all",
216                            placeholder: "{t(locale, TK::FormCityPlaceholder)}",
217                            value: "{city_input}",
218                            oninput: move |evt| city_input.set(evt.value()),
219                            onkeydown: on_city_keydown,
220                        }
221                        button {
222                            class: "bg-white/10 hover:bg-white/15 active:scale-95 text-slate-200 text-sm px-3 py-2 rounded-xl transition-all cursor-pointer border border-white/5",
223                            onclick: on_city_geocode,
224                            "🔍"
225                        }
226                    }
227                    if !search_results.read().is_empty() {
228                        div { class: "absolute top-full left-0 mt-1.5 w-72 bg-[#0e0f22]/95 border border-white/10 rounded-xl shadow-2xl backdrop-blur-2xl z-50 overflow-hidden",
229                            {search_results.read().iter().map(|result| {
230                                let r = result.clone();
231                                let display = if let Some(ko) = &r.name_ko {
232                                    format!("{} ({}), {} - {}", ko, r.name, r.country, r.tz)
233                                } else {
234                                    format!("{}, {} - {}", r.name, r.country, r.tz)
235                                };
236                                rsx! {
237                                    div {
238                                        class: "px-3.5 py-3 text-xs text-slate-300 hover:bg-violet-600/30 hover:text-violet-200 border-b border-white/5 last:border-0 cursor-pointer transition-colors truncate",
239                                        onclick: move |_| select_city(r.clone()),
240                                        "{display}"
241                                    }
242                                }
243                            })}
244                        }
245                    }
246                    // Geocoding result / coordinate display
247                    div { class: "text-[10px] text-slate-500 mt-0.5 tracking-wider font-semibold",
248                        if geo_status.read().is_empty() {
249                            "📍 {lat_display:.4}°N, {lon_display:.4}°E"
250                        } else {
251                            "{geo_status}"
252                        }
253                    }
254                }
255
256                // Form Checkboxes Section
257                div { class: "flex items-center gap-4 flex-wrap pb-1 h-9",
258                    // Lunar calendar checkbox
259                    div { class: "flex items-center gap-2",
260                        input {
261                            r#type: "checkbox", id: "is_lunar",
262                            class: "w-4 h-4 rounded border-white/10 bg-white/5 accent-violet-500 cursor-pointer",
263                            checked: "{state.form.read().is_lunar}",
264                            onchange: move |evt| state.form.write().is_lunar = evt.value() == "true"
265                        }
266                        label { r#for: "is_lunar", class: "text-sm text-slate-300 select-none cursor-pointer whitespace-nowrap hover:text-slate-200 transition-colors",
267                            "{t(locale, TK::FormLunar)}"
268                        }
269                    }
270                    // Male checkbox
271                    div { class: "flex items-center gap-2",
272                        input {
273                            r#type: "checkbox", id: "is_male",
274                            class: "w-4 h-4 rounded border-white/10 bg-white/5 accent-violet-500 cursor-pointer",
275                            checked: "{state.form.read().is_male}",
276                            onchange: move |evt| state.form.write().is_male = evt.value() == "true"
277                        }
278                        label { r#for: "is_male", class: "text-sm text-slate-300 select-none cursor-pointer whitespace-nowrap hover:text-slate-200 transition-colors",
279                            "{t(locale, TK::FormMale)}"
280                        }
281                    }
282                    // Night Rat Hour checkbox
283                    div { class: "flex items-center gap-2",
284                        input {
285                            r#type: "checkbox", id: "use_night_rat_hour",
286                            class: "w-4 h-4 rounded border-white/10 bg-white/5 accent-violet-500 cursor-pointer",
287                            checked: "{state.form.read().use_night_rat_hour}",
288                            onchange: move |evt| state.form.write().use_night_rat_hour = evt.value() == "true"
289                        }
290                        label { r#for: "use_night_rat_hour", class: "text-sm text-slate-300 select-none cursor-pointer whitespace-nowrap hover:text-slate-200 transition-colors",
291                            "{t(locale, TK::FormUseNightRatHour)}"
292                        }
293                    }
294                }
295            }
296        }
297    }
