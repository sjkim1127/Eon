364#[component]
365fn LuckCard(
366    title: String,
367    value: String,
368    sub: String,
369    color: String,
370    icon: &'static str,
371) -> Element {
372    rsx! {
373        div { class: "p-4 rounded-2xl bg-gradient-to-b {color} border flex flex-col gap-2",
374            div { class: "flex items-center gap-2",
375                span { class: "text-xl", "{icon}" }
376                span { class: "text-xs text-slate-400 font-medium", "{title}" }
377            }
378            p { class: "text-xl font-bold text-slate-100 font-serif", "{value}" }
379            if !sub.is_empty() {
380                p { class: "text-sm text-slate-400", "{sub}" }
381            }
382        }
383    }
