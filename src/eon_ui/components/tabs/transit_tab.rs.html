332#[component]
333fn LuckCard(title: String, value: String, sub: String, color: String, icon: &'static str) -> Element {
334    rsx! {
335        div { class: "p-4 rounded-2xl bg-gradient-to-b {color} border flex flex-col gap-2",
336            div { class: "flex items-center gap-2",
337                span { class: "text-xl", "{icon}" }
338                span { class: "text-xs text-slate-400 font-medium", "{title}" }
339            }
340            p { class: "text-xl font-bold text-slate-100 font-serif", "{value}" }
341            if !sub.is_empty() {
342                p { class: "text-sm text-slate-400", "{sub}" }
343            }
344        }
345    }
