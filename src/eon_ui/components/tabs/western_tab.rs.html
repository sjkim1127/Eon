297    let mut selected_house_sys = use_signal(|| "P".to_string());
298    let mut copied_feedback = use_signal(|| false);
299
300    // Reactive trigger for manual analysis runs
301    let mut analysis_trigger = use_signal(|| 0);
302
303    // Auto-run or manually triggered analysis when form, house system, or trigger changes
304    let state_cloned = state.clone();
305    use_effect(move || {
306        let form = state_cloned.form.read().clone();
307        let house_sys = selected_house_sys.read().clone();
308        let _trig = *analysis_trigger.read();
309
310        if form.year > 0 {
311            let mut state = state_cloned.clone();
312            spawn(async move {
313                state.western.write().status = TaskStatus::Loading;
314                let base_input = form.to_analysis_input();
315                let west_input = WesternAnalysisInput::new(base_input, house_sys);
316
317                match facade::analyze_western(west_input) {
318                    Ok(res) => {
319                        state.western.write().data = Some(res);
320                        state.western.write().status = TaskStatus::Success;
321                    }
322                    Err(e) => {
323                        state.western.write().error = Some(e.to_string());
324                        state.western.write().status = TaskStatus::Error(e.to_string());

331    rsx! {
