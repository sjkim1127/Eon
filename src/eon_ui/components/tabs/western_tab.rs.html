9pub fn get_planet_emoji_and_name(name: &str, locale: Locale) -> (&'static str, String) {
10    let (emoji, val) = match name {
11        "Sun" => ("☀️", t(locale, TK::WestPlanetSun)),
12        "Moon" => ("🌙", t(locale, TK::WestPlanetMoon)),
13        "Mercury" => ("☿", t(locale, TK::WestPlanetMercury)),
14        "Venus" => ("♀", t(locale, TK::WestPlanetVenus)),
15        "Mars" => ("♂", t(locale, TK::WestPlanetMars)),
16        "Jupiter" => ("♃", t(locale, TK::WestPlanetJupiter)),
17        "Saturn" => ("♄", t(locale, TK::WestPlanetSaturn)),
18        "Uranus" => ("♅", t(locale, TK::WestPlanetUranus)),
19        "Neptune" => ("♆", t(locale, TK::WestPlanetNeptune)),
20        "Pluto" => ("♇", t(locale, TK::WestPlanetPluto)),
21        "Chiron" => ("🔑", t(locale, TK::WestPlanetChiron)),
22        "True Node" => ("☊", t(locale, TK::WestNodeNorth)),
23        "South Node" => ("☋", t(locale, TK::WestNodeSouth)),
24        "NorthNode" => ("☊", t(locale, TK::WestNodeNorth)),
25        "SouthNode" => ("☋", t(locale, TK::WestNodeSouth)),
26        _ => ("🪐", name),
27    };
28    (emoji, val.to_string())
29}

49pub fn get_aspect_emoji_and_name(
