18#[component]
19pub fn ZwdsTab() -> Element {
20    let state = use_context::<AnalysisState>();
21    let locale = *state.locale.read();
22    let target_year = use_signal(|| 2026i32);
23    let selected_palace_idx = use_signal(|| None::<usize>);
24    let hovered_palace_idx = use_signal(|| None::<usize>);
25
26    // Copy signals to avoid borrowing state mutably in closures
27    let zwds_signal = state.zwds;
28    let form_signal = state.form;
29    let target_year_sig = target_year;
30
31    let run_analysis_with_year = move |year: i32| {
32        spawn(async move {
33            let mut zwds_signal = zwds_signal;
34            let form_signal = form_signal;
35            zwds_signal.write().status = TaskStatus::Loading;
36            let form = form_signal.read().clone();
37            let input = ZwdsAnalysisInput::new(form.to_analysis_input(), form.is_male, Some(year));
38            match facade::analyze_zwds(input) {
39                Ok(res) => {
40                    zwds_signal.write().data = Some(res);
41                    zwds_signal.write().status = TaskStatus::Success;
42                }
43                Err(e) => {
44                    zwds_signal.write().error = Some(e.to_string());
45                    zwds_signal.write().status = TaskStatus::Error(e.to_string());
46                }
47            }
48        });
49    };
50
51    let update_year = move |new_year: i32| {
52        let mut target_year_sig = target_year_sig;
53        target_year_sig.set(new_year);
54        let is_success = matches!(zwds_signal.read().status, TaskStatus::Success);
55        if is_success {
56            run_analysis_with_year(new_year);
57        }
58    };
59
60    // Auto-run analysis when form or target_year changes
61    use_effect(move || {
62        let form = state.form.read().clone();
63        let year = *target_year.read();
64        if form.year > 0 {
65            run_analysis_with_year(year);
66        }
67    });
68
69    let zwds_status = state.zwds.read().status.clone();
70    let zwds_data = state.zwds.read().data.clone();
71
72    rsx! {
73        div { class: "space-y-6 animate-in fade-in slide-in-from-bottom-4 duration-700",
74            // 1. 공통 출생 정보 입력 폼

79                div { class: "space-y-1.5",
80                    h2 { class: "text-2xl font-bold text-transparent bg-clip-text bg-gradient-to-r from-violet-400 to-indigo-300",
81                        "{t(locale, TK::ZwdsReportTitle)}"
82                    }
83                    p { class: "text-sm text-slate-400 font-medium",
84                        match locale {
85                            Locale::Ko => "{t(locale, TK::ZwdsAnalyzeDesc)}",
86                            Locale::Zh => "分析紫微斗数命盘，推算大限及流年运势。",
87                            Locale::En => "Analyze ZWDS chart, deduce Da-Xian and Liu-Nian fortunes.",
88                            Locale::Ru => "Анализ карты Цзы Вэй Доу Шу, расчет периодов Да-Сянь и Лю-Нянь.",
89                        }
90                    }
91                }
92
93                // 컨트롤바: 대상 연도 조절 및 새로고침 버튼
94                div { class: "flex items-center gap-3",
95                    div { class: "flex items-center bg-slate-950/80 border border-slate-800/80 px-3 py-1.5 rounded-xl gap-2",
96                        span { class: "text-xs font-bold text-slate-400 uppercase tracking-wider",
97                            match locale {
98                                Locale::Ko => "{t(locale, TK::ZwdsTargetYear)}",
99                                Locale::Zh => "目标年份",
100                                Locale::En => "Target Year",
101                                Locale::Ru => "Целевой год",
102                            }
103                        }
104                        button {
105                            class: "text-slate-400 hover:text-white transition-colors cursor-pointer text-xs font-bold px-1.5 py-0.5 hover:bg-slate-800 rounded-md",
106                            onclick: move |_| update_year(*target_year.read() - 1),
107                            "◀"
108                        }
109                        span { class: "text-sm font-bold text-violet-400 font-mono min-w-[3.5rem] text-center",
110                            {t(locale, TK::ZwdsTargetYearVal).replace("{}", &target_year.to_string())}
111                        }
112                        button {
113                            class: "text-slate-400 hover:text-white transition-colors cursor-pointer text-xs font-bold px-1.5 py-0.5 hover:bg-slate-800 rounded-md",
114                            onclick: move |_| update_year(*target_year.read() + 1),
115                            "▶"
116                        }
117                    }
118
119                    button {
120                        class: "p-2.5 bg-slate-800 hover:bg-slate-700 active:bg-slate-600 border border-slate-700/50 rounded-xl text-slate-300 hover:text-white transition-all cursor-pointer flex items-center justify-center active:scale-95",
121                        onclick: move |_| run_analysis_with_year(*target_year.read()),
122                        title: "{t(locale, TK::ZwdsAnalyzeBtn)}",
123                        svg {
124                            class: "w-5 h-5",
125                            fill: "none",
126                            stroke: "currentColor",
127                            view_box: "0 0 24 24",
128                            path {
129                                stroke_linecap: "round",
130                                stroke_linejoin: "round",
131                                stroke_width: "2",
132                                d: "M4 4v5h.582m15.356 2A8.001 8.001 0 1121.21 7.89M21 3v5h-5"
133                            }
134                        }
135                    }
136                }
137            }
138
139            // 3. 분석 결과 영역
140            {
141                match &zwds_status {
142                    TaskStatus::Idle => rsx! {
143                        div { class: "flex flex-col items-center justify-center py-20 gap-3 text-slate-500 bg-slate-900/20 border border-slate-800/40 rounded-3xl",
144                            span { class: "text-5xl animate-bounce", "🔮" }
145                            p { class: "text-lg font-medium", "{t(locale, TK::ZwdsIdleHint)}" }
146                        }
147                    },
148                    TaskStatus::Loading => rsx! {
149                        div { class: "flex flex-col items-center justify-center py-20 gap-3 bg-slate-900/20 border border-slate-800/40 rounded-3xl",
150                            div { class: "w-12 h-12 rounded-full border-4 border-violet-500/30 border-t-violet-400 animate-spin" }
151                            p { class: "text-violet-400 font-medium animate-pulse", "{t(locale, TK::ZwdsLoadingHint)}" }

153                    },
154                    TaskStatus::Error(e) => rsx! {
155                        div { class: "p-4 rounded-xl bg-red-900/20 border border-red-800/50 text-red-400",

161                            rsx! {
162                                div { class: "space-y-6",

169                            }
170                        } else {

173                    }
174                }
175            }
176
177            // 4. 궁위 클릭 상세 모달
178            {
179                if let Some(p_idx) = *selected_palace_idx.read() {
180                    if let Some(ref res) = zwds_data {
181                        rsx! {
182                            PalaceDetailModal {
183                                palace_idx: p_idx,
184                                data: res.clone(),
185                                selected_palace_idx,
186                            }
187                        }
188                    } else {
189                        rsx! {}
190                    }
191                } else {
192                    rsx! {}
193                }
194            }
195        }
196    }
197}

205) -> Element {
206    // 4x4 Grid의 16칸 셀 배치 (외곽 12칸 궁 매핑, 중앙 4칸 비움)

211        Some(4),
212        Some(5),
213        Some(6),

221        None,

232
233    rsx! {
234        div { class: "grid grid-cols-4 gap-3 bg-slate-950 p-4 rounded-3xl border border-slate-800/50 shadow-2xl relative overflow-hidden",
235            {
236                grid_cells.into_iter().enumerate().map(|(idx, cell)| {
237                    match cell {
238                        Some(p_idx) => {
239                            let palace = &chart.palaces[p_idx];
240                            rsx! {
241                                PalaceCard {
242                                    key: "{p_idx}",
243                                    palace: palace.clone(),
244                                    is_soul: p_idx == chart.soul_idx,
245                                    is_body: p_idx == chart.body_idx,
246                                    current_daxian: Some(data.current_daxian.clone()),
247                                    current_liu_nian: Some(data.current_liu_nian.clone()),
248                                    onclick: move |_| {
249                                        selected_palace_idx.set(Some(p_idx));
250                                    },
251                                    onmouseenter: move |_| {
252                                        hovered_palace_idx.set(Some(p_idx));
253                                    },
254                                    onmouseleave: move |_| {
255                                        hovered_palace_idx.set(None);
256                                    }
257                                }
258                            }
259                        },
260                        None => {
261                            // Row 2, Col 2 위치(인덱스 5)에서만 중앙 카드 렌더링
262                            if idx == 5 {
263                                rsx! {
264                                    CenterCard {
265                                        key: "center-card",
266                                        data: data.clone(),
267                                    }
268                                }
269                            } else {
270                                rsx! {}
271                            }
272                        }
273                    }
274                })
275            }
276
277            // SVG flying stars layer
278            if let Some(from_idx) = *hovered_palace_idx.read() {
279                {
280                    let chart = &data.chart;
281                    let hovered_p_name = chart.palaces[from_idx].name;

284

286                        .filter(|fs| fs.from_palace == hovered_p_name)
287                        .filter_map(|fs| {
288                            let to_idx = chart.palaces.iter().position(|p| p.name == fs.to_palace)?;
289                            Some((fs, to_idx))
290                        })
291                        .collect::<Vec<_>>();
292
293                    if !flying_lines.is_empty() {
294                        rsx! {
295                            svg {
296                                class: "absolute inset-0 w-full h-full pointer-events-none z-10",
297                                view_box: "0 0 100 100",
298                                defs {
299                                    marker {
300                                        id: "arrow-lu",
301                                        view_box: "0 0 10 10",
302                                        ref_x: "8",
303                                        ref_y: "5",
304                                        marker_width: "6",
305                                        marker_height: "6",
306                                        orient: "auto-start-reverse",
307                                        path { d: "M 0 2 L 8 5 L 0 8 z", fill: "#10b981" }
308                                    }
309                                    marker {
310                                        id: "arrow-quan",
311                                        view_box: "0 0 10 10",
312                                        ref_x: "8",
313                                        ref_y: "5",
314                                        marker_width: "6",
315                                        marker_height: "6",
316                                        orient: "auto-start-reverse",
317                                        path { d: "M 0 2 L 8 5 L 0 8 z", fill: "#3b82f6" }
318                                    }
319                                    marker {
320                                        id: "arrow-ke",
321                                        view_box: "0 0 10 10",
322                                        ref_x: "8",
323                                        ref_y: "5",
324                                        marker_width: "6",
325                                        marker_height: "6",
326                                        orient: "auto-start-reverse",
327                                        path { d: "M 0 2 L 8 5 L 0 8 z", fill: "#8b5cf6" }
328                                    }
329                                    marker {
330                                        id: "arrow-ji",
331                                        view_box: "0 0 10 10",
332                                        ref_x: "8",
333                                        ref_y: "5",
334                                        marker_width: "6",
335                                        marker_height: "6",
336                                        orient: "auto-start-reverse",
337                                        path { d: "M 0 2 L 8 5 L 0 8 z", fill: "#ef4444" }
338                                    }
339                                }
340                                {
341                                    flying_lines.into_iter().map(|(fs, to_idx)| {
342                                        let (x1, y1) = get_palace_center(from_idx);
343                                        let (x2, y2) = get_palace_center(to_idx);
344
345                                        let (color, marker_id) = match fs.sihua_type {
346                                            SiHuaType::HuaLu => ("#10b981", "arrow-lu"),
347                                            SiHuaType::HuaQuan => ("#3b82f6", "arrow-quan"),
348                                            SiHuaType::HuaKe => ("#8b5cf6", "arrow-ke"),
349                                            SiHuaType::HuaJi => ("#ef4444", "arrow-ji"),
350                                        };
351
352                                        let star_label = translate_zwds_star(locale, fs.star);
353                                        let sihua_emoji = fs.sihua_type.emoji();
354
355                                        if from_idx == to_idx {
356                                            let path_d = format!(
357                                                "M {} {} C {} {}, {} {}, {} {}",
358                                                x1 - 2.0, y1 - 2.0,
359                                                x1 - 5.0, y1 - 8.0,
360                                                x1 + 5.0, y1 - 8.0,
361                                                x1 + 2.0, y1 - 2.0
362                                            );
363                                            let mx = x1;
364                                            let my = y1 - 9.0;
365                                            let label_len = star_label.chars().count();
366                                            let rect_w = 4.0 + (label_len as f64) * 1.5;
367                                            let rect_h = 3.6;
368
369                                            rsx! {
370                                                g { key: "self-{fs.star.korean()}-{sihua_emoji}",
371                                                    path {
372                                                        d: "{path_d}",
373                                                        fill: "none",
374                                                        stroke: "{color}",
375                                                        stroke_width: "1.2",
376                                                        marker_end: "url(#{marker_id})"
377                                                    }
378                                                    rect {
379                                                        x: "{mx - rect_w / 2.0}",
380                                                        y: "{my - rect_h / 2.0}",
381                                                        width: "{rect_w}",
382                                                        height: "{rect_h}",
383                                                        rx: "1.0",
384                                                        fill: "#0b0f19",
385                                                        stroke: "{color}",
386                                                        stroke_width: "0.4"
387                                                    }
388                                                    text {
389                                                        x: "{mx}",
390                                                        y: "{my + 1.0}",
391                                                        fill: "#cbd5e1",
392                                                        font_size: "2.0",
393                                                        font_weight: "bold",
394                                                        text_anchor: "middle",
395                                                        "{star_label}{sihua_emoji}"
396                                                    }
397                                                }
398                                            }
399                                        } else {
400                                            let dx = x2 - x1;
401                                            let dy = y2 - y1;
402                                            let dist = (dx*dx + dy*dy).sqrt();
403                                            let (x2_short, y2_short) = if dist > 0.0 {
404                                                let offset = 8.5;
405                                                if dist > offset {
406                                                    (x2 - dx * (offset / dist), y2 - dy * (offset / dist))
407                                                } else {
408                                                    (x2, y2)
409                                                }
410                                            } else {
411                                                (x2, y2)
412                                            };
413
414                                            let mx = (x1 + x2) / 2.0;
415                                            let my = (y1 + y2) / 2.0;
416                                            let label_len = star_label.chars().count();
417                                            let rect_w = 4.0 + (label_len as f64) * 1.5;
418                                            let rect_h = 3.6;
419
420                                            rsx! {
421                                                g { key: "line-{from_idx}-{to_idx}-{fs.star.korean()}",
422                                                    line {
423                                                        x1: "{x1}",
424                                                        y1: "{y1}",
425                                                        x2: "{x2_short}",
426                                                        y2: "{y2_short}",
427                                                        stroke: "{color}",
428                                                        stroke_width: "1.2",
429                                                        stroke_dasharray: "3 1.5",
430                                                        marker_end: "url(#{marker_id})"
431                                                    }
432                                                    rect {
433                                                        x: "{mx - rect_w / 2.0}",
434                                                        y: "{my - rect_h / 2.0}",
435                                                        width: "{rect_w}",
436                                                        height: "{rect_h}",
437                                                        rx: "1.0",
438                                                        fill: "#0b0f19",
439                                                        stroke: "{color}",
440                                                        stroke_width: "0.4"
441                                                    }
442                                                    text {
443                                                        x: "{mx}",
444                                                        y: "{my + 1.0}",
445                                                        fill: "#cbd5e1",
446                                                        font_size: "2.0",
447                                                        font_weight: "bold",
448                                                        text_anchor: "middle",
449                                                        "{star_label}{sihua_emoji}"
450                                                    }
451                                                }
452                                            }
453                                        }
454                                    })
455                                }
456                            }
457                        }
458                    } else {
459                        rsx! {}
460                    }
461                }
462            }
463        }
464    }

467/// 개별 궁위 카드 컴포넌트
468#[component]
469fn PalaceCard(
470    palace: PalaceData,
471    is_soul: bool,
472    is_body: bool,
473    current_daxian: Option<eon_zwds::types::DaXian>,
474    current_liu_nian: Option<eon_zwds::types::LiuNian>,
475    onclick: EventHandler<MouseEvent>,
476    onmouseenter: EventHandler<MouseEvent>,
477    onmouseleave: EventHandler<MouseEvent>,
478) -> Element {
479    let state = use_context::<AnalysisState>();
480    let locale = *state.locale.read();
481
482    // 대운 동적 궁위명 구하기
483    let daxian_palace_name = if let Some(ref dx) = current_daxian {
484        let name = eon_zwds::palace::get_palace_name(dx.palace_idx, palace.index);
485        let prefix = match locale {
486            Locale::Ko => "{t(locale, TK::ZwdsDaxianPrefix)}",
487            Locale::Zh => "大",
488            Locale::En => "D-",
489            Locale::Ru => "Д-",
490        };
491        let abbr = crate::i18n::translate_zwds_palace_abbr(locale, name);
492        Some(format!("{}{}", prefix, abbr))
493    } else {
494        None
495    };
496
497    // 유년 동적 궁위명 구하기
498    let liunian_palace_name = if let Some(ref ln) = current_liu_nian {
499        let name = eon_zwds::palace::get_palace_name(ln.palace_idx, palace.index);
500        let prefix = match locale {
501            Locale::Ko => "{t(locale, TK::ZwdsLiuNianPrefix)}",
502            Locale::Zh => "流",
503            Locale::En => "A-",
504            Locale::Ru => "Г-",
505        };
506        let abbr = crate::i18n::translate_zwds_palace_abbr(locale, name);
507        Some(format!("{}{}", prefix, abbr))
508    } else {
509        None
510    };
511
512    // 테두리 및 배경 디자인
513    let border_cls = if palace.is_current_liu_nian {
514        "border-amber-500/80 shadow-lg shadow-amber-900/10 bg-amber-950/10"
515    } else if is_soul {
516        "border-violet-500/70 shadow-lg shadow-violet-900/10 bg-violet-950/10"
517    } else {
518        "border-slate-800/70 bg-slate-900/40 hover:border-slate-700/60"
519    };
520
521    let mut annual_stars = Vec::new();
522    if let Some(ref ln) = current_liu_nian {
523        if palace.index == ln.liu_lu {
524            annual_stars.push(ZwdsStar::LuCun);
525        }
526        if palace.index == ln.liu_yang {
527            annual_stars.push(ZwdsStar::QingYang);
528        }
529        if palace.index == ln.liu_tuo {
530            annual_stars.push(ZwdsStar::TuoLuo);
531        }
532        if palace.index == ln.liu_chang {
533            annual_stars.push(ZwdsStar::WenChang);
534        }
535        if palace.index == ln.liu_qu {
536            annual_stars.push(ZwdsStar::WenQu);
537        }
538    }
539
540    rsx! {
541        div {
542            class: "h-44 p-3 rounded-2xl border flex flex-col justify-between transition-all duration-200 cursor-pointer hover:scale-[1.02] hover:shadow-xl hover:shadow-violet-950/10 active:scale-98 {border_cls}",
543            onclick: move |e| onclick.call(e),
544            onmouseenter: move |e| onmouseenter.call(e),
545            onmouseleave: move |e| onmouseleave.call(e),
546            // 궁 헤더: 천간/지지 & 궁명
547            div { class: "flex justify-between items-start",
548                div { class: "flex flex-col",
549                    span { class: "text-[10px] text-slate-500 font-bold",
550                        "{palace.heavenly_stem}{palace.earthly_branch}"
551                    }
552                    span { class: "text-[10px] text-slate-400 font-bold",
553                        "{palace.earthly_branch}{t(locale, TK::ZwdsPalaceSuffix)}"
