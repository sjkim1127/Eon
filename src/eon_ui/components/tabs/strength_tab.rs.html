15fn planet_bar_color(p: VedicPlanet) -> &'static str {
16    match p {
17        VedicPlanet::Sun => "bg-orange-500",
18        VedicPlanet::Moon => "bg-slate-300",
19        VedicPlanet::Mars => "bg-red-600",
20        VedicPlanet::Mercury => "bg-emerald-400",
21        VedicPlanet::Jupiter => "bg-yellow-400",
22        VedicPlanet::Venus => "bg-pink-400",
23        VedicPlanet::Saturn => "bg-indigo-600",
24        VedicPlanet::Rahu => "bg-purple-500",
25        VedicPlanet::Ketu => "bg-amber-700",
26        VedicPlanet::Ascendant => "bg-white",
27    }
28}

386fn DeukDetail(label: &'static str, acquired: bool, desc: String) -> Element {
387    let (bg, icon_color) = if acquired {
388        (
389            "bg-emerald-900/20 border-emerald-700/40",

393        ("bg-slate-800/40 border-slate-700/40", "text-slate-500")

395    rsx! {
396        div { class: "p-3 rounded-xl border {bg}",
397            div { class: "flex items-center gap-1.5 mb-1",
398                span { class: "text-xs font-mono {icon_color}", if acquired { "○" } else { "✗" } }
399                span { class: "text-xs font-semibold text-slate-300", "{label}" }
400            }
401            p { class: "text-xs text-slate-500 leading-snug", "{desc}" }
402        }
403    }
