151fn get_d1_planet_tooltip(
152    locale: Locale,
153    lbl: &str,
154    d1_planets: &[eon_vedic::core::chart::VedicPosition],
155    d1_ascendant: &eon_vedic::core::chart::VedicPosition,
156    varga_label: &str,
157    shadbalas: Option<&[eon_vedic::analysis::strength::PlanetStrength]>,
158) -> Option<VedicTooltipTarget> {
159    let p_opt = planet_from_lbl(lbl);
160    if let Some(planet) = p_opt {
161        let pos = if planet == VedicPlanet::Ascendant {
162            d1_ascendant
163        } else {
164            d1_planets.iter().find(|p| p.planet == planet)?
165        };
166
167        let nak_name = nakshatra_name(pos.nakshatra);
168        let nak_lord = nakshatra_lord_localized(locale, pos.nakshatra);
169
170        let planet_name = translate_planet(locale, pos.planet).to_string();
171        let deg_within_sign = pos.sidereal_deg % 30.0;
172        let deg_floor = deg_within_sign.floor() as i32;
173        let min_val = ((deg_within_sign - deg_floor as f64) * 60.0).round() as i32;
174        let longitude_str = format!("{}° {:02}'", deg_floor, min_val);
175
176        Some(VedicTooltipTarget::Planet {
177            name: planet_name,
178            symbol: lbl.to_string(),
179            rasi_num: pos.rasi,
180            house_num: pos.house_index,
181            longitude_str,
182            nakshatra_name: match locale {
183                Locale::Ko => format!("{} ({}단계)", nak_name, pos.pada),
184                Locale::En => format!("{} (Pada {})", nak_name, pos.pada),
185                Locale::Zh => format!("{} ({}步)", nak_name, pos.pada),
186                Locale::Ru => format!("{} (Пада {})", nak_name, pos.pada),
187            },
188            nakshatra_lord: nak_lord.to_string(),
189            is_retrograde: pos.is_retrograde,
190            is_combust: pos.is_combust,
191            varga_label: varga_label.to_string(),
192            deity: None,
193            purpose: None,
194            shadbala: shadbalas.and_then(|sb| sb.iter().find(|s| s.planet == pos.planet).cloned()),
195        })
196    } else {
197        None
198    }
199}

248fn get_house_tooltip(
249    locale: Locale,
250    house_num: u8,
251    lagna_rasi: u8,
252    varga_label: &str,
253    bhava_strengths: Option<&[eon_vedic::analysis::bhava::BhavaStrength]>,
254) -> Option<VedicTooltipTarget> {
255    let rasi_num = (lagna_rasi + house_num - 2) % 12 + 1;
256    let rasi_name_str = rasi_name(locale, rasi_num).to_string();
257    let rasi_lord = rasi_lord_localized(locale, rasi_num).to_string();
258
259    let score = bhava_strengths.and_then(|strengths| {
260        strengths
261            .iter()
262            .find(|s| s.house == house_num)
263            .map(|s| s.total_score)
264    });
265
266    Some(VedicTooltipTarget::House {
267        house_num,
268        rasi_num,
269        rasi_name: rasi_name_str,
270        rasi_lord,
271        score,
272        varga_label: varga_label.to_string(),
273    })
274}
