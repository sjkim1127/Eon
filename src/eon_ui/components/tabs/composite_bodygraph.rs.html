1#![allow(non_snake_case)]
2use dioxus::prelude::*;
3use eon_human_design::connection::HumanDesignConnectionResult;
4use eon_human_design::HdCenter;
5
6#[derive(Props, PartialEq, Clone)]
7pub struct CompositeBodyGraphProps {
8    pub result: HumanDesignConnectionResult,
9}
10
11#[component]
12pub fn CompositeBodyGraph(props: CompositeBodyGraphProps) -> Element {
13    let defined_fill = "#f8fafc";
14    let open_fill = "#1e293b";
15    let outline_color = "#334155";
16    let gate_text_color = "fill-slate-500";
17
18    let has_center = |c: HdCenter| -> &'static str {
19        if props.result.defined_centers.contains(&c) {
20            defined_fill
21        } else {
22            open_fill
23        }
24    };
25
26    let get_channel_color = |g1: u8, g2: u8| -> &'static str {
27        if props.result.electromagnetic_channels.contains(&(g1, g2))
28            || props.result.electromagnetic_channels.contains(&(g2, g1))
29        {
30            "#facc15" // Yellow/Spark for Electromagnetic
31        } else if props.result.compromise_channels.contains(&(g1, g2))
32            || props.result.compromise_channels.contains(&(g2, g1))
33        {
34            "#a855f7" // Purple for Compromise
35        } else if props.result.dominance_channels.contains(&(g1, g2))
36            || props.result.dominance_channels.contains(&(g2, g1))
37        {
38            "#3b82f6" // Blue for Dominance (A over B or B over A)
39        } else if props.result.companionship_channels.contains(&(g1, g2))
40            || props.result.companionship_channels.contains(&(g2, g1))
41        {
42            "#22c55e" // Green for Companionship
43        } else {
44            "#e2e8f0" // Default undefined
45        }
46    };
47
48    rsx! {
49        div { class: "w-full flex justify-center items-center py-6 bg-slate-900/30 rounded-2xl border border-slate-800",
50            svg {
51                width: "100%", height: "auto", view_box: "0 0 400 600", class: "drop-shadow-2xl max-w-sm",
52                // Example Single Channel to demonstrate the concept.
53
54                // Channel 1-8
55                line { x1: "200", y1: "310", x2: "200", y2: "270", stroke: get_channel_color(1, 8), stroke_width: "6", stroke_linecap: "round" }
56                text { x: "196", y: "308", class: "text-[8px] font-bold {gate_text_color}", "1" }
57                text { x: "196", y: "276", class: "text-[8px] font-bold {gate_text_color}", "8" }
58
59                // Head
60                polygon { points: "200,30 260,80 140,80", fill: has_center(HdCenter::Head), stroke: outline_color, stroke_width: "4" }
61                // Ajna
62                polygon { points: "200,160 260,110 140,110", fill: has_center(HdCenter::Ajna), stroke: outline_color, stroke_width: "4" }
63                // Throat
64                rect { x: "170", y: "210", width: "60", height: "60", rx: "8", fill: has_center(HdCenter::Throat), stroke: outline_color, stroke_width: "4" }
65                // Self
66                polygon { points: "200,310 240,350 200,390 160,350", fill: has_center(HdCenter::SelfG), stroke: outline_color, stroke_width: "4" }
67                // Heart
68                polygon { points: "250,280 290,260 290,300", fill: has_center(HdCenter::Heart), stroke: outline_color, stroke_width: "4" }
69                // Sacral
70                rect { x: "170", y: "410", width: "60", height: "60", rx: "8", fill: has_center(HdCenter::Sacral), stroke: outline_color, stroke_width: "4" }
71                // Root
72                rect { x: "170", y: "510", width: "60", height: "60", rx: "8", fill: has_center(HdCenter::Root), stroke: outline_color, stroke_width: "4" }
73                // Spleen
74                polygon { points: "90,440 130,420 130,480", fill: has_center(HdCenter::Spleen), stroke: outline_color, stroke_width: "4" }
75                // Solar Plexus
76                polygon { points: "310,440 270,420 270,480", fill: has_center(HdCenter::SolarPlexus), stroke: outline_color, stroke_width: "4" }
77            }
78        }
79    }
80}
