249/// 생애 인생 점수 SVG 라인차트
250fn LifelineChart(timeline: &[YearlyScore]) -> Element {
251    if timeline.is_empty() {
252        return rsx! { div {} };
253    }
254
255    let width = 800.0f64;
256    let height = 200.0f64;
257    let pad_left = 40.0f64;
258    let pad_right = 20.0f64;
259    let pad_top = 16.0f64;
260    let pad_bottom = 24.0f64;
261
262    let chart_w = width - pad_left - pad_right;
263    let chart_h = height - pad_top - pad_bottom;
264
265    let min_score = timeline
266        .iter()
267        .map(|t| t.total_score)
268        .fold(f64::INFINITY, f64::min);
269    let max_score = timeline
270        .iter()
271        .map(|t| t.total_score)
272        .fold(f64::NEG_INFINITY, f64::max);
273    let avg_score = timeline.iter().map(|t| t.total_score).sum::<f64>() / timeline.len() as f64;
274    let score_range = (max_score - min_score).max(1.0);
275
276    let n = timeline.len() as f64;
277    let points: Vec<(f64, f64)> = timeline
278        .iter()
279        .enumerate()
280        .map(|(i, t)| {
281            let x = pad_left + (i as f64 / (n - 1.0).max(1.0)) * chart_w;
282            let y = pad_top + (1.0 - (t.total_score - min_score) / score_range) * chart_h;
283            (x, y)
284        })
285        .collect();
286
287    let polyline_pts = points
288        .iter()
289        .map(|(x, y)| format!("{x:.1},{y:.1}"))
290        .collect::<Vec<_>>()
291        .join(" ");
292
293    let avg_y = pad_top + (1.0 - (avg_score - min_score) / score_range) * chart_h;
294
295    let threshold = min_score + score_range * 0.8;
296
297    let last_pt = points
298        .last()
299        .cloned()
300        .unwrap_or((pad_left, pad_top + chart_h));
301    let first_pt = points
302        .first()
303        .cloned()
304        .unwrap_or((pad_left, pad_top + chart_h));
305    let area_path = format!(
306        "M{:.1},{:.1} {} L{:.1},{:.1} L{:.1},{:.1} Z",
307        first_pt.0,
308        pad_top + chart_h,
309        polyline_pts,
310        last_pt.0,
311        pad_top + chart_h,
312        first_pt.0,
313        pad_top + chart_h,
314    );
315
316    let max_lbl = format!("{max_score:.0}");
317    let min_lbl = format!("{min_score:.0}");
318    let avg_y_str = format!("{avg_y:.1}");
319    let area_d = area_path.clone();
320    let line_pts = polyline_pts.clone();
321
322    rsx! {
323        svg {
324            view_box: "0 0 {width} {height}",
325            class: "w-full h-auto",
326            xmlns: "http://www.w3.org/2000/svg",
327
328            defs {
329                linearGradient { id: "area-grad", x1: "0", y1: "0", x2: "0", y2: "1",
330                    stop { offset: "0%", style: "stop-color:#a855f7;stop-opacity:0.4" }
331                    stop { offset: "100%", style: "stop-color:#a855f7;stop-opacity:0.02" }
332                }
333            }
334
335            {(0..5u32).map(|i| {
336                let y = pad_top + (i as f64 / 4.0) * chart_h;
337                rsx! {
338                    line { x1: "{pad_left}", y1: "{y:.1}", x2: "{width - pad_right}", y2: "{y:.1}", stroke: "#1e293b", stroke_width: "1" }
339                }
340            })}
341
342            line {
343                x1: "{pad_left}", y1: "{avg_y_str}", x2: "{width - pad_right}", y2: "{avg_y_str}",
344                stroke: "#d97706", stroke_width: "1", stroke_dasharray: "4 3", opacity: "0.5"
345            }
346
347            path { d: "{area_d}", fill: "url(#area-grad)" }
348
349            polyline {
350                points: "{line_pts}",
351                fill: "none",
352                stroke: "#a855f7",
353                stroke_width: "2",
354                stroke_linejoin: "round",
355                stroke_linecap: "round",
356            }
357
358            {points.iter().zip(timeline.iter()).filter_map(|((x, y), t)| {
359                if t.total_score >= threshold {
360                    let cx = format!("{x:.1}");
361                    let cy = format!("{y:.1}");
362                    Some(rsx! {
363                        circle { cx: "{cx}", cy: "{cy}", r: "4", fill: "#fbbf24", opacity: "0.8" }
364                    })
365                } else {
366                    None
367                }
368            })}
369
370            {
371                let locale = *use_context::<AnalysisState>().locale.read();
372                rsx! {
373                    {(0..=10u32).map(|i| {
374                        let idx = ((i as usize) * (timeline.len() - 1)) / 10;
375                        let idx = idx.min(timeline.len() - 1);
376                        let x = pad_left + (idx as f64 / (n - 1.0).max(1.0)) * chart_w;
377                        let year_lbl = timeline[idx].age;
378                        let age_str = format_age(locale, year_lbl as i32);
379                        rsx! {
380                            text {
381                                x: "{x:.1}", y: "{height - 6.0}",
382                                text_anchor: "middle", font_size: "9",
383                                fill: "#64748b",
384                                "{age_str}"
385                            }
386                        }
387                    })}
388                }
389            }
390
391            text { x: "{pad_left - 5.0}", y: "{pad_top + 4.0}", text_anchor: "end", font_size: "9", fill: "#64748b", "{max_lbl}" }
392            text { x: "{pad_left - 5.0}", y: "{pad_top + chart_h}", text_anchor: "end", font_size: "9", fill: "#64748b", "{min_lbl}" }
393        }
394    }
395}

397#[component]
398fn EnergyBar(value: f64) -> Element {
399    let pct = value.clamp(0.0, 100.0) as u32;
400    let color = if value > 60.0 {
401        "bg-emerald-500"
402    } else if value < 30.0 {
403        "bg-red-500"
404    } else {
405        "bg-amber-500"
406    };
407    rsx! {
408        div { class: "w-20 h-2 bg-slate-700 rounded-full overflow-hidden",
409            div { class: "h-full {color} rounded-full", style: "width: {pct}%" }
410        }
411    }
412}
