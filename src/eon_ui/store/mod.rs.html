19#[derive(Clone)]
20pub struct AnalysisTaskState<T: Clone + 'static> {
21    pub status: TaskStatus,
22    pub data: Option<T>,
23    pub error: Option<String>,
24}
25
26impl<T: Clone + 'static> Default for AnalysisTaskState<T> {
27    fn default() -> Self {
28        Self {
29            status: TaskStatus::Idle,
30            data: None,
31            error: None,
32        }
33    }
34}
