17const DB_NAME: &str = "eon_db";
18const STORE_NAME: &str = "profiles";

20pub async fn init_db() -> Result<IdbDatabase, String> {
21    let mut db_req =
22        IdbDatabase::open_f64(DB_NAME, 1.0).map_err(|e| format!("Failed to open DB: {:?}", e))?;
23
24    db_req.set_on_upgrade_needed(Some(|evt: &IdbVersionChangeEvent| -> Result<(), JsValue> {
25        if let None = evt.db().object_store_names().find(|n| n == STORE_NAME) {
26            let mut params = IdbObjectStoreParameters::new();
27            let key_path = IdbKeyPath::str("id");
28            params.key_path(Some(&key_path));
29            evt.db()
30                .create_object_store_with_params(STORE_NAME, &params)?;
31        }
32        Ok(())
33    }));
34
35    db_req
36        .into_future()
37        .await
38        .map_err(|e| format!("DB init failed: {:?}", e))
39}

41pub async fn save_profile(name: String, form_state: FormState) -> Result<UserProfile, String> {
42    let db = init_db().await?;
43    let tx = db
44        .transaction_on_one_with_mode(STORE_NAME, IdbTransactionMode::Readwrite)
45        .map_err(|e| format!("Transaction error: {:?}", e))?;
46    let store = tx
47        .object_store(STORE_NAME)
48        .map_err(|e| format!("Store error: {:?}", e))?;
49
50    let profile = UserProfile {
51        id: Uuid::new_v4().to_string(),
52        name,
53        form_state,
54        created_at: Utc::now().timestamp(),
55    };
56
57    let js_val = serde_wasm_bindgen::to_value(&profile)
58        .map_err(|e| format!("Serialization error: {:?}", e))?;
59
60    store
61        .add_val(&js_val)
62        .map_err(|e| format!("Failed to save: {:?}", e))?;
63
64    tx.await
65        .into_result()
66        .map_err(|e| format!("Transaction failed: {:?}", e))?;
67
68    Ok(profile)
69}

71pub async fn load_all_profiles() -> Result<Vec<UserProfile>, String> {
72    let db = init_db().await?;
73    let tx = db
74        .transaction_on_one_with_mode(STORE_NAME, IdbTransactionMode::Readonly)
75        .map_err(|e| format!("Transaction error: {:?}", e))?;
76    let store = tx
77        .object_store(STORE_NAME)
78        .map_err(|e| format!("Store error: {:?}", e))?;
79
80    let values = store
81        .get_all()
82        .map_err(|e| format!("Get all error: {:?}", e))?
83        .await
84        .map_err(|e| format!("Get all async error: {:?}", e))?;
85
86    let mut profiles: Vec<UserProfile> = Vec::new();
87    for js_val in values {
88        if let Ok(profile) = serde_wasm_bindgen::from_value(js_val) {
89            profiles.push(profile);
90        }
91    }
92
93    // Sort by created_at desc
94    profiles.sort_by(|a, b| b.created_at.cmp(&a.created_at));
95
96    Ok(profiles)
97}

100pub async fn delete_profile(id: &str) -> Result<(), String> {
101    let db = init_db().await?;
102    let tx = db
103        .transaction_on_one_with_mode(STORE_NAME, IdbTransactionMode::Readwrite)
104        .map_err(|e| format!("Transaction error: {:?}", e))?;
105    let store = tx
106        .object_store(STORE_NAME)
107        .map_err(|e| format!("Store error: {:?}", e))?;
108
109    store
110        .delete(&JsValue::from_str(id))
111        .map_err(|e| format!("Delete error: {:?}", e))?;
112
113    tx.await
114        .into_result()
115        .map_err(|e| format!("Transaction failed: {:?}", e))?;
116
117    Ok(())
118}
